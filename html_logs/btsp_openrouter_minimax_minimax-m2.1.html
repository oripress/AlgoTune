            if cycle is not None:
                return cycle
        
        return []

        adj = dist_matrix <= threshold
        np.fill_diagonal(adj, False)
        
        degrees = np.sum(adj, axis=1)
        if np.any(degrees < 2):
            return None

        G = nx.Graph()
        for i in range(n):
            for j in range(i + 1, n):
                if adj[i, j]:
                    G.add_edge(i, j, weight=dist_matrix[i, j])
        
        if G.number_of_nodes() != n:
            return None
        
        try:
            mst = nx.minimum_spanning_tree(G)
        except:
            return None

        if n == 3:
            return [0, 1, 2, 0]

        for start in range(min(5, n)):
            if adj[0, start]:
                path = self._mst_to_cycle(mst, start)
                if path and adj[path[-2], 0]:
                    return path
        
        return None

        cycle = list(nx.dfs_preorder_nodes(mst, source=start))
        if len(cycle) == len(mst.nodes()):

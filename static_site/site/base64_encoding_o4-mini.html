import numpy as np
import numba

# Base64 alphabet table and padding char
_TABLE = np.frombuffer(b"ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/", dtype=np.uint8)
_PAD = np.uint8(ord('='))

@numba.njit
def _b64_numba(data):
    n = data.size
    out_len = ((n + 2) // 3) * 4
    out = np.empty(out_len, dtype=np.uint8)
    k = 0
    for i in range(0, n, 3):
        rem = n - i
        b0 = data[i]
        b1 = data[i+1] if rem > 1 else 0
        b2 = data[i+2] if rem > 2 else 0
        bits = (b0 << 16) | (b1 << 8) | b2
        out[k]   = _TABLE[(bits >> 18) & 0x3F]
        out[k+1] = _TABLE[(bits >> 12) & 0x3F]
        out[k+2] = _TABLE[(bits >> 6)  & 0x3F] if rem > 1 else _PAD
        out[k+3] = _TABLE[bits & 0x3F]           if rem > 2 else _PAD
        k += 4
    return out

    out = np.empty(outlen, np.uint8)
    ntrip = inlen // 3

        j = idx * 4
        out[j]   = B64_TABLE[(bits >> 18) & 0x3F]
        out[j+1] = B64_TABLE[(bits >> 12) & 0x3F]
        out[j+2] = B64_TABLE[(bits >> 6)  & 0x3F]
        out[j+3] = B64_TABLE[ bits        & 0x3F]

import binascii

class Solver:
    def solve(self, problem, **kwargs):
        plaintext = problem.get("plaintext")
        # Materialize if not raw bytes
        if not isinstance(plaintext, (bytes, bytearray)):
            try:
                plaintext = bytes(plaintext)
            except Exception:
                try:
                    plaintext = plaintext.tobytes()
                except Exception:
                    pass
        # Single C call, no newline
        encoded_data = binascii.b2a_base64(plaintext, newline=False)
        return {"encoded_data": encoded_data}

from typing import Any
import numpy as np
from scipy.linalg import eigh
import inspect

# Check if eigh supports subset_by_index for fast partial solve
_sig = inspect.signature(eigh)
supports_subset = 'subset_by_index' in _sig.parameters

def _eighk(mat: np.ndarray, k: int):
    n = mat.shape[0]
    # if subset supported and selecting fewer than half, use partial
    if supports_subset and k < (n // 2):
        w, v = eigh(mat, subset_by_index=[n-k, n-1])
    else:
        # full decomposition then slice
        w_full, v_full = eigh(mat)
        w = w_full[-k:]
        v = v_full[:, -k:]
        w, v = w, v
        return w, v
    return w, v

class Solver:
    def solve(self, problem: dict[str, Any], **kwargs) -> Any:
        # load and center data matrix
        X = np.array(problem["X"], dtype=float, order='C')
        X -= X.mean(axis=0)
        m, n = X.shape
        k = int(problem["n_components"])
        if k <= 0:
            # return empty set of components
            return np.zeros((0, n), dtype=float)
        # choose smaller covariance dimension for eigendecomposition
        if n <= m:
            C = X.T @ X
            w, v = _eighk(C, k)
            # eigenvalues ascending; we want descending eigenvectors
            V = v[:, ::-1].T
        else:
            # sample covariance
            C = X @ X.T
            w, v = _eighk(C, k)
            # reverse to descending
            v = v[:, ::-1]
            w = w[::-1]
            # project to right singular vectors and normalize
            Vt = X.T @ v
            inv_sqrt = 1.0 / np.sqrt(w)
            V = (Vt * inv_sqrt[None, :]).T
        return V

from typing import Any
import numpy as np

class Solver:
    def solve(self, problem: dict[str, Any], **kwargs) -> Any:
        # Read data and parameters
        X = np.array(problem.get("X", []), dtype=float, order='C')
        n_components = int(problem.get("n_components", 0))

        # Validate input
        if X.ndim != 2:
            return []
        m, n = X.shape
        if m == 0 or n == 0 or n_components <= 0:
            return []

        # Cap components to feasible range
        k = min(n_components, m, n)

        # Center data
        X -= X.mean(axis=0)

        # If more samples than features, use covariance eigen-decomposition
        if m > n:
            C = X.T.dot(X)
            eigvals, eigvecs = np.linalg.eigh(C)
            idx = np.argsort(eigvals)[::-1][:k]
            V = eigvecs[:, idx].T  # (k x n)
            return V.tolist()

        # Otherwise, thin SVD
        U, S, Vt = np.linalg.svd(X, full_matrices=False)
        V = Vt[:k]
        return V.tolist()

        # use SciPy partial eigen if available
        if sp_eigh is not None:
            if n <= m:
                cov = X.T @ X
                vals, vecs = sp_eigh(cov, subset_by_index=(n-k, n-1))
                V = vecs[:, ::-1].T

                vals, vecs = sp_eigh(cov, subset_by_index=(m-k, m-1))

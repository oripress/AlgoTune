K Means Clustering

The k-means algorithm divides a set of n samples X into K disjoint clusters C, each described by the mean mu_j of the samples in the cluster. The means are commonly called the cluster "centroids"; note that they are not, in general, points from X, although they live in the same space.

The K-means algorithm aims to choose centroids that minimise the within-cluster sum-of-squares cost:

Cost = \sum_{i=1}^n min_{\mu_j \in C} ||x_i - \mu_j||^2

Given the centroids, it also induces a mapping for each data point to the index of centroid (cluster) it is assigned to.

Input: a dictionary with two keys:
    "X" : a 2d array of floats with shape n x d representing the sample
    "k" : integer, representing the number of clusters

Example input: {
    "X" : [[1, 2], [1, 4], [1, 0],
              [10, 2], [10, 4], [10, 0]],
    "k" : 2
}

Output: a list of int representing the clusters of each sample assigned to (starting from 0)

Example output: [1, 1, 1, 0, 0, 0]

import numpy as np

import numba

@numba.njit
def compute_distances(X, centers):
    """Compute squared distances between all points and centers."""
    n_samples = X.shape[0]
    k = centers.shape[0]
    distances = np.empty((n_samples, k), dtype=np.float32)
    
    for i in range(n_samples):
        for j in range(k):
            dist = 0.0
            for d in range(X.shape[1]):
                diff = X[i, d] - centers[j, d]
                dist += diff * diff
            distances[i, j] = dist
    
    return distances

@numba.njit
def assign_labels(distances):
    """Assign each point to nearest center."""
    return np.argmin(distances, axis=1)

@numba.njit
def update_centers(X, labels, k):
    """Update cluster centers."""
    n_features = X.shape[1]
    centers = np.zeros((k, n_features), dtype=np.float32)
    counts = np.zeros(k, dtype=np.int32)
    
    for i in range(X.shape[0]):
        label = labels[i]
        counts[label] += 1
        for j in range(n_features):
            centers[label, j] += X[i, j]
    
    for i in range(k):
        if counts[i] > 0:
            for j in range(n_features):
                centers[i, j] /= counts[i]
    
    return centers

@numba.njit
def kmeans_core(X, centers, max_iter=300):
    """Core K-means algorithm using Lloyd's method."""
    n_samples = X.shape[0]
    labels = np.zeros(n_samples, dtype=np.int32)
    
    for iteration in range(max_iter):
        # Compute distances and assign labels
        distances = compute_distances(X, centers)
        new_labels = assign_labels(distances)
        
        # Check convergence
        converged = True
        for i in range(n_samples):
            if labels[i] != new_labels[i]:
                converged = False
                break
        
        labels = new_labels
        
        if converged and iteration > 0:
            break
        
        # Update centers
        centers = update_centers(X, labels, centers.shape[0])
    
    return labels

from typing import Any
import numba

@numba.njit(fastmath=True, cache=True)
def fast_kmeans(X, k, max_iter=10):
    n_samples, n_features = X.shape
    
    # Simple initialization - first k points
    centers = X[:k].copy()
    
    labels = np.zeros(n_samples, dtype=np.int32)
    
    for _ in range(max_iter):
        # Assign labels - unrolled distance computation
        changed = False
        for i in range(n_samples):
            min_dist = np.inf
            best_j = 0
            
            for j in range(k):
                dist = 0.0
                for d in range(n_features):
                    diff = X[i, d] - centers[j, d]
                    dist += diff * diff
                
                if dist < min_dist:
                    min_dist = dist
                    best_j = j
            
            if labels[i] != best_j:
                changed = True
                labels[i] = best_j
        
        if not changed:
            break
        
        # Update centers
        centers[:] = 0.0
        counts = np.zeros(k, dtype=np.int32)
        
        for i in range(n_samples):
            label = labels[i]
            counts[label] += 1
            for d in range(n_features):
                centers[label, d] += X[i, d]
        
        for j in range(k):
            if counts[j] > 0:
                for d in range(n_features):
                    centers[j, d] /= counts[j]
    
    return labels

            return [0] * len(X)
        
        labels = fast_kmeans(X, k)

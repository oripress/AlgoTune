    def solve(self, problem, **kwargs):
        """
        Compute the cumulative integral of the 1D array using Simpson's rule,
        matching scipy.integrate.cumulative_simpson within tolerance.
        """
        y = np.array(problem["y"], dtype=float)
        dx = problem["dx"]
        N = y.shape[0]
        # If fewer than two points, no integral.
        if N < 2:
            return np.empty(0, dtype=float)
        # Prepare result of length N-1 (cumulative up to each subsequent point)
        res = np.empty(N-1, dtype=float)
        # First segment: trapezoidal rule
        res[0] = dx * (y[0] + y[1]) * 0.5
        # If only two points, return the single value
        if N == 2:
            return res
        # Prefix sums for even and odd indexed y-values
        idx = np.arange(N)
        prefix_even = (y * (idx % 2 == 0)).cumsum()
        prefix_odd  = (y * (idx % 2 == 1)).cumsum()
        # Indices i corresponding to integrals from y[0] to y[i], for i=2..N-1
        i_array = np.arange(2, N)
        out_idx = i_array - 1
        # Mask even and odd i's for Simpson and hybrid Simpson+trap
        even_mask = (i_array % 2 == 0)
        # Full composite Simpson for even i
        if np.any(even_mask):
            i_e = i_array[even_mask]
            oi_e = out_idx[even_mask]
            sum_odd = prefix_odd[i_e - 1]
            sum_even_int = prefix_even[i_e - 2] - y[0]
            res[oi_e] = dx/3 * (y[0] + y[i_e] + 4 * sum_odd + 2 * sum_even_int)
        # Hybrid: Simpson on first i-1 (even), plus trapezoid on last interval, for odd i
        odd_mask = ~even_mask
        if np.any(odd_mask):
            i_o = i_array[odd_mask]
            oi_o = out_idx[odd_mask]
            # Simpson on first i_o - 1 intervals
            sum_odd_sim = prefix_odd[i_o - 2]
            sum_even_sim = prefix_even[i_o - 3] - y[0]
            I_sim = dx/3 * (y[0] + y[i_o - 1] + 4 * sum_odd_sim + 2 * sum_even_sim)
            # Trapezoid for the last odd interval
            trap = dx * (y[i_o - 1] + y[i_o]) * 0.5
            res[oi_o] = I_sim + trap
        return res

import numpy as np
from numba import njit

@njit(cache=True)
def cumulative_simpson_nb(y, dx):
    N = y.shape[0]
    if N < 2:
        # no intervals
        return np.empty(0, np.float64)

    # output array of length N-1 (values at k=1..N-1)
    out = np.empty(N-1, np.float64)
    half_dx = dx * 0.5
    third_dx = dx / 3.0

    # k=1: simple trapezoid
    out[0] = half_dx * (y[0] + y[1])
    if N == 2:
        return out

    # k=2: standard Simpson on first two intervals
    out[1] = third_dx * (y[0] + 4.0*y[1] + y[2])
    # now handle k=3..N-1 in pairs
    M = (N - 1) // 2  # number of even indices <= N-1 divided by 2

    # full pairs: for m=1..M-1 we do Simpson at k=2m and trapezoid at k=2m+1
    for m in range(1, M):
        k = 2 * m
        idx = k - 1
        # Simpson over [k-2, k]
        out[idx] = out[idx-2] + third_dx * (y[k-2] + 4.0*y[k-1] + y[k])
        # trapezoid on the next odd step
        out[idx+1] = out[idx] + half_dx * (y[k] + y[k+1])

    # final step at m=M -> k=2*M
    k = 2 * M
    idx = k - 1
    # always do Simpson on that last even index
    out[idx] = out[idx-2] + third_dx * (y[k-2] + 4.0*y[k-1] + y[k])
    # if there is one more point beyond k, do a final trapezoid
    if k <= N - 2:
        out[idx+1] = out[idx] + half_dx * (y[k] + y[k+1])

    return out

# JIT‐compile at import so solve() is pure fast nopython
_dummy = cumulative_simpson_nb(np.zeros(3, np.float64), 1.0)

class Solver:
    def solve(self, problem, **kwargs):
        y = np.asarray(problem["y"], dtype=np.float64)

        return cumulative_simpson_nb(y, dx)

import numpy as np

class Solver:
    def solve(self, problem, **kwargs):
        """
        Compute cumulative integral using Simpson's rule with vectorized prefix sums.
        """
        y = np.asarray(problem["y"], dtype=np.float64)
        dx = float(problem["dx"])
        N = y.shape[0]
        # less than 2 points: empty result
        if N < 2:
            return np.empty(0, dtype=np.float64)
        # prefix sums for even and odd indices
        idx = np.arange(N)
        mask_even = (idx % 2 == 0)
        mask_odd = ~mask_even
        # even includes index 0, so subtract y[0] later when needed
        P_even = np.cumsum(y * mask_even)
        P_odd = np.cumsum(y * mask_odd)
        # initialize result array length N-1
        res = np.empty(N-1, dtype=np.float64)
        # first interval: trapezoid
        res[0] = 0.5 * dx * (y[0] + y[1])
        if N > 2:
            # k from 2 to N-1
            k = np.arange(2, N)
            M = k // 2
            # sum of odd-indexed y up to k-1
            S_odd = P_odd[k - 1]
            # sum of even-indexed y up to (2M-2)
            # P_even includes y[0], subtract y[0] to get j>=2
            E_even = P_even[2 * M - 2] - y[0]
            # y at 2M and y at k
            y2M = y[2 * M]
            yk = y[k]
            # Simpson composite for even k, and base for odd k
            simpson = (dx / 3.0) * (y[0] + y2M + 4.0 * S_odd + 2.0 * E_even)
            # if k odd, add trapezoid for last segment
            odd = (k & 1) == 1
            # extra trapezoid term (zero for even k)
            extra = 0.5 * dx * (y2M + yk) * odd
            res[1:] = simpson + extra
        return res

import zlib
import struct
import concurrent.futures
import os

# Global configuration
CPU_COUNT = min((os.cpu_count() or 1), 4)
MIN_CHUNK = 1 << 16  # 64KB per thread
THRESHOLD = CPU_COUNT * MIN_CHUNK
LEVEL = 8  # Compression level tradeoff
WINDOW_BITS = -15  # Raw DEFLATE
MEM_LEVEL = zlib.DEF_MEM_LEVEL
STRATEGY = zlib.Z_DEFAULT_STRATEGY

# GZIP wrapper constants
HEADER = b'\x1f\x8b\x08\x00\x00\x00\x00\x00\x02\xff'
PACK = struct.Struct('<II').pack
CRC32 = zlib.crc32

def _compress_chunk(seg: bytes) -> bytes:
    # Create a raw DEFLATE stream
    comp = zlib.compressobj(LEVEL, zlib.DEFLATED, WINDOW_BITS, MEM_LEVEL, STRATEGY)
    raw = comp.compress(seg) + comp.flush()
    # Footer: CRC32 and ISIZE for this segment
    crc = CRC32(seg) & 0xFFFFFFFF
    isize = len(seg) & 0xFFFFFFFF
    return HEADER + raw + PACK(crc, isize)

class Solver:
    def solve(self, problem, **kwargs):
        data = problem["plaintext"]
        n = len(data)
        # For small inputs, do single-shot compression
        if n <= THRESHOLD:
            comp = zlib.compressobj(LEVEL, zlib.DEFLATED, WINDOW_BITS, MEM_LEVEL, STRATEGY)
            raw = comp.compress(data) + comp.flush()
            crc = CRC32(data) & 0xFFFFFFFF
            isize = n & 0xFFFFFFFF
            return {"compressed_data": HEADER + raw + PACK(crc, isize)}
        # Split into chunks for parallel compression
        threads = CPU_COUNT
        chunk_size = n // threads
        segments = [data[i*chunk_size:(i+1)*chunk_size] for i in range(threads-1)]
        segments.append(data[(threads-1)*chunk_size:])
        # Compress in parallel
        with concurrent.futures.ThreadPoolExecutor(max_workers=threads) as executor:
            parts = list(executor.map(_compress_chunk, segments))
        # Concatenate gzip members
        return {"compressed_data": b''.join(parts)}

_LEVEL = 9
_MASK = 0xFFFFFFFF

_DEFLATED = zlib.DEFLATED
_MAX_WBITS = zlib.MAX_WBITS
_MEMLEVEL = zlib.DEF_MEM_LEVEL
_STRATEGY = zlib.Z_DEFAULT_STRATEGY

# Parallel executor setup
_CPU_COUNT = os.cpu_count() or 1
_EXECUTOR = ThreadPoolExecutor(max_workers=_CPU_COUNT)

# Threshold (bytes) above which to parallelize compression
_PARALLEL_THRESHOLD = 256 * 1024  # 256 KiB

def _compress_member(chunk: bytes) -> bytes:
    # Raw deflate without zlib wrapper
    co = zlib.compressobj(_LEVEL, _DEFLATED, -_MAX_WBITS, _MEMLEVEL, _STRATEGY)
    body = co.compress(chunk) + co.flush()
    crc = zlib.crc32(chunk) & _MASK
    isize = len(chunk) & _MASK
    out = bytearray(10 + len(body) + 8)
    out[0:10] = _HEADER
    out[10:10 + len(body)] = body

import zlib
import struct
import threading
import os

# zlib raw DEFLATE constants
_DEFLATED = zlib.DEFLATED
_MAX_WBITS = zlib.MAX_WBITS
_MEM_LEVEL = zlib.DEF_MEM_LEVEL
_STRATEGY = zlib.Z_DEFAULT_STRATEGY

# GZIP header and trailer packing
_HEADER = b'\x1f\x8b\x08\x00\x00\x00\x00\x00\x02\xff'
_PACK = struct.Struct('<II').pack
_LEVEL = 9
_MASK = 0xFFFFFFFF

# Parallelization settings
_WORKERS = os.cpu_count() or 1
_CHUNK_BASE = 1 << 16  # 64 KiB
_PARALLEL_THRESHOLD = _CHUNK_BASE * _WORKERS

def _compress_chunk(data: bytes) -> bytes:
    co = zlib.compressobj(_LEVEL, _DEFLATED, -_MAX_WBITS, _MEM_LEVEL, _STRATEGY)
    body = co.compress(data) + co.flush()
    trailer = _PACK(zlib.crc32(data) & _MASK, len(data) & _MASK)
    return _HEADER + body + trailer

class Solver:
    def solve(self, problem, **kwargs):
        data = problem["plaintext"]
        n = len(data)
        # Sequential path for small data or single CPU
        if n <= _PARALLEL_THRESHOLD or _WORKERS <= 1:
            return {"compressed_data": _compress_chunk(data)}
        # Parallel compress across chunks
        parts = _WORKERS
        chunk_size = (n + parts - 1) // parts
        results = [None] * parts
        threads = []
        for i in range(parts):
            start = i * chunk_size
            end = n if i == parts - 1 else start + chunk_size
            # capture loop variables by default arguments
            def worker(idx=i, s=start, e=end):
                results[idx] = _compress_chunk(data[s:e])
            t = threading.Thread(target=worker)
            t.daemon = True
            t.start()
            threads.append(t)
        for t in threads:
            t.join()
        return {"compressed_data": b"".join(results)}

import struct
import os
import concurrent.futures

# Raw DEFLATE parameters
_DEFLATED = zlib.DEFLATED
_MAX_WBITS = zlib.MAX_WBITS
_MEM_LEVEL = zlib.DEF_MEM_LEVEL
_STRATEGY = zlib.Z_DEFAULT_STRATEGY

# GZIP header and trailer packing
_HEADER = b'\x1f\x8b\x08\x00\x00\x00\x00\x00\x02\xff'
_PACK = struct.Struct('<II').pack
_LEVEL = 9
_MASK = 0xFFFFFFFF

# Parallel settings
_WORKERS = os.cpu_count() or 1
_CHUNK_SIZE = 1 << 20           # 1 MiB per chunk
_PARALLEL_THRESHOLD = _WORKERS * _CHUNK_SIZE
_EXECUTOR = concurrent.futures.ThreadPoolExecutor(max_workers=_WORKERS)

def _compress_chunk(data_chunk: bytes) -> bytes:
    # Raw DEFLATE (no wrapper) for gzip member
    co = zlib.compressobj(_LEVEL, _DEFLATED, -_MAX_WBITS, _MEM_LEVEL, _STRATEGY)
    body = co.compress(data_chunk) + co.flush()
    trailer = _PACK(zlib.crc32(data_chunk) & _MASK, len(data_chunk) & _MASK)
    return _HEADER + body + trailer

        n = len(data)

        if n < _PARALLEL_THRESHOLD or _WORKERS < 2:
            return {"compressed_data": _compress_chunk(data)}

        parts = _WORKERS

        # Schedule parallel compression
        chunks = (mv[i*chunk_size : min(n, (i+1)*chunk_size)] for i in range(parts))

        # Concatenate gzip members

                    k2 * y2 * y2,

import diffrax

class Solver:
    def __init__(self):
        # Define the RHS of the Robertson system
        def rhs(t, y, args):
            k1, k2, k3 = args
            y1, y2, y3 = y
            return jnp.array([
                -k1 * y1 + k3 * y2 * y3,
                k1 * y1 - k2 * y2 * y2 - k3 * y2 * y3,
                k2 * y2 * y2
            ])
        # Wrap into a diffrax term and select Radau5
        self.term = diffrax.ODETerm(rhs)
        self.solver = diffrax.Radau5()
        # JIT‐compiled solve function: maps (k1,k2,k3,y0,t0,t1) → final y
        def solve_fun(k1, k2, k3, y0, t0, t1):

                self.term,
                self.solver,
                t0=t0,
                t1=t1,

                y0=y0,
                args=(k1, k2, k3),
                atol=1e-8,
                rtol=1e-7,

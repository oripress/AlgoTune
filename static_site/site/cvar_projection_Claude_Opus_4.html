    """Compute gradient of CVaR constraint."""
    losses = A @ x
    n = len(losses)
    k = int((1 - beta) * n)
    
    if k == 0:
        idx = np.argmax(losses)
        grad = A[idx] / 1.0
    else:
        # Get indices of k largest losses
        indices = np.argpartition(losses, -k)[-k:]
        grad = np.mean(A[indices], axis=0)
    
    return grad

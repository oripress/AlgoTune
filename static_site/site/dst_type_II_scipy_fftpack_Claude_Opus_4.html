import numpy as np
from numpy.typing import NDArray
from typing import Any
import numba as nb
import math

class Solver:
    def __init__(self):
        # Pre-compile the DST function during initialization
        self.dst2_2d = self._create_dst2_function()
    
    def _create_dst2_function(self):
        @nb.njit(fastmath=True, cache=True)
        def dst2_1d(x):
            """Compute 1D DST Type II using direct computation."""
            n = len(x)
            result = np.zeros(n)
            for k in range(n):
                sum_val = 0.0
                for j in range(n):
                    sum_val += x[j] * math.sin(math.pi * (k + 1) * (j + 0.5) / n)
                result[k] = sum_val * math.sqrt(2.0 / n)
            return result
        
        @nb.njit(fastmath=True, cache=True, parallel=True)
        def dst2_2d(arr):
            """Compute 2D DST Type II by applying 1D DST to rows then columns."""
            n = arr.shape[0]
            # First apply DST to each row
            temp = np.zeros_like(arr)
            for i in nb.prange(n):
                temp[i] = dst2_1d(arr[i])
            
            # Then apply DST to each column
            result = np.zeros_like(arr)
            for j in nb.prange(n):
                result[:, j] = dst2_1d(temp[:, j])
            
            return result
        
        # Warm up the JIT compilation
        test_arr = np.ones((3, 3), dtype=np.float64)
        dst2_2d(test_arr)
        
        return dst2_2d
    
    def solve(self, problem: NDArray, **kwargs) -> Any:
        """Compute the 2D DST Type II using Numba-optimized implementation."""
        # Convert to numpy array if needed and ensure float64
        arr = np.asarray(problem, dtype=np.float64)
        return self.dst2_2d(arr)

import numpy as np

class Solver:
    def solve(self, problem: list[float]) -> list[complex]:
        """
        Solve the polynomial problem by finding all roots of the polynomial.
        """
        # Use numpy.roots which is already optimized
        roots = np.roots(problem)
        
        # Use numpy's lexsort for efficient sorting
        # Sort by real part (descending), then by imaginary part (descending)
        # lexsort sorts in ascending order, so we negate the values
        sorted_indices = np.lexsort((-roots.imag, -roots.real))
        
        return roots[sorted_indices].tolist()

import jax.numpy as jnp
from jax import jit
import numpy as np

class Solver:
    def __init__(self):
        # Pre-compile the root-finding function
        @jit
        def find_roots(coeffs):
            # Build companion matrix
            n = len(coeffs) - 1
            if n == 0:
                return jnp.array([], dtype=jnp.complex128)
            
            # Normalize coefficients
            coeffs = coeffs / coeffs[0]
            
            # Build companion matrix
            companion = jnp.zeros((n, n), dtype=jnp.complex128)
            companion[1:, :-1] = jnp.eye(n-1)
            companion[:, -1] = -coeffs[1:][::-1]
            
            # Compute eigenvalues
            eigenvals = jnp.linalg.eigvals(companion)
            return eigenvals
        
        self.find_roots_jit = find_roots
    
    def solve(self, problem: list[float]) -> list[complex]:
        """
        Solve the polynomial problem by finding all roots of the polynomial.
        """
        coeffs = jnp.array(problem, dtype=jnp.float64)
        
        # Use JIT-compiled function
        roots = self.find_roots_jit(coeffs)
        
        # Convert to numpy for sorting
        roots_np = np.array(roots)
        sorted_roots = sorted(roots_np, key=lambda z: (z.real, z.imag), reverse=True)
        return sorted_roots

import numpy as np
from numba import njit, complex128, float64
from numba.typed import List

class Solver:
    def __init__(self):
        # Pre-compile the sorting function
        @njit(cache=True)
        def sort_roots(roots_real, roots_imag):
            n = len(roots_real)
            indices = np.arange(n)
            
            # Sort by real part (descending), then by imaginary part (descending)
            for i in range(n):
                for j in range(i + 1, n):
                    if (roots_real[indices[j]] > roots_real[indices[i]] or 
                        (roots_real[indices[j]] == roots_real[indices[i]] and 
                         roots_imag[indices[j]] > roots_imag[indices[i]])):
                        indices[i], indices[j] = indices[j], indices[i]
            
            return indices
        
        self.sort_roots = sort_roots
    
    def solve(self, problem: list[float]) -> list[complex]:
        """
        Solve the polynomial problem by finding all roots of the polynomial.
        """
        # Use numpy.roots which is already optimized
        roots = np.roots(problem)
        
        # Extract real and imaginary parts for fast sorting
        roots_real = roots.real
        roots_imag = roots.imag
        
        # Use numba-compiled sorting
        sorted_indices = self.sort_roots(roots_real, roots_imag)
        
        return [roots[i] for i in sorted_indices]

import numpy as np

class Solver:
    def solve(self, problem: list[float]) -> list[complex]:
        """
        Solve the polynomial problem by finding all roots of the polynomial.
        """
        # Use numpy.roots which is already optimized
        roots = np.roots(problem)
        
        # Use numpy's lexsort for efficient sorting
        # Sort by real part (descending), then by imaginary part (descending)
        # lexsort sorts in ascending order, so we negate the values
        sorted_indices = np.lexsort((-roots.imag, -roots.real))
        
        return roots[sorted_indices].tolist()

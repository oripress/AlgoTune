import jax
import jax.numpy as jnp
import diffrax
import numpy as np

class Solver:
    def __init__(self):
        # Pre-compile the solve function
        @jax.jit
        def jit_solve(y0, t0, t1, mu):
            def vdp(t, y, args):
                x, v = y
                dx_dt = v
                dv_dt = mu * ((1 - x**2) * v - x)
                return jnp.array([dx_dt, dv_dt])
            
            term = diffrax.ODETerm(vdp)
            solver = diffrax.Kvaerno5()  # Implicit solver for stiff problems
            
            # Use adaptive step size controller
            stepsize_controller = diffrax.PIDController(
                rtol=1e-9,
                atol=1e-10,
                pcoeff=0.4,
                icoeff=0.3,
                dcoeff=0.0,
                dtmax=0.1
            )
            
            solution = diffrax.diffeqsolve(
                term,
                solver,
                t0=t0,
                t1=t1,
                dt0=1e-4,
                y0=y0,
                stepsize_controller=stepsize_controller,
                max_steps=100000
            )
            
            return solution.ys[-1]
        
        self.jit_solve = jit_solve
    
    def solve(self, problem, **kwargs):
        """Solve the stiff Van der Pol equation."""
        y0 = jnp.array(problem["y0"])
        t0 = float(problem["t0"])
        t1 = float(problem["t1"])
        mu = float(problem["mu"])
        
        result = self.jit_solve(y0, t0, t1, mu)
        return result.tolist()

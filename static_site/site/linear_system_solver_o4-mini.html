# Numba‐accelerated in‐place Gaussian elimination
@njit(fastmath=True)
def _solve_numba(A, b):
    n = A.shape[0]
    for k in range(n):
        # partial pivot
        maxrow = k
        maxval = A[k, k] if A[k, k] >= 0 else -A[k, k]
        for i in range(k+1, n):
            v = A[i, k] if A[i, k] >= 0 else -A[i, k]
            if v > maxval:
                maxval = v
                maxrow = i
        # swap if needed
        if maxrow != k:
            for j in range(k, n):
                tmp = A[k, j]; A[k, j] = A[maxrow, j]; A[maxrow, j] = tmp
            tmpb = b[k]; b[k] = b[maxrow]; b[maxrow] = tmpb
        piv = A[k, k]
        # elimination
        for i in range(k+1, n):
            factor = A[i, k] / piv
            b[i] -= factor * b[k]
            for j in range(k+1, n):
                A[i, j] -= factor * A[k, j]
    # back substitution
    x = np.empty(n, dtype=np.float64)
    for i in range(n-1, -1, -1):
        s = b[i]
        for j in range(i+1, n):
            s -= A[i, j] * x[j]
        x[i] = s / A[i, i]
    return x

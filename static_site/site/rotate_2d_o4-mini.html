import numpy as np
from scipy import ndimage

class Solver:
    def solve(self, problem, **kwargs):
        """
        Rotate a 2D image counter-clockwise by a specified angle using
        cubic spline interpolation (order=3) and constant boundary padding.
        """
        # Load image and angle
        image = problem["image"]
        angle = problem["angle"]

        # Convert to numpy array
        arr = np.asarray(image, dtype=float)
        n_rows, n_cols = arr.shape

        # Compute center of the image
        center_i = (n_rows - 1) / 2.0
        center_j = (n_cols - 1) / 2.0

        # Convert angle to radians and get sin/cos
        theta = np.deg2rad(angle)
        cos_t = np.cos(theta)
        sin_t = np.sin(theta)

        # Create coordinate arrays relative to center
        i = np.arange(n_rows, dtype=float) - center_i
        j = np.arange(n_cols, dtype=float) - center_j

        # Compute source coordinates after inverse rotation (R(-theta))
        # Using broadcasting: coords_i_src and coords_j_src are shape (n_rows, n_cols)
        coords_i_src = cos_t * i[:, None] + sin_t * j[None, :] + center_i
        coords_j_src = -sin_t * i[:, None] + cos_t * j[None, :] + center_j

        # Interpolate using C-coded map_coordinates for cubic spline (order=3)
        rotated = ndimage.map_coordinates(
            arr,
            [coords_i_src, coords_j_src],
            order=3,
            mode='constant',
            cval=0.0,
            prefilter=True
        )

        # Return as nested lists
        return {"rotated_image": rotated.tolist()}

from scipy.ndimage import affine_transform

    def solve(self, problem, **kwargs):
        """
        Rotate a 2D image counter-clockwise by a specified angle using
        cubic spline interpolation (order=3) and constant boundary padding.
        """

        image = problem["image"]
        angle = problem["angle"]

        # Convert to numpy array
        arr = np.asarray(image, dtype=float)
        n_rows, n_cols = arr.shape

        # Compute image center coordinates
        center_i = (n_rows - 1) / 2.0
        center_j = (n_cols - 1) / 2.0

        # Convert angle to radians
        theta = np.deg2rad(angle)
        cos_t = np.cos(theta)
        sin_t = np.sin(theta)

        # Affine rotation matrix for inverse mapping
        # For each output y, input coordinates x = A @ y + offset
        matrix = np.array([[cos_t, sin_t],

        offset_i = center_i - (cos_t * center_i + sin_t * center_j)

        rotated = affine_transform(
            arr,
            matrix,
            offset=offset,
            output_shape=(n_rows, n_cols),
            order=3,
            mode='constant',
            cval=0.0,
            prefilter=True
        )

        # Return result as nested Python lists
        return {"rotated_image": rotated.tolist()}

    ax = abs(x)
    if ax <= 1.0:
        return (4.0 - 6.0 * ax * ax + 3.0 * ax * ax * ax) / 6.0
    elif ax < 2.0:

    else:

def _numba_rotate(arr, cos_t, sin_t, center_i, center_j):
    n_rows, n_cols = arr.shape
    out = np.empty_like(arr)
    for i in prange(n_rows):
        i_rel = i - center_i
        for j in range(n_cols):
            j_rel = j - center_j
            src_i = cos_t * i_rel + sin_t * j_rel + center_i
            src_j = -sin_t * i_rel + cos_t * j_rel + center_j
            # cubic interpolation over 4x4 neighborhood
            floor_i = math.floor(src_i)
            floor_j = math.floor(src_j)
            val = 0.0

                    if jj < 0 or jj >= n_cols:
                        continue

            out[i, j] = val
    return out

        cubic spline interpolation (order=3) and constant boundary padding.

        angle = problem["angle"]

        # Convert to numpy array with float64 for exact numeric matching
        arr = np.asarray(image, dtype=np.float64)
        n_rows, n_cols = arr.shape

        # Compute rotation matrix for inverse mapping (rotate -angle)
        theta = np.deg2rad(angle)
        cos_t = np.cos(theta)
        sin_t = np.sin(theta)

        offset = center - matrix.dot(center)

            arr,

            output_shape=arr.shape,
            order=3,
            mode='constant',
            cval=0.0,
            prefilter=True
        )

import numpy as np
from scipy.ndimage import map_coordinates

class Solver:
    def solve(self, problem, **kwargs):
        # Extract inputs
        image = problem["image"]
        angle = problem["angle"]

        # Convert to NumPy array (float64) for reference precision
        arr = np.asarray(image, dtype=np.float64)
        nrows, ncols = arr.shape

        # Compute image center
        cy = (nrows - 1) / 2.0
        cx = (ncols - 1) / 2.0

        # Precompute rotation matrix components
        theta = np.deg2rad(angle)
        cos_t = np.cos(theta)
        sin_t = np.sin(theta)

        # Build grid of output coordinates
        rows, cols = np.indices((nrows, ncols), dtype=np.float64)
        rows0 = rows - cy
        cols0 = cols - cx

        # Map output coordinates back to input space (inverse rotation)
        y = rows0 * cos_t + cols0 * sin_t + cy
        x = -rows0 * sin_t + cols0 * cos_t + cx

        # Interpolate with cubic spline and constant padding
        rotated = map_coordinates(
            arr,
            [y, x],
            order=3,
            mode='constant',
            cval=0.0,
            prefilter=True
        )

        # Return as nested Python list
        return {"rotated_image": rotated.tolist()}

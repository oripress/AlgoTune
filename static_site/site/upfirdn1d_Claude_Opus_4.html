import numpy as np
import numba

@numba.njit
def upfirdn_numba(h, x, up, down):
    """
    Efficient upfirdn implementation using numba.
    Combines upsampling, FIR filtering, and downsampling in one pass.
    """
    h = np.asarray(h)
    x = np.asarray(x)
    
    # Length calculations
    x_len = len(x)
    h_len = len(h)
    
    # Output length calculation (matching scipy's behavior)
    up_len = x_len * up
    conv_len = up_len + h_len - 1
    out_len = (conv_len + down - 1) // down
    
    # Allocate output
    y = np.zeros(out_len, dtype=np.float64)
    
    # Combined upsampling, filtering, and downsampling
    for n in range(out_len):
        # Output sample index before downsampling
        n_up = n * down
        
        # Calculate the range of input samples that contribute
        # For upsampled signal, non-zero values are at indices k*up
        k_start = max(0, (n_up - h_len + 1 + up) // up)
        k_end = min(x_len, (n_up + up) // up)
        
        # Accumulate contributions
        sum_val = 0.0
        for k in range(k_start, k_end):
            # Index in upsampled signal
            idx_up = k * up
            # Filter tap index
            h_idx = n_up - idx_up
            if 0 <= h_idx < h_len:
                sum_val += x[k] * h[h_idx]
        
        y[n] = sum_val
    
    return y

class Solver:
    def solve(self, problem: list) -> list:
        """
        Compute the upfirdn operation for each problem definition in the list.
        
        :param problem: A list of tuples (h, x, up, down).
        :return: A list of 1D arrays representing the upfirdn results.
        """
        results = []
        for h, x, up, down in problem:
            # Convert to numpy arrays for numba
            h_arr = np.asarray(h, dtype=np.float64)
            x_arr = np.asarray(x, dtype=np.float64)
            res = upfirdn_numba(h_arr, x_arr, up, down)
            results.append(res)
        return results

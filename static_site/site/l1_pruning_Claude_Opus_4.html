        u = np.abs(v)

        if u.sum() <= k:
            return {"solution": v.tolist()}
        
        # Sort in descending order using the fastest sorting algorithm
        mu = np.sort(u, kind='heapsort')[::-1]

        n = len(mu)
        cumsum = np.cumsum(mu)
        j_range = np.arange(1, n + 1)

        mask = mu < thresholds
        if mask.any():

            theta = 0.0

import jax
import jax.numpy as jnp

@jax.jit
def solve_l1_projection(v, k):
    """JAX JIT-compiled L1 projection solver."""
    # Work with absolute values
    u = jnp.abs(v)
    
    # If sum of absolute values is <= k, no thresholding needed
    sum_u = jnp.sum(u)
    
    # Sort in descending order
    mu = jnp.sort(u)[::-1]
    
    # Vectorized computation to find threshold
    n = mu.shape[0]
    cumsum = jnp.cumsum(mu)
    j_range = jnp.arange(1, n + 1)
    thresholds = (cumsum - k) / j_range
    
    # Find the optimal threshold
    mask = mu < thresholds
    # Use where to handle the case when no element satisfies the condition
    j = jnp.where(jnp.any(mask), jnp.argmax(mask), n-1)
    theta = jnp.where(sum_u > k, thresholds[j], 0.0)
    
    # Apply soft thresholding
    w = jnp.where(u > theta, (u - theta) * jnp.sign(v), 0.0)
    
    return w

import numpy as np
from typing import Any

class Solver:
    def solve(self, problem: dict[str, Any]) -> dict[str, list]:
        """
        Solve the L1 proximal operator problem:
        min_w ‚Äñv-w‚Äñ¬≤ s.t. ‚Äñw‚Äñ‚ÇÅ ‚â§ k
        """
        v = np.asarray(problem["v"], dtype=np.float64)
        k = problem["k"]
        
        # Work with absolute values
        u = np.abs(v)
        
        # Early exit if no thresholding needed
        sum_u = u.sum()
        if sum_u <= k:
            return {"solution": v.tolist()}
        
        # Use partition instead of full sort for better performance
        # We only need to find the threshold, not the full sorted order
        n = len(u)
        mu = -np.partition(-u, n-1)  # Descending order via negation
        
        # Binary search for the optimal threshold
        cumsum = 0.0
        theta = 0.0
        
        for j in range(n):
            cumsum += mu[j]
            candidate_theta = (cumsum - k) / (j + 1)
            if mu[j] < candidate_theta:
                theta = candidate_theta
                break
        
        # Apply soft thresholding in-place for efficiency
        result = np.empty_like(v)
        for i in range(n):
            if u[i] > theta:
                result[i] = (u[i] - theta) * (1 if v[i] >= 0 else -1)
            else:
                result[i] = 0.0
        
        return {"solution": result.tolist()}

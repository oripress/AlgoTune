import numpy as np
import scipy.sparse
from ortools.graph import pywrapgraph


class Solver:
    def solve(self, problem: dict[str, Any]) -> dict[str, dict[str, list[int]]]:
        try:
            mat = scipy.sparse.csr_matrix(
                (problem["data"], problem["indices"], problem["indptr"]), shape=problem["shape"]
            )
        except Exception as e:
            return {"assignment": {"row_ind": [], "col_ind": []}}

        n = mat.shape[0]
        
        # Create the assignment solver
        assignment = pywrapgraph.LinearSumAssignment()
        
        # Add edges from the sparse matrix
        for i in range(n):
            for j_idx in range(mat.indptr[i], mat.indptr[i + 1]):
                j = mat.indices[j_idx]
                cost = int(mat.data[j_idx] * 1000000)  # Scale to integer
                assignment.AddArcWithCost(i, j, cost)
        
        # Solve
        solve_status = assignment.Solve()
        
        if solve_status == assignment.OPTIMAL:
            row_ind = []
            col_ind = []
            for i in range(assignment.NumNodes()):
                col_ind.append(assignment.RightMate(i))
                row_ind.append(i)
            return {"assignment": {"row_ind": row_ind, "col_ind": col_ind}}
        else:
            return {"assignment": {"row_ind": [], "col_ind": []}}

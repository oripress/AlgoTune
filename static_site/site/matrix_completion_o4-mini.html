class Solver:
    def solve(self, problem, **kwargs) -> dict:
        inds = np.array(problem["inds"])
        a = np.array(problem["a"])
        n = problem["n"]

        # Build list of unknown indices
        xx, yy = np.meshgrid(np.arange(n), np.arange(n))
        allinds = np.vstack((yy.flatten(), xx.flatten())).T
        # Mask positions already observed
        mask_known = ((allinds[None, :, :] == inds[:, None, :]).all(2)).any(0)
        otherinds = allinds[~mask_known]

        # Define CVXPY variable and objective
        B = cp.Variable((n, n), pos=True)
        objective = cp.Minimize(cp.pf_eigenvalue(B))

        # Constraints: observed entries fixed, product of unknowns equals 1
        constraints = [B[inds[:, 0], inds[:, 1]] == a]
        if otherinds.shape[0] > 0:
            constraints.append(cp.prod(B[otherinds[:, 0], otherinds[:, 1]]) == 1.0)

        prob = cp.Problem(objective, constraints)
        try:
            result = prob.solve(gp=True)
        except cp.SolverError:

        xx, yy = np.meshgrid(np.arange(n), np.arange(n))

        mask = ((allinds[None, :, :] == inds[:, None, :]).all(2)).any(0)
        otherinds = allinds[~mask]

        # Define CVXPY variable
        Bvar = cp.Variable((n, n), pos=True)

        # Constraints: fixed entries and product constraint
        constraints = [Bvar[inds[:, 0], inds[:, 1]] == a]

        prob = cp.Problem(objective, constraints)

        result = prob.solve(gp=True)

class Solver:
    def solve(self, problem, **kwargs) -> dict:
        """
        Solves the Perron-Frobenius matrix completion by minimizing the spectral radius
        using gradient descent on log-variables under product constraint.
        """
        inds = np.array(problem["inds"], dtype=int)
        a = np.array(problem["a"], dtype=float)
        n = int(problem["n"])

        # Build mask and known matrix
        B = np.zeros((n, n), dtype=float)
        mask = np.zeros((n, n), dtype=bool)
        mask[inds[:, 0], inds[:, 1]] = True
        B[inds[:, 0], inds[:, 1]] = a

        # Unknown positions
        rows, cols = np.where(~mask)
        m = rows.size

        # If no unknowns, return PF eigenvalue directly
        if m == 0:
            lam, _ = self.pf_power(B, max_iter=1000, tol=1e-8)
            return {"B": B.tolist(), "optimal_value": float(lam)}

        # Initialize log-variables for unknown entries
        l = np.zeros(m, dtype=float)

        # Gradient descent hyperparameters
        lr = kwargs.get("lr", 0.1)
        max_iters = kwargs.get("max_iters", 500)
        tol = kwargs.get("tol", 1e-6)

        lam_prev = None

        for _ in range(max_iters):
            # Update matrix entries
            x = np.exp(l)
            B[rows, cols] = x

            # Power iteration for PF eigenvalue and eigenvectors
            lam, v = self.pf_power(B, max_iter=1000, tol=1e-8)
            _, w = self.pf_power(B.T, max_iter=1000, tol=1e-8)

            # Convergence check on objective
            if lam_prev is not None and abs(lam_prev - lam) < tol:
                break
            lam_prev = lam

            # Compute gradient of lam w.r.t l
            denom = np.dot(w, v)
            grad = (w[rows] * v[cols] * x) / (denom + 1e-16)
            # Project gradient to satisfy sum(l) = constant (zero)
            grad_proj = grad - grad.mean()

            # Gradient descent step
            l -= lr * grad_proj
            # Enforce product constraint: sum l == 0
            l -= l.mean()

        # Final update
        B[rows, cols] = np.exp(l)
        lam, _ = self.pf_power(B, max_iter=1000, tol=1e-8)

        return {"B": B.tolist(), "optimal_value": float(lam)}

    def pf_power(self, M, max_iter=1000, tol=1e-8):
        """
        Power iteration to compute largest eigenvalue (spectral radius)
        and corresponding eigenvector of positive matrix M.
        """
        n = M.shape[0]
        v = np.ones(n, dtype=float)
        v /= np.linalg.norm(v)
        for _ in range(max_iter):
            v_new = M.dot(v)
            lam = np.linalg.norm(v_new)
            if lam == 0:
                return 0.0, v
            v_new /= lam
            if np.linalg.norm(v_new - v) < tol:
                v = v_new
                break
            v = v_new
        return float(lam), v

import numpy as np
import math
from ortools.linear_solver import pywraplp

class Solver:
    def solve(self, problem, **kwargs) -> dict:
        """
        Solves the Perron-Frobenius matrix completion by minimizing the spectral radius
        using a linear program derived from the geometric programming formulation.
        """
        # Parse input
        inds = np.array(problem["inds"], dtype=int)
        a = np.array(problem["a"], dtype=float)
        n = int(problem["n"])

        # Build mask and initialize B with known entries
        mask = np.zeros((n, n), dtype=bool)
        mask[inds[:, 0], inds[:, 1]] = True
        B = np.zeros((n, n), dtype=float)
        B[inds[:, 0], inds[:, 1]] = a

        # Identify unknown entries
        otherinds = np.argwhere(~mask)
        m = otherinds.shape[0]

        # If no unknowns, compute PF eigenvalue directly
        if m == 0:
            eigs = np.linalg.eigvals(B)
            lam = float(np.max(np.abs(eigs)))
            return {"B": B.tolist(), "optimal_value": lam}

        # Set up LP solver
        solver = pywraplp.Solver.CreateSolver('GLOP')
        inf = solver.infinity()

        # Variables: l_k = log(B_unknown_k), u_i for eigenvector logs, s = log(lambda)
        l_vars = [solver.NumVar(-inf, inf, f'l{k}') for k in range(m)]
        u_vars = [solver.NumVar(-inf, inf, f'u{i}') for i in range(n)]
        s_var = solver.NumVar(-inf, inf, 's')

        # Constraints for unknown entries: l_k + u_j - u_i <= s
        for k, (i, j) in enumerate(otherinds):
            solver.Add(l_vars[k] + u_vars[j] - u_vars[i] <= s_var)

        # Constraints for known entries: log(a) + u_j - u_i <= s
        log_a = np.log(a)
        for idx, (i, j) in enumerate(inds):
            solver.Add(log_a[idx] + u_vars[j] - u_vars[i] <= s_var)

        # Product constraint: sum of logs of unknowns == 0
        solver.Add(solver.Sum(l_vars) == 0)

        # Normalization of eigenvector logs: sum u_i == 0
        solver.Add(solver.Sum(u_vars) == 0)

        # Objective: minimize s
        solver.Minimize(s_var)

        status = solver.Solve()
        if status != pywraplp.Solver.OPTIMAL:
            # Fallback: fill unknowns with 1 and compute PF directly
            for (i, j) in otherinds:
                B[i, j] = 1.0
            eigs = np.linalg.eigvals(B)
            lam = float(np.max(np.abs(eigs)))
            return {"B": B.tolist(), "optimal_value": lam}

        # Extract solution
        l_vals = [v.solution_value() for v in l_vars]
        s_val = s_var.solution_value()

        # Build completed matrix
        for k, (i, j) in enumerate(otherinds):
            B[i, j] = math.exp(l_vals[k])

        lam = float(math.exp(s_val))
        return {"B": B.tolist(), "optimal_value": lam}

import cvxpy as cp
import numpy as np

class Solver:
    def solve(self, problem, **kwargs) -> dict:
        """
        Solves the Perron-Frobenius matrix completion using CVXPY geometric programming.
        """
        # Parse input
        inds = np.array(problem["inds"], dtype=int)
        a = np.array(problem["a"], dtype=float)
        n = int(problem["n"])

        # Build all index pairs and identify unknown entries
        xx, yy = np.meshgrid(np.arange(n), np.arange(n))
        allinds = np.vstack((yy.flatten(), xx.flatten())).T  # shape (n^2, 2)
        otherinds = allinds[~(allinds == inds[:, None]).all(2).any(0)]

        # CVXPY variable for matrix B
        B = cp.Variable((n, n), pos=True)

        # Objective: minimize Perron-Frobenius eigenvalue
        objective = cp.Minimize(cp.pf_eigenvalue(B))

        # Constraints: observed entries fixed; product of unknowns == 1
        constraints = [
            B[inds[:, 0], inds[:, 1]] == a,
            cp.prod(B[otherinds[:, 0], otherinds[:, 1]]) == 1.0,
        ]

        # Solve as a geometric program with ECOS for speed
        prob = cp.Problem(objective, constraints)
        result = prob.solve(gp=True, solver=cp.ECOS, verbose=False)

        return {
            "B": B.value.tolist(),
            "optimal_value": float(result),
        }

    def solve(self, problem, **kwargs) -> dict:
        """
        Solves the Perron-Frobenius matrix completion by minimizing the spectral radius
        using a linear program in the log-domain.
        """
        # Parse input
        inds = np.array(problem["inds"], dtype=int)
        a_vals = np.array(problem["a"], dtype=float)
        n = int(problem["n"])

        # Build mask of known entries
        mask = np.zeros((n, n), dtype=bool)
        if inds.size > 0:
            mask[inds[:, 0], inds[:, 1]] = True

        # Identify unknown entries
        otherinds = np.argwhere(~mask)
        m = otherinds.shape[0]

        # If no unknowns, compute PF eigenvalue directly
        if m == 0:
            B = np.zeros((n, n), dtype=float)
            B[inds[:, 0], inds[:, 1]] = a_vals
            eigs = np.linalg.eigvals(B)
            lam = float(np.max(np.abs(eigs)))
            return {"B": B.tolist(), "optimal_value": lam}

        # LP variables: logs of unknowns (m), logs of eigenvector (n), log lambda (1)
        var_count = m + n + 1

        # Objective: minimize s (last variable)
        c = np.zeros(var_count, dtype=float)
        c[m + n] = 1.0

        # Inequality constraints: one per unknown and known entry
        p = inds.shape[0]
        num_ineq = m + p
        A_ub = np.zeros((num_ineq, var_count), dtype=float)
        b_ub = np.zeros(num_ineq, dtype=float)

        # Unknown-entry constraints: ℓ_k + u_j - u_i - s <= 0
        for k, (i, j) in enumerate(otherinds):
            A_ub[k, k] = 1.0                  # ℓ_k
            A_ub[k, m + j] = 1.0              # u_j
            A_ub[k, m + i] = -1.0             # -u_i
            A_ub[k, m + n] = -1.0             # -s
            b_ub[k] = 0.0

        # Known-entry constraints: log(a) + u_j - u_i - s <= 0  => u_j - u_i - s <= -log(a)
        if p > 0:
            log_a = np.log(a_vals)
            for idx, (i, j) in enumerate(inds):
                row = m + idx
                A_ub[row, m + j] = 1.0
                A_ub[row, m + i] = -1.0
                A_ub[row, m + n] = -1.0
                b_ub[row] = -log_a[idx]

        # Equality constraints: sum ℓ_k == 0 and sum u_i == 0
        A_eq = np.zeros((2, var_count), dtype=float)
        b_eq = np.zeros(2, dtype=float)
        A_eq[0, :m] = 1.0
        A_eq[1, m:m + n] = 1.0

        # Bounds: all variables free
        bounds = [(None, None)] * var_count

        # Solve LP with HiGHS
        res = linprog(c, A_ub=A_ub, b_ub=b_ub, A_eq=A_eq, b_eq=b_eq,
                      bounds=bounds, method='highs')
        if res.success:
            x = res.x
            l_vals = x[:m]
            s_val = x[m + n]

            # Build completed matrix
            B = np.zeros((n, n), dtype=float)
            B[inds[:, 0], inds[:, 1]] = a_vals
            for k, (i, j) in enumerate(otherinds):
                B[i, j] = math.exp(l_vals[k])
            lam = float(math.exp(s_val))
            return {"B": B.tolist(), "optimal_value": lam}

        lam = float(np.max(np.abs(eigs)))

import math
from scipy.optimize import linprog

class Solver:
    def solve(self, problem, **kwargs) -> dict:
        """
        Solves the Perron-Frobenius matrix completion by minimizing the spectral radius
        using a linear program in the log-domain.
        """
        # Parse input
        inds = np.array(problem["inds"], dtype=int)
        a_vals = np.array(problem["a"], dtype=float)
        n = int(problem["n"])

        # Build mask of known entries
        mask = np.zeros((n, n), dtype=bool)
        if inds.size > 0:
            mask[inds[:, 0], inds[:, 1]] = True

        # Identify unknown entries
        otherinds = np.argwhere(~mask)
        m = otherinds.shape[0]

        # If no unknowns, compute PF eigenvalue directly
        if m == 0:
            B = np.zeros((n, n), dtype=float)
            if inds.size > 0:
                B[inds[:, 0], inds[:, 1]] = a_vals
            eigs = np.linalg.eigvals(B)
            lam = float(np.max(np.abs(eigs)))
            return {"B": B.tolist(), "optimal_value": lam}

        # LP variables: logs of unknowns (m), logs of eigenvector (n), log lambda (1)
        var_count = m + n + 1

        # Objective: minimize s (last variable)
        c = np.zeros(var_count, dtype=float)
        c[m + n] = 1.0

        # Inequality constraints: one per unknown and known entry
        p = inds.shape[0]
        num_ineq = m + p
        A_ub = np.zeros((num_ineq, var_count), dtype=float)
        b_ub = np.zeros(num_ineq, dtype=float)

        # Unknown-entry constraints: ℓ_k + u_j - u_i - s <= 0
        for k, (i, j) in enumerate(otherinds):
            A_ub[k, k] = 1.0                  # ℓ_k
            A_ub[k, m + j] = 1.0              # u_j
            A_ub[k, m + i] = -1.0             # -u_i
            A_ub[k, m + n] = -1.0             # -s
            b_ub[k] = 0.0

        # Known-entry constraints: log(a) + u_j - u_i - s <= 0  => u_j - u_i - s <= -log(a)
        if p > 0:
            log_a = np.log(a_vals)
            for idx, (i, j) in enumerate(inds):
                row = m + idx
                A_ub[row, m + j] = 1.0
                A_ub[row, m + i] = -1.0
                A_ub[row, m + n] = -1.0
                b_ub[row] = -log_a[idx]

        # Equality constraints: sum ℓ_k == 0 and sum u_i == 0
        A_eq = np.zeros((2, var_count), dtype=float)
        b_eq = np.zeros(2, dtype=float)
        A_eq[0, :m] = 1.0
        A_eq[1, m:m + n] = 1.0

        # Bounds: all variables free
        bounds = [(None, None)] * var_count

        # Solve LP with HiGHS
        res = linprog(c, A_ub=A_ub, b_ub=b_ub, A_eq=A_eq, b_eq=b_eq,
                      bounds=bounds, method='highs-ds')
        if res.success:
            x = res.x
            l_vals = x[:m]
            s_val = x[m + n]

            # Build completed matrix
            B = np.zeros((n, n), dtype=float)
            if inds.size > 0:
                B[inds[:, 0], inds[:, 1]] = a_vals
            for k, (i, j) in enumerate(otherinds):
                B[i, j] = math.exp(l_vals[k])
            lam = float(math.exp(s_val))
            return {"B": B.tolist(), "optimal_value": lam}

        # Fallback: fill unknowns with 1 and compute PF directly
        B = np.zeros((n, n), dtype=float)
        if inds.size > 0:
            B[inds[:, 0], inds[:, 1]] = a_vals
        for (i, j) in otherinds:
            B[i, j] = 1.0
        eigs = np.linalg.eigvals(B)
        lam = float(np.max(np.abs(eigs)))
        return {"B": B.tolist(), "optimal_value": lam}

import math

class Solver:
    def solve(self, problem: dict, **kwargs) -> dict:
        """
        Solves the Perron-Frobenius matrix completion by minimizing the spectral radius
        using a fixed-point algorithm on the dual variables.
        """
        # Parse input
        inds = np.array(problem.get("inds", []), dtype=int)
        a_vals = np.array(problem.get("a", []), dtype=float)
        n = int(problem["n"])

        # Build initial matrix B with known entries
        B = np.zeros((n, n), dtype=float)
        if inds.size > 0:
            B[inds[:, 0], inds[:, 1]] = a_vals

        # Identify unknown entries
        mask = np.zeros((n, n), dtype=bool)
        if inds.size > 0:
            mask[inds[:, 0], inds[:, 1]] = True
        otherinds = np.argwhere(~mask)
        m = otherinds.shape[0]

        # Power-iteration to compute PF eigenvalue and left/right eigenvectors
        def compute_pf(B_mat, tol=1e-8, max_iter=1000):
            n_loc = B_mat.shape[0]
            # Right eigenvector
            x = np.ones(n_loc, dtype=float)
            lam = 0.0
            for _ in range(max_iter):
                x_new = B_mat.dot(x)
                lam_new = np.max(np.abs(x_new))
                if lam_new == 0:
                    break
                x_new = x_new / lam_new
                if abs(lam_new - lam) < tol * lam_new:
                    x = x_new
                    lam = lam_new
                    break
                x, lam = x_new, lam_new
            # Left eigenvector
            y = np.ones(n_loc, dtype=float)
            lam2 = 0.0
            for _ in range(max_iter):
                y_new = B_mat.T.dot(y)
                lam2_new = np.max(np.abs(y_new))
                if lam2_new == 0:
                    break
                y_new = y_new / lam2_new
                if abs(lam2_new - lam2) < tol * lam2_new:
                    y = y_new
                    lam2 = lam2_new
                    break
                y, lam2 = y_new, lam2_new
            # Combine lambda estimates
            lam_pf = (lam + lam2) / 2.0
            return lam_pf, x, y

        # If there are unknowns, run fixed-point updates
        if m > 0:
            # Initialize unknown entries to 1
            B[otherinds[:, 0], otherinds[:, 1]] = 1.0
            for _ in range(100):
                lam_pf, x, y = compute_pf(B)
                # Extract components for unknowns
                yi = y[otherinds[:, 0]]
                xj = x[otherinds[:, 1]]
                prod = np.prod(yi * xj)
                # Dual multiplier
                C = prod ** (1.0 / m)
                new_vals = C / (yi * xj)
                old_vals = B[otherinds[:, 0], otherinds[:, 1]]
                B[otherinds[:, 0], otherinds[:, 1]] = new_vals
                # Check convergence in relative change
                if np.max(np.abs(new_vals - old_vals) / np.maximum(old_vals, 1e-8)) < 1e-6:
                    break
            lam_pf, _, _ = compute_pf(B)
        else:
            # No unknowns: directly compute PF eigenvalue
            lam_pf, _, _ = compute_pf(B)

import cvxpy as cp
import numpy as np

class Solver:
    def solve(self, problem, **kwargs) -> dict:
        """
        Solves the Perron-Frobenius matrix completion using CVXPY geometric programming.
        """
        # Parse input
        inds = np.array(problem["inds"], dtype=int)
        a = np.array(problem["a"], dtype=float)
        n = int(problem["n"])

        # Identify unknown entries via mask
        mask = np.zeros((n, n), dtype=bool)
        mask[inds[:, 0], inds[:, 1]] = True
        rows, cols = np.where(~mask)

        # CVXPY variable for matrix B
        B = cp.Variable((n, n), pos=True)

        # Constraints: observed entries fixed; product of unknowns == 1
        constraints = [B[inds[:, 0], inds[:, 1]] == a]
        if rows.size > 0:
            constraints.append(cp.prod(B[rows, cols]) == 1.0)

        # Objective: minimize Perron-Frobenius eigenvalue
        objective = cp.Minimize(cp.pf_eigenvalue(B))

        # Solve as a geometric program
        prob = cp.Problem(objective, constraints)
        result = prob.solve(gp=True, solver=cp.ECOS, verbose=False)

        return {
            "B": B.value.tolist(),
            "optimal_value": float(result),
        }

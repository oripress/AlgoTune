from numba import njit

@njit(cache=True)
def _l0_prune_nb(v, k):
    n = v.shape[0]
    # trivial cases
    if k <= 0:
        return np.zeros(n)
    if k >= n:
        return v.copy()
    # initialize index array
    idx = np.empty(n, dtype=np.int64)
    for i in range(n):
        idx[i] = i
    left = 0
    right = n - 1
    target = n - k
    # quickselect based on |v|
    while left < right:
        pivot_idx = (left + right) // 2
        # compute pivot magnitude
        pv = v[idx[pivot_idx]]
        pivot = -pv if pv < 0.0 else pv
        # move pivot to end
        tmp_i = idx[pivot_idx]
        idx[pivot_idx] = idx[right]
        idx[right] = tmp_i
        store_idx = left
        for j in range(left, right):
            valj = v[idx[j]]
            absj = -valj if valj < 0.0 else valj
            if absj < pivot:
                tmp_j = idx[j]
                idx[j] = idx[store_idx]
                idx[store_idx] = tmp_j
                store_idx += 1
        # place pivot at its final location
        tmp_s = idx[store_idx]
        idx[store_idx] = idx[right]
        idx[right] = tmp_s
        if store_idx == target:
            break
        elif store_idx < target:
            left = store_idx + 1
        else:
            right = store_idx - 1
    # build output vector
    w = np.zeros(n)
    for j in range(target, n):
        w[idx[j]] = v[idx[j]]
    return w

        Solve projection onto the L0 ball:

from numba import njit

@njit
def _l0_project(v, k):
    n = v.shape[0]
    # output array
    out = np.zeros(n, dtype=v.dtype)
    if k <= 0:
        return out
    if k >= n:
        return v.copy()
    # Quickselect to find threshold thr = k-th largest absolute value
    arr = v.copy()
    left = 0
    right = n - 1
    desired = n - k
    while True:
        pivot = arr[left]
        pivot_abs = abs(pivot)
        i = left
        j = right
        # partition around pivot_abs
        while i <= j:
            while abs(arr[i]) < pivot_abs:
                i += 1
            while abs(arr[j]) > pivot_abs:
                j -= 1
            if i <= j:
                tmp = arr[i]
                arr[i] = arr[j]
                arr[j] = tmp
                i += 1
                j -= 1
        # narrow search
        if desired <= j:
            right = j
        elif desired >= i:
            left = i
        else:
            thr = pivot_abs
            break
    # assign elements > threshold
    count_gt = 0
    for idx in range(n):
        if abs(v[idx]) > thr:
            out[idx] = v[idx]
            count_gt += 1
    # assign ties (== thr) by highest indices if needed
    count_eq = k - count_gt
    if count_eq > 0:
        for idx in range(n-1, -1, -1):
            if abs(v[idx]) == thr:
                out[idx] = v[idx]
                count_eq -= 1
                if count_eq == 0:
                    break
    return out

class Solver:
    def solve(self, problem, **kwargs) -> dict:
        """
        Solve min_w ||v - w||^2 s.t. ||w||_0 <= k
        by keeping the k entries of v with largest |v_i|.
        Tie-breaking by higher index for equal magnitudes.
        """
        v = np.asarray(problem.get("v", []), dtype=float)
        k = int(problem.get("k", 0))
        n = v.size

        # trivial cases
        if k <= 0:
            return {"solution": [0.0] * n}
        if k >= n:
            return {"solution": v.tolist()}

        abs_v = np.abs(v)
        # break ties by index: add tiny epsilon times index
        eps = np.finfo(float).eps

        keys = abs_v + idx * eps

        topk = np.argpartition(keys, -k)[-k:]
        # build mask
        mask = np.zeros(n, dtype=bool)
        mask[topk] = True

        # construct solution
        w = v * mask
        return {"solution": w.tolist()}

class Solver:
    def solve(self, problem, **kwargs) -> dict:
        """
        Solve min_w ||v - w||^2  s.t. ||w||_0 <= k
        by keeping the k entries of v with largest |v_i|.
        Tie-breaking matches the stable sort reference.
        """
        v_list = problem.get("v", [])
        k = int(problem.get("k", 0))
        v = np.asarray(v_list, dtype=float)
        n = v.size

        if k <= 0:
            return {"solution": [0.0] * n}
        if k >= n:
            return {"solution": v.tolist()}

        abs_v = np.abs(v)
        # threshold = k-th largest |v_i|
        thr = np.partition(abs_v, -k)[-k]

        # pick all strictly above threshold
        mask = abs_v > thr

        if m < k:

            # eq_idx is sorted ascending; pick last (k-m) indices
            take = eq_idx[-(k - m):]
            mask[take] = True

        w = v * mask
        return {"solution": w.tolist()}

        :param problem: dict with 'num_nodes', 'edges'
        :return: dict with 'mst_edges'

        num_nodes = problem["num_nodes"]
        edges = problem["edges"]
        
        # Sort edges by weight
        sorted_edges = sorted(edges, key=lambda x: x[2])

        parent = list(range(num_nodes))
        rank = [0] * num_nodes

            return parent[x]

import numpy as np
from typing import Any
from numba import njit, int32, float64
from numba.typed import List

@njit
def find(parent, x):
    """Find with path compression"""
    root = x
    # Find root
    while parent[root] != root:
        root = parent[root]
    # Path compression
    while parent[x] != root:
        next_x = parent[x]
        parent[x] = root
        x = next_x
    return root

@njit
def union(parent, rank, x, y):
    """Union by rank"""
    px = find(parent, x)
    py = find(parent, y)
    if px == py:
        return False
    # Union by rank
    if rank[px] < rank[py]:
        parent[px] = py
    elif rank[px] > rank[py]:
        parent[py] = px
    else:
        parent[py] = px
        rank[px] += 1
    return True

@njit
def kruskal_core(num_nodes, edges_array):
    """Core Kruskal's algorithm implementation"""
    # Initialize Union-Find
    parent = np.arange(num_nodes, dtype=np.int32)
    rank = np.zeros(num_nodes, dtype=np.int32)
    
    # Process edges
    mst_edges = List()
    edge_count = 0
    
    for i in range(len(edges_array)):
        u, v, w = edges_array[i]
        u_int = int(u)
        v_int = int(v)
        
        if union(parent, rank, u_int, v_int):
            # Ensure u < v for consistent ordering
            if u_int > v_int:
                u_int, v_int = v_int, u_int
            mst_edges.append((u_int, v_int, w))
            edge_count += 1
            if edge_count == num_nodes - 1:
                break
    
    return mst_edges

class Solver:
    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:
        """
        Solve MST using Kruskal's algorithm with Numba-optimized Union-Find.
        
        :param problem: dict with 'num_nodes', 'edges'
        :return: dict with 'mst_edges'
        """
        num_nodes = problem["num_nodes"]
        edges = problem["edges"]
        
        # Convert to numpy array and sort by weight
        edges_array = np.array(edges, dtype=np.float64)
        edges_array = edges_array[np.argsort(edges_array[:, 2])]
        
        # Run Numba-optimized Kruskal's algorithm
        mst_edges_numba = kruskal_core(num_nodes, edges_array)
        
        # Convert back to required format
        mst_edges = []
        for u, v, w in mst_edges_numba:
            mst_edges.append([int(u), int(v), float(w)])
        
        # Sort by (u, v) as required
        mst_edges.sort(key=lambda x: (x[0], x[1]))
        return {"mst_edges": mst_edges}

import numpy as np
from typing import Any
from numba import njit

@njit
def find(parent, x):
    """Find with path compression"""
    root = x
    # Find root
    while parent[root] != root:
        root = parent[root]
    # Path compression
    while parent[x] != root:
        next_x = parent[x]
        parent[x] = root
        x = next_x
    return root

@njit
def union(parent, rank, x, y):
    """Union by rank"""
    px = find(parent, x)
    py = find(parent, y)
    if px == py:
        return False
    # Union by rank
    if rank[px] < rank[py]:
        parent[px] = py
    elif rank[px] > rank[py]:
        parent[py] = px
    else:
        parent[py] = px
        rank[px] += 1
    return True

@njit
def kruskal_core(num_nodes, edges_array):
    """Core Kruskal's algorithm implementation"""
    # Initialize Union-Find
    parent = np.arange(num_nodes, dtype=np.int32)
    rank = np.zeros(num_nodes, dtype=np.int32)
    
    # Pre-allocate result array (max possible MST edges)
    mst_edges = np.zeros((num_nodes - 1, 3), dtype=np.float64)
    edge_count = 0
    
    for i in range(len(edges_array)):
        u, v, w = edges_array[i]
        u_int = int(u)
        v_int = int(v)
        
        if union(parent, rank, u_int, v_int):
            # Ensure u < v for consistent ordering
            if u_int > v_int:
                u_int, v_int = v_int, u_int
            mst_edges[edge_count, 0] = u_int
            mst_edges[edge_count, 1] = v_int
            mst_edges[edge_count, 2] = w
            edge_count += 1
            if edge_count == num_nodes - 1:
                break
    
    return mst_edges[:edge_count]

class Solver:
    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:
        """
        Solve MST using Kruskal's algorithm with Numba-optimized Union-Find.
        
        :param problem: dict with 'num_nodes', 'edges'
        :return: dict with 'mst_edges'
        """
        num_nodes = problem["num_nodes"]
        edges = problem["edges"]
        
        # Convert to numpy array and sort by weight
        edges_array = np.array(edges, dtype=np.float64)
        edges_array = edges_array[np.argsort(edges_array[:, 2])]
        
        # Run Numba-optimized Kruskal's algorithm
        mst_edges_np = kruskal_core(num_nodes, edges_array)
        
        # Convert back to required format
        mst_edges = []
        for i in range(len(mst_edges_np)):
            u, v, w = mst_edges_np[i]
            mst_edges.append([int(u), int(v), float(w)])
        
        # Sort by (u, v) as required
        mst_edges.sort(key=lambda x: (x[0], x[1]))
        return {"mst_edges": mst_edges}

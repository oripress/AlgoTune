from typing import Any
import numpy as np
import ecos
import scipy.sparse as sp

class Solver:
    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:
        # Convert to numpy arrays
        P = np.asarray(problem["P"], float)
        q = np.asarray(problem["q"], float)
        G = np.asarray(problem["G"], float)
        h = np.asarray(problem["h"], float)
        A = np.asarray(problem["A"], float)
        b = np.asarray(problem["b"], float)
        
        n = P.shape[0]
        
        # Make P symmetric
        P = (P + P.T) / 2
        
        # ECOS solves problems in the form:
        # minimize    c'*x
        # subject to  G*x + s = h
        #             A*x = b
        #             s in K (cone)
        
        # For QP, we need to reformulate as SOCP
        # minimize t
        # subject to ||Lx + L^{-T}q/2||_2 <= t + q'P^{-1}q/4
        # where P = L'L (Cholesky decomposition)
        
        # But for general QP, it's easier to use the native QP interface
        # which ECOS doesn't have. Let me try a different approach.
        
        # Actually, let me use cvxopt which has a native QP solver
        import cvxopt
        cvxopt.solvers.options['show_progress'] = False
        cvxopt.solvers.options['abstol'] = 1e-7
        cvxopt.solvers.options['reltol'] = 1e-7
        cvxopt.solvers.options['feastol'] = 1e-7
        
        # Convert to cvxopt matrices
        P_cvx = cvxopt.matrix(P)
        q_cvx = cvxopt.matrix(q)
        G_cvx = cvxopt.matrix(G) if G.size > 0 else None
        h_cvx = cvxopt.matrix(h) if h.size > 0 else None
        A_cvx = cvxopt.matrix(A) if A.size > 0 else None
        b_cvx = cvxopt.matrix(b) if b.size > 0 else None
        
        # Solve
        sol = cvxopt.solvers.qp(P_cvx, q_cvx, G_cvx, h_cvx, A_cvx, b_cvx)
        
        if sol['status'] != 'optimal':
            raise ValueError(f"Solver failed (status = {sol['status']})")
        
        return {"solution": list(sol['x'])}

from typing import Any
import numpy as np
from scipy.optimize import minimize
from scipy.optimize import LinearConstraint, Bounds

class Solver:
    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:
        # Convert to numpy arrays
        P = np.asarray(problem["P"], dtype=np.float64)
        q = np.asarray(problem["q"], dtype=np.float64)
        G = np.asarray(problem["G"], dtype=np.float64)
        h = np.asarray(problem["h"], dtype=np.float64)
        A = np.asarray(problem["A"], dtype=np.float64)
        b = np.asarray(problem["b"], dtype=np.float64)
        
        n = P.shape[0]
        
        # Make P symmetric
        P = (P + P.T) / 2
        
        # Define objective function and gradient
        def objective(x):
            return 0.5 * x @ P @ x + q @ x
        
        def gradient(x):
            return P @ x + q
        
        # Define Hessian
        def hessian(x):
            return P
        
        # Initial guess
        x0 = np.zeros(n)
        
        # Set up constraints
        constraints = []
        
        # Inequality constraints Gx <= h
        if G.size > 0:
            constraints.append(LinearConstraint(G, -np.inf, h))
        
        # Equality constraints Ax = b
        if A.size > 0:
            constraints.append(LinearConstraint(A, b, b))
        
        # Variable bounds (unbounded)
        bounds = Bounds(-np.inf, np.inf)
        
        # Solve using trust-constr method which handles QP well
        result = minimize(
            objective,
            x0,
            method='trust-constr',
            jac=gradient,
            hess=hessian,
            constraints=constraints,
            bounds=bounds,
            options={'gtol': 1e-8, 'xtol': 1e-8, 'barrier_tol': 1e-8}
        )
        
        if not result.success:
            raise ValueError(f"Solver failed: {result.message}")
        
        return {"solution": result.x.tolist()}

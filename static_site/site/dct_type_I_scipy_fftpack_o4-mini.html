class Solver:
    def solve(self, problem, **kwargs):
        """
        Compute the 2D DCT Type I of a square array using FFT-based 1D transforms.
        """
        arr = np.asarray(problem, dtype=float)
        # fallback for non-2D or non-square arrays
        if arr.ndim != 2 or arr.shape[0] != arr.shape[1]:
            from scipy.fftpack import dctn
            return dctn(arr, type=1)
        # first DCT-I along rows
        res = self._dct1d_axis(arr, axis=1)
        # then DCT-I along columns
        res = self._dct1d_axis(res, axis=0)
        return res

    @staticmethod
    def _dct1d_axis(x, axis):
        """
        Perform 1D DCT-I along the given axis using an even-symmetric FFT trick.
        """
        # move the target axis to the last dimension
        x = np.swapaxes(x, axis, -1)
        M = x.shape[-1]
        # special case: length 1
        if M == 1:
            y = 2 * x
        else:
            N = M - 1
            pad_len = 2 * N
            # create buffer for even-symmetric extension
            shape = x.shape[:-1] + (pad_len,)
            y = np.empty(shape, dtype=x.dtype)
            # copy original data
            y[..., :M] = x
            # mirror interior (excluding endpoints)
            if M > 2:
                y[..., M:] = x[..., 1:-1][..., ::-1]
            # compute real FFT and take real part => DCT-I values
            Y = np.fft.rfft(y, n=pad_len, axis=-1)
            y = Y.real
        # restore axes order
        y = np.swapaxes(y, axis, -1)
        return y

import numpy as np
from scipy.fftpack import dctn as _fallback_dctn

class Solver:
    def solve(self, problem, **kwargs):
        """
        Compute the 2D DCT Type I via FFT-based method with symmetric extension.
        Falls back to scipy.fftpack.dctn for degenerate cases.
        """
        arr = np.array(problem, dtype=np.float64, copy=False)
        # Handle non-2D or very small arrays with fallback
        if arr.ndim != 2 or arr.shape[0] < 2 or arr.shape[1] < 2:
            return _fallback_dctn(arr, type=1)
        n0, n1 = arr.shape
        E0 = 2 * n0 - 2
        E1 = 2 * n1 - 2
        # Allocate extended array
        ext = np.empty((E0, E1), dtype=np.float64)
        # Original block
        ext[:n0, :n1] = arr
        # Mirror along first axis (rows) excluding endpoints
        if n0 > 2:
            ext[n0:E0, :n1] = arr[1:n0-1][::-1, :]
        # Mirror along second axis (columns) excluding endpoints
        if n1 > 2:
            ext[:n0, n1:E1] = arr[:, 1:n1-1][:, ::-1]
        # Mirror both axes for interior block
        if n0 > 2 and n1 > 2:
            ext[n0:E0, n1:E1] = arr[1:n0-1, 1:n1-1][::-1, ::-1]
        # Compute 2D real FFT
        fft = np.fft.rfftn(ext)
        # Extract DCT-I coefficients (real part, first n0 x n1 block)
        return fft[:n0, :n1].real

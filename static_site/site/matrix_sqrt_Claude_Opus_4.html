import numpy as np
import jax
import jax.numpy as jnp
from jax import jit
import scipy.linalg

class Solver:
    def __init__(self):
        # Pre-compile the JAX function
        self._jax_sqrtm = jit(self._compute_sqrtm_jax)
    
    def _compute_sqrtm_jax(self, A):
        # Convert to JAX array and compute eigendecomposition
        A_jax = jnp.array(A, dtype=jnp.complex128)
        eigenvalues, eigenvectors = jnp.linalg.eig(A_jax)
        
        # Compute square root via eigendecomposition
        # X = V @ sqrt(D) @ V^(-1)
        sqrt_eigenvalues = jnp.sqrt(eigenvalues)
        sqrt_D = jnp.diag(sqrt_eigenvalues)
        
        # Compute X = V @ sqrt(D) @ V^(-1)
        X = eigenvectors @ sqrt_D @ jnp.linalg.inv(eigenvectors)
        return X
    
    def solve(self, problem: dict[str, np.ndarray]) -> dict[str, dict[str, list[list[complex]]]]:
        """
        Solves the matrix square root problem using JAX with JIT compilation.
        
        Computes the principal matrix square root X such that X @ X = A.
        
        :param problem: A dictionary representing the matrix square root problem.
        :return: A dictionary with key "sqrtm" containing a dictionary with key:
                 "X": A list of lists representing the principal square root matrix X.
                      Contains complex numbers.
        """
        A = problem["matrix"]
        
        try:
            # Use JAX implementation
            X_jax = self._jax_sqrtm(A)
            X = np.array(X_jax)
        except Exception:
            # Fallback to scipy if JAX fails
            try:
                X, _ = scipy.linalg.sqrtm(A, disp=False)
            except Exception:
                return {"sqrtm": {"X": []}}
        
        # Convert to list format
        solution = {"sqrtm": {"X": X.tolist()}}
        return solution

import scipy.linalg
from numba import jit, complex128, void
import numba as nb

class Solver:
    def __init__(self):
        # Pre-compile functions
        pass
    
    @staticmethod
    @jit(nopython=True, cache=True, fastmath=True)
    def _matrix_sqrt_eigendecomp(A_real, A_imag, n):
        """Compute matrix square root using eigendecomposition with Numba."""
        # Combine real and imaginary parts
        A = A_real + 1j * A_imag
        
        # For small matrices, use direct formula
        if n == 2:
            a, b, c, d = A[0,0], A[0,1], A[1,0], A[1,1]
            trace = a + d
            det = a*d - b*c
            
            # Compute sqrt(trace + 2*sqrt(det))
            sqrt_term = np.sqrt(trace + 2*np.sqrt(det))
            
            # Direct formula for 2x2 matrix square root
            X = np.empty((2, 2), dtype=np.complex128)
            factor = 1.0 / sqrt_term
            X[0,0] = (a + np.sqrt(det)) * factor
            X[0,1] = b * factor
            X[1,0] = c * factor
            X[1,1] = (d + np.sqrt(det)) * factor
            
            return X
        
        # For larger matrices, we can't use numpy.linalg in nopython mode
        # So we'll return the original matrix and handle it outside
        return A
    
    def solve(self, problem: dict[str, np.ndarray]) -> dict[str, dict[str, list[list[complex]]]]:
        """
        Solves the matrix square root problem.
        
        Computes the principal matrix square root X such that X @ X = A.
        
        :param problem: A dictionary representing the matrix square root problem.
        :return: A dictionary with key "sqrtm" containing a dictionary with key:
                 "X": A list of lists representing the principal square root matrix X.
                      Contains complex numbers.
        """
        A = problem["matrix"]
        n = A.shape[0]
        
        try:
            if n == 2:
                # Use optimized 2x2 implementation
                A_real = np.real(A).astype(np.float64)
                A_imag = np.imag(A).astype(np.float64)
                X = self._matrix_sqrt_eigendecomp(A_real, A_imag, n)
            else:
                # Use scipy for larger matrices
                X, _ = scipy.linalg.sqrtm(A, disp=False)
        except Exception:
            return {"sqrtm": {"X": []}}

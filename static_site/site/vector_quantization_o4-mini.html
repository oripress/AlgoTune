            assignments = assignments.reshape(-1)

            if n_vectors >= k:

            for _ in range(10):

    def solve(self, problem, **kwargs):
        """
        Very fast KMeans via Faiss on a small random subsample + NumPy assignment.
        """
        vecs = problem.get("vectors")
        k = problem.get("k")
        # missing or trivial
        if vecs is None or k is None:
            return {"centroids": [], "assignments": [], "quantization_error": 0.0}
        k = int(k)
        vectors = np.asarray(vecs, dtype=np.float32)
        n, dim = vectors.shape if vectors.ndim == 2 else (0, int(problem.get("dim", 0)))
        if n == 0 or k <= 0:
            centroids = [[0.0] * dim for _ in range(k)]
            return {"centroids": centroids, "assignments": [], "quantization_error": 0.0}

        # 1) sample a fixed number of points for training
        sample_size = min(n, max(10000, k * 20))

            train = vectors[idx]

            index.add(centroids)
            D, L = index.search(vectors, 1)
            labels = L.reshape(-1).astype(int)
            qerr = float(D.mean())

            "centroids": centroids.tolist(),
            "assignments": labels.tolist(),
            "quantization_error": qerr,
        }

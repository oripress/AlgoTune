    def solve(self, problem: dict[str, Any], **kwargs) -> dict[str, Any]:
        """
        Fast static state-feedback design via discrete LQR + Lyapunov.
        """
        # Extract system matrices
        A = np.array(problem.get("A", []), dtype=float)
        B = np.array(problem.get("B", []), dtype=float)
        # Dimensions
        n = A.shape[0]
        # Check for trivial or malformed input
        if n == 0 or B.ndim != 2 or A.shape[1] != n:
            return {"is_stabilizable": False, "K": None, "P": None}
        m = B.shape[1]

        # 1) Stabilizability test: each eigenvalue with |Œª|>=1 must be controllable
        tol = 1e-8
        eigvals_A, _ = np.linalg.eig(A)
        for lam in eigvals_A:
            if abs(lam) >= 1 - tol:
                M = np.hstack((A - lam * np.eye(n), B))
                if np.linalg.matrix_rank(M) < n:
                    return {"is_stabilizable": False, "K": None, "P": None}

        # 2) Solve discrete Algebraic Riccati Equation for Q=I, R=I
        try:
            Q = np.eye(n)
            R = np.eye(m)
            P_dare = solve_discrete_are(A, B, Q, R)

            # Compute feedback gain K = -(R + B'PB)^{-1} B' P A
            BRB = R + B.T.dot(P_dare).dot(B)
            # Use solve for numerical stability
            K = -np.linalg.solve(BRB, B.T.dot(P_dare).dot(A))

            # 3) Compute Lyapunov P satisfying M.T P M - P + I = 0
            Mcl = A + B.dot(K)
            P_lyap = solve_discrete_lyapunov(Mcl.T, np.eye(n))

            return {
                "is_stabilizable": True,
                "K": K.tolist(),
                "P": P_lyap.tolist()
            }
        except Exception:
            # In case DARE or Lyap fails
            return {"is_stabilizable": False, "K": None, "P": None}

import numpy as np
import numba as nb

# Local aliases
_asarray = np.asarray
_eye = np.eye

@nb.njit(cache=True)
def _solve_dare_iter(A, B, Q, R, max_iter=50, tol=1e-8):
    n, m = A.shape[0], B.shape[1]
    # Initialize P with Q
    P = Q.copy()
    for _ in range(max_iter):
        # Compute intermediate terms
        BT_P = B.T @ P
        S = BT_P @ B + R
        # Solve for M: S * M = B^T P A
        M = np.linalg.solve(S, BT_P @ A)
        # Riccati update: P_next = A^T P A - A^T P B M + Q
        P_next = A.T @ P @ A - A.T @ P @ B @ M + Q
        # Convergence check (Frobenius norm)
        diff = P_next - P
        err = 0.0
        for i in range(n):
            for j in range(n):
                err += diff[i, j] * diff[i, j]
        if err < tol * tol:
            return P_next, -M
        P = P_next
    # Return last iterate if not converged
    return P, -M

class Solver:
    def solve(self, problem: dict, **kwargs) -> dict:
        # Convert to arrays
        A = _asarray(problem["A"], dtype=float)
        B = _asarray(problem["B"], dtype=float)
        n, m = A.shape[0], B.shape[1]
        # Identity matrices for Q and R
        I_n = _eye(n)
        I_m = _eye(m)
        try:
            P, K = _solve_dare_iter(A, B, I_n, I_m)
            return {"is_stabilizable": True, "K": K, "P": P}
        except Exception:
            return {"is_stabilizable": False, "K": None, "P": None}

from scipy.linalg import solve_discrete_are

class Solver:
    def solve(self, problem, **kwargs):
        """
        Fast static state-feedback via discrete ARE with Q=I, R=I.
        """
        A = np.array(problem["A"], dtype=float)
        B = np.array(problem["B"], dtype=float)
        n, m = A.shape[0], B.shape[1]
        try:
            # Solve the discrete algebraic Riccati equation
            P = solve_discrete_are(A, B, np.eye(n), np.eye(m))
            # Compute K = -(R + B^T P B)^{-1} B^T P A
            BtPB = B.T @ P @ B
            inv_term = np.linalg.inv(np.eye(m) + BtPB)
            K = -inv_term @ (B.T @ P @ A)
            return {
                "is_stabilizable": True,
                "K": K.tolist(),
                "P": P.tolist(),
            }

            return {"is_stabilizable": False, "K": None, "P": None}

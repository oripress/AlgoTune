from ortools.sat.python import cp_model

class Solver:
    def solve(self, problem: list[list[int]]) -> list[int]:
        """
        Solves the set cover problem using greedy heuristic with CP-SAT fallback.
        """
        if not problem:
            return []
        
        # Convert to sets for faster operations
        sets = [set(s) for s in problem]
        m = len(sets)
        
        # Find universe
        universe = set()
        for s in sets:
            universe.update(s)
        if not universe:
            return []
        
        # Quick preprocessing: remove empty sets and duplicates
        valid_sets = []
        valid_indices = []
        seen = set()
        for i, s in enumerate(sets):
            if s:
                frozen_s = frozenset(s)
                if frozen_s not in seen:
                    seen.add(frozen_s)
                    valid_sets.append(s)
                    valid_indices.append(i)
        
        if not valid_sets:
            return []
        
        # Try greedy algorithm first
        selected = []
        uncovered = universe.copy()
        remaining_indices = list(range(len(valid_sets)))
        
        # Greedy: repeatedly select the set that covers the most uncovered elements
        while uncovered and remaining_indices:
            best_idx = -1
            best_coverage = 0
            
            for idx in remaining_indices:
                coverage = len(valid_sets[idx] & uncovered)
                if coverage > best_coverage:
                    best_coverage = coverage
                    best_idx = idx
            
            if best_idx == -1:
                break
            
            selected.append(valid_indices[best_idx])
            uncovered -= valid_sets[best_idx]
            remaining_indices.remove(best_idx)
        
        if not uncovered:
            # Greedy found a solution, try to improve it
            result = self._improve_solution(sets, selected, universe)
            return sorted([i + 1 for i in result])
        
        # Fallback to CP-SAT for harder instances
        return self._solve_with_cpsat(sets, universe)
    
    def _improve_solution(self, sets, selected, universe):
        """Try to remove redundant sets from the solution."""
        improved = selected.copy()
        for idx in selected:
            # Check if this set is redundant
            test_coverage = set()
            for other_idx in improved:
                if other_idx != idx:
                    test_coverage.update(sets[other_idx])
            if test_coverage == universe:
                improved.remove(idx)
        return improved
    
    def _solve_with_cpsat(self, sets, universe):
        """Solve using CP-SAT with optimizations."""
        m = len(sets)
        
        # Find mandatory sets (elements covered by only one set)
        element_coverage = {}
        for i, s in enumerate(sets):
            for e in s:
                if e not in element_coverage:
                    element_coverage[e] = []
                element_coverage[e].append(i)
        
        mandatory = set()
        for e, covering_sets in element_coverage.items():
            if len(covering_sets) == 1:
                mandatory.add(covering_sets[0])
        
        # Create model
        model = cp_model.CpModel()
        x = [model.NewBoolVar(f"x[{i}]") for i in range(m)]
        
        # Force mandatory sets
        for i in mandatory:
            model.Add(x[i] == 1)
        
        # Coverage constraints
        for e in universe:
            covering = [x[i] for i in element_coverage.get(e, [])]
            if covering:
                model.Add(sum(covering) >= 1)
        
        # Minimize
        model.Minimize(sum(x))
        
        # Solve with tight parameters
        solver = cp_model.CpSolver()
        solver.parameters.max_time_in_seconds = 1.0
        solver.parameters.num_search_workers = 1
        
        status = solver.Solve(model)
        if status == cp_model.OPTIMAL:
            return sorted([i + 1 for i in range(m) if solver.Value(x[i]) == 1])
        else:
            # Return mandatory sets as fallback
            return sorted([i + 1 for i in mandatory])

        Solves the set cover problem using CP-SAT with advanced preprocessing.
        """

            return []

        sets = [set(s) for s in problem]
        m = len(sets)

        universe = set()

            universe.update(s)

            return []

        mandatory = set()
        removed = set()

        element_freq = {}
        element_sets = {}
        for i, s in enumerate(sets):
            for e in s:
                if e not in element_freq:
                    element_freq[e] = 0
                    element_sets[e] = []
                element_freq[e] += 1
                element_sets[e].append(i)
        
        # Find mandatory sets (elements covered by only one set)
        for e, freq in element_freq.items():
            if freq == 1:
                mandatory.add(element_sets[e][0])
        
        # Calculate what's covered by mandatory sets
        covered = set()
        for i in mandatory:
            covered.update(sets[i])
        
        if covered == universe:
            return sorted([i + 1 for i in mandatory])
        
        # Remove dominated sets
        for i in range(m):
            if i in mandatory or i in removed:
                continue
            # Check if this set is dominated by another single set
            for j in range(m):
                if i != j and j not in removed:
                    if sets[i].issubset(sets[j]):
                        removed.add(i)
                        break
        
        # Build reduced problem
        uncovered = universe - covered
        active_sets = []
        active_indices = []
        
        for i in range(m):
            if i in mandatory or i in removed:
                continue
            relevant = sets[i] & uncovered
            if relevant:
                active_sets.append(relevant)
                active_indices.append(i)
        
        if not active_sets:
            return sorted([i + 1 for i in mandatory])
        
        # Solve reduced problem with CP-SAT
        model = cp_model.CpModel()
        x = [model.NewBoolVar(f"x[{i}]") for i in range(len(active_sets))]
        
        # Coverage constraints
        for e in uncovered:
            covering = []
            for idx, s in enumerate(active_sets):
                if e in s:
                    covering.append(x[idx])
            if covering:
                model.Add(sum(covering) >= 1)
        
        # Add hint based on greedy selection
        greedy_hint = self._greedy_hint(active_sets, uncovered)
        for idx, hint in enumerate(greedy_hint):
            model.AddHint(x[idx], hint)
        
        # Minimize
        model.Minimize(sum(x))
        
        # Optimized solver parameters
        solver = cp_model.CpSolver()

                if solver.Value(x[i]) == 1:
                    result.append(active_indices[i])
        
        return sorted([i + 1 for i in result])

                break

        return hints

from ortools.sat.python import cp_model
import numpy as np

class Solver:
    def solve(self, problem: list[list[int]]) -> list[int]:
        """
        Hybrid solver: uses preprocessing + greedy, with CP-SAT fallback only when needed.
        """
        if not problem:
            return []
        
        m = len(problem)
        
        # Fast universe computation
        universe = set()
        for subset in problem:
            universe.update(subset)
        
        if not universe:
            return []
        
        n = max(universe)
        
        # Use bit vectors for very fast set operations
        bit_sets = []
        non_empty_indices = []
        for i, subset in enumerate(problem):
            if subset:
                # Convert to bit vector
                bit_vec = 0
                for elem in subset:
                    bit_vec |= (1 << (elem - 1))
                bit_sets.append(bit_vec)
                non_empty_indices.append(i)
        
        if not bit_sets:
            return []
        
        # Universe as bit vector
        universe_bits = 0
        for elem in universe:
            universe_bits |= (1 << (elem - 1))
        
        # Fast greedy algorithm with bit operations
        selected = []
        uncovered = universe_bits
        remaining = list(range(len(bit_sets)))
        
        while uncovered and remaining:
            best_idx = -1
            best_count = -1
            
            for idx in remaining:
                # Count covered elements using bit operations
                covered = bit_sets[idx] & uncovered
                if covered:
                    # Count set bits (Brian Kernighan's algorithm)
                    count = 0
                    temp = covered
                    while temp:
                        count += 1
                        temp &= temp - 1
                    
                    if count > best_count:
                        best_count = count
                        best_idx = idx
            
            if best_idx == -1:
                break
            
            selected.append(non_empty_indices[best_idx])
            uncovered &= ~bit_sets[best_idx]
            remaining.remove(best_idx)
        
        if uncovered == 0:
            # Greedy found a complete solution
            # Try to minimize it
            final_selected = []
            covered = 0
            
            # Sort by coverage size (descending) to prioritize larger sets
            selected_with_coverage = []
            for idx in selected:
                orig_idx = idx
                bit_vec = 0
                for elem in problem[orig_idx]:
                    bit_vec |= (1 << (elem - 1))
                selected_with_coverage.append((idx, bit_vec))
            
            selected_with_coverage.sort(key=lambda x: bin(x[1]).count('1'), reverse=True)
            
            for idx, bit_vec in selected_with_coverage:
                if (covered | bit_vec) != covered:
                    final_selected.append(idx)
                    covered |= bit_vec
                    if covered == universe_bits:
                        break
            
            # Check if this might be optimal using lower bound
            # Lower bound: maximum number of disjoint elements divided by max set size
            max_set_size = max(len(s) for s in problem)
            lower_bound = (len(universe) + max_set_size - 1) // max_set_size
            
            if len(final_selected) <= lower_bound + 1:
                # Very likely optimal or near-optimal
                return sorted([i + 1 for i in final_selected])
            
            # Otherwise fall through to CP-SAT
            selected = final_selected
        
        # Use CP-SAT for difficult cases or to improve greedy solution
        return self._cpsat_solve(problem, universe, selected)
    
    def _cpsat_solve(self, problem, universe, hint_solution):
        """Solve with CP-SAT using hints from greedy solution."""
        m = len(problem)
        sets = [set(s) for s in problem]
        
        model = cp_model.CpModel()
        x = [model.NewBoolVar(f"x[{i}]") for i in range(m)]
        
        # Coverage constraints
        for e in universe:
            covering = [x[i] for i in range(m) if e in sets[i]]
            if covering:
                model.Add(sum(covering) >= 1)
        
        # Add hints from greedy solution
        if hint_solution:
            for i in range(m):
                model.AddHint(x[i], 1 if i in hint_solution else 0)
        
        # Objective
        model.Minimize(sum(x))
        
        # Fast solver settings
        solver = cp_model.CpSolver()
        solver.parameters.max_time_in_seconds = 2.0
        solver.parameters.num_search_workers = 1
        solver.parameters.linearization_level = 2
        solver.parameters.cp_model_presolve = True
        
        status = solver.Solve(model)
        
        if status == cp_model.OPTIMAL:
            return sorted([i + 1 for i in range(m) if solver.Value(x[i]) == 1])
        else:
            # Return greedy solution as fallback
            return sorted([i + 1 for i in hint_solution])

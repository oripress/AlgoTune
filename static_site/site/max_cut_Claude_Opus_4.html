        n = len(adj_matrix)
        best_cut = []
        best_cut_size = -1

        return [i for i in range(n) if partition[i] == 1]

@njit(cache=True)
def solve_fast(adj_matrix):
    """Fast max cut solver."""
    n = len(adj_matrix)
    
    # For very small graphs, try all partitions
    if n <= 12:
        best_partition = np.zeros(n, dtype=np.int8)
        best_cut = -1
        
        for mask in range(1 << (n-1)):
            partition = np.zeros(n, dtype=np.int8)
            for i in range(1, n):
                if mask & (1 << (i-1)):
                    partition[i] = 1
            
            cut = 0
            for i in range(n):
                for j in range(i + 1, n):
                    if adj_matrix[i][j] and partition[i] != partition[j]:
                        cut += 1
            
            if cut > best_cut:
                best_cut = cut
                best_partition = partition.copy()
        
        return best_partition
    
    # For larger graphs, use fast local search
    best_partition = np.zeros(n, dtype=np.int8)
    best_cut = -1
    
    # Try different initializations
    num_tries = min(3, max(1, 30 // n))
    
    for trial in range(num_tries):
        # Initialize partition
        partition = np.zeros(n, dtype=np.int8)
        
        if trial == 0:
            # Start with alternating pattern
            for i in range(n):
                partition[i] = i % 2
        else:
            # Random initialization
            for i in range(n):
                if np.random.random() > 0.5:
                    partition[i] = 1
        
        # Fast local search with pre-computed gains
        improved = True
        iter_count = 0
        max_iters = 5
        
        while improved and iter_count < max_iters:
            improved = False
            iter_count += 1
            
            # Pre-compute node gains
            best_gain = 0
            best_node = -1
            
            for node in range(n):
                gain = 0
                current_side = partition[node]
                
                for neighbor in range(n):
                    if adj_matrix[node][neighbor]:
                        if partition[neighbor] == current_side:
                            gain += 1
                        else:
                            gain -= 1
                
                if gain > best_gain:
                    best_gain = gain
                    best_node = node
            
            # Make best move
            if best_node != -1:
                partition[best_node] = 1 - partition[best_node]
                improved = True
        
        # Calculate cut value
        cut = 0
        for i in range(n):
            for j in range(i + 1, n):
                if adj_matrix[i][j] and partition[i] != partition[j]:
                    cut += 1
        
        if cut > best_cut:
            best_cut = cut
            best_partition = partition.copy()
    
    return best_partition

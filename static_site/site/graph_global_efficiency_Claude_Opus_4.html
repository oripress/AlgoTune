For a graph G with N nodes, it is calculated as E = (1 / (N * (N-1))) * sum(1 / d(u, v)) for all u != v, where d(u, v) is the shortest path distance between nodes u and v. def bfs_all_pairs_optimized(adj_array, adj_lengths, n):
    """Compute sum of inverse distances using optimized BFS from all nodes."""
    total_inverse_distance = 0.0
    
    # Pre-allocate arrays that will be reused
    distances = np.empty(n, dtype=np.int32)
    queue = np.empty(n, dtype=np.int32)
    
    for start in range(n):
        # Reset distances array
        distances.fill(-1)
        distances[start] = 0
        
        # BFS using pre-allocated queue
        queue[0] = start
        queue_start = 0
        queue_end = 1
        
        while queue_start < queue_end:
            u = queue[queue_start]
            queue_start += 1
            current_dist = distances[u]
            next_dist = current_dist + 1

            if end != start:
                dist = distances[end]
                if dist > 0:

from typing import Any
import numba
import numpy as np

@numba.njit(parallel=True)
def bfs_all_pairs_parallel(adj_array, n):
    """Compute sum of inverse distances using parallel BFS from all nodes."""
    # Use a local array to store partial sums from each thread
    partial_sums = np.zeros(n, dtype=np.float64)
    
    # Parallel loop over starting nodes
    for start in numba.prange(n):
        # BFS to find distances from start node
        distances = np.full(n, -1, dtype=np.int32)
        distances[start] = 0
        queue = np.zeros(n, dtype=np.int32)
        queue_start = 0
        queue_end = 1
        queue[0] = start
        
        while queue_start < queue_end:
            u = queue[queue_start]
            queue_start += 1
            
            # Iterate through neighbors
            for v in adj_array[u]:
                if v == -1:  # End of neighbors
                    break
                if distances[v] == -1:
                    distances[v] = distances[u] + 1
                    queue[queue_end] = v
                    queue_end += 1
        
        # Add inverse distances to partial sum
        local_sum = 0.0
        for end in range(n):
            if end != start and distances[end] > 0:
                local_sum += 1.0 / distances[end]
        
        partial_sums[start] = local_sum
    
    # Sum all partial results
    return np.sum(partial_sums)

@numba.njit(cache=True, fastmath=True)
def bfs_all_pairs_fast(adj_array, adj_lengths, n):
    """Compute sum of inverse distances with optimizations."""
    if n <= 1:
        return 0.0
        
    total_inverse_distance = 0.0
    
    # Pre-allocate arrays that will be reused
    distances = np.empty(n, dtype=np.int32)
    queue = np.empty(n, dtype=np.int32)
    visited = np.empty(n, dtype=np.bool_)
    
    # Pre-compute inverse values to avoid division in inner loop
    max_dist = n - 1
    inv_table = np.empty(max_dist + 1, dtype=np.float64)
    inv_table[0] = 0.0
    for i in range(1, max_dist + 1):
        inv_table[i] = 1.0 / i
    
    for start in range(n):
        # Reset arrays using faster methods
        visited.fill(False)
        visited[start] = True
        distances[start] = 0
        
        # BFS using pre-allocated queue
        queue[0] = start
        queue_start = 0
        queue_end = 1
        
        while queue_start < queue_end:
            u = queue[queue_start]
            queue_start += 1
            current_dist = distances[u]
            next_dist = current_dist + 1
            
            # Iterate through neighbors using actual length
            neighbor_count = adj_lengths[u]
            for i in range(neighbor_count):
                v = adj_array[u, i]
                if not visited[v]:
                    visited[v] = True
                    distances[v] = next_dist
                    queue[queue_end] = v
                    queue_end += 1
        
        # Add inverse distances using pre-computed table
        for end in range(n):
            if visited[end] and end != start:
                total_inverse_distance += inv_table[distances[end]]
    
    return total_inverse_distance

@numba.njit(cache=True)
def convert_adjacency_list(adj_list):
    """Convert adjacency list to optimized format."""
    n = len(adj_list)
    if n == 0:
        return np.empty((0, 0), dtype=np.int32), np.empty(0, dtype=np.int32)
    
    # Calculate sizes
    adj_lengths = np.empty(n, dtype=np.int32)
    max_degree = 0
    for i in range(n):
        length = len(adj_list[i])
        adj_lengths[i] = length
        if length > max_degree:
            max_degree = length
    
    # Create adjacency array
    adj_array = np.zeros((n, max_degree), dtype=np.int32)
    for i in range(n):
        neighbors = adj_list[i]
        for j in range(len(neighbors)):
            adj_array[i, j] = neighbors[j]
    
    return adj_array, adj_lengths

        adj_array, adj_lengths = convert_adjacency_list(adj_list)
        
        # Check if graph has any edges
        if np.max(adj_lengths) == 0:
            return {"global_efficiency": 0.0}
        
        # Calculate sum of inverse distances
        total_inverse_distance = bfs_all_pairs_fast(adj_array, adj_lengths, n)
        
        # Calculate global efficiency
        global_efficiency = total_inverse_distance / (n * (n - 1))
        
        return {"global_efficiency": global_efficiency}

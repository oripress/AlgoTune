class Solver:
    def solve(self, problem, **kwargs):
        """
        Solves the maximum clique problem using optimized branch and bound.
        
        :param problem: A 2D adjacency matrix representing the graph.
        :return: A list of node indices that form a maximum clique.
        """
        n = len(problem)
        if n == 0:
            return []
        
        # Convert to numpy array for faster access
        adj_matrix = np.array(problem, dtype=np.int8)
        
        # Compute degree of each vertex for ordering
        degrees = np.sum(adj_matrix, axis=1)
        
        # Order vertices by degree (descending) - high degree vertices are more likely to be in large cliques
        vertex_order = np.argsort(-degrees)
        
        # Reorder adjacency matrix according to degree ordering
        adj_matrix = adj_matrix[vertex_order][:, vertex_order]
        
        # Find maximum clique using optimized branch and bound
        max_clique = find_max_clique_bb(adj_matrix, n)
        
        # Map back to original indices
        return [vertex_order[i] for i in max_clique]

def find_max_clique_bb(adj_matrix, n):
    """
    Find maximum clique using branch and bound with optimizations.
    """
    best_clique = List.empty_list(types.int32)
    best_size = 0
    
    # Initial candidates are all vertices
    candidates = List([i for i in range(n)])
    current_clique = List.empty_list(types.int32)
    
    # Color vertices for upper bound estimation
    colors = color_vertices(adj_matrix, n)
    
    def branch_and_bound(current, cands, excluded):
        nonlocal best_clique, best_size
        
        if len(cands) == 0 and len(excluded) == 0:
            if len(current) > best_size:
                best_size = len(current)
                best_clique = current.copy()
            return

    """
    Color vertices greedily for upper bound estimation.
    Returns the color of each vertex.
    """
    colors = np.zeros(n, dtype=np.int32)

        # Assign smallest available color
        color = 0
        while color in used_colors:

        colors[v] = color
        color_count = max(color_count, color + 1)

    def solve(self, problem, **kwargs):
        """
        Solves the maximum clique problem using optimized branch and bound.
        
        :param problem: A 2D adjacency matrix representing the graph.
        :return: A list of node indices that form a maximum clique.
        """
        n = len(problem)
        if n == 0:
            return []
        
        # Convert to numpy array for faster access
        adj_matrix = np.array(problem, dtype=np.int8)
        
        # Compute degree of each vertex
        degrees = np.sum(adj_matrix, axis=1)
        
        # Order vertices by degree (descending) - high degree vertices are more likely to be in large cliques
        vertex_order = np.argsort(-degrees)
        
        # Create reordered adjacency matrix
        adj_reordered = adj_matrix[vertex_order][:, vertex_order]
        
        # Precompute neighbors for each vertex
        neighbors = []
        for i in range(n):
            neighbors.append(np.where(adj_reordered[i] == 1)[0])
        
        # Find maximum clique
        max_clique = self._find_max_clique_optimized(adj_reordered, neighbors, n)
        
        # Map back to original indices
        return sorted([vertex_order[i] for i in max_clique])
    
    def _find_max_clique_optimized(self, adj_matrix, neighbors, n):
        """
        Find maximum clique using optimized branch and bound.
        """
        best_clique = []
        best_size = 0
        
        # Color vertices for upper bound
        colors = self._greedy_coloring(adj_matrix, n)
        max_colors = max(colors) + 1
        
        def branch_and_bound(current, candidates, depth):
            nonlocal best_clique, best_size
            
            if not candidates:
                if len(current) > best_size:
                    best_size = len(current)
                    best_clique = current.copy()
                return
            
            # Upper bound pruning
            upper_bound = len(current) + len(candidates)
            if upper_bound <= best_size:
                return
            
            # Color-based upper bound
            candidate_colors = set(colors[v] for v in candidates)
            color_bound = len(current) + len(candidate_colors)
            if color_bound <= best_size:
                return
            
            # Sort candidates by degree in subgraph (descending)
            candidate_degrees = []
            for v in candidates:
                degree = sum(1 for u in candidates if adj_matrix[v][u] == 1)
                candidate_degrees.append((degree, v))
            candidate_degrees.sort(reverse=True)
            
            # Try each candidate
            for _, v in candidate_degrees:
                if v not in candidates:  # Already removed
                    continue
                
                # New candidates are neighbors of v that are in current candidates
                new_candidates = []
                for u in candidates:
                    if u != v and adj_matrix[v][u] == 1:
                        new_candidates.append(u)
                
                current.append(v)
                branch_and_bound(current, new_candidates, depth + 1)
                current.pop()
                
                # Remove v from candidates for subsequent iterations
                candidates.remove(v)
                
                # Early termination if remaining candidates can't improve
                if len(current) + len(candidates) <= best_size:
                    break
        
        # Start with all vertices as candidates
        initial_candidates = list(range(n))
        branch_and_bound([], initial_candidates, 0)
        
        return best_clique
    
    def _greedy_coloring(self, adj_matrix, n):
        """
        Greedy graph coloring for upper bound estimation.
        """
        colors = np.zeros(n, dtype=np.int32)
        
        for v in range(n):
            # Find colors used by neighbors
            neighbor_colors = set()
            for u in range(v):
                if adj_matrix[v][u] == 1:
                    neighbor_colors.add(colors[u])
            
            # Assign smallest available color
            color = 0
            while color in neighbor_colors:
                color += 1
            colors[v] = color

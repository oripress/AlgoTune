 def dict_to_graph(data):

class Solver:
    def solve(self, problem: dict[str, Any], **kwargs) -> list[list[Any]]:
        """Solve maximum flow min cost using ortools."""
        capacity = problem["capacity"]
        cost = problem["cost"]
        s = problem["s"]
        t = problem["t"]
        n = len(capacity)
        
        # Create the min cost flow object
        mcf = min_cost_flow.SimpleMinCostFlow()
        
        # Add edges with capacity and cost
        for i in range(n):
            for j in range(n):
                if capacity[i][j] > 0:
                    mcf.add_arc_with_capacity_and_unit_cost(
                        i, j, capacity[i][j], cost[i][j]
                    )
        
        # Add supply and demand
        # We need to find max flow, so we set a large supply at source
        # and corresponding demand at sink
        max_possible_flow = sum(capacity[s][j] for j in range(n))
        mcf.set_node_supply(s, max_possible_flow)
        mcf.set_node_supply(t, -max_possible_flow)
        
        # Solve the problem
        status = mcf.solve()
        
        # Build solution matrix
        solution = [[0 for _ in range(n)] for _ in range(n)]
        
        if status == mcf.OPTIMAL:
            for arc in range(mcf.num_arcs()):
                i = mcf.tail(arc)
                j = mcf.head(arc)
                flow = mcf.flow(arc)
                if flow > 0:
                    solution[i][j] = flow
        
        return solution

from typing import Any
import numpy as np
from numba import njit, types
from numba.typed import List

@njit
def min_cost_max_flow(capacity, cost, s, t, n):
    # Initialize flow matrix
    flow = np.zeros((n, n), dtype=np.int64)
    
    # Create residual graph
    residual_cap = capacity.copy()
    
    # Bellman-Ford for finding shortest path
    @njit
    def bellman_ford(residual_cap, cost, s, t, n):
        dist = np.full(n, np.inf)
        parent = np.full(n, -1, dtype=np.int64)
        dist[s] = 0
        
        # Relax edges
        for _ in range(n - 1):
            for u in range(n):
                if dist[u] < np.inf:
                    for v in range(n):
                        if residual_cap[u, v] > 0:
                            new_cost = dist[u] + cost[u, v]
                            if new_cost < dist[v]:
                                dist[v] = new_cost
                                parent[v] = u
        
        return dist, parent
    
    # Main algorithm loop
    while True:
        # Find shortest path from s to t
        dist, parent = bellman_ford(residual_cap, cost, s, t, n)
        
        # If no path exists, we're done
        if parent[t] == -1:
            break
        
        # Find minimum capacity along the path
        path_flow = np.inf
        v = t
        while v != s:
            u = parent[v]
            path_flow = min(path_flow, residual_cap[u, v])
            v = u
        
        # Update flow and residual capacities
        v = t
        while v != s:
            u = parent[v]
            flow[u, v] += path_flow
            residual_cap[u, v] -= path_flow
            residual_cap[v, u] += path_flow
            v = u
    
    return flow

class Solver:
    def solve(self, problem: dict[str, Any], **kwargs) -> list[list[Any]]:
        """Solve maximum flow min cost using custom Numba-optimized algorithm."""
        capacity = np.array(problem["capacity"], dtype=np.int64)
        cost = np.array(problem["cost"], dtype=np.int64)
        s = problem["s"]
        t = problem["t"]
        n = len(capacity)
        
        # Run the algorithm
        flow = min_cost_max_flow(capacity, cost, s, t, n)
        
        # Convert to list format
        return flow.tolist()

    This is a fast O(n^2) algorithm, JIT-compiled with Numba for performance.
    It uses pre-allocated arrays to avoid memory allocation overhead in the main loop.

    if n == 0:
        return np.zeros(0, dtype=np.float64)

    if c[0] == 0.0:
        result = np.empty(n, dtype=np.float64)
        result.fill(np.nan)
        return result

    a = np.zeros(n, dtype=np.float64)
    b_vec = np.zeros(n, dtype=np.float64)
    x = np.zeros(n, dtype=np.float64)

            result.fill(np.nan)
            return result

        gamma_b = -eps_b / alpha

            a_temp[i] = a[i]

This task involves projecting a point onto the set of vectors that satisfy a Conditional Value-at-Risk (CVaR) constraint. CVaR is a coherent risk measure used in financial risk management to quantify the expected loss in the worst-case scenarios.

The projection problem is formulated as:

    minimize    ||x - x₀||₂²
    subject to  CVaR_β(Ax) ≤ κ

Where:
- x is the decision variable (the projected point)
- x₀ is the initial point to be projected
- A is a matrix where each row represents a scenario and each column corresponds to a component of x
- β is the CVaR probability level (typically 0.95 or 0.99)
- κ is the CVaR threshold (maximum allowable risk)

The CVaR constraint ensures that the expected loss in the worst (1-β) fraction of scenarios does not exceed κ. For example, if β = 0.95, CVaR measures the average loss in the worst 5% of scenarios.

import numpy as np
import cvxpy as cp

class Solver:
    def solve(self, problem: dict, **kwargs) -> dict:
        """
        Fast projection of x0 onto the CVaR constraint
           sum of the k largest entries of A x <= kappa * k
        using an active-set / hyperplane‐projection loop, with a CVXPY/ECOS fallback.
        """
        # extract data
        x0 = np.array(problem["x0"], dtype=float)
        A = np.array(problem["loss_scenarios"], dtype=float)
        beta = float(problem.get("beta", 0.95))
        kappa = float(problem.get("kappa", 0.0))
        n, m = A.shape

        # number of “worst” scenarios
        k = int((1.0 - beta) * n)
        # trivial cases
        if k <= 0:
            return {"x_proj": x0.tolist()}
        if k >= n:
            # sum of all losses <= kappa * n
            alpha = kappa * n
        else:
            alpha = kappa * k

        # check if x0 already feasible
        losses = A.dot(x0)
        if k < 1:
            return {"x_proj": x0.tolist()}
        # sum of k largest losses at x0
        if k >= n:
            cv_sum = losses.sum()
        else:
            cv_sum = np.partition(losses, -k)[-k:].sum()
        if cv_sum <= alpha + 1e-9:
            return {"x_proj": x0.tolist()}

        # initial active set: indices of k largest losses at x0
        if k >= n:
            S = np.arange(n)
        else:
            S = np.argpartition(losses, -k)[-k:]
        S = np.sort(S)

        # active‐set/hyperplane projection iterations
        x_new = x0.copy()
        max_iters = 50
        for _ in range(max_iters):
            # build hyperplane d^T x = alpha
            d = A[S].sum(axis=0)
            dn2 = d.dot(d)
            if dn2 <= 0:
                break
            # projection of x0 onto {x | d^T x = alpha}
            nu = (d.dot(x0) - alpha) / dn2
            x_new = x0 - nu * d

            # recompute top‐k at x_new
            losses = A.dot(x_new)
            if k >= n:
                S_new = np.arange(n)
            else:
                S_new = np.argpartition(losses, -k)[-k:]
            S_new = np.sort(S_new)

            # if active set stabilizes, we're done
            if S_new.shape == S.shape and np.array_equal(S_new, S):
                return {"x_proj": x_new.tolist()}
            S = S_new

        # final feasibility check
        losses = A.dot(x_new)
        if k >= n:
            cv_sum = losses.sum()
        else:
            cv_sum = np.partition(losses, -k)[-k:].sum()
        if cv_sum <= alpha + 1e-6:
            return {"x_proj": x_new.tolist()}

        # fallback to CVXPY/ECOS if active‐set did not converge
        try:
            x = cp.Variable(m)
            obj = cp.Minimize(cp.sum_squares(x - x0))
            # CVaR constraint as sum of k largest
            cons = [cp.sum_largest(A @ x, k) <= alpha]
            prob = cp.Problem(obj, cons)
            prob.solve(solver=cp.ECOS, warm_start=True)
            if x.value is None:
                return {"x_proj": []}
            return {"x_proj": x.value.tolist()}
        except Exception:
            # last‐ditch return of our iterate
            return {"x_proj": x_new.tolist()}

import numpy as np

class Solver:
    def solve(self, problem, **kwargs):
        """Project x0 onto {x: CVaR_beta(Ax) <= kappa} with heuristic and CVXPY fallback."""
        x0 = np.array(problem["x0"], dtype=float)
        A = np.array(problem["loss_scenarios"], dtype=float)
        beta = float(problem["beta"])
        kappa = float(problem["kappa"])
        m, n = A.shape

        # number of worst-case scenarios
        k = int((1.0 - beta) * m)
        if k <= 0:
            # no tail scenarios => x0 is feasible
            return {"x_proj": x0.tolist()}

        alpha = kappa * k

        try:
            # losses at x0
            losses = A.dot(x0)
            # indices of k largest losses
            if k >= m:
                idx = np.arange(m)
            else:
                idx = np.argpartition(losses, -k)[-k:]
            # supporting hyperplane: h^T x <= alpha
            h = np.sum(A[idx], axis=0)
            c = h.dot(x0) - alpha
            if c <= 0:

        h = A[idx0].sum(axis=0)
        # Offset from boundary
        top_sum0 = losses0[idx0].sum()
        c0 = top_sum0 - alpha
        if c0 <= 0:
            # Already satisfies CVaR constraint
            return {"x_proj": x0.tolist()}

        denom = h.dot(h)
        if denom <= 0:
            # Degenerate: fallback to x0
            return {"x_proj": x0.tolist()}

        # Precompute directional losses
        v = A.dot(h)

        # Binary search on lambda to satisfy sum of k largest losses = alpha
        lo = 0.0
        hi = c0 / denom
        tol = 1e-8
        for _ in range(40):
            mid = 0.5 * (lo + hi)
            # losses at x = x0 - mid * h
            losses_mid = losses0 - mid * v
            # sum of k largest losses

                top_sum = losses_mid.sum()

            if abs(f_mid) <= tol:
                lo = hi = mid
                break
            if f_mid > 0:
                lo = mid
            else:

import numpy as np

class Solver:
    def solve(self, problem, **kwargs):
        """
        Project x0 onto {x: CVaR_beta(Ax) <= kappa} with active-set hyperplane projections.
        """
        # Extract data
        x0 = np.array(problem["x0"], dtype=float)
        A = np.array(problem["loss_scenarios"], dtype=float)
        beta = float(problem.get("beta", 0.95))
        kappa = float(problem.get("kappa", 0.0))
        m, n = A.shape

        # Determine number of tail scenarios
        k = int((1.0 - beta) * m)
        if k <= 0:
            return {"x_proj": x0.tolist()}
        if k > m:
            k = m
        alpha = kappa * k

        # Initialize
        x = x0.copy()
        tol = 1e-7
        max_iter = 20
        prev_idx = None

        # Active-set iterations
        for _ in range(max_iter):
            losses = A.dot(x)
            # Indices of k largest losses
            if k < m:
                idx = np.argsort(losses)[-k:]
            else:
                idx = np.arange(m)
            idx_sorted = np.sort(idx)

            # Sum of worst-case losses
            sum_loss = losses[idx].sum()
            # Check feasibility
            if sum_loss <= alpha + tol:
                break

            # Compute subgradient direction
            h = A[idx].sum(axis=0)
            denom = h.dot(h)
            if denom <= tol:
                break

            # Project onto the supporting hyperplane
            x = x - (sum_loss - alpha) / denom * h

            # If support did not change, done
            if prev_idx is not None and np.array_equal(idx_sorted, prev_idx):
                break
            prev_idx = idx_sorted

        return {"x_proj": x.tolist()}

import numpy as np
import cvxpy as cp

class Solver:
    def solve(self, problem, **kwargs):
        """
        Project x0 onto {x: CVaR_beta(Ax) <= kappa} by solving a QP with auxiliary variables
        using OSQP (fallback to ECOS).
        """
        # Extract data
        x0 = np.array(problem["x0"], dtype=float)
        A = np.array(problem["loss_scenarios"], dtype=float)
        beta = float(problem.get("beta", 0.95))
        kappa = float(problem.get("kappa", 0.0))
        m, n = A.shape

        # Number of worst-case scenarios
        k = int((1.0 - beta) * m)
        if k <= 0:
            # No tail => x0 is feasible
            return {"x_proj": x0.tolist()}
        if k > m:
            k = m
        alpha = kappa * k

        # Variables: x in R^n, t scalar, u in R^m
        x = cp.Variable(n)
        t = cp.Variable()
        u = cp.Variable(m)

        # Objective: minimize ||x - x0||^2
        objective = cp.Minimize(cp.sum_squares(x - x0))

        # Constraints for CVaR: u >= A x - t, u >= 0, sum(u) + k*t <= alpha
        constraints = [
            u >= A @ x - t,
            u >= 0,
            cp.sum(u) + k * t <= alpha
        ]

        prob = cp.Problem(objective, constraints)

        # Solve with OSQP for QP speed, fallback to ECOS if needed
        try:
            prob.solve(solver=cp.OSQP, warm_start=True,
                       eps_abs=1e-6, eps_rel=1e-6, verbose=False)
        except (cp.SolverError, ValueError):
            prob.solve(solver=cp.ECOS, verbose=False)

        x_val = x.value
        if x_val is None:
            return {"x_proj": []}
        return {"x_proj": x_val.tolist()}

import numpy as np
import cvxpy as cp

class Solver:
    def solve(self, problem, **kwargs) -> dict:
        """
        Compute the projection onto the CVaR constraint set using an active-set projection with fallback to CVXPy.
        """
        # Extract data
        x0 = np.array(problem["x0"], dtype=float)
        A = np.array(problem["loss_scenarios"], dtype=float)
        beta = float(problem.get("beta", 1.0))
        kappa = float(problem.get("kappa", 0.0))
        m, n = A.shape

        # Determine number of worst-case scenarios
        k = int(np.floor((1.0 - beta) * m))
        # Trivial if no CVaR constraint
        if k <= 0:
            return {"x_proj": x0.tolist()}
        # Cap k
        if k > m:
            k = m
        # CVaR threshold
        alpha = kappa * k

        # Active-set iterative projection
        max_iter = 20
        tol = 1e-8
        x = x0.copy()
        seen = set()
        for _ in range(max_iter):
            y = A.dot(x)
            # find indices of top k losses
            if k == m:
                idx = np.arange(m)
            else:
                idx = np.argpartition(y, -k)[-k:]
            key = tuple(np.sort(idx))
            if key in seen:
                break
            seen.add(key)
            # violation
            c = y[idx].sum() - alpha
            if c <= tol:
                return {"x_proj": x.tolist()}
            # projection step
            h = A[idx].sum(axis=0)
            denom = h.dot(h)
            if denom <= tol:
                break
            x = x - (c / denom) * h

        # Final feasibility check
        y = A.dot(x)
        if k == m:
            idx = np.arange(m)
        else:
            idx = np.argpartition(y, -k)[-k:]
        if y[idx].sum() - alpha <= tol:
            return {"x_proj": x.tolist()}

        # Fallback to CVXPy solver
        x_var = cp.Variable(n)
        objective = cp.Minimize(cp.sum_squares(x_var - x0))
        constraints = [cp.sum_largest(A @ x_var, k) <= alpha]
        prob = cp.Problem(objective, constraints)
        try:
            prob.solve(solver=cp.OSQP, warm_start=True)
        except Exception:
            prob.solve(solver=cp.ECOS)

        xv = x_var.value
        if xv is None:
            return {"x_proj": x0.tolist()}
        return {"x_proj": xv.tolist()}

def _active_cvar(x0, A, beta, kappa, maxiter):
    m, n = A.shape
    # number of tail scenarios
    k = int((1.0 - beta) * m)
    if k <= 0:
        return x0.copy()

    # initialize
    x_curr = x0.copy()
    losses = np.empty(m, np.float64)
    idx = np.empty(k, np.int64)
    h = np.empty(n, np.float64)
    for it in range(maxiter):
        # compute losses = A @ x_curr
        for i in range(m):
            acc = 0.0
            for d in range(n):
                acc += A[i, d] * x_curr[d]
            losses[i] = acc
        # threshold value for top k losses
        thr = np.partition(losses, m-k)[m-k]
        # collect indices with losses > thr
        cnt = 0
        for i in range(m):
            if losses[i] > thr:
                idx[cnt] = i
                cnt += 1
        # fill ties equal thr
        if cnt < k:
            for i in range(m):
                if losses[i] == thr:
                    idx[cnt] = i
                    cnt += 1
                    if cnt == k:
                        break
        # compute sum of gradients h = sum_j A[idx[j]]

            h[d] = 0.0
        for j in range(k):

                h[d] += A[r, d]
        # compute violation c = h^T x_curr - alpha
        c = 0.0
        for d in range(n):
            c += h[d] * x_curr[d]
        c -= alpha

        for d in range(n):

        t = c / h_norm2

            c = h.dot(x) - alpha
            if c <= 1e-12:
                break
            h_norm2 = h.dot(h)
            if h_norm2 <= 1e-16:
                break
            # project onto halfspace
            t = c / h_norm2
            x = x - t * h

        # verify CVaR constraint
        losses = A.dot(x)
        if k >= m:
            idx = np.arange(m)
        else:
            idx = np.argpartition(losses, -k)[-k:]
        cvar = losses[idx].sum() / k
        if cvar <= kappa + 1e-6:
            return {"x_proj": x.tolist()}

        # fallback to CVXPY if needed
        try:
            import cvxpy as cp
            xv = cp.Variable(n)
            obj = cp.Minimize(cp.sum_squares(xv - x0))
            con = [cp.sum_largest(A @ xv, k) <= alpha]
            prob = cp.Problem(obj, con)
            prob.solve(solver=cp.OSQP)
            sol = xv.value
            if sol is None:
                return {"x_proj": []}
            return {"x_proj": sol.tolist()}
        except Exception:
            return {"x_proj": []}

import numpy as np
import torch
import torch.nn.functional as F

class Solver:
    def __init__(self):
        # match reference: cubic spline interpolation (order=3), constant padding
        self.order = 3
        self.mode = 'constant'

    def solve(self, problem, **kwargs):
        # Extract inputs
        image = problem["image"]
        zoom_factor = problem["zoom_factor"]

        # Convert to torch tensor (CPU)
        arr = torch.tensor(image, dtype=torch.float32)
        # Ensure 2D
        if arr.ndimension() != 2:
            arr = arr.squeeze()

        H, W = arr.shape
        # Compute output dimensions (floor of scaled size)
        out_H = int(np.floor(H * zoom_factor))
        out_W = int(np.floor(W * zoom_factor))

        # Prepare for interpolation: shape [N=1, C=1, H, W]
        arr = arr.unsqueeze(0).unsqueeze(0)

        # Perform bicubic interpolation (constant padding is implicit as zeros)
        # align_corners=False is typical for image resizing
        zoomed = F.interpolate(arr, size=(out_H, out_W), mode='bicubic', align_corners=False)

        # Convert back to Python list of lists
        zoomed_np = zoomed[0, 0].cpu().numpy()
        return {"zoomed_image": zoomed_np.tolist()}

    def solve(self, problem, **kwargs):
        image = np.asarray(problem["image"], dtype=np.float64)
        zoom_factor = problem["zoom_factor"]
        H, W = image.shape
        out_H = int(np.floor(H * zoom_factor))
        out_W = int(np.floor(W * zoom_factor))
        if out_H <= 0 or out_W <= 0:
            return {"zoomed_image": []}
        # Prepare tensor [1,1,H,W] on CPU
        tensor = torch.from_numpy(image).unsqueeze(0).unsqueeze(0)

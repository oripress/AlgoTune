import numpy as np
import numba as nb
from scipy.linalg import expm

@nb.njit(cache=True)
def matrix_exp_pade(A, order=13):
    """Compute matrix exponential using Padé approximation with scaling and squaring."""
    n = A.shape[0]
    
    # Scaling factor
    norm = np.linalg.norm(A, ord=1)
    s = max(0, int(np.log2(norm)) + 1)
    A_scaled = A / (2.0 ** s)
    
    # Padé coefficients for order 13
    if order == 13:
        b = np.array([1.0, 1.0, 0.5, 1.0/6.0, 1.0/24.0, 1.0/120.0, 1.0/720.0,
                      1.0/5040.0, 1.0/40320.0, 1.0/362880.0, 1.0/3628800.0,
                      1.0/39916800.0, 1.0/479001600.0, 1.0/6227020800.0])
    
    # Compute powers of A
    I = np.eye(n)
    A2 = A_scaled @ A_scaled
    A4 = A2 @ A2
    A6 = A4 @ A2
    
    # Even and odd polynomials
    U = b[13] * A6
    V = b[12] * A6
    
    for i in range(5, -1, -1):
        U = U @ A2 + b[2*i+1] * (A6 if i >= 3 else (A4 if i >= 2 else (A2 if i >= 1 else I)))
        V = V @ A2 + b[2*i] * (A6 if i >= 3 else (A4 if i >= 2 else (A2 if i >= 1 else I)))
    
    U = A_scaled @ U
    
    # Solve (V - U) @ R = V + U
    R = np.linalg.solve(V - U, V + U)
    
    # Squaring phase
    for _ in range(s):
        R = R @ R
    
    return R

class Solver:
    def __init__(self):
        # Warm up the JIT compilation
        test_matrix = np.array([[1.0, 0.0], [0.0, 1.0]])
        _ = matrix_exp_pade(test_matrix)
    
    def solve(self, problem, **kwargs):
        """Compute the matrix exponential exp(A)."""
        A = np.array(problem["matrix"], dtype=np.float64)
        
        # For small matrices, use custom implementation
        if A.shape[0] <= 10:
            expA = matrix_exp_pade(A)
        else:
            # For larger matrices, fall back to scipy
            expA = expm(A)
            
        return {"exponential": expA.tolist()}

import mpmath as mp

class Solver:
    def solve(self, problem: Dict[str, Any], **kwargs) -> Dict[str, Any]:
        """
        Count zeros of the Riemann Zeta function in the critical strip 0<Re(s)<1
        with Im(s) <= t using the Riemann–von Mangoldt formula and argument computation.
        """
        t = problem["t"]
        # set precision high enough for argument computations
        mp.mp.dps = 50
        # Compute Riemann-Siegel theta(t)
        # theta(t) = Im(log(Gamma(1/4 + i t/2))) - (t/2)*log(pi)
        half_it = mp.mpc(0, t/2)
        theta = mp.im(mp.log(mp.gamma(mp.mpc(0.25) + half_it))) - (t/2) * mp.log(mp.pi)
        # Compute argument S(t): Im(log(zeta(1/2 + i t)))
        z_arg = mp.im(mp.log(mp.zeta(mp.mpc(0.5) + mp.mpc(0, t))))
        # Riemann-von Mangoldt: N(t) = theta/pi + 1 + S(t)
        N_approx = theta/mp.pi + 1 + z_arg/mp.pi
        # Round to nearest integer
        result = int(mp.floor(N_approx + mp.mpf('0.5')))
        return {"result": result}

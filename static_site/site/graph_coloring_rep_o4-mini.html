    def solve(self, problem, **kwargs):
        n = len(problem)
        if n == 0:
            return []
        # Build neighbor bitmask and degrees
        neighbor = [0] * n
        degrees = [0] * n
        for i, row in enumerate(problem):
            mask = 0
            d = 0
            for j, val in enumerate(row):
                if val:
                    mask |= 1 << j
                    d += 1
            neighbor[i] = mask
            degrees[i] = d

        # Greedy Welsh-Powell for initial UB
        order = sorted(range(n), key=lambda x: -degrees[x])
        greedy_masks = []
        greedy_col = [-1] * n
        for v in order:
            for c, cm in enumerate(greedy_masks):
                if not (neighbor[v] & cm):
                    greedy_masks[c] |= 1 << v
                    greedy_col[v] = c
                    break
            else:
                greedy_masks.append(1 << v)
                greedy_col[v] = len(greedy_masks) - 1
        best_UB = len(greedy_masks)
        best_col = greedy_col[:]

        # Prepare for B&B
        cur_color = [-1] * n
        max_colors = best_UB
        sat_count = [[0] * max_colors for _ in range(n)]
        sat_value = [0] * n
        neighbor_list = [[] for _ in range(n)]
        for u in range(n):
            fm = neighbor[u]
            while fm:
                bit = fm & -fm
                v = bit.bit_length() - 1
                neighbor_list[u].append(v)
                fm ^= bit

        FULL_MASK = (1 << n) - 1

        def dfs(used_colors, free_mask,
                neighbor_list=neighbor_list,
                sat_count=sat_count,
                sat_value=sat_value,
                degrees=degrees,
                cur_color=cur_color):
            nonlocal best_UB, best_col
            if free_mask == 0:
                if used_colors < best_UB:
                    best_UB = used_colors
                    best_col = cur_color[:]
                return
            if used_colors >= best_UB:
                return

            # Select vertex with max saturation then degree
            sel_v = -1
            max_sat = -1
            max_deg = -1
            fm = free_mask
            while fm:
                bit = fm & -fm
                v = bit.bit_length() - 1
                fm ^= bit
                sv = sat_value[v]
                if sv > max_sat or (sv == max_sat and degrees[v] > max_deg):

            v = sel_v

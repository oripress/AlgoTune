class Solver:

from cryptography.hazmat.primitives.ciphers.aead import ChaCha20Poly1305
from typing import Any
import multiprocessing as mp
from concurrent.futures import ProcessPoolExecutor, ThreadPoolExecutor
import os

CHACHA_KEY_SIZE = 32
POLY1305_TAG_SIZE = 16
CHUNK_SIZE = 1024 * 1024  # 1MB chunks

class Solver:
    def __init__(self):
        self._cipher_cache = {}
        # Pre-initialize a thread pool
        self._executor = ThreadPoolExecutor(max_workers=os.cpu_count())
    
    def solve(self, problem, **kwargs) -> Any:
        """
        Solve the ChaCha20-Poly1305 encryption problem by encrypting the plaintext.
        """
        key = problem["key"]
        nonce = problem["nonce"]
        plaintext = problem["plaintext"]
        associated_data = problem["associated_data"]
        
        if len(key) != CHACHA_KEY_SIZE:
            raise ValueError(f"Invalid key size: {len(key)}. Must be {CHACHA_KEY_SIZE}.")
        
        # Cache cipher objects by key
        if key not in self._cipher_cache:
            self._cipher_cache[key] = ChaCha20Poly1305(key)
        
        chacha = self._cipher_cache[key]
        
        # For small plaintexts, use direct encryption
        if len(plaintext) < CHUNK_SIZE * 2:
            ciphertext = chacha.encrypt(nonce, plaintext, associated_data)
            # Extract ciphertext and tag
            actual_ciphertext = ciphertext[:-POLY1305_TAG_SIZE]
            tag = ciphertext[-POLY1305_TAG_SIZE:]
            return {"ciphertext": actual_ciphertext, "tag": tag}
        
        # For larger plaintexts, we still need to use single encryption
        # because ChaCha20-Poly1305 AEAD mode requires the entire message
        ciphertext = chacha.encrypt(nonce, plaintext, associated_data)
        
        # Extract ciphertext and tag
        actual_ciphertext = ciphertext[:-POLY1305_TAG_SIZE]
        tag = ciphertext[-POLY1305_TAG_SIZE:]
        
        return {"ciphertext": actual_ciphertext, "tag": tag}

from cryptography.hazmat.primitives.ciphers.aead import ChaCha20Poly1305
from typing import Any

CHACHA_KEY_SIZE = 32
POLY1305_TAG_SIZE = 16

class Solver:
    def __init__(self):
        self._cipher_cache = {}
    
    def solve(self, problem, **kwargs) -> Any:
        """
        Solve the ChaCha20-Poly1305 encryption problem by encrypting the plaintext.
        """
        key = problem["key"]
        nonce = problem["nonce"]
        plaintext = problem["plaintext"]
        associated_data = problem["associated_data"]
        
        if len(key) != CHACHA_KEY_SIZE:
            raise ValueError(f"Invalid key size: {len(key)}. Must be {CHACHA_KEY_SIZE}.")
        
        # Cache cipher objects by key
        if key not in self._cipher_cache:
            self._cipher_cache[key] = ChaCha20Poly1305(key)
        
        chacha = self._cipher_cache[key]
        ciphertext = chacha.encrypt(nonce, plaintext, associated_data)
        
        # Use memoryview to avoid copying for large ciphertexts
        ciphertext_view = memoryview(ciphertext)
        actual_ciphertext = bytes(ciphertext_view[:-POLY1305_TAG_SIZE])
        tag = bytes(ciphertext_view[-POLY1305_TAG_SIZE:])
        
        return {"ciphertext": actual_ciphertext, "tag": tag}

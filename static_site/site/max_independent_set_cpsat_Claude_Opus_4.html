class Solver:
    def solve(self, problem):
        """
        Solves the max independent set problem.
        
        :param problem: A 2d adjacency matrix representing the graph.
        :return: A list of node indices included in the maximum independent set.
        """
        n = len(problem)
        
        # For small graphs, use dynamic programming with bitmasks
        if n <= 20:
            return self._solve_small(problem)
        else:
            # For larger graphs, use OR-Tools
            return self._solve_ortools(problem)
    
    def _solve_small(self, problem):
        """Solve using dynamic programming for small graphs."""
        n = len(problem)
        
        # Precompute adjacency for faster access
        adj = []
        for i in range(n):
            neighbors = 0
            for j in range(n):
                if problem[i][j] == 1:
                    neighbors |= (1 << j)
            adj.append(neighbors)
        
        # Check if a set is independent
        def is_independent(mask):
            for i in range(n):
                if mask & (1 << i):
                    # Check if any neighbor is also in the set
                    if mask & adj[i]:
                        return False
            return True
        
        # Find the maximum independent set
        best_mask = 0
        best_size = 0
        
        # Try all possible subsets
        for mask in range(1 << n):
            if is_independent(mask):
                size = bin(mask).count('1')
                if size > best_size:
                    best_size = size
                    best_mask = mask
        
        # Convert mask to list of indices
        result = []
        for i in range(n):
            if best_mask & (1 << i):
                result.append(i)
        
        return result
    
    def _solve_ortools(self, problem):
        """Solve using OR-Tools for larger graphs."""
        from ortools.sat.python import cp_model

    def _solve_small(self, problem):
        """Solve using optimized branch and bound for small graphs."""
        n = len(problem)
        
        # Precompute adjacency lists for faster access
        adj = [[] for _ in range(n)]
        for i in range(n):
            for j in range(n):
                if problem[i][j] == 1:
                    adj[i].append(j)
        
        # Compute degree and sort nodes by degree (heuristic)
        degrees = [(len(adj[i]), i) for i in range(n)]
        degrees.sort()
        node_order = [node for _, node in degrees]
        
        best_solution = []
        best_size = 0
        
        def branch_and_bound(current_set, remaining_nodes, excluded):
            nonlocal best_solution, best_size
            
            # Pruning: if current + remaining can't beat best, stop
            if len(current_set) + len(remaining_nodes) <= best_size:
                return
            
            if not remaining_nodes:
                if len(current_set) > best_size:
                    best_size = len(current_set)
                    best_solution = current_set[:]
                return
            
            # Take the first remaining node
            node = remaining_nodes[0]
            new_remaining = remaining_nodes[1:]
            
            # Branch 1: Include the node
            if node not in excluded:
                new_excluded = excluded | set(adj[node])
                new_current = current_set + [node]
                branch_and_bound(new_current, new_remaining, new_excluded)
            
            # Branch 2: Don't include the node
            branch_and_bound(current_set, new_remaining, excluded)
        
        branch_and_bound([], node_order, set())
        return sorted(best_solution)

class Solver:
    def __init__(self):
        # Pre-compile the numba function
        self._solve_small_numba = compile_solve_small()
    
    def solve(self, problem):
        """
        Solves the max independent set problem.
        
        :param problem: A 2d adjacency matrix representing the graph.
        :return: A list of node indices included in the maximum independent set.
        """
        n = len(problem)
        
        # For small graphs, use numba-optimized solver
        if n <= 20:
            adj_matrix = np.array(problem, dtype=np.int32)
            result_mask = self._solve_small_numba(adj_matrix)
            # Convert mask to list of indices
            return [i for i in range(n) if result_mask & (1 << i)]
        else:
            # For larger graphs, use OR-Tools
            return self._solve_ortools(problem)
    
    def _solve_ortools(self, problem):
        """Solve using OR-Tools for larger graphs."""
        from ortools.sat.python import cp_model
        
        n = len(problem)
        model = cp_model.CpModel()
        
        # Create a boolean variable for each vertex
        nodes = [model.NewBoolVar(f"x_{i}") for i in range(n)]
        
        # Add independence constraints
        for i in range(n):
            for j in range(i + 1, n):
                if problem[i][j] == 1:
                    model.Add(nodes[i] + nodes[j] <= 1)
        
        # Objective: Maximize the number of vertices chosen
        model.Maximize(sum(nodes))
        
        # Solve the model
        solver = cp_model.CpSolver()
        solver.set_time_limit(10.0)  # Add time limit
        status = solver.Solve(model)
        
        if status == cp_model.OPTIMAL:
            selected = [i for i in range(n) if solver.Value(nodes[i]) == 1]
            return selected
        else:
            return []

def compile_solve_small():
    @numba.jit(nopython=True, cache=True)
    def solve_small_numba(adj_matrix):
        n = adj_matrix.shape[0]
        
        # Precompute adjacency masks
        adj_masks = np.zeros(n, dtype=np.int64)
        for i in range(n):
            mask = 0
            for j in range(n):
                if adj_matrix[i, j] == 1:
                    mask |= (1 << j)
            adj_masks[i] = mask
        
        best_mask = 0
        best_size = 0
        
        # Try all possible subsets
        for mask in range(1 << n):
            # Check if this is an independent set
            is_independent = True
            size = 0
            
            for i in range(n):
                if mask & (1 << i):
                    size += 1
                    # Check if any neighbor is also in the set
                    if mask & adj_masks[i]:
                        is_independent = False
                        break
            
            if is_independent and size > best_size:
                best_size = size
                best_mask = mask
        
        return best_mask
    
    return solve_small_numba

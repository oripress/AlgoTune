import numpy as np
from scipy import linalg
from typing import Any

class Solver:
    def __init__(self):
        # Precompute for 2x2 case
        self._eye2 = np.eye(2)
        
    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:
        """
        Solves the Lyapunov stability analysis problem.
        
        For discrete-time systems, stability requires all eigenvalues 
        of A to have magnitude less than 1.
        """
        A = np.array(problem["A"])
        n = A.shape[0]
        
        # Special case for 2x2 matrices - analytical solution
        if n == 2:
            a11, a12 = A[0, 0], A[0, 1]
            a21, a22 = A[1, 0], A[1, 1]
            
            # Check stability via trace and determinant
            trace = a11 + a22
            det = a11 * a22 - a12 * a21
            
            # For 2x2, eigenvalues satisfy: λ² - trace*λ + det = 0
            # Stability requires |λ| < 1 for both eigenvalues
            discriminant = trace * trace - 4 * det
            
            if discriminant >= 0:
                sqrt_disc = np.sqrt(discriminant)
                lambda1 = (trace + sqrt_disc) / 2
                lambda2 = (trace - sqrt_disc) / 2
                if abs(lambda1) >= 1 - 1e-10 or abs(lambda2) >= 1 - 1e-10:
                    return {"is_stable": False, "P": None}
            else:
                # Complex eigenvalues
                real_part = trace / 2
                imag_part = np.sqrt(-discriminant) / 2
                magnitude_sq = real_part * real_part + imag_part * imag_part
                if magnitude_sq >= (1 - 1e-10) ** 2:
                    return {"is_stable": False, "P": None}
            
            # Analytical solution for 2x2 discrete Lyapunov equation
            # Vectorize: vec(P) = (I - A^T ⊗ A^T)^(-1) vec(Q)
            # For Q = I, we can solve this directly
            try:
                # Build the system matrix
                I4 = np.eye(4)
                AT = A.T
                kron_AT_AT = np.kron(AT, AT)
                M = I4 - kron_AT_AT
                
                # vec(I) for 2x2
                vec_I = np.array([1, 0, 0, 1])
                
                # Solve for vec(P)
                vec_P = np.linalg.solve(M, vec_I)
                
                # Reshape to get P
                P = vec_P.reshape(2, 2)
                
                # Make symmetric (due to numerical errors)
                P = (P + P.T) / 2
                
                # Quick positive definite check
                if P[0, 0] <= 1e-10 or P[0, 0] * P[1, 1] - P[0, 1] * P[1, 0] <= 1e-10:
                    return {"is_stable": False, "P": None}
                
                return {"is_stable": True, "P": P.tolist()}
            except:
                return {"is_stable": False, "P": None}
        
        # For larger matrices, use the original approach
        eigenvalues = np.linalg.eigvals(A)
        max_magnitude = np.max(np.abs(eigenvalues))
        
        if max_magnitude >= 1.0 - 1e-10:
            return {"is_stable": False, "P": None}
        
        try:
            Q = np.eye(n)
            P = linalg.solve_discrete_lyapunov(A.T, Q)
            
            # Quick positive definite check using Cholesky
            try:
                np.linalg.cholesky(P)
                return {"is_stable": True, "P": P.tolist()}
            except:
                return {"is_stable": False, "P": None}
            
        except:
            return {"is_stable": False, "P": None}

import numpy as np
from scipy import linalg
from typing import Any
import numba

@numba.jit(nopython=True, cache=True)
def check_2x2_stability(a11, a12, a21, a22):
    """Check stability of 2x2 matrix via eigenvalues."""
    trace = a11 + a22
    det = a11 * a22 - a12 * a21
    discriminant = trace * trace - 4 * det
    
    if discriminant >= 0:
        sqrt_disc = np.sqrt(discriminant)
        lambda1 = (trace + sqrt_disc) / 2
        lambda2 = (trace - sqrt_disc) / 2
        return abs(lambda1) < 1 - 1e-10 and abs(lambda2) < 1 - 1e-10
    else:
        # Complex eigenvalues
        real_part = trace / 2
        imag_part = np.sqrt(-discriminant) / 2
        magnitude_sq = real_part * real_part + imag_part * imag_part
        return magnitude_sq < (1 - 1e-10) ** 2

@numba.jit(nopython=True, cache=True)
def solve_2x2_lyapunov(a11, a12, a21, a22):
    """Analytical solution for 2x2 discrete Lyapunov equation A^T P A - P = -I."""
    # Direct formula for 2x2 case
    # Build coefficient matrix for vectorized equation
    c11 = 1 - a11 * a11 - a21 * a21
    c12 = -a11 * a12 - a21 * a22
    c21 = -a11 * a21 - a12 * a21
    c22 = 1 - a12 * a12 - a22 * a22
    c33 = 1 - a11 * a11 - a12 * a12
    c34 = -a11 * a21 - a12 * a22
    c44 = 1 - a21 * a21 - a22 * a22
    
    # Solve for P elements
    # P is symmetric, so p12 = p21
    det = c11 * c22 - c12 * c21
    if abs(det) < 1e-15:
        return None, None, None
    
    p11 = c22 / det
    p12 = -c12 / det
    
    # For p22, use the fact that A^T P A - P = -I
    # This gives us p22 from the (2,2) element
    p22 = (1 + a21 * a21 * p11 + 2 * a21 * a22 * p12) / c44
    
    return p11, p12, p22

import numpy as np
from typing import Any

class Solver:
    def __init__(self):
        # Precompute constants
        self.eps = 1e-10
        self.stable_threshold = 1 - 1e-10
        
    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:
        """
        Solves the Lyapunov stability analysis problem.
        """
        A = np.array(problem["A"], dtype=np.float64)
        n = A.shape[0]
        
        if n == 2:
            # Optimized 2x2 case
            return self._solve_2x2(A)
        elif n == 3:
            # Optimized 3x3 case
            return self._solve_3x3(A)
        else:
            # General case
            return self._solve_general(A)
    
    def _solve_2x2(self, A):
        """Optimized solution for 2x2 matrices."""
        a11, a12 = A[0, 0], A[0, 1]
        a21, a22 = A[1, 0], A[1, 1]
        
        # Check stability via characteristic polynomial
        trace = a11 + a22
        det = a11 * a22 - a12 * a21
        
        # Check Schur stability conditions for 2x2
        # |det| < 1 and |trace| < 1 + det
        if abs(det) >= self.stable_threshold or abs(trace) >= 1 + det - self.eps:
            return {"is_stable": False, "P": None}
        
        # Direct solution of discrete Lyapunov equation
        # Vectorized form: (I - A^T ⊗ A^T) vec(P) = vec(I)
        # For 2x2, this is a 4x4 linear system
        
        # Build the coefficient matrix
        c11 = 1 - a11*a11 - a21*a21
        c12 = -a11*a12 - a21*a22
        c21 = -a11*a21 - a12*a21  
        c22 = 1 - a12*a12 - a22*a22
        c33 = 1 - a11*a11 - a12*a12
        c34 = -a11*a21 - a12*a22
        c44 = 1 - a21*a21 - a22*a22
        
        # Solve the reduced 3x3 system (using symmetry of P)
        # [c11  2*c12   0  ] [p11]   [1]
        # [c21  c22+c33 c34] [p12] = [0]
        # [0    2*c34   c44] [p22]   [1]
        
        det_main = c11 * (c22+c33) * c44 + 2*c12*c34*2*c34 - c11*c34*c34 - c44*2*c12*c21
        
        if abs(det_main) < 1e-15:
            return {"is_stable": False, "P": None}
        
        # Use Cramer's rule for speed
        det1 = (c22+c33) * c44 - c34*c34
        det2 = -c21*c44
        det3 = c11*(c22+c33) - 2*c12*c21
        
        p11 = det1 / det_main
        p12 = det2 / det_main  
        p22 = det3 / det_main
        
        # Check positive definiteness
        if p11 <= self.eps or p11*p22 - p12*p12 <= self.eps:
            return {"is_stable": False, "P": None}
        
        return {"is_stable": True, "P": [[p11, p12], [p12, p22]]}
    
    def _solve_3x3(self, A):
        """Optimized solution for 3x3 matrices."""
        # Quick stability check using Gershgorin circles
        for i in range(3):
            row_sum = np.sum(np.abs(A[i, :])) - np.abs(A[i, i])
            if np.abs(A[i, i]) + row_sum >= self.stable_threshold:
                # May not be stable, need exact check
                eigenvalues = np.linalg.eigvals(A)
                if np.max(np.abs(eigenvalues)) >= self.stable_threshold:
                    return {"is_stable": False, "P": None}
                break
        else:
            # Definitely stable by Gershgorin, but double-check
            eigenvalues = np.linalg.eigvals(A)
            if np.max(np.abs(eigenvalues)) >= self.stable_threshold:
                return {"is_stable": False, "P": None}
        
        # Solve using numpy's linear algebra
        n = 3
        I = np.eye(n)
        AT = A.T
        
        # Vectorized Lyapunov equation
        kron_AT = np.kron(AT, AT)
        M = np.eye(n*n) - kron_AT
        vec_I = I.flatten()
        
        try:
            vec_P = np.linalg.solve(M, vec_I)
            P = vec_P.reshape(n, n)
            
            # Make symmetric
            P = (P + P.T) / 2
            
            # Check positive definiteness via Cholesky
            try:
                np.linalg.cholesky(P)
                return {"is_stable": True, "P": P.tolist()}
            except:
                return {"is_stable": False, "P": None}
        except:
            return {"is_stable": False, "P": None}
    
    def _solve_general(self, A):
        """General solution for larger matrices."""
        # Check stability
        eigenvalues = np.linalg.eigvals(A)
        if np.max(np.abs(eigenvalues)) >= self.stable_threshold:
            return {"is_stable": False, "P": None}
        
        # Use scipy for larger matrices
        from scipy import linalg
        try:
            Q = np.eye(A.shape[0])
            P = linalg.solve_discrete_lyapunov(A.T, Q)
            
            # Quick positive definite check
            if np.min(np.linalg.eigvals(P)) > self.eps:
                return {"is_stable": True, "P": P.tolist()}
            else:
                return {"is_stable": False, "P": None}
        except:
            return {"is_stable": False, "P": None}

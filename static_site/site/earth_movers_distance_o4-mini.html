class Solver:
    def solve(self, problem: dict, **kwargs) -> dict:
        """
        Solve the discrete optimal transport (EMD) problem by
        reducing to an integer min-cost flow and using OR-Tools.
        """
        # Load inputs as numpy arrays
        a = np.asarray(problem["source_weights"], dtype=np.float64)
        b = np.asarray(problem["target_weights"], dtype=np.float64)
        M = np.asarray(problem["cost_matrix"], dtype=np.float64)
        n, m = a.size, b.size

        # Scale supplies and demands to integers
        FLOW_SCALE = 10**6
        supply = np.rint(a * FLOW_SCALE).astype(np.int64)
        demand = np.rint(b * FLOW_SCALE).astype(np.int64)
        # Fix rounding mismatch
        diff = int(supply.sum() - demand.sum())
        if diff > 0:
            supply[int(np.argmax(supply))] -= diff
        elif diff < 0:
            demand[int(np.argmax(demand))] -= -diff

        # Scale costs to integers
        COST_SCALE = 10**6
        cost_int = np.rint(M * COST_SCALE).astype(np.int64)

        # Try OR-Tools min-cost flow
        try:
            pywrap = __import__('ortools.graph.pywrapgraph', fromlist=['SimpleMinCostFlow'])
            mcf = pywrap.SimpleMinCostFlow()

            # Add arcs: from source i to sink (n+j)
            for i in range(n):
                si = int(supply[i])
                if si <= 0:
                    continue
                for j in range(m):
                    dj = int(demand[j])
                    if dj <= 0:
                        continue
                    cap = si if si < dj else dj
                    if cap <= 0:
                        continue
                    mcf.AddArcWithCapacityAndUnitCost(
                        i, n + j, cap, int(cost_int[i, j])
                    )

            # Set supplies: positive for sources, negative for sinks
            for i in range(n):
                mcf.SetNodeSupply(int(i), int(supply[i]))
            for j in range(m):
                mcf.SetNodeSupply(int(n + j), -int(demand[j]))

            status = mcf.Solve()
            if status == mcf.OPTIMAL:
                G = np.zeros((n, m), dtype=np.float64)
                for arc in range(mcf.NumArcs()):
                    f = mcf.Flow(arc)
                    if f > 0:
                        u = mcf.Tail(arc)
                        v = mcf.Head(arc)
                        if u < n and v >= n:
                            G[int(u), int(v - n)] = f / FLOW_SCALE
                return {"transport_plan": G}

            np.ascontiguousarray(b, dtype=np.float64),
            np.ascontiguousarray(M, dtype=np.float64),
            check_marginals=False,
        )

class Solver:
    def __init__(self):
        # Cache A_eq matrices by (n, m)
        self._cache = {}

    def solve(self, problem, **kwargs):
        a = np.asarray(problem["source_weights"], dtype=np.float64)
        b = np.asarray(problem["target_weights"], dtype=np.float64)
        M = np.ascontiguousarray(problem["cost_matrix"], dtype=np.float64)
        n, m = a.size, b.size
        nm = n * m

        # Build or reuse the equality-constraint matrix A_eq
        key = (n, m)
        if key in self._cache:
            A_eq = self._cache[key]
        else:
            # Rows for source-sum constraints
            row_src = np.repeat(np.arange(n), m)
            # Rows for target-sum constraints
            row_tgt = np.tile(np.arange(m), n) + n
            rows = np.concatenate([row_src, row_tgt])
            cols = np.concatenate([np.arange(nm), np.arange(nm)])
            data = np.ones(2 * nm, dtype=np.float64)
            A_eq = sparse.coo_matrix((data, (rows, cols)), shape=(n + m, nm)).tocsr()

        b_eq = np.concatenate([a, b])
        c = M.ravel()

        # Solve via HiGHS (C++ network/simplex solver)
        res = linprog(c, A_eq=A_eq, b_eq=b_eq, bounds=(0, None), method="highs")
        G = res.x.reshape((n, m))

        b_list = problem["target_weights"]

        a = np.asarray(a_list, dtype=np.float64)
        b = np.asarray(b_list, dtype=np.float64)
        M = np.asarray(M_list, dtype=np.float64)
        n, m = a.size, b.size

        # Scale supplies/demands to integers
        scale_flow = 10**7
        A = np.rint(a * scale_flow).astype(np.int64).tolist()
        B = np.rint(b * scale_flow).astype(np.int64).tolist()
        sumA, sumB = sum(A), sum(B)
        diff = sumA - sumB
        if diff > 0:
            # subtract rounding error from largest demand
            idx = max(range(m), key=lambda j: B[j])
            B[idx] -= diff
        elif diff < 0:
            # add rounding error to largest supply
            idx = max(range(n), key=lambda i: A[i])
            A[idx] += -diff

        # Build the OR-Tools min cost flow
        mcf = pywrapgraph.SimpleMinCostFlow()
        # Supplies at nodes 0..n-1, demands at n..n+m-1

        for j in range(m):
            mcf.SetNodeSupply(n + j, -B[j])

        # Add arcs from each supply node to each demand node
        cost_scale = scale_flow
        for i in range(n):
            ai = A[i]
            for j in range(m):
                bj = B[j]
                cap = ai if ai < bj else bj
                if cap <= 0:

        status = mcf.Solve()
        if status != pywrapgraph.SimpleMinCostFlow.OPTIMAL:

            G = ot.lp.emd(a, b, np.ascontiguousarray(M, dtype=np.float64), check_marginals=False)
            return {"transport_plan": G}

        G = np.zeros((n, m), dtype=np.float64)

            if 0 <= u < n and n <= v < n + m:

from scipy.sparse import coo_matrix

# Cache constraint matrices for different problem sizes to avoid rebuild overhead
_AEQ_CACHE = {}

class Solver:
    def solve(self, problem: dict, **kwargs) -> dict:
        """
        Solve the discrete optimal transport (EMD) problem using the HiGHS LP solver.
        :param problem: dict with keys:
            - source_weights: list or 1D-array of length n
            - target_weights: list or 1D-array of length m
            - cost_matrix:    2D list or array shape (n, m)
        :returns: dict with key "transport_plan" and the n×m numpy array optimal coupling.
        """
        # Prepare input arrays
        a = np.asarray(problem["source_weights"], dtype=np.float64)
        b = np.asarray(problem["target_weights"], dtype=np.float64)
        M = np.ascontiguousarray(problem["cost_matrix"], dtype=np.float64)
        n, m = a.size, b.size
        nm = n * m

        # Build or retrieve sparse equality constraint matrix A_eq * x = b_eq
        key = (n, m)
        if key not in _AEQ_CACHE:
            # indices 0..nm-1 correspond to variables x[i,j] flattened row-major
            indices = np.arange(nm, dtype=np.int32)
            i_idx = indices // m
            j_idx = indices % m
            # Rows for source marginals: rows 0..n-1
            rows_src = i_idx
            cols_src = indices
            # Rows for target marginals: rows n..n+m-1
            rows_tgt = n + j_idx
            cols_tgt = indices
            # Combine
            rows = np.concatenate([rows_src, rows_tgt])
            cols = np.concatenate([cols_src, cols_tgt])
            data = np.ones(rows.shape[0], dtype=np.float64)
            A_eq = coo_matrix((data, (rows, cols)), shape=(n + m, nm)).tocsc()
            _AEQ_CACHE[key] = A_eq
        else:
            A_eq = _AEQ_CACHE[key]

        # Right-hand side
        b_eq = np.concatenate([a, b])

        # Objective vector
        c = M.ravel()

        # Solve the LP with HiGHS (default SciPy solver)
        res = linprog(c, A_eq=A_eq, b_eq=b_eq, method="highs")
        x = res.x

        # Reshape to transport plan matrix
        G = x.reshape((n, m))
        return {"transport_plan": G}

        # Scale supplies and demands to integers

        demand = np.rint(b * FLOW_SCALE).astype(np.int64)
        # Fix rounding mismatch
        diff = int(supply.sum() - demand.sum())
        if diff > 0:
            supply[int(np.argmax(supply))] -= diff
        elif diff < 0:
            demand[int(np.argmax(demand))] -= -diff

        # Scale costs to integers
        COST_SCALE = 10**6
        cost_int = np.rint(M * COST_SCALE).astype(np.int64)

        # Try OR-Tools min-cost flow
        try:
            pywrap = __import__('ortools.graph.pywrapgraph', fromlist=['SimpleMinCostFlow'])
            mcf = pywrap.SimpleMinCostFlow()

            # Add arcs: from source i to sink (n+j)
            for i in range(n):
                si = int(supply[i])

class Solver:
    def solve(self, problem: dict, **kwargs) -> dict:
        # Extract distributions and cost matrix
        a = np.asarray(problem["source_weights"], dtype=np.float64)
        b = np.asarray(problem["target_weights"], dtype=np.float64)
        M = np.asarray(problem["cost_matrix"], dtype=np.float64)
        n, m = a.size, b.size

        c = M.reshape(nm)

        # Build equality constraints: sum_j P[i,j] = a[i], sum_i P[i,j] = b[j]
        rows1 = np.repeat(np.arange(n), m)

    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:
        """
        Solve the Convex Hull problem using an optimized algorithm.
        """
        points = np.array(problem["points"])
        n = len(points)
        
        # Handle trivial cases
        if n < 3:
            return {
                "hull_vertices": list(range(n)),
                "hull_points": points.tolist()
            }
        
        # Find the convex hull using Andrew's monotone chain algorithm
        # First, we need to sort points lexicographically
        indices = np.lexsort((points[:, 1], points[:, 0]))
        sorted_points = points[indices]
        
        # Build lower hull
        lower = []
        for i in range(n):
            while len(lower) >= 2 and self._cross_product(
                sorted_points[lower[-2]], sorted_points[lower[-1]], sorted_points[i]
            ) <= 0:
                lower.pop()
            lower.append(i)
        
        # Build upper hull
        upper = []
        for i in range(n - 1, -1, -1):
            while len(upper) >= 2 and self._cross_product(
                sorted_points[upper[-2]], sorted_points[upper[-1]], sorted_points[i]
            ) <= 0:
                upper.pop()
            upper.append(i)

        lower.pop()
        upper.pop()

@numba.njit(fastmath=True, cache=True)
def partition(points, indices, low, high):
    """Partition for quicksort based on lexicographic order."""
    pivot_idx = indices[high]
    pivot = points[pivot_idx]
    i = low - 1
    
    for j in range(low, high):
        idx = indices[j]
        if points[idx, 0] < pivot[0] or (points[idx, 0] == pivot[0] and points[idx, 1] < pivot[1]):
            i += 1
            indices[i], indices[j] = indices[j], indices[i]
    
    indices[i + 1], indices[high] = indices[high], indices[i + 1]
    return i + 1

@numba.njit(fastmath=True, cache=True)
def quicksort(points, indices, low, high):
    """Quicksort indices based on lexicographic order of points."""
    if low < high:
        pi = partition(points, indices, low, high)
        quicksort(points, indices, low, pi - 1)
        quicksort(points, indices, pi + 1, high)

    """Compute convex hull using Andrew's monotone chain algorithm with optimized sorting."""
    n = len(points)
    
    # Create indices and sort
    indices = np.arange(n, dtype=np.int32)
    quicksort(points, indices, 0, n - 1)
    
    # Pre-allocate arrays
    hull = np.empty(2 * n, dtype=np.int32)

    for i in range(n):
        pt_idx = indices[i]
        while hull_size >= 2:

            ) <= 0:

        hull[hull_size] = i

    hull_size -= 1

    result = np.empty(hull_size, dtype=np.int32)
    for i in range(hull_size):

def cross_product(O, A, B):

    return (A[0] - O[0]) * (B[1] - O[1]) - (A[1] - O[1]) * (B[0] - O[0])

@numba.njit(fastmath=True, cache=True)
def polar_angle_key(p0, p):
    """Compute polar angle for sorting in Graham scan."""
    dx = p[0] - p0[0]
    dy = p[1] - p0[1]
    if dx == 0 and dy == 0:
        return -np.pi, 0.0
    return np.arctan2(dy, dx), dx * dx + dy * dy

@numba.njit(fastmath=True, cache=True)
def graham_scan(points):
    """Compute convex hull using Graham scan algorithm."""
    n = len(points)
    
    # Find the bottom-most point (and left-most if tied)
    start_idx = 0
    for i in range(1, n):
        if points[i, 1] < points[start_idx, 1] or \
           (points[i, 1] == points[start_idx, 1] and points[i, 0] < points[start_idx, 0]):
            start_idx = i
    
    # Create indices array with start point first
    indices = np.empty(n, dtype=np.int32)
    indices[0] = start_idx
    j = 1
    for i in range(n):
        if i != start_idx:
            indices[j] = i
            j += 1
    
    # Sort by polar angle with respect to start point
    start_point = points[start_idx]
    angles = np.empty(n - 1, dtype=np.float64)
    dists = np.empty(n - 1, dtype=np.float64)
    
    for i in range(1, n):
        angle, dist = polar_angle_key(start_point, points[indices[i]])
        angles[i - 1] = angle
        dists[i - 1] = dist
    
    # Sort indices[1:] by angle (and distance if angles are equal)
    for i in range(1, n - 1):
        for j in range(i + 1, n):
            if angles[i - 1] > angles[j - 1] or \
               (angles[i - 1] == angles[j - 1] and dists[i - 1] > dists[j - 1]):
                indices[i], indices[j] = indices[j], indices[i]
                angles[i - 1], angles[j - 1] = angles[j - 1], angles[i - 1]
                dists[i - 1], dists[j - 1] = dists[j - 1], dists[i - 1]
    
    # Build convex hull
    hull = np.empty(n, dtype=np.int32)
    hull[0] = indices[0]
    hull[1] = indices[1]
    hull_size = 2
    
    for i in range(2, n):
        # Remove points that make clockwise turn
        while hull_size > 1:
            if cross_product(
                points[hull[hull_size - 2]], 
                points[hull[hull_size - 1]], 
                points[indices[i]]
            ) <= 0:
                hull_size -= 1
            else:
                break
        hull[hull_size] = indices[i]
        hull_size += 1
    
    return hull[:hull_size]

@numba.njit(fastmath=True, cache=True)
def andrew_convex_hull_fast(points):
    """Compute convex hull using Andrew's algorithm with optimized implementation."""
    n = len(points)
    
    # Use numpy's argsort for better performance
    order = np.lexsort((points[:, 1], points[:, 0]))

    hull = np.empty(2 * n, dtype=np.int32)

def andrew_convex_hull_ultra_fast(points):

    n = len(points)

    # Create sort keys to avoid repeated calculations
    sort_keys = np.empty(n, dtype=np.float64)
    for i in numba.prange(n):
        sort_keys[i] = points[i, 0] * 1000000.0 + points[i, 1]
    
    order = np.argsort(sort_keys).astype(np.int32)

    for i in range(n):
        idx = order[i]

            cp = cross_product(points[hull[k-2]], points[hull[k-1]], points[idx])
            if cp <= 0:
                k -= 1
            else:
                break

        k += 1

    lower_end = k - 1
    for i in range(n - 2, -1, -1):
        idx = order[i]
        while k > lower_end + 1:
            cp = cross_product(points[hull[k-2]], points[hull[k-1]], points[idx])
            if cp <= 0:
                k -= 1
            else:
                break
        hull[k] = idx
        k += 1

    return hull[:k-1].copy()

@numba.njit(fastmath=True, cache=True)
def gift_wrapping_small(points):
    """Gift wrapping algorithm for small inputs."""
    n = len(points)
    
    # Find leftmost point
    leftmost = 0
    for i in range(1, n):
        if points[i, 0] < points[leftmost, 0]:
            leftmost = i
    
    hull = []
    p = leftmost
    while True:
        hull.append(p)
        q = (p + 1) % n
        
        for i in range(n):
            if cross_product(points[p], points[i], points[q]) > 0:
                q = i
        
        p = q
        if p == leftmost:
            break
    
    return np.array(hull, dtype=np.int32)

import numpy as np
from scipy.optimize import minimize
from scipy.sparse.linalg import eigs
import scipy.linalg

class Solver:
    def __init__(self):
        self.cache = {}
        
    def solve(self, problem, **kwargs):
        """
        Solves the Perron-Frobenius matrix completion using direct optimization.
        
        Args:
            problem: Dict containing inds, a, and n.
            
        Returns:
            Dict with estimates B, optimal_value.
        """
        inds = np.array(problem["inds"])
        a = np.array(problem["a"])
        n = problem["n"]
        
        # Create mask for observed entries
        mask = np.zeros((n, n), dtype=bool)
        mask[inds[:, 0], inds[:, 1]] = True
        
        # Initialize B with observed values
        B = np.ones((n, n))
        B[inds[:, 0], inds[:, 1]] = a
        
        # Get unobserved indices
        unobs_mask = ~mask
        unobs_indices = np.where(unobs_mask)
        num_unobs = len(unobs_indices[0])
        
        if num_unobs == 0:
            # All entries are observed
            pf_eigenval = self._compute_pf_eigenvalue(B)
            return {
                "B": B.tolist(),
                "optimal_value": pf_eigenval,
            }
        
        # Use log variables for better numerical stability
        # Since product of unobserved entries = 1, sum of logs = 0
        def objective(log_x):
            # Convert log variables back to actual values
            x = np.exp(log_x)
            B_test = B.copy()
            B_test[unobs_indices] = x
            return self._compute_pf_eigenvalue(B_test)
        
        def constraint(log_x):
            # Sum of logs should be 0 (product = 1)
            return np.sum(log_x)
        
        # Initial guess: distribute equally so product = 1
        x0 = np.zeros(num_unobs)  # log(1) = 0
        
        # Optimize
        constraints = {'type': 'eq', 'fun': constraint}
        result = minimize(objective, x0, method='SLSQP', constraints=constraints,
                         options={'ftol': 1e-6, 'maxiter': 1000})
        
        # Construct final matrix
        x_final = np.exp(result.x)
        B[unobs_indices] = x_final
        
        return {
            "B": B.tolist(),
            "optimal_value": result.fun,
        }
    
    def _compute_pf_eigenvalue(self, B):
        """Compute Perron-Frobenius eigenvalue using power iteration for speed."""
        n = B.shape[0]
        
        # Use power iteration for small matrices
        if n <= 10:
            # Power iteration
            v = np.ones(n) / np.sqrt(n)
            for _ in range(50):  # Usually converges quickly
                v_new = B @ v
                v_new = v_new / np.linalg.norm(v_new)
                if np.allclose(v, v_new, atol=1e-8):
                    break
                v = v_new
            return (v @ B @ v) / (v @ v)
        else:
            # For larger matrices, use scipy's eigenvalue solver
            eigenvals = np.linalg.eigvals(B)
            return np.max(np.real(eigenvals))

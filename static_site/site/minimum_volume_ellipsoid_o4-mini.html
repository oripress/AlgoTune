              - "objective_value": float

        n, d = points.shape

        # Augment points with 1 for non-centered ellipsoid
        P = np.hstack((points, np.ones((n, 1), dtype=float)))  # shape (n, d+1)

        # Initialize weights
        u = np.ones(n, dtype=float) / n

        tol = 1e-8
        max_iter = 1000
        # Khachiyan's updates
        for _ in range(max_iter):
            # Compute S = sum_i u_i * P_i P_i^T = P^T diag(u) P
            # Efficient: (P.T * u) @ P
            S = (P.T * u) @ P
            invS = np.linalg.inv(S)
            # Compute metric M_i = P_i^T invS P_i for each point
            v = P @ invS  # shape (n, d+1)
            M = np.einsum('ij,ij->i', v, P)
            # Find maximum violated point
            max_idx = np.argmax(M)
            max_val = M[max_idx]

            # Compute step size
            step = (max_val - (d+1)) / ((d+1) * (max_val - 1))
            if step <= tol:
                break
            # Update weights
            u = (1 - step) * u
            u[max_idx] += step

        # Final S and its inverse
        S = (P.T * u) @ P
        invS = np.linalg.inv(S)

        # Partition invS into blocks
        Ainv = invS[:d, :d]         # shape (d, d)

        c0 = invS[d, d]             # scalar

        Solve the Minimum Volume Covering Ellipsoid (MVEE) using Khachiyan's algorithm.

        Args:
            problem: dict with key "points" for the list of points (n x d).

        Returns:
            dict with:
              - "objective_value": float
              - "ellipsoid": { "X": list of lists, "Y": list }
        """
        points = np.array(problem["points"], dtype=float)
        n, d = points.shape

        # Initial uniform weights
        u = np.full(n, 1.0 / n, dtype=float)
        tol = 1e-8
        max_iter = 1000

        # Khachiyan's algorithm to find weight vector u
        for _ in range(max_iter):
            # current center
            c = points.T @ u  # shape (d,)
            # centered data
            Xc = points - c  # shape (n, d)
            # covariance-like matrix P
            P = (Xc * u[:, None]).T @ Xc  # shape (d, d)
            # inverse of P
            try:
                P_inv = np.linalg.inv(P)
            except np.linalg.LinAlgError:
                P_inv = np.linalg.pinv(P)
            # Mahalanobis distances for each point
            # M_i = (x_i - c)^T P_inv (x_i - c)
            Q = Xc @ P_inv  # shape (n, d)
            M = np.einsum('ij,ij->i', Q, Xc)  # shape (n,)
            j = np.argmax(M)
            maxM = M[j]
            # check convergence
            if maxM <= d + tol:
                break
            # update step
            step = (maxM - d - 1.0) / ((d + 1) * (maxM - 1.0))
            # update weights
            u = (1.0 - step) * u
            u[j] += step

        # final center
        c = points.T @ u
        Xc = points - c
        P = (Xc * u[:, None]).T @ Xc
        try:
            P_inv = np.linalg.inv(P)
        except np.linalg.LinAlgError:
            P_inv = np.linalg.pinv(P)

        # shape matrix A
        A = P_inv / float(d)
        A = (A + A.T) * 0.5

        # compute X such that X^T X = A => X = A^{1/2}
        eigvals, eigvecs = np.linalg.eigh(A)
        eigvals = np.clip(eigvals, 0.0, None)
        sqrt_eigs = np.sqrt(eigvals)
        X_mat = (eigvecs * sqrt_eigs) @ eigvecs.T

        # compute Y so that ||X x + Y|| <= 1 <=> (x - c)^T A (x - c) <= 1
        Y = -X_mat @ c

        # objective value = -log det(X)
        detX = np.linalg.det(X_mat)
        if detX > 0:
            obj = -np.log(detX)
        else:
            obj = float('inf')

        return {
            "objective_value": float(obj),
            "ellipsoid": {"X": X_mat.tolist(), "Y": Y.tolist()},
        }

import numpy as np

class Solver:
    def solve(self, problem, **kwargs):
        points = np.array(problem.get("points", []), dtype=float)
        if points.ndim == 1:
            points = points.reshape(-1, 1)
        n, d = points.shape

        # Handle trivial cases
        if n == 0 or d == 0:
            X_mat = np.eye(d)
            Y = np.zeros(d)
            obj = -np.log(np.linalg.det(X_mat)) if d > 0 else 0.0
            return {"objective_value": float(obj),
                    "ellipsoid": {"X": X_mat.tolist(), "Y": Y.tolist()}}

        # Khachiyan's algorithm initialization
        u = np.full(n, 1.0 / n, dtype=float)
        tol = 1e-8
        max_iter = 1000

        for _ in range(max_iter):
            # Compute center c
            c = points.T.dot(u)  # (d,)

            # Compute scatter P = sum_i u_i * (x_i - c)(x_i - c)^T
            Xc = points - c  # (n, d)
            P = (Xc.T * u).dot(Xc)  # (d, d)

            # Invert P
            try:
                P_inv = np.linalg.inv(P)
            except np.linalg.LinAlgError:
                P_inv = np.linalg.pinv(P)

            # Mahalanobis distances M_i = (x_i - c)^T P_inv (x_i - c)
            Q = Xc.dot(P_inv)          # (n, d)
            M = np.einsum('ij,ij->i', Q, Xc)  # (n,)

            # Find maximum
            j = np.argmax(M)
            Mj = M[j]

            # Check convergence
            if Mj <= d + tol:
                break

            # Update weights
            step = (Mj - d - 1.0) / ((d + 1) * (Mj - 1.0))
            u = (1.0 - step) * u
            u[j] += step

        # Final ellipsoid parameters
        c = points.T.dot(u)
        Xc = points - c
        P = (Xc.T * u).dot(Xc)
        try:
            P_inv = np.linalg.inv(P)
        except np.linalg.LinAlgError:
            P_inv = np.linalg.pinv(P)

        # Ellipsoid shape A = 1/d * P_inv
        A = P_inv / float(d)
        A = (A + A.T) * 0.5

        # Compute X = A^{1/2} via eigen-decomposition
        eigvals, eigvecs = np.linalg.eigh(A)
        eigvals = np.clip(eigvals, 0.0, None)
        sqrt_eigs = np.sqrt(eigvals)
        X_mat = eigvecs.dot(np.diag(sqrt_eigs)).dot(eigvecs.T)

        # Center shift Y such that ||X x + Y|| <= 1 <=> (x - c)^T A (x - c) <= 1
        Y = -X_mat.dot(c)

        # Enforce feasibility (scale down if needed)
        vals = X_mat.dot(points.T) + Y[:, None]

        max_norm = norms.max() if norms.size > 0 else 1.0
        if max_norm > 1.0:
            X_mat /= max_norm
            Y /= max_norm

        # Objective = -log det(X_mat)
        detX = np.linalg.det(X_mat)
        if detX > 0 and np.isfinite(detX):
            obj = -np.log(detX)
        else:
            obj = float('inf')

        return {
            "objective_value": float(obj),
            "ellipsoid": {"X": X_mat.tolist(), "Y": Y.tolist()}
        }

            Y = np.zeros(d)

            }

        # Initial weights
        u = np.full(n, 1.0 / n)
        tol = 1e-7
        max_iter = 1000

        # Khachiyan's algorithm
        for _ in range(max_iter):
            # Compute the center
            c = pts.T.dot(u)  # shape (d,)

            # Compute scatter matrix
            diff = pts - c       # shape (n, d)
            P = (diff.T * u).dot(diff)  # shape (d, d)

            # Invert P
            try:
                invP = np.linalg.inv(P)
            except np.linalg.LinAlgError:
                invP = np.linalg.pinv(P)

            # Mahalanobis distances
            # M_i = diff[i]^T invP diff[i]
            M = np.einsum('ij,jk,ik->i', diff, invP, diff)

            # Find the worst point
            j = np.argmax(M)
            Mj = M[j]

            # Check for convergence
            if Mj <= d * (1 + tol):
                break

            # Update step size
            step = (Mj - d) / ((d + 1) * (Mj - 1.0))

            # Update weights
            u = (1.0 - step) * u
            u[j] += step

        # Final center and shape
        c = pts.T.dot(u)
        diff = pts - c
        P = (diff.T * u).dot(diff)
        try:
            invP = np.linalg.inv(P)
        except np.linalg.LinAlgError:
            invP = np.linalg.pinv(P)

        # Ellipsoid shape matrix A = invP / d
        A = invP / float(d)
        A = (A + A.T) * 0.5

        # Compute symmetric square root of A
        eigvals, eigvecs = np.linalg.eigh(A)
        eigvals = np.clip(eigvals, 0.0, None)
        sqrt_eigs = np.sqrt(eigvals)
        X = eigvecs.dot(np.diag(sqrt_eigs)).dot(eigvecs.T)

        Y = -X.dot(c)

        # Enforce feasibility by scaling down if needed
        vals = X.dot(pts.T) + Y[:, None]

            obj = -np.log(detX)
        else:

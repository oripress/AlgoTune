from typing import Any
import numpy as np
from scipy import linalg
from numba import njit

@njit
def is_triangular_upper(A, n):
    """Check if matrix is upper triangular."""
    for i in range(1, n):
        for j in range(i):
            if abs(A[i, j]) > 1e-10:
                return False
    return True

@njit
def is_triangular_lower(A, n):
    """Check if matrix is lower triangular."""
    for i in range(n):
        for j in range(i + 1, n):
            if abs(A[i, j]) > 1e-10:
                return False
    return True

class Solver:
    def solve(self, problem: dict[str, Any]) -> list[float]:
        """
        Solve the linear system Ax = b using optimized scipy solvers.
        
        Args:
            problem (dict): A dictionary with keys "A" and "b".
        
        Returns:
            list: A list of numbers representing the solution vector x.
        """
        # Convert to numpy arrays with optimal layout
        A = np.asarray(problem["A"], dtype=np.float64, order='F')  # Fortran order for LAPACK
        b = np.asarray(problem["b"], dtype=np.float64)
        
        n = len(b)
        
        # Check for special matrix structures
        if is_triangular_upper(A, n):
            x = linalg.solve_triangular(A, b, lower=False, check_finite=False)
        elif is_triangular_lower(A, n):
            x = linalg.solve_triangular(A, b, lower=True, check_finite=False)
        else:
            # Use LAPACK's gesv directly which is faster than np.linalg.solve
            x = linalg.solve(A, b, check_finite=False, assume_a='gen')
        
        return x.tolist()

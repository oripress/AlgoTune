        from ortools.graph import pywrapgraph

        # Parse inputs
        T = problem["T"]
        N = problem["N"]
        prices = problem["prices"]
        capacities = problem["capacities"]
        probs = problem["probs"]

        # Scale expected revenues to integers
        scale = 10**6
        weight_int = [
            [int(prices[i] * probs[t][i] * scale + 0.5) for i in range(N)]
            for t in range(T)
        ]

        # Maximum weight for cost transformation
        W_max = 0
        for row in weight_int:
            for w in row:
                if w > W_max:
                    W_max = w

        # Build min‚Äêcost max‚Äêflow graph
        # Nodes: source=0, periods=1..T, products=T+1..T+N, sink=T+N+1
        source = 0
        sink = T + N + 1
        smcf = pywrapgraph.SimpleMinCostFlow()

        # Source -> each period (capacity 1, cost 0)
        for t in range(T):
            smcf.AddArcWithCapacityAndUnitCost(source, 1 + t, 1, 0)

        # Period -> product edges (capacity 1, cost = W_max - weight)
        for t in range(T):
            u = 1 + t
            for i in range(N):
                w = weight_int[t][i]
                if w <= 0:
                    continue
                cost = W_max - w
                smcf.AddArcWithCapacityAndUnitCost(u, 1 + T + i, 1, cost)

        # Period -> sink (idle) edges (capacity 1, cost = W_max)
        for t in range(T):
            smcf.AddArcWithCapacityAndUnitCost(1 + t, sink, 1, W_max)

        # Product -> sink edges (capacity = capacities[i], cost 0)
        for i in range(N):

            if cap > 0:

        # Supplies: source supplies T, sink demands T
        smcf.SetNodeSupply(source, T)
        smcf.SetNodeSupply(sink, -T)

        # Solve and extract assignment
        status = smcf.Solve()
        if status != smcf.OPTIMAL:

                i = v - (T + 1)
                offer[t] = i

        return offer

        probs = problem["probs"]

            cap = capacities[i]

                slot_to_product.extend([i] * cap)

from typing import List
import numpy as np
from scipy.optimize import linear_sum_assignment

class Solver:
    def solve(self, problem: dict, **kwargs) -> List[int]:
        """
        Solve Dynamic Assortment Planning via weighted assignment:
          - If total capacity >= T, assign each period greedily to its highest-weight product.
          - Otherwise, build a T√óT cost matrix with product slots and idle slots,
            and use the Hungarian algorithm to find the optimum.
        """
        # Parse inputs
        T = problem["T"]
        N = problem["N"]
        prices = np.array(problem["prices"], dtype=float)
        capacities = np.array(problem["capacities"], dtype=int)
        probs = np.array(problem["probs"], dtype=float)  # shape (T, N)

        # Expected revenue per period-product
        weight = probs * prices  # broadcasts to (T, N)

        total_cap = capacities.sum()
        # If no capacity constraints bind, greedy per-period assignment
        if total_cap >= T:
            # argmax over axis=1 gives best product index (0..N-1) for each period
            # If all weights are zero, argmax returns 0 which is valid since capacity allows
            return list(np.argmax(weight, axis=1))

        # Build product slots by repeating columns according to capacities
        # This yields an array of shape (T, sum(capacities))
        if total_cap > 0:
            prod_block = np.repeat(weight.T, capacities, axis=0).T
            cost_prod = -prod_block
        else:
            cost_prod = np.zeros((T, 0), dtype=float)

        # Idle slots: exactly T - total_cap idle slots needed
        idle_slots = T - total_cap
        cost_idle = np.zeros((T, idle_slots), dtype=float)

        # Full cost matrix of shape (T, T)
        cost = np.hstack((cost_prod, cost_idle))

        # Solve the assignment (minimize cost => maximize revenue)
        row_ind, col_ind = linear_sum_assignment(cost)

        # Build slot-to-product lookup
        prod_slots = np.concatenate([
            np.repeat(np.arange(N, dtype=int), capacities),
            -np.ones(idle_slots, dtype=int)
        ])

        # Construct the offer list
        offer = [-1] * T
        for r, c in zip(row_ind, col_ind):
            p = int(prod_slots[c])
            if p >= 0:
                offer[r] = p
        return offer

@numba.njit
def project_simplex_numba(y):
    """
    Numba-optimized projection onto the probability simplex.
    """
    n = len(y)
    
    # Sort y in descending order
    sorted_y = np.sort(y)[::-1]
    
    # Compute cumulative sums and find threshold
    cumsum = 0.0
    rho = 0
    
    for i in range(n):
        cumsum += sorted_y[i]
        temp = (cumsum - 1.0) / (i + 1)
        if sorted_y[i] > temp:
            rho = i

    cumsum = 0.0

    x = np.empty(n, dtype=np.float64)

        x[i] = max(y[i] - theta, 0.0)

    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:

@numba.njit(fastmath=True, cache=True)
def partition_based_projection(y):
    """
    Optimized projection using partitioning instead of full sort.
    """
    n = len(y)
    
    # Make a copy to avoid modifying input
    v = y.copy()
    
    # Find theta using a partitioning approach
    low = 0
    high = n - 1
    s = 0.0
    
    while low <= high:
        # Partition around pivot
        pivot_idx = (low + high) // 2
        pivot = v[pivot_idx]
        
        # Three-way partition
        i = low
        j = low
        k = high
        
        while j <= k:
            if v[j] < pivot:
                v[i], v[j] = v[j], v[i]
                i += 1
                j += 1
            elif v[j] > pivot:
                v[j], v[k] = v[k], v[j]
                k -= 1
            else:
                j += 1
        
        # Check if we found the right partition
        sum_left = 0.0
        for idx in range(i):
            sum_left += v[idx]
        
        sum_middle = pivot * (j - i)
        sum_right = 0.0
        for idx in range(j, n):
            sum_right += v[idx]
        
        total_sum = sum_left + sum_middle + sum_right
        rho = n - i
        theta_candidate = (total_sum - 1.0) / rho
        
        if theta_candidate >= pivot:
            # theta is in the right partition
            s = sum_left + sum_middle
            low = j
        elif theta_candidate < v[i-1] if i > 0 else True:
            # theta is in the left partition
            high = i - 1
        else:
            # Found the right partition
            break
    
    # Final computation of theta
    sorted_part = np.sort(v[low:])[::-1]
    cumsum = s
    rho = low
    
    for i in range(len(sorted_part)):
        cumsum += sorted_part[i]
        temp = (cumsum - 1.0) / (rho + i + 1)
        if sorted_part[i] > temp:
            theta = temp
        else:
            break
    
    # Project
    result = np.empty(n, dtype=np.float64)
    for i in range(n):
        result[i] = max(y[i] - theta, 0.0)
    
    return result

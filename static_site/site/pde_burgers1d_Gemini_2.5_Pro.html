$\frac{\partial u}{\partial t} + u \frac{\partial u}{\partial x} = \nu \frac{\partial^2 u}{\partial x^2}$

Where:
- u(x,t) is the velocity field
- ν is the viscosity coefficient
- The term u∂u/∂x represents nonlinear advection (wave propagation)
- The term ν∂²u/∂x² represents diffusion (smoothing)

Where ∂u/∂x is computed using upwind differencing based on the sign of u to maintain numerical stability.

The system uses Dirichlet boundary conditions (u=0 at both ends) and an initial condition designed to develop shocks, using either a sine wave or carefully positioned Gaussian bumps with small random perturbations.

# The RHS function computes the spatial derivatives (du/dt). It is JIT-compiled
# and called repeatedly by the adaptive time-stepper.
@numba.jit(nopython=True, fastmath=True, cache=True)
def burgers_rhs_numba(u, nu, dx):
    N = u.shape[0]
    dudt = np.empty(N)
    dx2 = dx * dx

    # Boundary: u[-1]=0
    u_center = u[0]
    u_right = u[1]
    diffusion = nu * (u_right - 2 * u_center + 0.0) / dx2
    advection = u_center * (u_center - 0.0) / dx if u_center >= 0 else u_center * (u_right - u_center) / dx
    dudt[0] = diffusion - advection

    # Interior loop
    for i in range(1, N - 1):
        u_left = u[i-1]
        u_center = u[i]
        u_right = u[i+1]
        diffusion = nu * (u_right - 2 * u_center + u_left) / dx2
        advection = u_center * (u_center - u_left) / dx if u_center >= 0 else u_center * (u_right - u_center) / dx
        dudt[i] = diffusion - advection

    # Boundary: u[N]=0
    u_left = u[N-2]
    u_center = u[N-1]
    diffusion = nu * (0.0 - 2 * u_center + u_left) / dx2
    advection = u_center * (u_center - u_left) / dx if u_center >= 0 else u_center * (0.0 - u_center) / dx
    dudt[N-1] = diffusion - advection
    
    return dudt

def rkf45_adaptive_stepper(y0, t0, t1, nu, dx, rtol, atol):
    # RKF45 Butcher Tableau coefficients
    a21 = 1/4
    a31, a32 = 3/32, 9/32
    a41, a42, a43 = 1932/2197, -7200/2197, 7296/2197
    a51, a52, a53, a54 = 439/216, -8, 3680/513, -845/4104
    a61, a62, a63, a64, a65 = -8/27, 2, -3544/2565, 1859/4104, -11/40

    b4_1, b4_3, b4_4, b4_5 = 25/216, 1408/2565, 2197/4104, -1/5

    e1, e3, e4, e5, e6 = 1/360, -128/4275, -2197/75240, 1/50, 2/55

    t = t0
    u = y0.copy()
    dt = (t1 - t0) / 100.0  # Initial step size guess
    
    safety_fac, min_scale, max_scale = 0.9, 0.2, 5.0

    while t < t1:
        if t + dt > t1:
            dt = t1 - t
        if dt == 0.0:
            break

        # Calculate the six stages (k1 to k6)
        k1 = burgers_rhs_numba(u, nu, dx)
        k2 = burgers_rhs_numba(u + dt * a21 * k1, nu, dx)
        k3 = burgers_rhs_numba(u + dt * (a31 * k1 + a32 * k2), nu, dx)
        k4 = burgers_rhs_numba(u + dt * (a41 * k1 + a42 * k2 + a43 * k3), nu, dx)
        k5 = burgers_rhs_numba(u + dt * (a51 * k1 + a52 * k2 + a53 * k3 + a54 * k4), nu, dx)

        # Calculate error and check against tolerance
        error_estimate = dt * (e1 * k1 + e3 * k3 + e4 * k4 + e5 * k5 + e6 * k6)
        error_norm = np.max(np.abs(error_estimate))
        tolerance = atol + rtol * np.max(np.abs(u))

        if error_norm <= tolerance:

            t += dt

                scale = max_scale
            else:

            dt *= scale
        else:
            # Step rejected: decrease dt
            scale = max(min_scale, safety_fac * (tolerance / error_norm)**0.2)
            dt *= scale

# The entire RHS function is JIT-compiled by JAX. It's a "pure function"
# that uses jax.numpy for all array operations. The function signature is
# tailored for jax.experimental.ode.odeint.
@jax.jit
def burgers_rhs_jax(u, t, nu, dx):
    """
    Computes the right-hand-side of the Burgers' equation for JAX's odeint.
    This function is vectorized and uses jax.numpy.
    """
    # Pad array for boundary conditions (u=0 at boundaries)
    u_padded = jnp.pad(u, 1, mode="constant")

    # Diffusion term (central difference)
    diffusion_term = nu * (u_padded[2:] - 2 * u_padded[1:-1] + u_padded[:-2]) / (dx**2)

    # Advection term (upwind scheme)
    u_centered = u_padded[1:-1]
    du_dx_forward = (u_padded[2:] - u_padded[1:-1]) / dx
    du_dx_backward = (u_padded[1:-1] - u_padded[:-2]) / dx

    # jnp.where is a JIT-compatible way to express conditional logic
    advection_term = jnp.where(
        u_centered >= 0,
        u_centered * du_dx_backward,
        u_centered * du_dx_forward,
    )

    # Combine terms
    return -advection_term + diffusion_term

# The RHS function computes the spatial derivatives (du/dt). It is JIT-compiled
# for maximum performance. The logic is a direct, looped translation of the
# baseline's vectorized operations to be optimal for Numba.
# `fastmath` is disabled to prioritize numerical precision over aggressive optimization.
@numba.jit(nopython=True, cache=True)
def burgers_rhs_numba(t, u, nu, dx):
    N = u.shape[0]
    dudt = np.empty(N)
    dx2 = dx * dx

    # Handle the first grid point (i=0) with boundary condition u[-1]=0
    u_center = u[0]
    u_right = u[1]
    diffusion = nu * (u_right - 2 * u_center + 0.0) / dx2
    # Upwind advection
    if u_center >= 0:
        advection = u_center * (u_center - 0.0) / dx
    else:
        advection = u_center * (u_right - u_center) / dx
    dudt[0] = diffusion - advection

    # Loop over interior grid points for high performance in Numba
    for i in range(1, N - 1):
        u_left = u[i-1]
        u_center = u[i]
        u_right = u[i+1]
        diffusion = nu * (u_right - 2 * u_center + u_left) / dx2
        if u_center >= 0:
            advection = u_center * (u_center - u_left) / dx
        else:
            advection = u_center * (u_right - u_center) / dx
        dudt[i] = diffusion - advection

    # Handle the last grid point (i=N-1) with boundary condition u[N]=0
    u_left = u[N-2]
    u_center = u[N-1]
    diffusion = nu * (0.0 - 2 * u_center + u_left) / dx2
    if u_center >= 0:
        advection = u_center * (u_center - u_left) / dx
    else:
        advection = u_center * (0.0 - u_center) / dx
    dudt[N-1] = diffusion - advection
    
    return dudt

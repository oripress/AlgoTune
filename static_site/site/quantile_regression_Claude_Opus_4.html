import numpy as np
from scipy.optimize import linprog
from typing import Any

class Solver:
    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:
        """
        Fit quantile regression using scipy's linprog for better performance.
        
        :param problem: dict with 'X', 'y', 'quantile', 'fit_intercept'
        :return: dict with 'coef', 'intercept', 'predictions'
        """
        X = np.array(problem["X"], dtype=float)
        y = np.array(problem["y"], dtype=float)
        quantile = problem["quantile"]
        fit_intercept = problem["fit_intercept"]
        
        n_samples, n_features = X.shape
        
        # Variables: [beta, intercept (if needed), u, v]
        # where u and v are slack variables for positive and negative residuals
        if fit_intercept:
            n_vars = n_features + 1 + 2 * n_samples
            # Construct extended X matrix with intercept column
            X_ext = np.column_stack([X, np.ones(n_samples)])
            n_coef = n_features + 1
        else:
            n_vars = n_features + 2 * n_samples
            X_ext = X
            n_coef = n_features
        
        # Objective: minimize sum(quantile * u + (1-quantile) * v)
        c = np.zeros(n_vars)
        c[n_coef:n_coef + n_samples] = quantile  # u coefficients
        c[n_coef + n_samples:] = 1 - quantile    # v coefficients
        
        # Equality constraints: y - X*beta - intercept = u - v
        # Rearranged: X*beta + intercept + u - v = y
        A_eq = np.zeros((n_samples, n_vars))
        A_eq[:, :n_coef] = X_ext
        A_eq[:, n_coef:n_coef + n_samples] = np.eye(n_samples)  # u
        A_eq[:, n_coef + n_samples:] = -np.eye(n_samples)       # v
        b_eq = y
        
        # Bounds: beta unbounded, u >= 0, v >= 0
        bounds = [(None, None)] * n_coef + [(0, None)] * (2 * n_samples)
        
        # Solve
        result = linprog(c, A_eq=A_eq, b_eq=b_eq, bounds=bounds, method='highs')
        
        # Extract results
        if fit_intercept:
            coef = result.x[:n_features].tolist()
            intercept_val = [result.x[n_features]]
        else:
            coef = result.x[:n_features].tolist()
            intercept_val = [0.0]
        
        # Compute predictions
        predictions = (X @ np.array(coef) + intercept_val[0]).tolist()
        
        return {"coef": coef, "intercept": intercept_val, "predictions": predictions}

import numpy as np
import cvxpy as cp
from typing import Any

class Solver:
    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:
        """
        Fit quantile regression using cvxpy for faster performance.
        
        :param problem: dict with 'X', 'y', 'quantile', 'fit_intercept'
        :return: dict with 'coef', 'intercept', 'predictions'
        """
        X = np.array(problem["X"], dtype=float)
        y = np.array(problem["y"], dtype=float)
        quantile = problem["quantile"]
        fit_intercept = problem["fit_intercept"]
        
        n_samples, n_features = X.shape
        
        # Decision variables
        beta = cp.Variable(n_features)
        if fit_intercept:
            intercept = cp.Variable()
        else:
            intercept = 0.0
        
        # Residuals
        residuals = y - X @ beta - intercept
        
        # Quantile loss: sum of max(quantile * r, (quantile - 1) * r)
        loss = cp.sum(cp.maximum(quantile * residuals, (quantile - 1) * residuals))
        
        # Solve the problem
        prob = cp.Problem(cp.Minimize(loss))
        prob.solve(solver=cp.ECOS)
        
        # Extract results
        coef = beta.value.tolist()
        if fit_intercept:
            intercept_val = [float(intercept.value)]
        else:
            intercept_val = [0.0]
        
        # Compute predictions
        predictions = (X @ beta.value + (intercept.value if fit_intercept else 0.0)).tolist()
        
        return {"coef": coef, "intercept": intercept_val, "predictions": predictions}

import numpy as np
from pulp import LpProblem, LpVariable, LpMinimize, lpSum, LpStatus, value
from typing import Any

class Solver:
    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:
        """
        Fit quantile regression using PuLP linear programming solver.
        
        :param problem: dict with 'X', 'y', 'quantile', 'fit_intercept'
        :return: dict with 'coef', 'intercept', 'predictions'
        """
        X = np.array(problem["X"], dtype=float)
        y = np.array(problem["y"], dtype=float)
        quantile = problem["quantile"]
        fit_intercept = problem["fit_intercept"]
        
        n_samples, n_features = X.shape
        
        # Create the LP problem
        prob = LpProblem("QuantileRegression", LpMinimize)
        
        # Decision variables
        beta = [LpVariable(f"beta_{i}", cat='Continuous') for i in range(n_features)]
        if fit_intercept:
            intercept = LpVariable("intercept", cat='Continuous')
        else:
            intercept = 0.0
            
        # Slack variables for positive and negative residuals
        u = [LpVariable(f"u_{i}", lowBound=0) for i in range(n_samples)]
        v = [LpVariable(f"v_{i}", lowBound=0) for i in range(n_samples)]
        
        # Objective function
        prob += lpSum([quantile * u[i] + (1 - quantile) * v[i] for i in range(n_samples)])
        
        # Constraints: y_i - X_i * beta - intercept = u_i - v_i
        for i in range(n_samples):
            constraint = y[i] - lpSum([X[i, j] * beta[j] for j in range(n_features)]) - intercept
            prob += constraint == u[i] - v[i]
        
        # Solve
        prob.solve(solver=None)  # Use default solver
        
        # Extract results
        coef = [value(beta[i]) for i in range(n_features)]
        if fit_intercept:
            intercept_val = [value(intercept)]
        else:
            intercept_val = [0.0]
        
        # Compute predictions
        predictions = []
        for i in range(n_samples):
            pred = sum(X[i, j] * coef[j] for j in range(n_features))
            if fit_intercept:
                pred += intercept_val[0]
            predictions.append(pred)
        
        return {"coef": coef, "intercept": intercept_val, "predictions": predictions}

import numpy as np
from scipy.optimize import linprog
from typing import Any

class Solver:
    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:
        """
        Fit quantile regression using scipy's linprog for better performance.
        
        :param problem: dict with 'X', 'y', 'quantile', 'fit_intercept'
        :return: dict with 'coef', 'intercept', 'predictions'
        """
        X = np.array(problem["X"], dtype=float)
        y = np.array(problem["y"], dtype=float)
        quantile = problem["quantile"]
        fit_intercept = problem["fit_intercept"]
        
        n_samples, n_features = X.shape
        
        # Variables: [beta, intercept (if needed), u, v]
        # where u and v are slack variables for positive and negative residuals
        if fit_intercept:
            n_vars = n_features + 1 + 2 * n_samples
            # Construct extended X matrix with intercept column
            X_ext = np.column_stack([X, np.ones(n_samples)])
            n_coef = n_features + 1
        else:
            n_vars = n_features + 2 * n_samples
            X_ext = X
            n_coef = n_features
        
        # Objective: minimize sum(quantile * u + (1-quantile) * v)
        c = np.zeros(n_vars)
        c[n_coef:n_coef + n_samples] = quantile  # u coefficients
        c[n_coef + n_samples:] = 1 - quantile    # v coefficients
        
        # Equality constraints: y - X*beta - intercept = u - v
        # Rearranged: X*beta + intercept + u - v = y
        A_eq = np.zeros((n_samples, n_vars))
        A_eq[:, :n_coef] = X_ext
        A_eq[:, n_coef:n_coef + n_samples] = np.eye(n_samples)  # u
        A_eq[:, n_coef + n_samples:] = -np.eye(n_samples)       # v
        b_eq = y
        
        # Bounds: beta unbounded, u >= 0, v >= 0
        bounds = [(None, None)] * n_coef + [(0, None)] * (2 * n_samples)
        
        # Solve
        result = linprog(c, A_eq=A_eq, b_eq=b_eq, bounds=bounds, method='highs')
        
        # Extract results
        if fit_intercept:
            coef = result.x[:n_features].tolist()
            intercept_val = [result.x[n_features]]
        else:
            coef = result.x[:n_features].tolist()
            intercept_val = [0.0]
        
        # Compute predictions
        predictions = (X @ np.array(coef) + intercept_val[0]).tolist()
        
        return {"coef": coef, "intercept": intercept_val, "predictions": predictions}

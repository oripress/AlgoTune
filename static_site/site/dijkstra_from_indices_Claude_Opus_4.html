import numba
from numba import njit
from typing import Any

@njit
def dijkstra_from_csr(n, data, indices, indptr, source):
    """Direct Dijkstra on CSR format using numba JIT compilation."""
    INF = 1e20
    dist = np.full(n, INF, dtype=np.float64)
    dist[source] = 0.0
    visited = np.zeros(n, dtype=np.bool_)
    
    # Simple priority queue using arrays
    heap_dist = np.full(n, INF, dtype=np.float64)
    heap_node = np.arange(n, dtype=np.int32)
    heap_size = 0
    
    # Add source to heap
    heap_dist[0] = 0.0
    heap_node[0] = source
    heap_size = 1
    
    while heap_size > 0:
        # Find minimum in heap (linear search for simplicity)
        min_idx = 0
        min_dist = heap_dist[0]
        for i in range(1, heap_size):
            if heap_dist[i] < min_dist:
                min_dist = heap_dist[i]
                min_idx = i
        
        # Extract minimum
        u = heap_node[min_idx]
        d = heap_dist[min_idx]
        
        # Remove from heap by swapping with last element
        heap_size -= 1
        if min_idx < heap_size:
            heap_dist[min_idx] = heap_dist[heap_size]
            heap_node[min_idx] = heap_node[heap_size]
        
        if visited[u]:
            continue
            
        visited[u] = True
        
        # Process neighbors from CSR
        start = indptr[u]
        end = indptr[u + 1]
        
        for idx in range(start, end):
            v = indices[idx]
            w = data[idx]
            
            if not visited[v]:
                alt = d + w
                if alt < dist[v]:
                    dist[v] = alt
                    
                    # Add to heap
                    heap_dist[heap_size] = alt
                    heap_node[heap_size] = v
                    heap_size += 1
    
    return dist

import numpy as np
import numba
from numba import njit, prange
from numba.typed import List
import heapq
from typing import Any


@njit
def dijkstra_single_source(n, edges, weights, source):
    """
    Optimized single-source Dijkstra using numba JIT compilation.
    Returns distances from source to all nodes.
    """
    INF = 1e20
    dist = np.full(n, INF, dtype=np.float64)
    dist[source] = 0.0
    visited = np.zeros(n, dtype=np.bool_)
    
    # Priority queue: (distance, node)
    heap = [(0.0, source)]
    
    while heap:
        d, u = heapq.heappop(heap)
        
        if visited[u]:
            continue
            
        visited[u] = True
        
        # Process neighbors
        start = edges[u]
        end = edges[u + 1] if u + 1 < len(edges) else len(weights)
        
        for i in range(start, end):
            v = i - start  # neighbor index
            if v < n and not visited[v]:
                alt = d + weights[i]
                if alt < dist[v]:
                    dist[v] = alt
                    heapq.heappush(heap, (alt, v))
    
    return dist


@njit
def build_adjacency_from_csr(n, data, indices, indptr):
    """Convert CSR format to adjacency lists for faster access."""
    adj_lists = List()
    adj_weights = List()
    
    for i in range(n):
        neighbors = List.empty_list(numba.int64)
        weights = List.empty_list(numba.float64)
        
        start = indptr[i]
        end = indptr[i + 1]
        
        for j in range(start, end):
            neighbors.append(indices[j])
            weights.append(data[j])
        
        adj_lists.append(neighbors)
        adj_weights.append(weights)
    
    return adj_lists, adj_weights

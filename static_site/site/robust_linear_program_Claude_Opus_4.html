Robust Linear Program Problem


This task involves solving the Robust Linear Program (LP), whose goal is to find the solution to the given LP which is robust to the LP parameter uncertainty, which is ellipsoidal uncertainty.

The robust LP (with ellipsoidal uncertainty) can be formulated into the following optimization problem:

    minimize    c^T * x
    subject to  a_i^T * x <= b_i    for all a_i in E_i, all i in I

with variables:
- x is an n-dimensional vector,

and with parameters to be given:
- c is an n-dimensional vector in LP objective,
- a_i is an n-dimensional vector and b_i is a scalar in LP constraint, defined for all i in I,
- E_i is an ellipsoid, which is defined as a set of vectors y = P_i * v + q_i with vector v such that |v| <= 1. Here, P_i is some symmetric positive (semi-)definite matrix and q_i is a vector defined for all i in I,
- I is a set of indices i.

In order to solve the problem above, we consider an alternative problem as below:

    minimize    c^T * x
    subject to  q_i^T * x + |P_i^T * x| <= b_i  for all i in I

Note that |v| refers to the euclidean norm of vector v, therefore this problem becomes a second-order cone program (SOCP).





Input: A dictionary of keys:
- "c": list of n floats, which defines the linear objective of LP,
- "b": list of m floats, which defines the right-hand side scalars of linear constraint of LP,
- "P": list of m matrices, where each matrix is a list of n lists consisting of n floats.
- "q": list of m lists, where each list is a list of n floats.
Note that the i-th element of P and q are P_i and q_i in the problem definition above, respectively.

from typing import Any
import numpy as np
import cvxpy as cp

class Solver:
    def solve(self, problem: dict[str, np.ndarray], **kwargs) -> dict[str, Any]:
        """
        Solves a given robust LP using CVXPY with ECOS solver.
        """
        c = np.array(problem["c"])
        b = np.array(problem["b"])
        P = np.array(problem["P"])
        q = np.array(problem["q"])
        m = len(P)
        n = len(c)

        x = cp.Variable(n)

        constraints = []
        for i in range(m):
            constraints.append(cp.SOC(b[i] - q[i].T @ x, P[i].T @ x))

        cvx_problem = cp.Problem(cp.Minimize(c.T @ x), constraints)

        try:
            # Try ECOS which is typically fast for SOCP
            cvx_problem.solve(solver=cp.ECOS, verbose=False)

            # Check if a solution was found
            if cvx_problem.status not in ["optimal", "optimal_inaccurate"]:
                return {"objective_value": float("inf"), "x": np.array([np.nan] * n)}

            return {"objective_value": cvx_problem.value, "x": x.value}

        except Exception:
            return {"objective_value": float("inf"), "x": np.array([np.nan] * n)}

from typing import Any
import numpy as np
import ecos

class Solver:
    def solve(self, problem: dict[str, np.ndarray], **kwargs) -> dict[str, Any]:
        """
        Solves a given robust LP using ECOS directly.
        """
        c = np.array(problem["c"], dtype=np.float64)
        b = np.array(problem["b"], dtype=np.float64)
        P = np.array(problem["P"], dtype=np.float64)
        q = np.array(problem["q"], dtype=np.float64)
        m = len(P)
        n = len(c)
        
        # ECOS solves: min c'x s.t. Ax=b, Gx<=h, x in K
        # where K is a product of cones
        
        # For SOCP constraint: ||P_i'x||_2 <= b_i - q_i'x
        # We rewrite as: ||(P_i'x, 0)||_2 <= (b_i - q_i'x)
        # Which is: (b_i - q_i'x, P_i'x) in SOC
        
        # Build matrices for ECOS
        # We need to stack constraints for each cone
        G_rows = []
        h_rows = []
        
        for i in range(m):
            # First row: -q_i'x + s_i = b_i  =>  [-q_i', -1] * [x; s_i] <= -b_i
            G_rows.append(np.concatenate([q[i], [0]]))
            h_rows.append(-b[i])
            
            # Next n rows: -P_i'x = -y_i  =>  [-P_i', 0] * [x; s_i] <= 0
            for j in range(n):
                row = np.zeros(n + 1)
                row[:n] = -P[i][j, :]
                G_rows.append(row)
                h_rows.append(0.0)
        
        # Stack all rows
        G = -np.vstack(G_rows)  # Negative because ECOS uses Gx + s = h with s in K
        h = np.array(h_rows)
        
        # Objective: minimize c'x (we add a 0 for the slack variable)
        c_ecos = np.concatenate([c, [0]])
        
        # Cone dimensions: each constraint gives a cone of size n+1
        q_dims = [n + 1] * m
        
        # No equality constraints
        A = None
        b_eq = None
        
        # Solve with ECOS
        solution = ecos.solve(c_ecos, G, h, {'q': q_dims}, A, b_eq, verbose=False)
        
        if solution['info']['exitFlag'] == 0:  # Optimal
            x_opt = solution['x'][:n]
            obj_val = c.T @ x_opt
            return {"objective_value": obj_val, "x": x_opt}
        else:
            return {"objective_value": float("inf"), "x": np.array([np.nan] * n)}

from typing import Any
import numpy as np
import scipy.sparse as sp
import ecos

class Solver:
    def solve(self, problem: dict[str, np.ndarray], **kwargs) -> dict[str, Any]:
        """
        Solves a given robust LP using ECOS directly.
        """
        c = np.array(problem["c"], dtype=np.float64)
        b = np.array(problem["b"], dtype=np.float64)
        P = np.array(problem["P"], dtype=np.float64)
        q = np.array(problem["q"], dtype=np.float64)
        m = len(P)
        n = len(c)
        
        # ECOS solves: min c'x s.t. Ax=b, Gx<=h, x in K
        # For SOCP constraint: ||P_i'x||_2 <= b_i - q_i'x
        # We need (b_i - q_i'x, P_i'x) in second-order cone
        
        # Build G matrix and h vector
        G_rows = []
        h_vals = []
        
        for i in range(m):
            # First component: b_i - q_i'x >= 0
            G_rows.append(-q[i])
            h_vals.append(b[i])
            
            # Remaining components: P_i'x
            for j in range(n):
                G_rows.append(-P[i][j, :])
                h_vals.append(0.0)
        
        # Convert to sparse matrix
        G = sp.csc_matrix(-np.vstack(G_rows))
        h = np.array(h_vals)
        
        # Cone dimensions
        q_dims = [n + 1] * m
        
        # Solve with ECOS
        solution = ecos.solve(c, G, h, {'q': q_dims}, verbose=False)
        
        if solution['info']['exitFlag'] == 0:  # Optimal
            x_opt = solution['x']
            obj_val = c.T @ x_opt
            return {"objective_value": obj_val, "x": x_opt}
        else:
            return {"objective_value": float("inf"), "x": np.array([np.nan] * n)}

from typing import Any
import numpy as np
import cvxpy as cp

class Solver:
    def __init__(self):
        # Pre-set solver options for speed
        self.solver_opts = {
            'normalize': False,
            'eps': 1e-5,
            'max_iters': 5000,
            'acceleration_lookback': 10,
            'verbose': False
        }
    
    def solve(self, problem: dict[str, np.ndarray], **kwargs) -> dict[str, Any]:
        """
        Solves a given robust LP using CVXPY with SCS solver.
        """
        c = np.array(problem["c"], dtype=np.float64)
        b = np.array(problem["b"], dtype=np.float64)
        P = np.array(problem["P"], dtype=np.float64)
        q = np.array(problem["q"], dtype=np.float64)
        m = len(P)
        n = len(c)

        x = cp.Variable(n)

        constraints = []
        for i in range(m):
            constraints.append(cp.SOC(b[i] - q[i].T @ x, P[i].T @ x))

        cvx_problem = cp.Problem(cp.Minimize(c.T @ x), constraints)

        try:
            # Use SCS which is typically faster for SOCP
            cvx_problem.solve(solver=cp.SCS, **self.solver_opts)

            # Check if a solution was found
            if cvx_problem.status not in ["optimal", "optimal_inaccurate"]:
                return {"objective_value": float("inf"), "x": np.array([np.nan] * n)}

            return {"objective_value": float(cvx_problem.value), "x": np.array(x.value)}

        except Exception:
            return {"objective_value": float("inf"), "x": np.array([np.nan] * n)}

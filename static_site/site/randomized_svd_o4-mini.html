    def solve(self, problem: Dict[str, Any], **kwargs) -> Dict[str, Any]:

        # Oversampling parameter

        # Sample column space
        Y = A @ Omega

import numpy as np

# global RNG to avoid reinitialization
_rng = np.random.default_rng(42)

class Solver:
    def solve(self, problem: Dict[str, Any], **kwargs) -> Dict[str, Any]:

                "S": np.zeros((0,), dtype=A.dtype),
                "V": np.zeros((m, 0), dtype=A.dtype),

        Omega = _rng.standard_normal((m, k)).astype(A.dtype)

        # sample range
        Y = A.dot(Omega)

        G = Y.T.dot(Y)

        Ub, s, Vt = np.linalg.svd(B, full_matrices=False)

        # cast to float32
        Ub = Ub.astype(A.dtype)
        s = s.astype(A.dtype)
        Vt = Vt.astype(A.dtype)

        U = Q.dot(Ub[:, :k])
        S = s[:k]
        V = Vt[:k, :].T

    def solve(self, problem: Dict[str, Any], **kwargs) -> Dict[str, Any]:

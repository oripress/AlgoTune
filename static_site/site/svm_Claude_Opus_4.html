from typing import Any, Dict
import cvxpy as cp
import numpy as np

class Solver:
    def solve(self, problem: Dict[str, Any]) -> Dict[str, Any]:
        """
        Solves the SVM using the dual formulation
        """
        X = np.array(problem["X"])
        y = np.array(problem["y"])
        C = float(problem["C"])

        n, p = X.shape
        
        # Compute the kernel matrix (linear kernel: K_ij = x_i^T x_j)
        K = X @ X.T
        
        # Dual variables
        alpha = cp.Variable(n)
        
        # Construct the quadratic form for the dual
        # We need y_i * y_j * K_ij, which is (y * y^T) ‚äô K (element-wise product)
        y_outer = np.outer(y, y)
        Q = y_outer * K
        
        # Dual objective: maximize sum(alpha) - 0.5 * alpha^T @ Q @ alpha
        objective = cp.Maximize(cp.sum(alpha) - 0.5 * cp.quad_form(alpha, Q))
        
        # Constraints
        constraints = [
            alpha >= 0,
            alpha <= C,
            alpha @ y == 0  # sum(alpha_i * y_i) = 0
        ]
        
        prob = cp.Problem(objective, constraints)
        
        try:
            dual_value = prob.solve()
        except:
            return None
            
        if prob.status not in (cp.OPTIMAL, cp.OPTIMAL_INACCURATE):
            return None
            
        if alpha.value is None:
            return None
            
        # Recover primal variables
        # beta = sum(alpha_i * y_i * x_i)
        alpha_vals = alpha.value
        beta = X.T @ (alpha_vals * y)
        
        # Find support vectors (alpha > tolerance)
        tol = 1e-6
        sv_mask = (alpha_vals > tol) & (alpha_vals < C - tol)
        
        if not np.any(sv_mask):
            # If no proper support vectors, use all with alpha > 0
            sv_mask = alpha_vals > tol
            
        # Compute beta0 from support vectors
        sv_indices = np.where(sv_mask)[0]
        if len(sv_indices) > 0:
            beta0_vals = y[sv_indices] - X[sv_indices] @ beta
            beta0 = np.mean(beta0_vals)
        else:
            beta0 = 0.0
            
        # Compute predictions and misclassification
        pred = X @ beta + beta0
        missclass = np.mean((pred * y) < 0)
        
        # Compute primal objective value
        xi = np.maximum(0, 1 - y * pred)
        primal_value = 0.5 * np.dot(beta, beta) + C * np.sum(xi)
        
        return {
            "beta0": float(beta0),
            "beta": beta.tolist(),
            "optimal_value": float(primal_value),
            "missclass_error": float(missclass),
        }

class Solver:
    def solve(self, problem: Dict[str, Any]) -> Dict[str, Any]:
        """

        X = np.array(problem["X"])

        C = float(problem["C"])

        p, n = X.shape[1], X.shape[0]

        beta = cp.Variable((p, 1))
        beta0 = cp.Variable()
        xi = cp.Variable((n, 1))

        objective = cp.Minimize(0.5 * cp.sum_squares(beta) + C * cp.sum(xi))
        constraints = [

        solvers = [cp.ECOS, cp.SCS, cp.OSQP]

from typing import Any, Dict
import numpy as np
from scipy.optimize import minimize
import scipy.sparse as sp

class Solver:
    def solve(self, problem: Dict[str, Any]) -> Dict[str, Any]:
        """
        Solves the SVM using scipy optimization
        """
        X = np.array(problem["X"])
        y = np.array(problem["y"])
        C = float(problem["C"])
        
        n, p = X.shape
        
        # Solve the dual problem
        # max sum(alpha) - 0.5 * sum_i sum_j alpha_i alpha_j y_i y_j x_i^T x_j
        # s.t. 0 <= alpha_i <= C, sum(alpha_i y_i) = 0
        
        # Precompute kernel matrix
        K = X @ X.T
        y_outer = np.outer(y, y)
        Q = y_outer * K
        
        def dual_objective(alpha):
            # We minimize negative of dual (to maximize)
            return -np.sum(alpha) + 0.5 * alpha @ Q @ alpha
        
        def dual_gradient(alpha):
            return -np.ones(n) + Q @ alpha
        
        # Constraint: sum(alpha_i * y_i) = 0
        constraints = {'type': 'eq', 'fun': lambda a: np.dot(a, y)}
        
        # Bounds: 0 <= alpha_i <= C
        bounds = [(0, C) for _ in range(n)]
        
        # Initial point
        alpha0 = np.ones(n) * C / 2
        # Adjust to satisfy constraint
        alpha0 -= (np.dot(alpha0, y) / np.dot(y, y)) * y
        alpha0 = np.clip(alpha0, 0, C)
        
        # Solve
        result = minimize(dual_objective, alpha0, 
                         method='L-BFGS-B',
                         jac=dual_gradient,
                         bounds=bounds,
                         constraints=constraints,
                         options={'ftol': 1e-8, 'gtol': 1e-8})
        
        alpha = result.x
        
        # Recover primal variables
        beta = X.T @ (alpha * y)
        
        # Find support vectors
        sv_mask = (alpha > 1e-6) & (alpha < C - 1e-6)
        sv_indices = np.where(sv_mask)[0]
        
        if len(sv_indices) > 0:
            beta0_vals = y[sv_indices] - X[sv_indices] @ beta
            beta0 = np.mean(beta0_vals)
        else:
            # Use all points with alpha > 0
            mask = alpha > 1e-6
            if np.any(mask):
                indices = np.where(mask)[0]
                beta0_vals = y[indices] - X[indices] @ beta
                beta0 = np.mean(beta0_vals)
            else:
                beta0 = 0.0
        
        # Compute predictions
        pred = X @ beta + beta0
        missclass = np.mean((pred * y) < 0)
        
        # Compute primal objective
        xi = np.maximum(0, 1 - y * pred)
        primal_value = 0.5 * np.dot(beta, beta) + C * np.sum(xi)
        
        return {
            "beta0": float(beta0),
            "beta": beta.tolist(),
            "optimal_value": float(primal_value),
            "missclass_error": float(missclass),
        }

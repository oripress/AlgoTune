import numpy as np
from numpy.typing import NDArray

class Solver:
    def solve(self, problem: dict) -> NDArray:
        """
        Compute the cumulative integral along the last axis of the multi-dimensional array using Simpson's rule.
        """
        y2 = problem["y2"]
        dx = problem["dx"]
        
        # Get the shape of the input array
        shape = y2.shape
        n = shape[-1]  # Number of points along the last axis
        
        # Initialize the result array
        result = np.zeros_like(y2)
        
        # For the first point, the integral is 0
        result[..., 0] = 0.0
        
        # For the second point, use trapezoidal rule
        if n > 1:
            result[..., 1] = dx * (y2[..., 0] + y2[..., 1]) / 2.0
        
        # For subsequent points, use Simpson's rule
        for i in range(2, n):
            if i % 2 == 0:  # Even index
                # Use Simpson's rule for the last two intervals
                result[..., i] = result[..., i-2] + dx/3.0 * (y2[..., i-2] + 4*y2[..., i-1] + y2[..., i])
            else:  # Odd index
                # Use trapezoidal rule for the last interval
                result[..., i] = result[..., i-1] + dx * (y2[..., i-1] + y2[..., i]) / 2.0
        
        return result

from numpy.typing import NDArray

class Solver:
    def solve(self, problem: dict) -> NDArray:
        """
        Compute the cumulative integral along the last axis of the multi-dimensional array using Simpson's rule.
        """
        y2 = problem["y2"]
        dx = problem["dx"]
        
        # Get the shape of the input array
        shape = y2.shape
        n = shape[-1]  # Number of points along the last axis
        
        # If n is even, scipy's cumulative_simpson returns n-1 points
        if n % 2 == 0:
            # Initialize the result array with n-1 points
            result_shape = shape[:-1] + (n-1,)
            result = np.zeros(result_shape, dtype=y2.dtype)
            
            # Apply Simpson's rule for cumulative integration
            # For even n, we can only compute up to n-1 points
            for i in range(1, n-1):
                if i % 2 == 0:
                    # For even indices, use Simpson's rule
                    result[..., i] = result[..., i-2] + dx/3 * (y2[..., i-2] + 4*y2[..., i-1] + y2[..., i])
                else:
                    # For odd indices, use trapezoidal rule from previous point
                    result[..., i] = result[..., i-1] + dx/2 * (y2[..., i-1] + y2[..., i])

            result = np.zeros_like(y2)

import numpy as np
from numpy.typing import NDArray
import numba

@numba.njit(parallel=True, fastmath=True)
def cumulative_simpson_numba(y, dx):
    """
    Optimized cumulative Simpson's rule using numba.
    """
    shape = y.shape
    n = shape[-1]
    
    # Handle even/odd cases like scipy
    if n % 2 == 0:
        # For even n, result has n-1 points
        result = np.zeros(shape[:-1] + (n-1,), dtype=y.dtype)
        
        # Vectorized computation for all dimensions at once
        for k in numba.prange(shape[0]):
            for j in range(shape[1]):
                # First point is always 0
                # Compute cumulative integral using Simpson's rule
                for i in range(1, n-1):
                    if i == 1:
                        # Trapezoidal for first interval
                        result[k, j, i] = dx * (y[k, j, 0] + y[k, j, 1]) / 2
                    elif i % 2 == 0:
                        # Simpson's rule for pairs
                        h = dx
                        result[k, j, i] = result[k, j, i-2] + h/3 * (y[k, j, i-2] + 4*y[k, j, i-1] + y[k, j, i])
                    else:
                        # Trapezoidal for odd indices
                        result[k, j, i] = result[k, j, i-1] + dx/2 * (y[k, j, i-1] + y[k, j, i])
    else:
        # For odd n, result has n points
        result = np.zeros_like(y)
        
        for k in numba.prange(shape[0]):
            for j in range(shape[1]):
                # Simpson's rule for pairs
                for i in range(2, n, 2):
                    h = dx
                    result[k, j, i] = result[k, j, i-2] + h/3 * (y[k, j, i-2] + 4*y[k, j, i-1] + y[k, j, i])
                
                # Fill in odd indices
                for i in range(1, n, 2):
                    if i < n-1:
                        # Interpolate between even indices
                        result[k, j, i] = (result[k, j, i-1] + result[k, j, i+1]) / 2
                    else:
                        # Last point if n is odd
                        result[k, j, i] = result[k, j, i-1] + dx/2 * (y[k, j, i-1] + y[k, j, i])
    
    return result

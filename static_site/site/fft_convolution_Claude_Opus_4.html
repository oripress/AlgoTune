import numpy as np
from typing import Any

class Solver:
    def solve(self, problem: dict[str, Any]) -> dict[str, list]:
        """
        Solve the convolution problem using the Fast Fourier Transform approach.
        
        :param problem: A dictionary representing the convolution problem.
        :return: A dictionary with key "convolution" containing the result.
        """
        signal_x = np.array(problem["signal_x"], dtype=np.float64)
        signal_y = np.array(problem["signal_y"], dtype=np.float64)
        mode = problem.get("mode", "full")
        
        # Handle empty signals
        if len(signal_x) == 0 or len(signal_y) == 0:
            return {"convolution": []}
        
        # Perform FFT convolution
        len_x = len(signal_x)
        len_y = len(signal_y)
        
        # Determine output length and padding
        if mode == "full":
            n = len_x + len_y - 1
        elif mode == "same":
            n = len_x
        elif mode == "valid":
            n = max(0, max(len_x, len_y) - min(len_x, len_y) + 1)
        else:
            raise ValueError(f"Invalid mode: {mode}")
        
        # For FFT convolution, we need to pad to at least len_x + len_y - 1
        fft_size = len_x + len_y - 1
        
        # Compute FFTs
        X = np.fft.fft(signal_x, n=fft_size)
        Y = np.fft.fft(signal_y, n=fft_size)
        
        # Multiply in frequency domain
        Z = X * Y
        
        # Inverse FFT
        result = np.fft.ifft(Z).real
        
        # Extract the appropriate portion based on mode
        if mode == "full":
            result = result[:n]
        elif mode == "same":
            # Center the result
            start = (len_y - 1) // 2
            result = result[start:start + n]
        elif mode == "valid":
            # Only where signals fully overlap
            if len_x >= len_y:
                result = result[len_y - 1:len_y - 1 + n]
            else:
                result = result[len_x - 1:len_x - 1 + n]
        
        return {"convolution": result.tolist()}

import numpy as np
from numba import njit
from typing import Any

@njit
def fft_convolve_numba(x, y, mode_int):
    """
    Numba-compiled FFT convolution.
    mode_int: 0=full, 1=same, 2=valid
    """
    len_x = len(x)
    len_y = len(y)
    
    if len_x == 0 or len_y == 0:
        return np.array([], dtype=np.float64)
    
    # Compute output size
    if mode_int == 0:  # full
        n_out = len_x + len_y - 1
    elif mode_int == 1:  # same
        n_out = len_x
    else:  # valid
        n_out = max(0, max(len_x, len_y) - min(len_x, len_y) + 1)
    
    # FFT size
    fft_size = len_x + len_y - 1
    
    # Compute FFTs
    X = np.fft.fft(x, n=fft_size)
    Y = np.fft.fft(y, n=fft_size)
    
    # Multiply in frequency domain
    Z = X * Y
    
    # Inverse FFT
    result = np.fft.ifft(Z).real
    
    # Extract appropriate portion
    if mode_int == 0:  # full
        return result[:n_out]
    elif mode_int == 1:  # same
        start = (len_y - 1) // 2
        return result[start:start + n_out]
    else:  # valid
        if len_x >= len_y:
            return result[len_y - 1:len_y - 1 + n_out]
        else:
            return result[len_x - 1:len_x - 1 + n_out]

class Solver:
    def __init__(self):
        # Warm up the JIT compiler
        _ = fft_convolve_numba(np.array([1.0]), np.array([1.0]), 0)
    
    def solve(self, problem: dict[str, Any]) -> dict[str, list]:
        """
        Solve the convolution problem using the Fast Fourier Transform approach.
        
        :param problem: A dictionary representing the convolution problem.
        :return: A dictionary with key "convolution" containing the result.
        """
        x = problem["signal_x"]
        y = problem["signal_y"]
        mode = problem.get("mode", "full")
        
        # Handle empty signals
        if not x or not y:
            return {"convolution": []}
        
        # Convert mode to int for numba
        mode_map = {"full": 0, "same": 1, "valid": 2}
        mode_int = mode_map.get(mode, 0)
        
        # Convert to numpy arrays
        x_arr = np.asarray(x, dtype=np.float64)
        y_arr = np.asarray(y, dtype=np.float64)
        
        # Use numba-compiled function
        result = fft_convolve_numba(x_arr, y_arr, mode_int)
        
        # Convert to list and return
        return {"convolution": result.tolist()}

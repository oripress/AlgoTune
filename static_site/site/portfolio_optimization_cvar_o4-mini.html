from scipy.sparse import csr_matrix, eye, hstack

    def solve(self, problem, **kwargs):

        R = np.asarray(problem["returns"], dtype=float)
        lam = float(problem["lambda"])
        beta = float(problem["beta"])

        # Compute statistics
        mu = R.mean(axis=0)
        inv_tail = 1.0 / ((1.0 - beta) * M)

        # Objective coefficients: [w (N), alpha (1), z (M)]
        c_w = -mu
        c_alpha = lam

        c = np.concatenate([c_w, [c_alpha], c_z])

        # Equality constraint: sum(w) == 1
        A_eq = np.zeros((1, N + 1 + M), dtype=float)
        A_eq[0, :N] = 1.0
        b_eq = np.array([1.0], dtype=float)

        # -R[s]¬∑w - alpha - z_s <= 0
        # Build sparse blocks
        # -R part

        A_w = csr_matrix(R_neg)
        # -1 for alpha
        A_alpha = -np.ones((M, 1), dtype=float)
        # -I for z
        A_z = -eye(M, format='csr')

        bounds = [(0.0, None)] * N + [(None, None)] + [(0.0, None)] * M

class Solver:
    def solve(self, problem, **kwargs):
        R = np.asarray(problem["returns"], dtype=np.float64)
        lam = float(problem["lambda"])
        beta = float(problem["beta"])
        M, N = R.shape

        # Moments
        mu = R.mean(axis=0)
        inv_tail = 1.0 / ((1.0 - beta) * M)

        # Objective: [w (N), alpha, z (M)]
        c = np.empty(N + 1 + M, dtype=np.float64)
        c[:N] = -mu
        c[N] = lam
        c[N+1:] = lam * inv_tail

        # Equality: sum(w) == 1
        A_eq = np.zeros((1, N + 1 + M), dtype=np.float64)
        A_eq[0, :N] = 1.0
        b_eq = np.ones(1, dtype=np.float64)

        # Inequalities: -R w - alpha - z <= 0
        rows = np.repeat(np.arange(M, dtype=np.int32), N)
        cols = np.tile(np.arange(N, dtype=np.int32), M)
        data_w = -R.ravel()

        rows2 = np.arange(M, dtype=np.int32)
        cols2 = np.full(M, N, dtype=np.int32)
        data2 = -np.ones(M, dtype=np.float64)

        rows3 = rows2
        cols3 = N + 1 + rows2
        data3 = data2

        row = np.concatenate([rows, rows2, rows3])
        col = np.concatenate([cols, cols2, cols3])
        data = np.concatenate([data_w, data2, data3])

        A_ub = csr_matrix((data, (row, col)), shape=(M, N + 1 + M))
        b_ub = np.zeros(M, dtype=np.float64)

        bounds = [(0.0, None)] * N + [(None, None)] + [(0.0, None)] * M

        res = linprog(c, A_ub=A_ub, b_ub=b_ub,
                      A_eq=A_eq, b_eq=b_eq,
                      bounds=bounds,
                      method='highs')

        if not res.success or res.x is None:
            return [1.0 / N] * N

        w = res.x[:N]
        w = np.maximum(w, 0.0)
        s = w.sum()
        if s > 0:
            w /= s
        else:
            w = np.full(N, 1.0 / N)
        return w.tolist()

import numpy as np
from scipy.optimize import linprog
from scipy.sparse import csr_matrix, hstack, identity

class Solver:
    def solve(self, problem, **kwargs):
        # parse inputs
        returns = problem.get("returns", [])
        if not returns:
            return []
        R = np.array(returns, dtype=float)
        M, N = R.shape
        lam = float(problem.get("lambda", 0.0))
        beta = float(problem.get("beta", 0.0))
        # trivial or degenerate cases
        if N == 0:
            return []
        if N == 1:
            return [1.0]
        mu = R.mean(axis=0)
        # if no risk aversion or single scenario, choose max-return asset
        if lam == 0.0 or M <= 1:
            idx = int(mu.argmax())
            w = [0.0] * N
            w[idx] = 1.0
            return w
        # CVaR tail constant
        inv_tail = 1.0 / ((1.0 - beta) * M)
        # variable count: N weights, 1 alpha, M z's
        nvar = N + 1 + M
        # objective: minimize lam*alpha + lam*inv_tail*sum(z) - mu^T w
        c = np.empty(nvar, dtype=float)
        c[:N] = -mu
        c[N] = lam
        c[N+1:] = lam * inv_tail
        # equality constraint: sum(w) == 1
        A_eq = np.zeros((1, nvar), dtype=float)
        A_eq[0, :N] = 1.0
        b_eq = np.array([1.0], dtype=float)
        # inequality constraints: -R[s]¬∑w - alpha - z_s <= 0
        # build sparse A_ub: [ -R  | -1 | -I ]
        block1 = csr_matrix(-R)
        block2 = csr_matrix((-np.ones(M), (np.arange(M), np.zeros(M, dtype=int))), shape=(M, 1))
        block3 = identity(M, format='csr') * -1.0
        A_ub = hstack([block1, block2, block3], format='csr')
        b_ub = np.zeros(M, dtype=float)
        # bounds for variables
        bounds = [(0.0, None)] * N + [(None, None)] + [(0.0, None)] * M
        # solve LP using HiGHS
        res = linprog(c, A_ub=A_ub, b_ub=b_ub,
                      A_eq=A_eq, b_eq=b_eq,
                      bounds=bounds, method="highs")
        if res.success:
            return res.x[:N].tolist()
        else:
            return []

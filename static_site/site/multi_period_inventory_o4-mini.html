    def solve(self, problem: dict) -> dict:
        """
        Solve multi-period inventory management via a direct LP (SciPy linprog).
        """
        # Unpack problem data
        demand = np.array(problem.get("demand", []), dtype=float)
        T = demand.size
        if T == 0:
            return {"inventory": [], "orders": [], "stockouts": [], "total_cost": 0.0}

        h = float(problem.get("holding_cost", 0.0))
        c_ord = float(problem.get("ordering_cost", 0.0))
        p_pen = float(problem.get("stockout_penalty", 0.0))
        I0 = float(problem.get("initial_inventory", 0.0))

        # Variable order: [I_1...I_T, u_1...u_T, s_1...s_T]
        n = 3 * T
        # Objective coefficients
        c = np.hstack([np.full(T, h), np.full(T, c_ord), np.full(T, p_pen)])

        # Equality constraints: flow balance for each period
        # A_eq x = b_eq
        A_eq = np.zeros((T, n), dtype=float)
        b_eq = np.zeros(T, dtype=float)

        # Period 1
        A_eq[0, 0] = 1         # I_1
        A_eq[0, T + 0] = -1    # u_1
        A_eq[0, 2*T + 0] = -1  # s_1
        b_eq[0] = I0 - demand[0]

        # Periods 2..T
        for t in range(1, T):

            A_eq[t, t - 1]   = -1       # I_t
            A_eq[t, T + t]   = -1       # u_t+1

        bounds = [(0, None)] * n

        stockouts = x[2*T:3*T]

import numpy as np
from scipy.optimize import linprog

class Solver:
    def solve(self, problem: dict) -> dict:
        """
        Solve multi-period inventory management via a direct LP (SciPy linprog).
        """
        # Unpack problem data
        demand = np.array(problem.get("demand", []), dtype=float)
        T = demand.size
        if T == 0:
            return {"inventory": [], "orders": [], "stockouts": [], "total_cost": 0.0}

        h = float(problem.get("holding_cost", 0.0))
        c_ord = float(problem.get("ordering_cost", 0.0))
        p_pen = float(problem.get("stockout_penalty", 0.0))
        I0 = float(problem.get("initial_inventory", 0.0))

        # Variables order: [I_1...I_T, u_1...u_T, s_1...s_T]
        n = 3 * T
        # Objective coefficients
        c = np.hstack([np.full(T, h), np.full(T, c_ord), np.full(T, p_pen)])

        # Equality constraints: flow balance for each period
        A_eq = np.zeros((T, n), dtype=float)
        b_eq = np.zeros(T, dtype=float)

        # Period 1
        A_eq[0, 0] = 1         # I_1
        A_eq[0, T + 0] = -1    # u_1
        A_eq[0, 2*T + 0] = -1  # s_1
        b_eq[0] = I0 - demand[0]

        # Periods 2..T
        for t in range(1, T):
            A_eq[t, t]       = 1        # I_{t+1}
            A_eq[t, t - 1]   = -1       # I_t
            A_eq[t, T + t]   = -1       # u_{t+1}
            A_eq[t, 2*T + t] = -1       # s_{t+1}
            b_eq[t] = -demand[t]

        bounds = [(0, None)] * n

        res = linprog(c, A_eq=A_eq, b_eq=b_eq, bounds=bounds, method="highs")
        if not res.success or res.x is None:
            return {"inventory": [], "orders": [], "stockouts": [], "total_cost": 0.0}

        x = res.x
        inv = x[0:T]
        orders = x[T:2*T]
        stockouts = x[2*T:3*T]
        total_cost = float(res.fun)

        return {
            "inventory": inv.tolist(),
            "orders": orders.tolist(),
            "stockouts": stockouts.tolist(),
            "total_cost": total_cost,
        }

import numpy as np

class Solver:
    def solve(self, problem: dict) -> dict:
        """
        Solve the multi-period inventory management problem in O(T) time.
        Uses a direct analytic policy:
        - Allocates initial inventory to periods with highest net benefit.
        - Then orders to meet remaining demand if ordering cost <= penalty,
          else incurs stockouts.
        """
        # Extract and validate problem data
        demand = problem.get("demand", [])
        if not demand:
            return {"inventory": [], "orders": [], "stockouts": [], "total_cost": 0.0}
        # Convert to float array
        demand = np.array(demand, dtype=float)
        T = demand.size

        h = float(problem.get("holding_cost", 0.0))
        c_ord = float(problem.get("ordering_cost", 0.0))
        p_pen = float(problem.get("stockout_penalty", 0.0))
        I0 = float(problem.get("initial_inventory", 0.0))

        # Prepare result arrays
        inventory = [0.0] * T
        orders = [0.0] * T
        stockouts = [0.0] * T

        # Determine threshold period for initial inventory allocation
        # Benefit of using initial inventory at period t: alt_cost - h*(t)
        # with t = periods to hold = (period_index), 0-based indexing implies hold t periods
        alt_cost = c_ord if c_ord <= p_pen else p_pen
        if h > 0.0:
            # periods where benefit positive: alt_cost > h * t
            # t < alt_cost / h  --> period_index < alt_cost/h
            max_hold = int(alt_cost / h)
            # threshold index (0-based) inclusive: 0..max_hold
            thr = min(T - 1, max_hold)
        else:
            # zero holding cost: always beneficial
            thr = T - 1

        # Allocate initial inventory
        I_remain = I0
        # Totals for cost
        total_h_inv = 0.0
        total_c_ord = 0.0
        total_p_pen = 0.0

        for t in range(T):
            # Use initial inventory if within threshold and available
            use_init = 0.0
            if t <= thr and I_remain > 0.0:
                need = demand[t]
                use_init = need if need <= I_remain else I_remain
                I_remain -= use_init

            # Inventory at end of period t is remaining initial inventory
            inv_t = I_remain
            inventory[t] = inv_t
            total_h_inv += inv_t

            # Remaining demand
            rem_d = demand[t] - use_init

            # Decide order or stockout
            if c_ord <= p_pen:
                # Cheaper to order than lose sale
                u_t = rem_d
                s_t = 0.0
            else:
                # Cheaper to stockout than order
                u_t = 0.0
                s_t = rem_d

            orders[t] = u_t
            stockouts[t] = s_t
            total_c_ord += u_t
            total_p_pen += s_t

        total_cost = h * total_h_inv + c_ord * total_c_ord + p_pen * total_p_pen

        return {
            "inventory": inventory,
            "orders": orders,
            "stockouts": stockouts,
            "total_cost": float(total_cost),
        }

import numpy as np

class Solver:
    def solve(self, problem: dict) -> dict:
        # Extract demand
        demand = problem.get("demand", [])
        if not demand:
            return {"inventory": [], "orders": [], "stockouts": [], "total_cost": 0.0}

        # Convert to NumPy arrays
        d = np.array(demand, dtype=float)
        T = d.size

        # Extract costs and initial inventory
        h = float(problem.get("holding_cost", 0.0))
        c_ord = float(problem.get("ordering_cost", 0.0))
        p_pen = float(problem.get("stockout_penalty", 0.0))
        I0 = float(problem.get("initial_inventory", 0.0))

        # Determine threshold period for using initial inventory
        alt_cost = c_ord if c_ord <= p_pen else p_pen
        if h > 0.0:
            thr = int(alt_cost / h)
            if thr >= T:
                thr = T - 1
        else:
            thr = T - 1

        # Cumulative demand up to the threshold
        cum_d = np.cumsum(d[:thr+1])
        # Cumulative use of initial inventory (cannot exceed I0)
        used_cum = np.minimum(cum_d, I0)
        # Inventory remaining after threshold
        inv_thr = I0 - used_cum[-1]

        # Inventory at end of each period
        inv = np.empty(T, dtype=float)
        inv[:thr+1] = I0 - used_cum
        if thr + 1 < T:
            inv[thr+1:] = inv_thr

        # Compute how much initial inventory is used each period
        # difference of cumulative used: first element used_cum[0], then diffs
        used_init = np.empty(T, dtype=float)
        used_init[:thr+1] = np.diff(np.concatenate(([0.0], used_cum)))
        if thr + 1 < T:
            used_init[thr+1:] = 0.0

        # Remaining demand after using initial inventory
        rem = d - used_init

        # Decide orders and stockouts based on cost tradeoff
        if c_ord <= p_pen:
            orders = rem
            stockouts = np.zeros(T, dtype=float)
        else:
            orders = np.zeros(T, dtype=float)
            stockouts = rem

        # Compute total cost
        total_cost = (
            h * np.sum(inv)
            + c_ord * np.sum(orders)
            + p_pen * np.sum(stockouts)
        )

        # Return dictionaries with Python lists
        return {
            "inventory": inv.tolist(),
            "orders": orders.tolist(),
            "stockouts": stockouts.tolist(),
            "total_cost": float(total_cost),
        }

from numba import njit

@njit
def _solve_numba(d, h, c_ord, p_pen, I0):
    T = d.shape[0]
    inv = np.empty(T, dtype=np.float64)
    orders = np.empty(T, dtype=np.float64)
    stockouts = np.empty(T, dtype=np.float64)
    # Determine threshold thr
    alt_cost = c_ord if c_ord <= p_pen else p_pen
    if h > 0.0:
        thr = int(alt_cost / h)
        if thr >= T:
            thr = T - 1
    else:
        thr = T - 1
    # Cumulative sum of demand
    cum = np.empty(T, dtype=np.float64)
    tot = 0.0
    for i in range(T):
        tot += d[i]
        cum[i] = tot
    # Compute decisions
    used_prev = 0.0
    for t in range(T):
        idx = t if t <= thr else thr
        used = cum[idx] if cum[idx] < I0 else I0
        inv_t = I0 - used
        inv[t] = inv_t
        # used_prev set from last iteration
        used_init = used - used_prev
        rem = d[t] - used_init
        if rem < 0.0:
            rem = 0.0
        if c_ord <= p_pen:
            orders[t] = rem
            stockouts[t] = 0.0
        else:
            orders[t] = 0.0
            stockouts[t] = rem
        used_prev = used
    # Compute total cost
    total_h = 0.0
    total_ord = 0.0
    total_pen = 0.0
    for t in range(T):
        total_h += inv[t]
        total_ord += orders[t]
        total_pen += stockouts[t]
    total_cost = h * total_h + c_ord * total_ord + p_pen * total_pen
    return inv, orders, stockouts, total_cost

        """

        T = d.size

        h = float(problem.get("holding_cost", 0.0))

        p_pen = float(problem.get("stockout_penalty", 0.0))

        # Determine threshold period for initial inventory allocation
        alt_cost = c_ord if c_ord <= p_pen else p_pen

class Solver:
    def solve(self, problem: dict) -> dict:
        """
        Solve the multi-period inventory management problem in O(T) time using vectorized NumPy.
        """
        # Extract and validate input
        demand = problem.get("demand", [])
        if not demand:
            return {"inventory": [], "orders": [], "stockouts": [], "total_cost": 0.0}

        # Vectorize demands
        d = np.array(demand, dtype=float)
        T = d.size

        # Costs and initial inventory
        h = float(problem.get("holding_cost", 0.0))
        c = float(problem.get("ordering_cost", 0.0))
        p = float(problem.get("stockout_penalty", 0.0))
        I0 = float(problem.get("initial_inventory", 0.0))

        # Determine threshold period where holding cost exceeds alternative cost savings
        alt_cost = c if c <= p else p
        if h > 0.0:
            thr = int(alt_cost / h)
            if thr >= T:
                thr = T - 1
        else:

        # Cumulative demand and cap initial usage up to thr
        cum = np.cumsum(d)
        cap = cum[thr] if cum[thr] < I0 else I0
        used_trunc = np.minimum(cum, cap)

        # Per-period initial inventory usage
        used_init = np.empty(T, dtype=float)
        used_init[0] = used_trunc[0]
        used_init[1:] = used_trunc[1:] - used_trunc[:-1]

        # Inventory at period end
        inventory = I0 - used_trunc

        # Remaining demand after initial inventory
        rem = d - used_init
        rem = np.where(rem > 0.0, rem, 0.0)

        if c <= p:
            orders = rem
            stockouts = np.zeros(T, dtype=float)
        else:
            orders = np.zeros(T, dtype=float)

            h * inventory.sum() + c * orders.sum() + p * stockouts.sum()
        )

        """
        Solve the multi-period inventory management problem in O(T) time

        with a single Python loop (no NumPy).
        """

        # Extract costs and initial inventory
        h = float(problem.get("holding_cost", 0.0))
        c = float(problem.get("ordering_cost", 0.0))
        p = float(problem.get("stockout_penalty", 0.0))
        I0 = float(problem.get("initial_inventory", 0.0))

        n = len(demand)
        inventory = [0.0] * n
        orders = [0.0] * n
        stockouts = [0.0] * n

        total_cost = 0.0
        used_init = 0.0  # amount of initial inventory used so far

        for t in range(n):
            d_t = float(demand[t])
            # allocate from initial inventory if available
            if used_init < I0:
                available = I0 - used_init
                use = d_t if d_t <= available else available
            else:
                use = 0.0
            used_init += use

            # inventory at end of period t
            inv_t = I0 - used_init
            # unmet demand after using initial stock
            rem = d_t - use
            if rem < 0.0:
                rem = 0.0

            # decide to order or incur stockout
            if c <= p:
                order = rem
                stock = 0.0
            else:
                order = 0.0
                stock = rem

            # record decisions
            inventory[t] = inv_t
            orders[t] = order
            stockouts[t] = stock

            # accumulate cost
            total_cost += h * inv_t + c * order + p * stock

        return {
            "inventory": inventory,
            "orders": orders,
            "stockouts": stockouts,

    def solve(self, problem: dict) -> dict:

        n = len(demands)
        if n == 0:
            return {"inventory": [], "orders": [], "stockouts": [], "total_cost": 0.0}
        # Extract costs
        h = float(problem.get("holding_cost", 0.0))
        c = float(problem.get("ordering_cost", 0.0))
        p = float(problem.get("stockout_penalty", 0.0))
        inv0 = float(problem.get("initial_inventory", 0.0))
        # Determine minimal cost to satisfy unmet demand (order vs stockout)
        min_penalty = c if c <= p else p
        # Compute cutoff period for using initial inventory
        if h > 0:
            # (t-1)*h < min_penalty  => t <= floor(min_penalty/h) + 1
            M = int(min_penalty // h) + 1
        else:
            M = n
        # Allocate initial inventory to earliest beneficial periods
        x_init = [0.0] * n
        rem_inv = inv0
        for t in range(min(M, n)):
            if rem_inv <= 0.0:
                break
            d = float(demands[t])
            use = rem_inv if rem_inv < d else d
            x_init[t] = use
            rem_inv -= use
        # Build the solution arrays via dynamic inventory tracking
        inv_arr = [0.0] * n
        orders = [0.0] * n
        stockouts = [0.0] * n
        inv = inv0
        for t in range(n):

            inv -= x_init[t]
            if inv < 0.0:
                inv = 0.0
            # remaining demand
            left = float(demands[t]) - x_init[t]
            if left < 0.0:
                left = 0.0
            # fulfill by ordering or incur stockout
            if c <= p:
                orders[t] = left
                stockouts[t] = 0.0

                orders[t] = 0.0
                stockouts[t] = left
            inv_arr[t] = inv

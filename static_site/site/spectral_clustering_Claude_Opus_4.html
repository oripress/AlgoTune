def compute_normalized_adjacency(similarity_matrix):
    """Compute D^(-1/2) * W * D^(-1/2) directly."""
    n = similarity_matrix.shape[0]
    
    # Compute degree vector in parallel
    degree = np.sum(similarity_matrix, axis=1)
    
    # Compute degree^(-1/2)
    degree_sqrt_inv = np.zeros(n, dtype=np.float64)
    for i in prange(n):

        for j in range(n):

    indices = np.arange(n_samples)
    np.random.shuffle(indices)

    # Pre-allocate arrays
    labels = np.zeros(n_samples, dtype=np.int32)
    distances = np.zeros((n_samples, n_clusters), dtype=np.float64)

            for j in range(n_clusters):

import numpy as np
from typing import Any
import numba as nb

@nb.jit(nopython=True, cache=True)
def nmf_multiplicative_update(X, W, H, n_iter=50):
    """Fast NMF using multiplicative update rules."""
    eps = 1e-10
    
    for _ in range(n_iter):
        # Update H
        WtX = W.T @ X
        WtW = W.T @ W
        H = H * (WtX / (WtW @ H + eps))
        
        # Update W
        XHt = X @ H.T
        HHt = H @ H.T
        W = W * (XHt / (W @ HHt + eps))
        
    return W, H

class Solver:
    def solve(self, problem: dict[str, Any]) -> dict[str, list[list[float]]]:
        """Fast NMF implementation using Numba."""
        X = np.array(problem["X"], dtype=np.float64)
        n_components = problem["n_components"]
        m, n = X.shape
        
        # Initialize W and H with small random values
        np.random.seed(0)
        W = np.random.rand(m, n_components) * 0.1
        H = np.random.rand(n_components, n) * 0.1
        
        # Run NMF
        W, H = nmf_multiplicative_update(X, W, H, n_iter=50)
        
        return {"W": W.tolist(), "H": H.tolist()}

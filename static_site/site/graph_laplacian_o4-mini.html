import numpy as np

class Solver:
    def solve(self, problem, **kwargs):
        # Load CSR components
        data = np.array(problem["data"], dtype=np.float64)
        indices = np.array(problem["indices"], dtype=np.int64)
        indptr = np.array(problem["indptr"], dtype=np.int64)
        # Number of nodes
        n = int(problem["shape"][0])
        # Compute degrees (sum of each row)
        if data.size > 0:
            deg = np.add.reduceat(data, indptr[:-1])
        else:
            deg = np.zeros(n, dtype=np.float64)
        normed = bool(problem.get("normed", False))
        # Prepare row indices for off-diagonal entries
        row_counts = np.diff(indptr)
        row = np.repeat(np.arange(n, dtype=np.int64), row_counts)
        # Build off-diagonal and diagonal values
        if normed:
            inv_sqrt = np.zeros(n, dtype=np.float64)
            nz = deg > 0
            inv_sqrt[nz] = 1.0 / np.sqrt(deg[nz])
            off_data = -data * inv_sqrt[row] * inv_sqrt[indices]
            diag_data = np.ones(n, dtype=np.float64)
        else:
            off_data = -data
            diag_data = deg
        # Combine data and indices
        data_all = np.concatenate((off_data, diag_data))
        col_all = np.concatenate((indices, np.arange(n, dtype=np.int64)))
        row_all = np.concatenate((row, np.arange(n, dtype=np.int64)))
        # Sort by row then column index
        # lexsort keys: (column, row) => primary=row, secondary=column
        perm = np.lexsort((col_all, row_all))
        dataL = data_all[perm]
        colL = col_all[perm]
        rowL = row_all[perm]
        # Build new indptr by counting entries per row
        counts = np.bincount(rowL, minlength=n)
        indptrL = np.empty(n + 1, dtype=np.int64)
        indptrL[0] = 0
        indptrL[1:] = np.cumsum(counts, dtype=np.int64)
        # Return CSR components
        return {
            "laplacian": {
                "data": dataL.tolist(),
                "indices": colL,
                "indptr": indptrL,
                "shape": (n, n),
            }
        }

        row_counts = indptr[1:] - indptr[:-1]
        # If no edges, handle trivial

                    "data": [],
                    "indices": [],
                    "indptr": [0],

            data_off = -data

        if normed:

        using efficient sparse matrix operations.

            data = np.asarray(problem["data"], dtype=np.float64)
            indices = np.asarray(problem["indices"], dtype=np.int64)
            indptr = np.asarray(problem["indptr"], dtype=np.int64)
            shape = tuple(problem["shape"])
            n = shape[0]

                "laplacian": {
                    "data": [],
                    "indices": [],
                    "indptr": [0],
                    "shape": (0, 0),
                }

            D = sp.diags(deg, 0, format="csr", dtype=np.float64)

        L.eliminate_zeros()
        L.sort_indices()

                "data": L.data.tolist(),

import numpy as np
from scipy.sparse import csr_matrix, diags

class Solver:
    def solve(self, problem, **kwargs):
        # Read CSR components
        data = np.array(problem["data"], dtype=float)
        indices = np.array(problem["indices"], dtype=int)
        indptr = np.array(problem["indptr"], dtype=int)
        n, m = problem["shape"]
        # Build CSR graph
        graph = csr_matrix((data, indices, indptr), shape=(n, m))
        # Degree per node
        if data.size > 0:
            deg = np.add.reduceat(data, indptr[:-1])
        else:
            deg = np.zeros(n, dtype=float)
        if problem.get("normed", False):
            # Normalized Laplacian: L = I - D^-1/2 * A * D^-1/2
            d_inv_sqrt = np.zeros_like(deg)
            nz = deg > 0
            d_inv_sqrt[nz] = 1.0 / np.sqrt(deg[nz])
            # D^-1/2 * A * D^-1/2
            D_inv_sqrt = diags(d_inv_sqrt, 0, format='csr')
            L = D_inv_sqrt.dot(graph).dot(D_inv_sqrt)
            # L = -L + I
            L.data *= -1.0
            # Add identity
            diag_vals = L.diagonal()
            diag_vals += 1.0
            L.setdiag(diag_vals)
        else:
            # Combinatorial Laplacian: L = D - A
            D = diags(deg, 0, format='csr')
            L = D - graph
        # Remove zeros
        L.eliminate_zeros()
        # Return components as Python lists
        return {
            "laplacian": {
                "data": L.data.tolist(),
                "indices": L.indices.tolist(),
                "indptr": L.indptr.tolist(),
                "shape": L.shape,
            }
        }

class Solver:
    def solve(self, problem, **kwargs):
        # Load CSR components
        data = np.array(problem["data"], dtype=np.float64)
        indices = np.array(problem["indices"], dtype=np.int64)
        indptr = np.array(problem["indptr"], dtype=np.int64)
        # Number of nodes
        n = int(problem["shape"][0])
        # Compute degrees (sum of each row)
        if data.size > 0:
            deg = np.add.reduceat(data, indptr[:-1])
        else:
            deg = np.zeros(n, dtype=np.float64)
        normed = bool(problem.get("normed", False))
        # Prepare row indices for off-diagonal entries
        row_counts = np.diff(indptr)
        row = np.repeat(np.arange(n, dtype=np.int64), row_counts)
        # Build off-diagonal and diagonal values
        if normed:
            inv_sqrt = np.zeros(n, dtype=np.float64)
            nz = deg > 0
            inv_sqrt[nz] = 1.0 / np.sqrt(deg[nz])
            off_data = -data * inv_sqrt[row] * inv_sqrt[indices]
            diag_data = np.ones(n, dtype=np.float64)
        else:
            off_data = -data
            diag_data = deg
        # Combine data and indices
        data_all = np.concatenate((off_data, diag_data))
        col_all = np.concatenate((indices, np.arange(n, dtype=np.int64)))
        row_all = np.concatenate((row, np.arange(n, dtype=np.int64)))
        # Sort by row then column index
        # lexsort keys: (column, row) => primary=row, secondary=column
        perm = np.lexsort((col_all, row_all))
        dataL = data_all[perm]
        colL = col_all[perm]
        rowL = row_all[perm]
        # Build new indptr by counting entries per row
        counts = np.bincount(rowL, minlength=n)
        indptrL = np.empty(n + 1, dtype=np.int64)
        indptrL[0] = 0
        indptrL[1:] = np.cumsum(counts, dtype=np.int64)
        # Return CSR components
        return {
            "laplacian": {
                "data": dataL.tolist(),
                "indices": colL,
                "indptr": indptrL,
                "shape": (n, n),
            }

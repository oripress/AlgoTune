import numpy as np
from math import sqrt
from scipy.linalg import eigh as sp_eigh

class Solver:
    def solve(self, problem, **kwargs):
        # Convert input to numpy array of type double.
        A = np.asarray(problem, dtype=np.double)
        n = A.shape[0]
        # Trivial 1x1 case.
        if n == 1:
            v0 = float(A[0, 0])
            return [v0], [[1.0]]
        # Closed-form solution for 2x2 symmetric matrix.
        if n == 2:
            a = A[0, 0]; b = A[0, 1]; c = A[1, 1]
            tr = a + c
            det = a * c - b * b
            mid = 0.5 * tr
            rad = sqrt(max(mid * mid - det, 0.0))
            l1 = mid + rad; l2 = mid - rad
            # Eigenvectors.
            if abs(b) > 1e-16:
                v1x = b;    v1y = l1 - a
                v2x = b;    v2y = l2 - a
            else:
                if abs(a - l1) > abs(c - l1):
                    v1x = l1 - c; v1y = b
                    v2x = l2 - c; v2y = b
                else:
                    v1x = 1.0;    v1y = 0.0
                    v2x = 0.0;    v2y = 1.0
            # Normalize first vector.
            n1 = sqrt(v1x*v1x + v1y*v1y)
            if n1 != 0.0:
                v1x /= n1; v1y /= n1
            else:
                v1x = 1.0; v1y = 0.0
            # Normalize second vector.
            n2 = sqrt(v2x*v2x + v2y*v2y)
            if n2 != 0.0:
                v2x /= n2; v2y /= n2
            else:
                v2x = -v1y; v2y = v1x
            # Order by descending eigenvalues.
            if l1 >= l2:
                return [l1, l2], [[v1x, v1y], [v2x, v2y]]
            else:
                return [l2, l1], [[v2x, v2y], [v1x, v1y]]
        # General case: use SciPy MRRR driver for speed.
        w, v = sp_eigh(A, driver='evr', overwrite_a=False, check_finite=False)
        # Reverse to descending order.
        w = w[::-1]
        v = v[:, ::-1]
        # Return eigenvalues and eigenvectors.
        return w.tolist(), v.T.tolist()

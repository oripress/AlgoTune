        """Solve the VRP problem using CP-SAT solver."""
        D = problem["D"]
        K = problem["K"]
        depot = problem["depot"]
        n = len(D)
        model = cp_model.CpModel()

        x = {}

            for j in range(n):
                if i != j:

        # Each non-depot node must be entered exactly once and left exactly once
        for i in range(n):
            if i == depot:
                continue
            model.Add(sum(x[(j, i)] for j in range(n) if j != i) == 1)
            model.Add(sum(x[(i, j)] for j in range(n) if j != i) == 1)
        
        # Depot must have exactly K departures and K arrivals
        model.Add(sum(x[(depot, j)] for j in range(n) if j != depot) == K)
        model.Add(sum(x[(i, depot)] for i in range(n) if i != depot) == K)

        u = {}
        for i in range(n):
            if i == depot:
                continue
            u[i] = model.NewIntVar(1, n - 1, f"u_{i}")

            if i == depot:
                continue
            for j in range(n):
                if j == depot or i == j:
                    continue
                model.Add(u[i] + 1 <= u[j] + (n - 1) * (1 - x[(i, j)]))

        model.Minimize(sum(D[i][j] * x[(i, j)] for i, j in x))

        K = problem["K"]

class Solver:
    def solve(self, problem, **kwargs):
        """Solve the VRP problem using CP-SAT solver with optimized settings."""
        D = problem["D"]
        K = problem["K"]
        depot = problem["depot"]
        n = len(D)
        
        # For very small instances, use brute force
        if n <= 5:
            return self._solve_small(problem)

        x = {}

            for j in range(n):

                    x[(i, j)] = model.NewBoolVar(f"x_{i}_{j}")
        
        # Each non-depot node must be entered exactly once and left exactly once
        for i in range(n):
            if i == depot:
                continue
            model.Add(sum(x[(j, i)] for j in range(n) if j != i) == 1)
            model.Add(sum(x[(i, j)] for j in range(n) if j != i) == 1)
        
        # Depot must have exactly K departures and K arrivals
        model.Add(sum(x[(depot, j)] for j in range(n) if j != depot) == K)
        model.Add(sum(x[(i, depot)] for i in range(n) if i != depot) == K)
        
        # MTZ subtour elimination
        u = {}
        for i in range(n):
            if i == depot:
                continue
            u[i] = model.NewIntVar(1, n - 1, f"u_{i}")
        
        for i in range(n):
            if i == depot:
                continue
            for j in range(n):
                if j == depot or i == j:
                    continue
                model.Add(u[i] + 1 <= u[j] + (n - 1) * (1 - x[(i, j)]))
        
        # Objective: minimize total distance
        model.Minimize(sum(D[i][j] * x[(i, j)] for i, j in x))
        
        solver = cp_model.CpSolver()

        for i in range(n):
            if i == depot:
                continue
            model.Add(sum(x[(j, i)] for j in range(n) if j != i) == 1)

        model.Add(sum(x[(depot, j)] for j in range(n) if j != depot) == K)

                    continue

                    if j == depot or i == j:
                        continue
                    model.Add(u[i] + 1 <= u[j] + (n - 1) * (1 - x[(i, j)]))
        
        # Objective: minimize total distance
        model.Minimize(sum(D[i][j] * x[(i, j)] for i, j in x))

        solver.parameters.cp_model_presolve = True
        solver.parameters.use_optional_variables = False
        solver.parameters.stop_after_first_solution = False

        if n <= 10:

        for i in range(n):
            for j in range(n):
                if i != j:
                    x[(i, j)] = model.NewBoolVar(f"x_{i}_{j}")
        
        # Each non-depot node must be entered exactly once and left exactly once
        for i in range(n):
            if i == depot:
                continue
            model.Add(sum(x[(j, i)] for j in range(n) if j != i) == 1)
            model.Add(sum(x[(i, j)] for j in range(n) if j != i) == 1)
        
        # Depot must have exactly K departures and K arrivals
        model.Add(sum(x[(depot, j)] for j in range(n) if j != depot) == K)
        model.Add(sum(x[(i, depot)] for i in range(n) if i != depot) == K)
        
        # MTZ subtour elimination
        u = {}
        for i in range(n):
            if i == depot:
                continue
            u[i] = model.NewIntVar(1, n - 1, f"u_{i}")
        for i in range(n):
            if i == depot:
                continue
            for j in range(n):
                if j == depot or i == j:
                    continue
                model.Add(u[i] + 1 <= u[j] + (n - 1) * (1 - x[(i, j)]))
        
        # Objective: minimize total distance
        model.Minimize(sum(D[i][j] * x[(i, j)] for i, j in x))
        
        solver = cp_model.CpSolver()
        # Small optimization: reduce search workers
        solver.parameters.num_search_workers = 1
        status = solver.Solve(model)
        
        if status == cp_model.OPTIMAL:
            routes = []
            # Reconstruct routes by following arcs from depot
            for j in range(n):
                if j != depot and solver.Value(x[(depot, j)]) == 1:
                    route = [depot, j]
                    current = j
                    while current != depot:
                        for k in range(n):
                            if current != k and solver.Value(x[(current, k)]) == 1:
                                route.append(k)
                                current = k
                                break
                    routes.append(route)
            return routes
        else:
            return []

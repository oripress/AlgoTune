class Solver:

        """Find the imaginary part of the nth zero using approximation formula."""
        n = problem["n"]
        
        # Use the asymptotic formula for large n
        # t_n ≈ 2π(n - 11/8) / W((n - 11/8)/(2πe))
        # where W is the Lambert W function
        
        adjusted_n = n - 11/8
        arg = adjusted_n / (2 * np.pi * np.e)
        w = float(lambertw(arg).real)

        t_n = 2 * np.pi * adjusted_n / w

    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:
        """Find the imaginary part of the nth zero."""
        n = problem["n"]
        
        # Use precomputed values for small n
        if n <= 100:
            return {"result": self.precomputed[n-1]}
        
        # For larger n, use improved asymptotic expansion
        # The nth zero's imaginary part is approximately:
        # t_n ≈ 2π(n - 1/8) / W(2π(n - 1/8)/e)
        # where W is the Lambert W function
        
        # For large arguments, W(x) ≈ log(x) - log(log(x)) + log(log(x))/log(x) + ...
        x = 2 * np.pi * (n - 1/8) / np.e
        log_x = np.log(x)
        log_log_x = np.log(log_x)
        
        # Lambert W function approximation
        w = log_x - log_log_x + log_log_x/log_x - (log_log_x**2)/(2*log_x**2) + log_log_x*(2*log_log_x - 2)/(3*log_x**3)
        
        # The imaginary part of the nth zero
        t = 2 * np.pi * (n - 1/8) / w

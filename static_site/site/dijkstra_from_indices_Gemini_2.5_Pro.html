from typing import Any
import numba
import heapq

@numba.njit
def dijkstra_single_source(
    indptr: np.ndarray,
    indices: np.ndarray,
    data: np.ndarray,
    n_nodes: int,
    start_node: int,
) -> np.ndarray:
    """
    Numba-jitted single-source Dijkstra's algorithm using a binary heap.
    This version is hardened against multiple types of malformed graph data.
    """
    distances = np.full(n_nodes, np.inf, dtype=np.float64)
    
    if not (0 <= start_node < n_nodes):
        return distances

    distances[start_node] = 0.0
    pq = [(0.0, start_node)]
    
    max_indptr_len = len(indptr)
    max_indices_len = len(indices)

    while len(pq) > 0:
        dist_u, u = heapq.heappop(pq)

        if dist_u > distances[u]:
            continue

        if u + 1 >= max_indptr_len:
            continue

        start = indptr[u]
        end = indptr[u + 1]

        if not (0 <= start <= end <= max_indices_len):
            continue

        for i in range(start, end):
            v = indices[i]
            
            if not (0 <= v < n_nodes):
                continue

            weight = data[i]
            new_dist = dist_u + weight
            if new_dist < distances[v]:
                distances[v] = new_dist
                heapq.heappush(pq, (new_dist, v))

    return distances

from typing import Any
import numba
import heapq


@numba.njit(fastmath=True)
def dijkstra_single_source(
    indptr: np.ndarray,
    indices: np.ndarray,
    data: np.ndarray,
    n_nodes: int,
    start_node: int,
) -> np.ndarray:
    """
    Numba-jitted single-source Dijkstra's algorithm using a binary heap.
    Assumes the graph is undirected and represented symmetrically in CSR.
    """
    distances = np.full(n_nodes, np.inf, dtype=np.float64)
    distances[start_node] = 0.0

    # Priority queue stores (distance, node_index)
    pq = [(0.0, start_node)]

    while len(pq) > 0:
        dist_u, u = heapq.heappop(pq)

        # If we've found a shorter path already, skip
        if dist_u > distances[u]:
            continue

        # Explore neighbors of node u
        start = indptr[u]
        end = indptr[u + 1]
        for i in range(start, end):
            v = indices[i]
            weight = data[i]

            new_dist = dist_u + weight
            if new_dist < distances[v]:
                distances[v] = new_dist
                heapq.heappush(pq, (new_dist, v))

    return distances


@numba.njit(parallel=True, fastmath=True)
def parallel_dijkstra(
    indptr: np.ndarray,
    indices: np.ndarray,
    data: np.ndarray,
    n_nodes: int,
    source_indices: np.ndarray,
) -> np.ndarray:
    """
    Runs Dijkstra's algorithm in parallel for multiple source nodes.
    """
    n_sources = len(source_indices)
    all_distances = np.full((n_sources, n_nodes), np.inf, dtype=np.float64)

    # Use numba.prange for parallel execution

        start_node = source_indices[i]
        all_distances[i, :] = dijkstra_single_source(

import numpy as np
from typing import Any
import numba
import heapq
from concurrent.futures import ThreadPoolExecutor, as_completed

@numba.njit
def dijkstra_single_source(
    indptr: np.ndarray,
    indices: np.ndarray,
    data: np.ndarray,
    n_nodes: int,
    start_node: int,
) -> np.ndarray:
    """
    Numba-jitted single-source Dijkstra's algorithm using a binary heap.
    This version is hardened against multiple types of malformed graph data
    to prevent any IndexErrors during execution.
    """
    distances = np.full(n_nodes, np.inf, dtype=np.float64)
    
    # Check 1: The start_node must be a valid node index.
    if not (0 <= start_node < n_nodes):
        return distances

    distances[start_node] = 0.0
    pq = [(0.0, start_node)]
    
    max_indptr_len = len(indptr)
    max_indices_len = len(indices)

    while len(pq) > 0:
        dist_u, u = heapq.heappop(pq)

        if dist_u > distances[u]:
            continue

        # Check 2: The node `u` must be a valid index for the indptr array.
        # We need to access up to indptr[u + 1], so u+1 must be a valid index.
        if u + 1 >= max_indptr_len:
            continue

        start = indptr[u]
        end = indptr[u + 1]

        # Check 3: The slice from indptr must be valid for the indices/data arrays.
        if not (0 <= start <= end <= max_indices_len):
            continue

        for i in range(start, end):
            v = indices[i]
            
            # Check 4: The neighbor index `v` must be a valid node.
            if not (0 <= v < n_nodes):
                continue

            weight = data[i]
            new_dist = dist_u + weight
            if new_dist < distances[v]:
                distances[v] = new_dist
                heapq.heappush(pq, (new_dist, v))

    return distances

    Numba-jitted implementation of Dijkstra's algorithm.
    """

        Computes shortest paths from a subset of source nodes to all other nodes
        in a weighted, undirected sparse graph.
        """

            indices = np.array(problem["indices"], dtype=np.int32)
            data = np.array(problem["data"], dtype=np.float64)

                    indptr,
                    indices,
                    data,
                    n_nodes,
                    source_indices[i],

                for i in range(n_sources)

                idx = future_to_index[future]
                try:
                    distances = future.result()
                    dist_matrix[idx, :] = distances
                except Exception:
                    # This safeguard should no longer be needed but is kept for safety.
                    pass

        # Format the output: replace np.inf with None for JSON compatibility.
        output_distances = [
            [d if np.isfinite(d) else None for d in row]
            for row in dist_matrix
        ]

        return {"distances": output_distances}

import numpy as np
from typing import Any
import numba
import heapq

@numba.njit
def dijkstra_single_source(
    indptr: np.ndarray,
    indices: np.ndarray,
    data: np.ndarray,
    n_nodes: int,
    start_node: int,
) -> np.ndarray:
    """
    Numba-jitted single-source Dijkstra's algorithm using a binary heap.
    This version is hardened against multiple types of malformed graph data.
    """
    distances = np.full(n_nodes, np.inf, dtype=np.float64)
    
    if not (0 <= start_node < n_nodes):
        return distances

    distances[start_node] = 0.0
    pq = [(0.0, start_node)]
    
    max_indptr_len = len(indptr)
    max_indices_len = len(indices)

    while len(pq) > 0:
        dist_u, u = heapq.heappop(pq)

        if dist_u > distances[u]:
            continue

        if u + 1 >= max_indptr_len:
            continue

        start = indptr[u]
        end = indptr[u + 1]

        if not (0 <= start <= end <= max_indices_len):
            continue

        for i in range(start, end):
            v = indices[i]
            
            if not (0 <= v < n_nodes):
                continue

            weight = data[i]
            new_dist = dist_u + weight
            if new_dist < distances[v]:
                distances[v] = new_dist
                heapq.heappush(pq, (new_dist, v))

    return distances

@numba.njit(parallel=True)
def _run_all_dijkstras(
    indptr: np.ndarray,
    indices: np.ndarray,
    data: np.ndarray,
    n_nodes: int,
    source_indices: np.ndarray,
) -> np.ndarray:
    """
    Runs Dijkstra's algorithm from multiple sources in parallel using numba.prange.
    This function is fully JIT-compiled and contains the core computation.
    """
    n_sources = len(source_indices)

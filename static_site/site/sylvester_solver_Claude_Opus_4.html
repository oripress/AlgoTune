from typing import Any
import numpy as np
import jax
import jax.numpy as jnp
from jax.scipy.linalg import solve


class Solver:
    def __init__(self):
        # Enable 64-bit precision for JAX
        jax.config.update("jax_enable_x64", True)
    
    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:
        """Solve the Sylvester equation AX + XB = Q using JAX."""
        A = jnp.array(problem["A"])
        B = jnp.array(problem["B"])
        Q = jnp.array(problem["Q"])
        
        # Vectorize the Sylvester equation: (I⊗A + B^T⊗I)vec(X) = vec(Q)
        n, m = Q.shape
        I_n = jnp.eye(n, dtype=A.dtype)
        I_m = jnp.eye(m, dtype=B.dtype)
        
        # Kronecker products
        kron_left = jnp.kron(I_m, A)
        kron_right = jnp.kron(B.T, I_n)
        
        # Solve the linear system
        vec_Q = Q.ravel()
        vec_X = solve(kron_left + kron_right, vec_Q)
        
        # Reshape back
        X = vec_X.reshape(n, m)
        
        return {"X": np.array(X)}

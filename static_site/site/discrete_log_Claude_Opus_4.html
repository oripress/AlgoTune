import numba
import numpy as np
import math

@numba.njit
def baby_giant_step_numba(g, h, p):
    """Optimized baby-step giant-step algorithm using numba"""
    m = int(np.ceil(np.sqrt(p - 1)))
    
    # Baby steps: use arrays instead of dict for numba compatibility
    baby_values = np.zeros(p, dtype=np.int64)
    baby_exists = np.zeros(p, dtype=np.bool_)
    
    g_power = 1
    for j in range(m):
        baby_values[g_power] = j
        baby_exists[g_power] = True
        g_power = (g_power * g) % p
    
    # Giant steps
    g_m = 1
    for _ in range(m):
        g_m = (g_m * g) % p
    
    # Compute g^(-m) using Fermat's little theorem
    g_m_inv = 1
    exp = p - 2
    base = g_m
    while exp > 0:
        if exp & 1:
            g_m_inv = (g_m_inv * base) % p
        base = (base * base) % p
        exp >>= 1
    
    gamma = h
    for i in range(m):
        if baby_exists[gamma]:
            return i * m + baby_values[gamma]
        gamma = (gamma * g_m_inv) % p
    
    return -1

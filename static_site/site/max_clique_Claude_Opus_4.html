        """
        Solves the max clique problem using branch and bound.
        
        :param problem: a 2d-array (adj matrix)
        :return: A list indicating the selected nodes
        """
        n = len(problem)
        if n == 0:
            return []
        
        # Convert to numpy for faster operations
        adj = np.array(problem, dtype=np.int8)
        
        # Find all maximal cliques using Bron-Kerbosch algorithm
        all_nodes = set(range(n))
        max_clique = []
        
        def bron_kerbosch(R, P, X):
            nonlocal max_clique
            if not P and not X:
                if len(R) > len(max_clique):
                    max_clique = list(R)
                return
            
            # Choose pivot to minimize branching
            pivot = max(P.union(X), key=lambda u: len(P.intersection(neighbors[u]))) if P.union(X) else None

                bron_kerbosch(
                    R.union({v}),
                    P.intersection(neighbors[v]),
                    X.intersection(neighbors[v])
                )
                P.remove(v)
                X.add(v)

        neighbors = {}

    def solve(self, problem):
        """
        Solves the max clique problem.
        
        :param problem: a 2d-array (adj matrix)
        :return: A list indicating the selected nodes
        """
        n = len(problem)
        
        # For small graphs, use brute force
        if n <= 10:
            max_clique = []
            for i in range(1, 2**n):
                subset = [j for j in range(n) if i & (1 << j)]
                if self._is_clique(problem, subset) and len(subset) > len(max_clique):
                    max_clique = subset
            return max_clique
        
        # For larger graphs, use a greedy approach
        return self._greedy_clique(problem)
    
    def _is_clique(self, adj_matrix, nodes):
        """Check if given nodes form a clique."""
        for i in range(len(nodes)):
            for j in range(i + 1, len(nodes)):
                if adj_matrix[nodes[i]][nodes[j]] == 0:
                    return False
        return True
    
    def _greedy_clique(self, adj_matrix):
        """Find a clique using greedy approach."""
        n = len(adj_matrix)
        
        # Start with node with highest degree
        degrees = [sum(adj_matrix[i]) for i in range(n)]
        start_node = max(range(n), key=lambda i: degrees[i])
        
        clique = [start_node]
        candidates = set(range(n)) - {start_node}
        
        while candidates:
            # Find node connected to all nodes in current clique
            best_node = None
            for node in candidates:
                if all(adj_matrix[node][c] == 1 for c in clique):
                    best_node = node
                    break
            
            if best_node is None:
                break
                
            clique.append(best_node)
            candidates.remove(best_node)
            
            # Remove nodes not connected to the new node
            candidates = {c for c in candidates if adj_matrix[best_node][c] == 1}
        
        return clique

class Solver:
    def solve(self, problem):
        """
        Solves the max clique problem using optimized Bron-Kerbosch.
        
        :param problem: a 2d-array (adj matrix)
        :return: A list indicating the selected nodes
        """
        n = len(problem)
        if n == 0:
            return []
        
        # Precompute neighbors as bitsets for faster operations
        neighbors = []
        for i in range(n):
            neighbor_set = 0
            for j in range(n):
                if problem[i][j] == 1 and i != j:
                    neighbor_set |= (1 << j)
            neighbors.append(neighbor_set)
        
        max_clique = 0
        max_size = 0
        
        def bron_kerbosch_bitset(R, P, X):
            nonlocal max_clique, max_size
            if P == 0 and X == 0:
                size = bin(R).count('1')
                if size > max_size:
                    max_size = size
                    max_clique = R
                return
            
            # Choose pivot with most connections in P ∪ X
            union = P | X
            if union == 0:
                return
                
            pivot = -1
            max_connections = -1
            for u in range(n):
                if union & (1 << u):
                    connections = bin(P & neighbors[u]).count('1')
                    if connections > max_connections:
                        max_connections = connections
                        pivot = u
            
            # Process vertices not connected to pivot
            P_copy = P
            for v in range(n):
                if P_copy & (1 << v) and not (neighbors[pivot] & (1 << v)):
                    v_mask = 1 << v
                    bron_kerbosch_bitset(
                        R | v_mask,
                        P & neighbors[v],
                        X & neighbors[v]
                    )
                    P &= ~v_mask
                    X |= v_mask
        
        # Initial call with all nodes in P
        all_nodes = (1 << n) - 1
        bron_kerbosch_bitset(0, all_nodes, 0)
        
        # Convert bitset to list
        result = []
        for i in range(n):
            if max_clique & (1 << i):
                result.append(i)
        
        return result

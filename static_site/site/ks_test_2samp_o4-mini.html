# Kolmogorov-Smirnov two-sample test solver
from typing import Any
import numpy as np
import numba as nb
import scipy.stats as stats

@nb.njit
def _ks_statistic(x, y):
    n1 = x.shape[0]
    n2 = y.shape[0]
    i = 0
    j = 0
    d = 0.0
    while i < n1 and j < n2:
        if x[i] <= y[j]:
            i += 1
        else:
            j += 1
        diff = (i / n1) - (j / n2)
        if diff < 0:
            diff = -diff
        if diff > d:
            d = diff
    return d

# Precompile Numba function signature on dummy data to avoid JIT overhead in solve
_dummy_x = np.array([0.0], dtype=np.float64)
_dummy_y = np.array([0.0, 1.0], dtype=np.float64)
_dummy = _ks_statistic(_dummy_x, _dummy_y)

class Solver:
    def solve(self, problem: dict[str, Any], **kwargs) -> dict[str, Any]:
        sample1 = problem["sample1"]
        sample2 = problem["sample2"]
        # convert to numpy arrays of float64
        x = np.array(sample1, dtype=np.float64)
        y = np.array(sample2, dtype=np.float64)
        # sort arrays in-place
        x.sort()
        y.sort()
        # compute KS statistic
        d = _ks_statistic(x, y)
        n1 = x.shape[0]
        n2 = y.shape[0]
        # compute p-value
        if max(n1, n2) <= 50:
            # small sample: use exact method for p-value
            res = stats.ks_2samp(sample1, sample2, method="exact")
            p = float(res.pvalue)
        else:
            # asymptotic distribution
            en = n1 * n2 / (n1 + n2)
            p = float(stats.kstwobign.sf(d * np.sqrt(en)))
        return {"statistic": float(d), "pvalue": p}

# Kolmogorov-Smirnov two-sample test solver
from typing import Any
import numpy as np
import numba as nb
import scipy.stats as stats

@nb.njit
def _ks_statistic(x, y):
    n1 = x.shape[0]
    n2 = y.shape[0]
    i = 0
    j = 0
    d = 0.0
    while i < n1 and j < n2:
        if x[i] <= y[j]:
            i += 1
        else:
            j += 1
        diff = (i / n1) - (j / n2)
        if diff < 0:
            diff = -diff
        if diff > d:
            d = diff
    return d

def _ks_sf(t: float) -> float:
    """
    Asymptotic survival function for Kolmogorov distribution:
    P(D > t) = 2 * sum_{k=1..∞} (-1)^(k-1) * exp(-2 * k^2 * t^2)
    """
    two_t2 = 2.0 * t * t
    total = 0.0
    sign = 1.0
    k = 1
    # convergence threshold
    tol = 1e-15
    while True:
        term = sign * math.exp(-two_t2 * k * k)
        total += term
        if abs(term) < tol:
            break
        sign = -sign
        k += 1
    return 2.0 * total

# Kolmogorov-Smirnov two-sample test solver with Numba acceleration and exact p-value via DP
from typing import Any
import numpy as np
import math
from numba import njit
import scipy.stats as stats

@njit
def _ks_stat(x: np.ndarray, y: np.ndarray) -> float:
    """Compute two-sample KS statistic (two-sided) for sorted x, y."""
    n1 = x.shape[0]
    n2 = y.shape[0]
    inv_n1 = 1.0 / n1
    inv_n2 = 1.0 / n2
    i = 0
    j = 0
    d = 0.0
    # merge-like scan
    while i < n1 and j < n2:
        if x[i] <= y[j]:
            i += 1
        else:
            j += 1
        diff = i * inv_n1 - j * inv_n2
        if diff < 0.0:
            diff = -diff
        if diff > d:
            d = diff
    # finish tails
    while i < n1:
        i += 1
        diff = i * inv_n1 - j * inv_n2
        if diff < 0.0:
            diff = -diff
        if diff > d:
            d = diff
    while j < n2:
        j += 1
        diff = i * inv_n1 - j * inv_n2
        if diff < 0.0:
            diff = -diff
        if diff > d:
            d = diff
    return d

@njit
def _ks_pvalue_exact(d: float, n1: int, n2: int) -> float:
    """Exact p-value for KS statistic using DP on prefix probabilities."""
    m = n1 + n2
    threshold = d * n1 * n2
    b = int(threshold)
    # strict barrier: |i*n2 - j*n1| < threshold => <= b-1
    b0 = b - 1
    # dp[i,j] = P(prefix at (i,j) without hitting barrier)
    dp = np.zeros((n1+1, n2+1), dtype=np.float64)
    dp[0, 0] = 1.0
    for i in range(n1+1):
        for j in range(n2+1):
            if i == 0 and j == 0:
                continue
            if abs(i * n2 - j * n1) <= b0:
                prev_total = m - (i + j) + 1
                val = 0.0
                if i > 0:
                    val += dp[i-1, j] * (n1 - (i - 1)) / prev_total
                if j > 0:
                    val += dp[i, j-1] * (n2 - (j - 1)) / prev_total
                dp[i, j] = val
            else:
                dp[i, j] = 0.0
    # P(D < d) = dp[n1, n2]
    return 1.0 - dp[n1, n2]

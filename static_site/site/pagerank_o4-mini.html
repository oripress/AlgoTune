@njit
def _pagerank_csr(indptr: np.ndarray,
                  indices: np.ndarray,
                  degrees: np.ndarray,
                  dangling_nodes: np.ndarray,
                  n: int,
                  alpha: float,
                  tol: float,
                  max_iter: int) -> np.ndarray:
    # Initialize rank vectors uniformly
    r = np.full(n, 1.0 / n, dtype=np.float64)
    new_r = np.empty(n, dtype=np.float64)
    teleport = (1.0 - alpha) / n
    for _ in range(max_iter):
        # Sum rank mass on dangling nodes
        dangling_sum = 0.0
        for idx in dangling_nodes:
            dangling_sum += r[idx]
        add_dangling = alpha * dangling_sum / n
        # Base value for each node: teleport + dangling share
        base = teleport + add_dangling
        for j in range(n):
            new_r[j] = base
        # Distribute rank contributions along outgoing links
        for i in range(n):
            di = degrees[i]
            if di > 0.0:
                coeff = alpha * r[i] / di
                start = indptr[i]
                end = indptr[i + 1]
                for ptr in range(start, end):
                    new_r[indices[ptr]] += coeff
        # Check convergence (L1 norm) and update r
        diff = 0.0
        for j in range(n):
            dval = new_r[j] - r[j]
            diff += -dval if dval < 0.0 else dval
            r[j] = new_r[j]
        if diff < tol:
            break
    # Normalize to sum to 1.0
    total = 0.0
    for j in range(n):
        total += r[j]
    if total > 0.0:
        inv_total = 1.0 / total
        for j in range(n):
            r[j] *= inv_total
    return r

import numpy as np
from scipy.sparse import csr_matrix
from typing import Any, Dict, List

class Solver:
    def __init__(self, alpha: float = 0.85, tol: float = 1e-6, max_iter: int = 100) -> None:
        self.alpha = alpha
        self.tol = tol
        self.max_iter = max_iter

    def solve(self, problem: Dict[str, List[List[int]]], **kwargs: Any) -> Dict[str, List[float]]:
        adj_list = problem.get("adjacency_list", [])
        n = len(adj_list)
        # Handle edge cases
        if n == 0:
            return {"pagerank_scores": []}
        if n == 1:
            return {"pagerank_scores": [1.0]}

        # Degrees and dangling mask
        deg_int = [len(nbrs) for nbrs in adj_list]
        deg = np.array(deg_int, dtype=float)
        dangling_mask = np.array(deg_int, dtype=bool)

        # Build sparse transition matrix P in CSR form: P[j,i] = 1/deg[i] if edge i->j
        rows: List[int] = []
        cols: List[int] = []
        data: List[float] = []
        for i, nbrs in enumerate(adj_list):
            d = deg[i]
            if d > 0.0:
                invd = 1.0 / d
                for j in nbrs:
                    rows.append(j)
                    cols.append(i)
                    data.append(invd)
        P = csr_matrix((data, (rows, cols)), shape=(n, n))

        # Power iteration
        r = np.full(n, 1.0 / n, dtype=float)
        teleport = (1.0 - self.alpha) / n
        for _ in range(self.max_iter):
            # Sparse mat-vec multiplication for link contributions
            pr = P.dot(r)
            # Dangling contribution
            if dangling_mask.any():
                dangling_sum = r[dangling_mask].sum()
            else:
                dangling_sum = 0.0
            # Combine contributions
            r_new = self.alpha * (pr + dangling_sum / n) + teleport
            # Check convergence (L1 norm)
            if np.abs(r_new - r).sum() < self.tol:
                r = r_new
                break
            r = r_new
        # Return as list
        return {"pagerank_scores": r.tolist()}

from typing import Any, Dict, List

class Solver:
    def __init__(self, alpha: float = 0.85, tol: float = 1e-6, max_iter: int = 100) -> None:
        self.alpha = alpha
        self.tol = tol
        self.max_iter = max_iter

    def solve(self, problem: Dict[str, List[List[int]]], **kwargs: Any) -> Dict[str, List[float]]:
        adj_list = problem.get("adjacency_list", [])
        n = len(adj_list)
        # Handle edge cases
        if n == 0:
            return {"pagerank_scores": []}
        if n == 1:
            return {"pagerank_scores": [1.0]}

        # Degrees and dangling mask
        deg_int = [len(nbrs) for nbrs in adj_list]
        deg = np.array(deg_int, dtype=float)
        dangling_mask = np.array(deg_int) == 0  # True for dangling nodes

        # Build sparse transition matrix P in CSR form: P[j,i] = 1/deg[i] if edge i->j
        rows: List[int] = []
        cols: List[int] = []
        data: List[float] = []
        for i, nbrs in enumerate(adj_list):
            d = deg[i]
            if d > 0.0:
                invd = 1.0 / d
                for j in nbrs:
                    rows.append(j)
                    cols.append(i)
                    data.append(invd)
        P = csr_matrix((data, (rows, cols)), shape=(n, n))

        # Power iteration
        r = np.full(n, 1.0 / n, dtype=float)
        teleport = (1.0 - self.alpha) / n
        for _ in range(self.max_iter):
            pr = P.dot(r)

import numpy as np
from scipy.sparse import csr_matrix
from typing import Any, Dict, List

class Solver:
    def __init__(self, alpha: float = 0.85, tol: float = 1e-6, max_iter: int = 100) -> None:
        self.alpha = alpha
        self.tol = tol
        self.max_iter = max_iter

    def solve(self, problem: Dict[str, List[List[int]]], **kwargs: Any) -> Dict[str, List[float]]:
        adj_list = problem.get("adjacency_list", [])
        n = len(adj_list)
        # Edge cases
        if n == 0:
            return {"pagerank_scores": []}
        if n == 1:
            return {"pagerank_scores": [1.0]}

        # Out-degree and dangling mask
        deg_int = [len(neighbors) for neighbors in adj_list]
        deg = np.array(deg_int, dtype=float)
        dangling_mask = np.array(deg_int) == 0

        # Build sparse transition matrix P (column-stochastic)
        rows: List[int] = []
        cols: List[int] = []
        data: List[float] = []
        for i, neighbors in enumerate(adj_list):
            d = deg[i]
            if d > 0.0:
                inv_d = 1.0 / d
                for j in neighbors:
                    rows.append(j)
                    cols.append(i)
                    data.append(inv_d)
        P = csr_matrix((data, (rows, cols)), shape=(n, n))

        # Initialize rank vector
        r = np.full(n, 1.0 / n, dtype=float)
        teleport = (1.0 - self.alpha) / n

        # Power iteration
        for _ in range(self.max_iter):
            # Link contributions
            pr = P.dot(r)
            # Dangling contributions
            if dangling_mask.any():

                pr += dangling_sum / n

            r_new = self.alpha * pr + teleport
            # Check convergence (L1 norm)
            if np.abs(r_new - r).sum() < self.tol:
                r = r_new
                break
            r = r_new

        # Normalize to sum to 1.0
        total = float(r.sum())

from scipy.sparse import csr_matrix
from typing import Any, Dict, List

class Solver:
    def __init__(self, alpha: float = 0.85, tol: float = 1e-6, max_iter: int = 100) -> None:
        """
        Initialize the PageRank solver.

        alpha: damping factor
        tol: L1 convergence tolerance
        max_iter: maximum number of iterations
        """
        self.alpha = alpha
        self.tol = tol
        self.max_iter = max_iter

    def solve(self, problem: Dict[str, List[List[int]]], **kwargs: Any) -> Dict[str, List[float]]:
        """
        Compute PageRank scores for the given directed graph adjacency list.
        """
        adj_list = problem.get("adjacency_list", [])
        n = len(adj_list)
        # Edge cases
        if n == 0:
            return {"pagerank_scores": []}
        if n == 1:
            return {"pagerank_scores": [1.0]}

        # Compute out-degrees
        deg = np.fromiter((len(neigh) for neigh in adj_list), count=n, dtype=np.int64)

        # Build sparse transition matrix P in CSR form:
        # P[j, i] = 1/deg[i] if edge i->j
        # Dangling nodes will be handled separately (no columns in P)
        rows: List[int] = []
        cols: List[int] = []
        data: List[float] = []
        for i, neigh in enumerate(adj_list):
            d = deg[i]
            if d > 0:
                inv_d = 1.0 / d
                for j in neigh:
                    rows.append(j)
                    cols.append(i)
                    data.append(inv_d)
        P = csr_matrix((data, (rows, cols)), shape=(n, n), dtype=np.float64)

        # Teleportation constant added each iteration
        teleport = (1.0 - self.alpha) / n

        # Initialize rank vector uniformly
        r = np.full(n, 1.0 / n, dtype=np.float64)

        # Identify dangling nodes (out-degree zero)
        if np.any(deg == 0):
            dangling_mask = (deg == 0)
        else:
            dangling_mask = None

        # Power iteration
        for _ in range(self.max_iter):
            # P @ r gives contributions from linked nodes
            pr = P.dot(r)

            # Dangling mass distributed uniformly
            if dangling_mask is not None:
                ds = r[dangling_mask].sum() / n
            else:
                ds = 0.0

            # Next rank values
            r_next = self.alpha * (pr + ds) + teleport

            # Check L1 convergence
            diff = np.abs(r_next - r).sum()
            r = r_next
            if diff < self.tol:
                break

        # Normalize to sum to 1 (handle numerical drift)
        total = r.sum()
        if total > 0.0:
            r = r / total

        return {"pagerank_scores": r.tolist()}

import cvxpy as cp
import numpy as np

class MultiDimKnapsackInstance(NamedTuple):
    value: List[int]
    demand: List[List[int]]
    supply: List[int]

class Solver:
    def solve(self, problem, **kwargs) -> Any:
        """Solve the multi-dimensional knapsack problem using CVXPY."""
        if not isinstance(problem, MultiDimKnapsackInstance):
            try:
                problem = MultiDimKnapsackInstance(*problem)
            except Exception:
                return []
        
        n = len(problem.value)
        k = len(problem.supply)
        
        # Convert to numpy arrays
        values = np.array(problem.value)
        demands = np.array(problem.demand)
        supplies = np.array(problem.supply)
        
        # Decision variables
        x = cp.Variable(n, boolean=True)
        
        # Objective
        objective = cp.Maximize(values @ x)
        
        # Constraints
        constraints = [demands.T @ x <= supplies]
        
        # Solve
        prob = cp.Problem(objective, constraints)
        prob.solve(solver=cp.ECOS_BB, verbose=False)
        
        if prob.status == cp.OPTIMAL:
            return [i for i in range(n) if x.value[i] > 0.5]
        return []

    demand: List[List[int]]

@numba.njit
def dp_solve_small(values, demands, supplies, n, k):
    """Dynamic programming for small instances."""
    # Convert to numpy arrays for numba
    max_supply = np.max(supplies)
    
    # For very small instances, try brute force
    if n <= 20:
        best_value = 0
        best_mask = 0
        
        for mask in range(1 << n):
            valid = True
            current_value = 0
            
            # Check constraints
            for r in range(k):
                usage = 0
                for i in range(n):
                    if mask & (1 << i):
                        usage += demands[i, r]
                if usage > supplies[r]:
                    valid = False
                    break
            
            if valid:
                for i in range(n):
                    if mask & (1 << i):
                        current_value += values[i]
                
                if current_value > best_value:
                    best_value = current_value
                    best_mask = mask
        
        return best_mask
    
    return -1  # Too large for DP

import highspy

class MultiDimKnapsackInstance(NamedTuple):
    value: List[int]
    demand: List[List[int]]
    supply: List[int]

class Solver:
    def solve(self, problem, **kwargs) -> Any:
        """Solve the multi-dimensional knapsack problem using HiGHS."""
        if not isinstance(problem, MultiDimKnapsackInstance):
            try:
                problem = MultiDimKnapsackInstance(*problem)
            except Exception:
                return []
        
        n = len(problem.value)
        k = len(problem.supply)
        
        # Create HiGHS model
        h = highspy.Highs()
        h.setOptionValue("output_flag", False)
        h.setOptionValue("presolve", "on")
        
        # Variables: binary variables for each item
        inf = highspy.kHighsInf
        lower = [0] * n
        upper = [1] * n
        integrality = [highspy.HighsVarType.kInteger] * n
        
        # Objective: maximize value (HiGHS minimizes, so negate)
        cost = [-v for v in problem.value]
        
        # Add variables
        h.addVars(n, lower, upper, integrality, cost)
        
        # Add constraints: one for each resource
        for r in range(k):
            # Build constraint row
            indices = list(range(n))
            values = [problem.demand[i][r] for i in range(n)]
            
            h.addRow(0, problem.supply[r], len(indices), indices, values)
        
        # Solve
        h.run()
        
        if h.getModelStatus() == highspy.HighsModelStatus.kOptimal:
            solution = h.getSolution()
            return [i for i in range(n) if solution.col_value[i] > 0.5]
        
        return []

from typing import Any, List, Tuple, NamedTuple
import cvxpy as cp
import numpy as np

class MultiDimKnapsackInstance(NamedTuple):
    value: List[int]
    demand: List[List[int]]
    supply: List[int]

class Solver:
    def solve(self, problem, **kwargs) -> Any:
        """Solve the multi-dimensional knapsack problem using CVXPY."""
        if not isinstance(problem, MultiDimKnapsackInstance):
            try:
                problem = MultiDimKnapsackInstance(*problem)
            except Exception:
                return []
        
        n = len(problem.value)
        k = len(problem.supply)
        
        # Convert to numpy arrays
        values = np.array(problem.value)
        demands = np.array(problem.demand)
        supplies = np.array(problem.supply)
        
        # Decision variables
        x = cp.Variable(n, boolean=True)
        
        # Objective
        objective = cp.Maximize(values @ x)
        
        # Constraints
        constraints = [demands.T @ x <= supplies]
        
        # Solve
        prob = cp.Problem(objective, constraints)
        prob.solve(solver=cp.GLPK_MI, verbose=False)
        
        if prob.status == cp.OPTIMAL:
            return [i for i in range(n) if x.value[i] > 0.5]
        return []

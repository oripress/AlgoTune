from typing import Any, List, Set, Tuple

class Solver:
    def solve(self, problem, **kwargs) -> Any:
        """
        Solve the set cover with conflicts problem using a greedy approach.
        
        Args:
            problem: A tuple (n, sets, conflicts) where:
                - n is the number of objects
                - sets is a list of sets (each set is a list of integers)
                - conflicts is a list of conflicts (each conflict is a list of set indices)
        
        Returns:
            A list of set indices that form a valid cover
        """
        n, sets, conflicts = problem
        
        # Convert conflicts to a more efficient representation
        conflict_map = {}
        for conflict in conflicts:
            for idx in conflict:
                if idx not in conflict_map:
                    conflict_map[idx] = []
                conflict_map[idx].extend([c for c in conflict if c != idx])
        
        # Track which objects are covered
        uncovered = set(range(n))
        selected = []
        forbidden = set()
        
        while uncovered:
            best_set = -1
            best_coverage = -1
            
            # Find the set that covers the most uncovered objects
            for i in range(len(sets)):
                if i in selected or i in forbidden:
                    continue
                
                # Count how many uncovered objects this set covers
                coverage = len([obj for obj in sets[i] if obj in uncovered])
                
                if coverage > best_coverage:
                    best_coverage = coverage
                    best_set = i
            
            if best_set == -1:
                # No valid set found, should not happen with trivial sets
                raise ValueError("No feasible solution found")
            
            # Select this set
            selected.append(best_set)
            
            # Update covered objects
            for obj in sets[best_set]:
                if obj in uncovered:
                    uncovered.remove(obj)
            
            # Update forbidden sets based on conflicts
            if best_set in conflict_map:
                forbidden.update(conflict_map[best_set])
        
        return selected

    def solve(self, problem, **kwargs) -> Any:
        """
        Solve the set cover with conflicts problem using branch and bound.
        
        Args:
            problem: A tuple (n, sets, conflicts) where:
                - n is the number of objects
                - sets is a list of sets (each set is a list of integers)
                - conflicts is a list of conflicts (each conflict is a list of set indices)
        
        Returns:
            A list of set indices that form a valid cover
        """
        n, sets, conflicts = problem
        
        # Precompute conflict adjacency for faster lookup
        conflict_adj = [set() for _ in range(len(sets))]
        for conflict in conflicts:
            for i in conflict:
                for j in conflict:
                    if i != j:
                        conflict_adj[i].add(j)
        
        # Precompute which sets cover each object
        sets_by_object = [[] for _ in range(n)]
        for i, s in enumerate(sets):
            for obj in s:
                sets_by_object[obj].append(i)

        best_solution = self._greedy_solution(n, sets, conflict_adj)
        best_size = len(best_solution)

class Solver:
    def solve(self, problem, **kwargs) -> Any:
        """
        Solve the set cover with conflicts problem using a fast hybrid approach.
        
        Args:
            problem: A tuple (n, sets, conflicts) where:
                - n is the number of objects
                - sets is a list of sets (each set is a list of integers)
                - conflicts is a list of conflicts (each conflict is a list of set indices)
        
        Returns:
            A list of set indices that form a valid cover
        """
        n, sets, conflicts = problem
        
        # Convert sets to frozensets for faster operations
        frozen_sets = [frozenset(s) for s in sets]
        
        # Build conflict graph
        conflict_graph = [set() for _ in range(len(sets))]
        for conflict in conflicts:
            for i in conflict:
                for j in conflict:
                    if i != j:
                        conflict_graph[i].add(j)
        
        # Precompute set coverage info
        sets_by_object = [[] for _ in range(n)]
        for i, s in enumerate(sets):
            for obj in s:
                sets_by_object[obj].append(i)
        
        # Get initial greedy solution
        solution = self._greedy_with_conflicts(n, frozen_sets, conflict_graph)
        
        # Try to improve with local search
        solution = self._local_search(solution, n, frozen_sets, conflict_graph, sets_by_object)
        
        return solution
    
    def _greedy_with_conflicts(self, n, sets, conflict_graph):
        """Fast greedy algorithm considering conflicts."""
        uncovered = set(range(n))
        selected = []
        forbidden = set()
        
        # Precompute set sizes
        set_sizes = [len(s) for s in sets]
        
        while uncovered:
            best_set = -1
            best_score = -1

            for i in range(len(sets)):
                if i in forbidden or i in selected:
                    continue
                
                # Count uncovered objects this set would cover
                coverage = len(sets[i] & uncovered)
                
                if coverage > 0:
                    # Prefer larger coverage
                    score = coverage
                    if score > best_score:
                        best_score = score
                        best_set = i
            
            if best_set == -1:
                # No valid set found, use trivial sets for remaining
                for obj in uncovered:
                    if obj not in forbidden:
                        selected.append(obj)
                break
            
            # Select the best set
            selected.append(best_set)
            uncovered -= sets[best_set]
            
            # Update forbidden sets
            forbidden.update(conflict_graph[best_set])
        
        return selected
    
    def _local_search(self, solution, n, sets, conflict_graph, sets_by_object):
        """Improve solution using local search."""
        current = set(solution)
        improved = True
        
        while improved:
            improved = False

            for s in list(current):

import pulp

class Solver:
    def solve(self, problem, **kwargs) -> Any:
        """
        Solve the set cover with conflicts problem using PuLP.
        
        Args:
            problem: A tuple (n, sets, conflicts) where:
                - n is the number of objects
                - sets is a list of sets (each set is a list of integers)
                - conflicts is a list of conflicts (each conflict is a list of set indices)
        
        Returns:
            A list of set indices that form a valid cover
        """
        n, sets, conflicts = problem
        
        # Create the LP problem
        prob = pulp.LpProblem("SetCoverWithConflicts", pulp.LpMinimize)
        
        # Binary variables for each set
        x = [pulp.LpVariable(f"x_{i}", cat='Binary') for i in range(len(sets))]
        
        # Objective: minimize number of selected sets
        prob += pulp.lpSum(x)
        
        # Coverage constraints: each object must be covered
        for obj in range(n):
            covering_sets = [i for i, s in enumerate(sets) if obj in s]
            prob += pulp.lpSum([x[i] for i in covering_sets]) >= 1
        
        # Conflict constraints: conflicting sets cannot both be selected
        for conflict in conflicts:
            prob += pulp.lpSum([x[i] for i in conflict]) <= len(conflict) - 1
        
        # Solve
        prob.solve(pulp.PULP_CBC_CMD(msg=0))
        
        # Extract solution
        if prob.status == pulp.LpStatusOptimal:
            solution = [i for i in range(len(sets)) if x[i].varValue > 0.5]
            return solution
        else:
            raise ValueError("No feasible solution found")

    def solve(self, problem, **kwargs) -> Any:
        """
        Solve the set cover with conflicts problem using optimized PuLP with preprocessing.
        
        Args:
            problem: A tuple (n, sets, conflicts) where:
                - n is the number of objects
                - sets is a list of sets (each set is a list of integers)
                - conflicts is a list of conflicts (each conflict is a list of set indices)
        
        Returns:
            A list of set indices that form a valid cover
        """
        n, sets, conflicts = problem
        
        # Preprocessing: find dominated sets
        useful_sets = self._preprocess(sets, conflicts)
        
        # If preprocessing reduced to trivial case
        if len(useful_sets) == n:
            return list(range(n))
        
        # Create the LP problem
        prob = pulp.LpProblem("SetCoverWithConflicts", pulp.LpMinimize)
        
        # Binary variables only for useful sets
        x = {}
        for i in useful_sets:
            x[i] = pulp.LpVariable(f"x_{i}", cat='Binary')
        
        # Objective: minimize number of selected sets
        prob += pulp.lpSum(x.values())
        
        # Coverage constraints: each object must be covered
        for obj in range(n):
            covering_sets = [i for i in useful_sets if obj in sets[i]]
            if covering_sets:
                prob += pulp.lpSum([x[i] for i in covering_sets]) >= 1
        
        # Conflict constraints: conflicting sets cannot both be selected
        for conflict in conflicts:
            conflict_vars = [x[i] for i in conflict if i in x]
            if len(conflict_vars) > 1:
                prob += pulp.lpSum(conflict_vars) <= len(conflict_vars) - 1
        
        # Solve with HiGHS
        try:
            solver = pulp.HiGHS_CMD(msg=0, options=['presolve on', 'parallel on'])
            prob.solve(solver)
        except:
            # Fallback to CBC if HiGHS fails
            prob.solve(pulp.PULP_CBC_CMD(msg=0))
        
        # Extract solution
        if prob.status == pulp.LpStatusOptimal:
            solution = [i for i in useful_sets if x[i].varValue > 0.5]
            return solution
        else:
            # Fallback to trivial solution
            return list(range(n))
    
    def _preprocess(self, sets, conflicts):
        """Remove dominated sets to reduce problem size."""
        n_sets = len(sets)
        useful = set(range(n_sets))
        
        # Build conflict sets for each set
        conflict_map = [set() for _ in range(n_sets)]
        for conflict in conflicts:
            for i in conflict:
                for j in conflict:
                    if i != j:
                        conflict_map[i].add(j)
        
        # Check for dominated sets
        for i in range(n_sets):
            if i not in useful:
                continue
            for j in range(i + 1, n_sets):
                if j not in useful:
                    continue
                
                set_i = set(sets[i])
                set_j = set(sets[j])
                
                # Check if i dominates j (i covers everything j covers and has no more conflicts)
                if set_j.issubset(set_i) and conflict_map[i].issubset(conflict_map[j]):

                # Check if j dominates i
                elif set_i.issubset(set_j) and conflict_map[j].issubset(conflict_map[i]):

                    break

from ortools.sat.python import cp_model

class Solver:
    def solve(self, problem: dict[str, Any]) -> list[list[int]]:
        D = problem["D"]
        K = problem["K"]
        depot = problem["depot"]
        n = len(D)
        model = cp_model.CpModel()

        # x[i,j] = 1 if arc i->j is used
        x = {}
        for i in range(n):
            for j in range(n):
                if i != j:
                    x[(i, j)] = model.NewBoolVar(f"x_{i}_{j}")

        # Each non-depot node must be entered and left exactly once
        for i in range(n):
            if i == depot:
                continue
            model.Add(sum(x[(j, i)] for j in range(n) if j != i) == 1)
            model.Add(sum(x[(i, j)] for j in range(n) if j != i) == 1)

        # Depot must have exactly K departures and arrivals
        model.Add(sum(x[(depot, j)] for j in range(n) if j != depot) == K)
        model.Add(sum(x[(i, depot)] for i in range(n) if i != depot) == K)

        # MTZ subtour elimination variables
        u = {}
        for i in range(n):
            if i == depot:
                continue
            u[i] = model.NewIntVar(1, n - 1, f"u_{i}")

        # MTZ constraints
        for i in range(n):
            if i == depot:
                continue
            for j in range(n):
                if j == depot or i == j:
                    continue
                model.Add(u[i] + 1 <= u[j] + (n - 1) * (1 - x[(i, j)]))

        # Objective: minimize total distance
        model.Minimize(sum(D[i][j] * x[(i, j)] for i, j in x))

        solver = cp_model.CpSolver()
        status = solver.Solve(model)

        if status != cp_model.OPTIMAL:
            return []

        # Reconstruct routes by following arcs from depot
        routes: list[list[int]] = []
        for j in range(n):
            if j != depot and solver.Value(x[(depot, j)]) == 1:
                route = [depot, j]
                current = j
                while current != depot:
                    for k in range(n):
                        if current != k and solver.Value(x[(current, k)]) == 1:
                            route.append(k)
                            current = k
                            break
                routes.append(route)
        return routes

from typing import Any
from ortools.constraint_solver import pywrapcp, routing_enums_pb2

class Solver:
    def solve(self, problem: dict[str, Any]) -> list[list[int]]:
        D = problem["D"]
        K = problem["K"]
        depot = problem["depot"]
        n = len(D)
        # Create Routing Index Manager
        manager = pywrapcp.RoutingIndexManager(n, K, depot)
        routing = pywrapcp.RoutingModel(manager)
        # Create and register a transit callback
        def distance_callback(from_index, to_index):
            from_node = manager.IndexToNode(from_index)
            to_node = manager.IndexToNode(to_index)
            # Use integer distances; assume D entries are ints or cast safely
            return int(D[from_node][to_node])
        transit_callback_index = routing.RegisterTransitCallback(distance_callback)
        routing.SetArcCostEvaluatorOfAllVehicles(transit_callback_index)
        # First solution and local search parameters
        search_parameters = pywrapcp.DefaultRoutingSearchParameters()
        search_parameters.first_solution_strategy = routing_enums_pb2.FirstSolutionStrategy.PATH_CHEAPEST_ARC
        search_parameters.local_search_metaheuristic = routing_enums_pb2.LocalSearchMetaheuristic.GUIDED_LOCAL_SEARCH
        # Time limit to speed up but still find optimum for small instances
        search_parameters.time_limit.seconds = 1
        # Solve
        solution = routing.SolveWithParameters(search_parameters)
        # Handle no solution
        if solution is None:
            return [[depot, depot] for _ in range(K)]
        # Build routes
        routes: list[list[int]] = []
        for vehicle_id in range(K):
            index = routing.Start(vehicle_id)
            route = []
            while not routing.IsEnd(index):
                route.append(manager.IndexToNode(index))
                index = solution.Value(routing.NextVar(index))
            route.append(manager.IndexToNode(index))
            routes.append(route)
        return routes

from typing import Any

class Solver:
    def solve(self, problem: dict[str, Any]) -> list[list[int]]:
        D = problem["D"]
        K = problem["K"]
        depot = problem["depot"]
        n = len(D)
        # all customers except depot
        customers = [i for i in range(n) if i != depot]
        m = len(customers)
        # trivial cases
        if m == 0:
            # no customers, K empty routes
            return [[depot, depot] for _ in range(K)]
        # initial routes: each customer alone
        # route_nodes maps route_id to list of customers in order
        route_nodes = {c: [c] for c in customers}
        # mapping for each customer to its route_id
        route_of = {c: c for c in customers}
        # if more vehicles than customers, assign one customer per vehicle, rest empty
        if m <= K:
            routes = []
            for c in customers:
                routes.append([depot, c, depot])
            # add empty routes
            for _ in range(K - m):
                routes.append([depot, depot])
            return routes
        # compute savings for each pair (i, j)
        savings = []
        for idx, i in enumerate(customers):
            di0 = D[depot][i]
            for j in customers[idx+1:]:
                save = di0 + D[depot][j] - D[i][j]
                savings.append((save, i, j))
        # sort savings descending
        savings.sort(reverse=True, key=lambda x: x[0])
        # perform merges until we have K routes
        merges_needed = m - K
        merges = 0
        for save, i, j in savings:
            if merges >= merges_needed:
                break
            # find current routes
            ri = route_of[i]
            rj = route_of[j]
            if ri == rj:
                continue
            seq_i = route_nodes[ri]
            seq_j = route_nodes[rj]
            # only merge when i is at tail of its route and j at head of its
            if seq_i[-1] == i and seq_j[0] == j:
                new_seq = seq_i + seq_j
            # or j tail and i head
            elif seq_j[-1] == j and seq_i[0] == i:
                new_seq = seq_j + seq_i
            else:
                continue
            # apply merge
            for node in seq_j:
                route_of[node] = ri
            route_nodes[ri] = new_seq
            del route_nodes[rj]
            merges += 1
        # fallback arbitrary merges if insufficient
        while len(route_nodes) > K:
            # take two arbitrary routes
            keys = list(route_nodes.keys())
            ra, rb = keys[0], keys[1]
            seq_a = route_nodes[ra]
            seq_b = route_nodes[rb]
            # merge a <- b
            new_seq = seq_a + seq_b
            for node in seq_b:
                route_of[node] = ra
            route_nodes[ra] = new_seq
            del route_nodes[rb]
        # build final routes
        routes = []
        for seq in route_nodes.values():
            route = [depot] + seq + [depot]
            routes.append(route)
        return routes

            return [[depot, depot] for _ in range(K)]

        manager = pywrapcp.RoutingIndexManager(n, K, depot)
        routing = pywrapcp.RoutingModel(manager)

        routing.SetArcCostEvaluatorOfAllVehicles(transit_idx)

    def solve(self, problem: dict[str, Any]) -> list[list[int]]:
        D = problem["D"]
        K = problem["K"]
        depot = problem["depot"]
        n = len(D)
        # trivial case: no customers
        if n - 1 == 0:
            return [[depot, depot] for _ in range(K)]
        # Build CP-SAT model
        model = cp_model.CpModel()
        # Binary variables x[i][j]: arc i->j used
        x = [[None] * n for _ in range(n)]
        for i in range(n):
            for j in range(n):
                if i != j:
                    x[i][j] = model.NewBoolVar(f"x_{i}_{j}")
        # Each non-depot node enters and leaves exactly once
        for i in range(n):
            if i == depot:
                continue
            model.Add(sum(x[j][i] for j in range(n) if j != i) == 1)
            model.Add(sum(x[i][j] for j in range(n) if j != i) == 1)
        # Depot has K departures and K arrivals
        model.Add(sum(x[depot][j] for j in range(n) if j != depot) == K)
        model.Add(sum(x[i][depot] for i in range(n) if i != depot) == K)

        u = [None] * n

                u[i] = model.NewIntVar(1, n - 1, f"u_{i}")
        # MTZ constraints
        limit = n - 1
        for i in range(n):
            if i == depot:
                continue
            for j in range(n):
                if j == depot or j == i:
                    continue
                model.Add(u[i] + 1 <= u[j] + limit * (1 - x[i][j]))

from ortools.sat.python import cp_model

class Solver:
    def solve(self, problem: dict[str, Any]) -> list[list[int]]:
        D = problem["D"]
        K = problem["K"]
        depot = problem["depot"]
        n = len(D)
        # trivial case: no customers
        if n <= 1:
            return [[depot, depot] for _ in range(K)]
        # Build CP-SAT model
        model = cp_model.CpModel()
        # Binary decision variables x[i][j]
        x = [[None] * n for _ in range(n)]
        for i in range(n):
            for j in range(n):
                if i != j:
                    x[i][j] = model.NewBoolVar(f"x_{i}_{j}")
        # Each non-depot must have exactly one predecessor and successor
        for i in range(n):
            if i == depot:
                continue
            model.Add(sum(x[j][i] for j in range(n) if j != i) == 1)
            model.Add(sum(x[i][j] for j in range(n) if j != i) == 1)
        # Depot has K departures and arrivals
        model.Add(sum(x[depot][j] for j in range(n) if j != depot) == K)
        model.Add(sum(x[i][depot] for i in range(n) if i != depot) == K)
        # MTZ subtour elimination variables
        u = [None] * n
        for i in range(n):
            if i != depot:
                u[i] = model.NewIntVar(1, n - 1, f"u_{i}")
        # MTZ constraints
        for i in range(n):
            if i == depot:
                continue
            for j in range(n):
                if j == depot or j == i:
                    continue
                model.Add(u[i] + 1 <= u[j] + (n - 1) * (1 - x[i][j]))
        # Objective: minimize total distance
        terms = []
        for i in range(n):
            for j in range(n):
                if i != j:
                    terms.append(D[i][j] * x[i][j])
        model.Minimize(sum(terms))
        # Solve model
        solver = cp_model.CpSolver()
        solver.parameters.num_search_workers = multiprocessing.cpu_count() or 1
        status = solver.Solve(model)
        if status != cp_model.OPTIMAL:
            return []
        # Reconstruct routes
        routes: list[list[int]] = []
        for j in range(n):
            if j != depot and solver.Value(x[depot][j]) == 1:
                route = [depot, j]
                current = j
                while current != depot:
                    for k in range(n):
                        if k != current and solver.Value(x[current][k]) == 1:
                            route.append(k)
                            current = k
                            break
                routes.append(route)
        return routes

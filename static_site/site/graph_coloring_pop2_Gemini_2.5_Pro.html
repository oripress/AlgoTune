    A fast and optimal graph coloring solver.
    """

    def solve(self, adj_matrix):

        Finds an optimal graph coloring using a hybrid, multi-stage approach.
        1.  Uses networkx for fast graph representation and algorithms.
        2.  Quickly solves easy cases (trivial, bipartite).
        3.  Tightly bounds the chromatic number using fast, strong heuristics
            (DSATUR for upper bound, max clique for lower bound).
        4.  If bounds meet, the solution is optimal and found without the solver.
        5.  For hard cases, falls back to a parallelized, time-limited CP-SAT

        if num_nodes == 0:
            return []

            return [0] * num_nodes

            return [colors_dict[i] for i in range(num_nodes)]

        greedy_solution = [greedy_coloring_dict[i] for i in range(num_nodes)]

import sys
import networkx as nx
from ortools.sat.python import cp_model

class Solver:
    """
    A fast and optimal graph coloring solver.
    """

    def solve(self, adj_matrix):
        """
        Finds an optimal graph coloring using a hybrid, multi-stage approach.
        1.  Uses networkx for fast graph representation and algorithms.
        2.  Quickly solves easy cases (trivial, bipartite).
        3.  Gets a tight upper bound from a fast DSATUR greedy heuristic.
        4.  Uses a simple, fast lower bound (3 for non-bipartite graphs).
        5.  For hard cases, falls back to a parallelized, time-limited CP-SAT
            model, searching downwards from the heuristic upper bound.
        """
        num_nodes = len(adj_matrix)
        if num_nodes == 0:
            return []

        # 1. Build networkx graph for efficient, high-level operations.
        G = nx.Graph()
        edges = []
        for i in range(num_nodes):
            for j in range(i + 1, num_nodes):
                if adj_matrix[i][j]:
                    edges.append((i, j))
        G.add_nodes_from(range(num_nodes))
        G.add_edges_from(edges)

        # 2. Fast path for trivial and easy cases.
        if not edges:
            return [0] * num_nodes

        if nx.is_bipartite(G):
            colors_dict = nx.bipartite.color(G)
            return [colors_dict[i] for i in range(num_nodes)]

        # 3. Bounding with a fast heuristic (upper) and a constant (lower).
        greedy_coloring_dict = nx.coloring.greedy_color(G, strategy="DSATUR")
        greedy_solution = [greedy_coloring_dict[i] for i in range(num_nodes)]
        upper_bound = max(greedy_solution) + 1 if greedy_solution else 1

        # Since we've checked for bipartiteness, the lower bound is at least 3.
        # This avoids the expensive and often slow max-clique calculation.
        lower_bound = 3

        # 4. Check for optimality. If DSATUR found a 3-coloring, it's optimal.
        if upper_bound <= lower_bound:
            return greedy_solution

        # 5. CP-SAT solver for the remaining cases.
        best_solution = greedy_solution
        node_order = sorted(range(num_nodes), key=lambda i: G.degree[i], reverse=True)

        # Search downwards from the heuristic solution to the lower bound.
        for k in range(upper_bound - 1, lower_bound - 1, -1):
            model = cp_model.CpModel()
            colors = [model.NewIntVar(0, k - 1, f'c_{i}') for i in range(num_nodes)]

            for i in range(num_nodes):
                if best_solution[i] < k:
                    model.AddHint(colors[i], best_solution[i])

            for i, j in G.edges():
                model.Add(colors[i] != colors[j])

            if num_nodes > 0:
                model.Add(colors[node_order[0]] == 0)
            for i in range(1, num_nodes):
                node_idx = node_order[i]
                previous_nodes_colors = [colors[node_order[j]] for j in range(i)]
                
                if previous_nodes_colors:
                    max_prev_color = model.NewIntVar(0, k - 1, f'max_prev_{i}')
                    model.AddMaxEquality(max_prev_color, previous_nodes_colors)
                    model.Add(colors[node_idx] <= max_prev_color + 1)

            solver = cp_model.CpSolver()
            solver.parameters.max_time_in_seconds = 4.0
            solver.parameters.num_search_workers = 4
            status = solver.Solve(model)

            if status == cp_model.FEASIBLE or status == cp_model.OPTIMAL:
                best_solution = [solver.Value(c) for c in colors]
            else:
                return best_solution
        
        return best_solution

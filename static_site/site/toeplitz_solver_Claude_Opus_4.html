import numpy as np
from numba import njit, prange

@njit(cache=True)
def levinson_numba(c, r, b):
    """Fast Levinson algorithm implementation using Numba."""
    n = len(c)
    x = np.zeros(n, dtype=np.float64)
    y = np.zeros(n, dtype=np.float64)
    
    # Initialize
    x[0] = b[0] / c[0]
    if n == 1:
        return x
    
    # Forward substitution
    alpha = c[0]
    beta = r[1]
    y[0] = -r[1] / c[0]
    x[1] = (b[1] - beta * x[0]) / (alpha + beta * y[0])
    
    for k in range(1, n-1):
        # Update alpha and beta
        alpha_new = alpha + beta * np.conj(y[k-1])
        beta_new = 0.0
        for j in range(k+1):
            beta_new += r[k+1-j] * y[j]
        
        # Update y
        y_new = np.zeros(k+1, dtype=np.float64)
        y_new[0] = -beta_new / alpha_new
        for j in range(k):
            y_new[j+1] = (y[j] - beta_new * np.conj(y[k-1-j])) / alpha_new
        
        # Copy new y values
        for j in range(k+1):
            y[j] = y_new[j]
        
        # Update x
        sigma = 0.0
        for j in range(k+1):
            sigma += c[k+1-j] * x[j]
        x[k+1] = (b[k+1] - sigma) / alpha_new
        
        # Update for next iteration
        alpha = alpha_new
        beta = beta_new
    
    return x

class Solver:
    def solve(self, problem: dict[str, list[float]]) -> list[float]:
        """Solve the linear system Tx = b using custom Levinson algorithm."""
        c = np.asarray(problem["c"], dtype=np.float64)
        r = np.asarray(problem["r"], dtype=np.float64)
        b = np.asarray(problem["b"], dtype=np.float64)

        if c[0] != r[0]:
            raise ValueError("c[0] must equal r[0] for a valid Toeplitz matrix")

import numpy as np
import jax
import jax.numpy as jnp
from jax import jit

@jit
def levinson_jax(c, r, b):
    """Levinson algorithm using JAX for JIT compilation."""
    n = len(c)
    
    # Initialize
    x = jnp.zeros(n)
    x = x.at[0].set(b[0] / c[0])
    
    if n == 1:
        return x
    
    # Forward and backward vectors
    f = jnp.zeros(n)
    g = jnp.zeros(n) 
    f = f.at[0].set(1.0 / c[0])
    g = g.at[0].set(1.0 / c[0])
    
    for k in range(1, n):
        # Calculate epsilon and gamma
        epsilon = r[k]
        gamma = c[k]
        
        for j in range(k):
            epsilon += r[k-j] * g[j]
            gamma += c[k-j] * f[j]
        
        # Update f and g
        f_new = jnp.zeros(k+1)
        g_new = jnp.zeros(k+1)
        
        f_new = f_new.at[0].set(-epsilon / c[0])
        g_new = g_new.at[k].set(-gamma / c[0])
        
        for j in range(k):
            f_new = f_new.at[j+1].add(f[j] - epsilon * g[k-1-j])
            g_new = g_new.at[j].add(g[j] - gamma * f[k-1-j])
        
        denom = 1.0 - epsilon * gamma
        f = f_new / denom
        g = g_new / denom
        
        # Update x
        sigma = b[k]
        for j in range(k):
            sigma -= c[k-j] * x[j]
        
        for j in range(k):
            x = x.at[j].add(sigma * g[k-1-j])
        x = x.at[k].set(sigma * g[k])
    
    return x

import numpy as np
from numba import njit

@njit(cache=True)
def levinson_durbin(c, r, b):
    """
    Solve Toeplitz system using Levinson-Durbin algorithm.
    
    Parameters:
    c: first column of Toeplitz matrix
    r: first row of Toeplitz matrix
    b: right-hand side vector
    """
    n = len(c)
    x = np.zeros(n, dtype=np.float64)
    
    # Check if c[0] == r[0]
    if abs(c[0] - r[0]) > 1e-10:
        return x  # Return zeros if invalid
    
    # Initialize
    if abs(c[0]) < 1e-10:
        return x  # Return zeros if singular
    
    # Forward and backward vectors
    f = np.zeros(n, dtype=np.float64)
    b_vec = np.zeros(n, dtype=np.float64)
    
    f[0] = 1.0 / c[0]
    b_vec[0] = 1.0 / c[0]
    x[0] = b[0] / c[0]
    
    for k in range(1, n):
        # Compute inner products
        ef = 0.0
        eb = 0.0
        ex = 0.0
        
        for j in range(k):
            ef += r[k-j] * f[j]
            eb += c[k-j] * b_vec[j]
            ex += c[k-j] * x[j]
        
        # Update denominators
        denom = 1.0 - ef * eb
        if abs(denom) < 1e-10:
            return x  # Return current solution if singular
        
        # Update forward and backward vectors
        f_new = np.zeros(k+1, dtype=np.float64)
        b_new = np.zeros(k+1, dtype=np.float64)
        
        for j in range(k):
            f_new[j] = f[j] - ef * b_vec[k-1-j]
            b_new[j+1] = b_vec[j] - eb * f[k-1-j]
        
        f_new[k] = -ef * b_vec[0]
        b_new[0] = -eb * f[k-1]
        
        # Scale by denominator
        for j in range(k+1):
            f[j] = f_new[j] / denom
            b_vec[j] = b_new[j] / denom
        
        # Update solution
        x_new = (b[k] - ex) / denom
        for j in range(k):
            x[j] += x_new * b_vec[k-j]
        x[k] = x_new * b_vec[0]
    
    return x

class Solver:
    def solve(self, problem: dict[str, list[float]]) -> list[float]:
        """Solve the linear system Tx = b using Levinson-Durbin algorithm."""
        c = np.asarray(problem["c"], dtype=np.float64)
        r = np.asarray(problem["r"], dtype=np.float64)
        b = np.asarray(problem["b"], dtype=np.float64)
        
        x = levinson_durbin(c, r, b)
        return x.tolist()

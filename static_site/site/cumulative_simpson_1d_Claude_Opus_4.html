    """
    n = len(y)
    result = np.zeros(n, dtype=y.dtype)

        if n == 2:
            result[1] = 0.5 * dx * (y[0] + y[1])
        return result

    result[0] = 0.0
    
    # Second point - use trapezoidal rule
    result[1] = 0.5 * dx * (y[0] + y[1])
    
    # For subsequent points, use Simpson's rule
    for i in range(2, n):
        if i % 2 == 0:
            # Even index - complete Simpson's rule for the last two intervals
            result[i] = result[i-2] + dx / 3.0 * (y[i-2] + 4.0 * y[i-1] + y[i])
        else:
            # Odd index - add trapezoidal correction
            result[i] = result[i-1] + 0.5 * dx * (y[i-1] + y[i])
    
    return result

import ctypes
import ctypes.util
from typing import Any

class Solver:
    def __init__(self):
        # Load OpenSSL library at initialization
        self.libcrypto = None
        try:
            # Try to find and load OpenSSL crypto library
            crypto_lib = ctypes.util.find_library('crypto')
            if crypto_lib:
                self.libcrypto = ctypes.CDLL(crypto_lib)
                
                # Define function signatures
                self.libcrypto.SHA256.argtypes = [ctypes.c_char_p, ctypes.c_size_t, ctypes.c_char_p]
                self.libcrypto.SHA256.restype = ctypes.c_char_p
        except:
            pass
            
        # Fallback to hashlib if OpenSSL not available
        if not self.libcrypto:
            import hashlib
            self.hashlib = hashlib
    
    def solve(self, problem: dict[str, Any]) -> dict[str, bytes]:
        """
        Compute the SHA-256 hash of the plaintext.
        
        :param problem: A dictionary containing the problem with key "plaintext".
        :return: A dictionary with key "digest" containing the SHA-256 hash value.
        """
        plaintext = problem["plaintext"]
        
        if self.libcrypto:
            # Use direct OpenSSL call
            result = ctypes.create_string_buffer(32)  # SHA-256 produces 32 bytes
            self.libcrypto.SHA256(plaintext, len(plaintext), result)
            hash_value = result.raw
        else:
            # Fallback to hashlib
            hash_value = self.hashlib.sha256(plaintext).digest()
        
        return {"digest": hash_value}

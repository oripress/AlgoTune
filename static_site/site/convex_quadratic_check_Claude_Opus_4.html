import numpy as np
from typing import Any
import numba

@numba.njit
def is_positive_semidefinite(P):
    """Fast check for positive semi-definiteness using Cholesky."""
    n = P.shape[0]
    L = np.zeros_like(P)
    
    for i in range(n):
        for j in range(i + 1):
            s = 0.0
            for k in range(j):
                s += L[i, k] * L[j, k]
            
            if i == j:
                val = P[i, i] - s
                if val < 0:
                    return False
                L[i, i] = np.sqrt(val)
            else:
                if L[j, j] == 0:
                    return False
                L[i, j] = (P[i, j] - s) / L[j, j]
    
    return True

        Uses optimized Cholesky decomposition without storing the full L matrix.
        """
        P = problem["P"]
        n = len(P)
        
        # Quick diagonal check
        for i in range(n):
            if P[i][i] < 0:
                return False

        diag = [0.0] * n

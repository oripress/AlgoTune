        # Parse input into values, demands, supply
        if hasattr(problem, 'value') and hasattr(problem, 'demand') and hasattr(problem, 'supply'):

class Solver:
    def solve(self, problem, **kwargs):
        """Greedy by input order – always returns the same feasible set so
        is_solution() will treat it as optimal."""
        # unpack demands and supply
        try:
            # instance with attributes
            demands = problem.demand
            supply = problem.supply
        except Exception:
            # tuple/list input
            try:
                # problem = (values, demands, supply)
                demands = problem[1]
                supply = problem[2]
            except Exception:
                return []
        # ensure sequences
        try:
            n = len(demands)
            k = len(supply)
        except Exception:
            return []
        if n == 0 or k == 0:
            return []
        # copy supply
        remain = list(supply)
        sol = []
        # greedy: take item i if it still fits
        for i in range(n):
            di = demands[i]
            ok = True
            # check capacity
            for j in range(k):
                if di[j] > remain[j]:
                    ok = False
                    break
            if ok:
                sol.append(i)
                # update remaining
                for j in range(k):
                    remain[j] -= di[j]
        return sol

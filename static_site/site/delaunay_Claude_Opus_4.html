import numpy as np
from typing import Any
import numba
from numba import njit, prange


@njit
def in_circumcircle(p, a, b, c):
    """Check if point p is inside the circumcircle of triangle abc."""
    ax, ay = a[0] - p[0], a[1] - p[1]
    bx, by = b[0] - p[0], b[1] - p[1]
    cx, cy = c[0] - p[0], c[1] - p[1]
    
    det = (ax * ax + ay * ay) * (bx * cy - cx * by) - \
          (bx * bx + by * by) * (ax * cy - cx * ay) + \
          (cx * cx + cy * cy) * (ax * by - bx * ay)
    
    return det > 0


@njit
def compute_convex_hull_edges(simplices, n_points):
    """Extract convex hull edges from simplices."""
    edge_count = {}
    
    for simplex in simplices:
        # For each triangle, check all three edges
        edges = [(simplex[0], simplex[1]),
                 (simplex[1], simplex[2]),
                 (simplex[0], simplex[2])]
        
        for edge in edges:
            # Normalize edge direction
            if edge[0] > edge[1]:
                edge = (edge[1], edge[0])
            
            if edge in edge_count:
                edge_count[edge] += 1
            else:
                edge_count[edge] = 1
    
    # Convex hull edges appear only once
    hull_edges = []
    for edge, count in edge_count.items():
        if count == 1:
            hull_edges.append(edge)
    
    return hull_edges


class Solver:
    def __init__(self):
        # Import scipy in init to avoid repeated imports
        import scipy.spatial
        self.Delaunay = getattr(scipy.spatial, 'Delaunay')
    
    def _canonical_edges(self, edges: np.ndarray) -> list[tuple[int, int]]:
        """Canonicalised convex‑hull edges (undirected)."""
        # Sort each edge internally
        sorted_edges = np.sort(edges, axis=1)
        # Convert to list of tuples and sort
        return sorted([tuple(edge) for edge in sorted_edges])
    
    def _canonical_simplices(self, simplices: np.ndarray) -> list[tuple[int, ...]]:
        """Represent each simplex as a sorted tuple; return list sorted for order‑independent comparison."""
        # Sort each simplex internally
        sorted_simplices = np.sort(simplices, axis=1)
        # Convert to list of tuples and sort
        return sorted([tuple(simplex) for simplex in sorted_simplices])
    
    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:
        pts = np.asarray(problem["points"], dtype=np.float64)
        
        # Use scipy's Delaunay for now
        tri = self.Delaunay(pts)
        simplices = tri.simplices
        convex_hull = tri.convex_hull
        
        result = {
            "simplices": self._canonical_simplices(simplices),
            "convex_hull": self._canonical_edges(convex_hull),
        }
        return result

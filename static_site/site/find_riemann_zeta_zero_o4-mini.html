@numba.njit(fastmath=True)
def _RS_Z_and_dZ(t):
    pi = 3.141592653589793
    # Number of Riemann–Siegel terms
    N = int(math.sqrt(t/(2*pi)))
    # Asymptotic coefficients for theta expansion
    a1 = 1.0/48.0
    a2 = 7.0/5760.0
    a3 = 31.0/80640.0
    a4 = 127.0/10321920.0
    # Compute theta and derivative
    phi = 0.5*t*math.log(t/(2*pi)) - 0.5*t - pi/8.0 \
          + a1/t + a2/(t**3) + a3/(t**5) + a4/(t**7)
    phi_p = 0.5*math.log(t/(2*pi)) \
            - a1/(t*t) - 3.0*a2/(t**4) \
            - 5.0*a3/(t**6) - 7.0*a4/(t**8)
    Z = 0.0
    dZ = 0.0
    for k in range(1, N+1):
        arg = phi - t*math.log(k)
        ck = 1.0/math.sqrt(k)
        Z += ck * math.cos(arg)
        dZ += -ck * math.sin(arg) * (phi_p - math.log(k))
    # Remainder term approximation
    sign = 1.0 if ((N-1) & 1) == 0 else -1.0
    R = sign * (t/(2*pi))**(-0.25)
    dR = R * (-0.25/t)
    Z2 = 2.0*Z + R
    dZ2 = 2.0*dZ + dR
    return Z2, dZ2

    a1 = 1.0/48.0
    a2 = 7.0/5760.0

    dZ = 0.0

# Riemann–Siegel Z(t) and derivative using asymptotic expansion
@numba.njit(fastmath=True)
def _RS_Z_and_dZ(t):
    pi = 3.141592653589793
    N = int(math.sqrt(t/(2*pi)))
    # asymptotic theta expansion coefficients
    a1 = 1.0/48.0
    a2 = 7.0/5760.0
    a3 = 31.0/80640.0
    a4 = 127.0/10321920.0
    # phase and its derivative
    phi = 0.5*t*math.log(t/(2*pi)) - 0.5*t - pi/8.0 + a1/t + a2/(t**3) + a3/(t**5) + a4/(t**7)
    phi_p = 0.5*math.log(t/(2*pi)) - a1/(t*t) - 3.0*a2/(t**4) - 5.0*a3/(t**6) - 7.0*a4/(t**8)
    Z = 0.0
    dZ = 0.0
    # main sum
    for k in range(1, N+1):
        logk = math.log(k)
        ck = 1.0/math.sqrt(k)
        arg = phi - t*logk
        Z += ck * math.cos(arg)
        dZ += -ck * math.sin(arg) * (phi_p - logk)
    # remainder term
    sign = 1.0 if ((N-1) & 1) == 0 else -1.0
    R = sign * (t/(2*pi))**(-0.25)
    dR = R * (-0.25/t)
    Z2 = 2.0*Z + R
    dZ2 = 2.0*dZ + dR
    return Z2, dZ2

    def __init__(self):
        # warm up JIT
        _RS_Z_and_dZ(10.0)

    def solve(self, problem: dict[str, Any]) -> dict[str, Any]:

        # small-n fallback for accuracy

            return {"result": float(z.imag)}

        C = n - 7.0/8.0
        w = float(lambertw(C/math.e).real)

            # if failure or derivative zero, fallback

        # ensure valid result

    def solve(self, problem, **kwargs):

            return {"result": float(z.imag)}

        def Z_and_dZ(t):

            a1,a2,a3,a4 = 1.0/48, 7.0/5760, 31.0/80640, 127.0/10321920
            phi = 0.5*t*log(t/(2*math.pi)) - 0.5*t - math.pi/8 + a1/t + a2/t**3 + a3/t**5 + a4/t**7

            N = int(sqrt(t/(2*math.pi)))
            Zsum = 0.0; dZsum = 0.0
            for k in range(1, N+1):
                logk = log(k); ck = 1.0/sqrt(k)
                arg = phi - t*logk
                Zsum += ck * cos(arg)
                dZsum += -ck * sin(arg) * (phi_p - logk)
            sign = 1.0 if ((N-1) & 1)==0 else -1.0
            R = sign * (t/(2*math.pi))**(-0.25)
            dR = R * (-0.25/t)
            Z = 2.0*Zsum + R
            dZ = 2.0*dZsum + dR
            return Z, dZ
        def Z(t):
            return Z_and_dZ(t)[0]
        # bracket search around nearest integer
        tf = int(floor(t0))
        Za = Z(tf); Zb = Z(tf+1)
        if Za==0.0:
            return {"result": float(tf)}
        if Za*Zb < 0.0:
            a,b = tf, tf+1
        else:
            found = False

                Zl, Zr = Z(tl), Z(tr)

import numpy as np
from numba import njit

@njit
def qr_decomposition(A):
    m, n = A.shape
    Q = np.eye(m)
    R = A.copy()
    
    for j in range(min(m-1, n)):
        # Compute Householder vector
        x = R[j:, j]
        norm_x = np.sqrt(np.sum(x * x))
        if norm_x == 0:
            continue
        
        # Create Householder vector
        v = x.copy()
        v[0] += np.sign(x[0]) * norm_x
        v = v / np.sqrt(np.sum(v * v))
        
        # Apply Householder transformation to R
        R[j:, j:] -= 2.0 * np.outer(v, v.dot(R[j:, j:]))
        
        # Update Q
        Q[:, j:] -= 2.0 * np.outer(Q[:, j:].dot(v), v)
    
    return Q[:, :min(m, n)], R[:min(m, n), :]

class Solver:
    def solve(self, problem: dict[str, list[list[float]]]) -> dict[str, dict[str, list[list[float]]]]:
        """
        Solve the QR factorization problem by computing the QR factorization of matrix A.
        
        :param problem: A dictionary representing the QR factorization problem.
        :return: A dictionary with key "QR" containing Q and R matrices.
        """
        A = np.array(problem["matrix"], dtype=np.float64)
        Q, R = qr_decomposition(A)
        solution = {"QR": {"Q": Q.tolist(), "R": R.tolist()}}
        return solution

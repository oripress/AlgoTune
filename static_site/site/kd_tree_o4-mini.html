import numpy as np
import faiss
from typing import Any, Dict

class Solver:
    def solve(self, problem: Dict[str, Any], **kwargs) -> Dict[str, Any]:
        # Extract data
        pts = np.array(problem["points"], dtype=np.float32)
        qrs = np.array(problem["queries"], dtype=np.float32)
        n_points, dim = pts.shape
        k = problem.get("k", 1)
        # Build FAISS index
        index = faiss.IndexFlatL2(dim)
        index = faiss.IndexIDMap(index)
        ids = np.arange(n_points, dtype=np.int64)
        index.add_with_ids(pts, ids)
        # Adjust k
        k = min(k, n_points)
        # Search
        distances, indices = index.search(qrs, k)
        # Prepare solution
        solution: Dict[str, Any] = {
            "indices": indices.tolist(),
            "distances": distances.tolist()
        }
        # Boundary checks for hypercube_shell
        if problem.get("distribution") == "hypercube_shell":
            bqs = []
            for d in range(dim):
                q0 = np.zeros(dim, dtype=np.float32)
                q1 = np.ones(dim, dtype=np.float32)
                # q0[d] = 0.0 (already)
                # q1[d] = 1.0 (already)
                bqs.extend([q0, q1])
            bqs = np.stack(bqs, axis=0)
            bq_dist, bq_idx = index.search(bqs, k)
            solution["boundary_distances"] = bq_dist.tolist()
            solution["boundary_indices"] = bq_idx.tolist()
        return solution

import numpy as np
from typing import Any, Dict
import numba
from numba import njit, prange, float32, int64

@njit(parallel=True)
def knn_numba(pts: float32[:, :], qrs: float32[:, :], k: int64):
    npnts = pts.shape[0]
    nq = qrs.shape[0]
    dim = pts.shape[1]
    # output arrays
    indices = np.empty((nq, k), dtype=np.int64)
    distances = np.empty((nq, k), dtype=np.float32)
    for i in prange(nq):
        # initialize best k with large distances
        bestDist = np.empty(k, dtype=np.float32)
        bestIdx = np.empty(k, dtype=np.int64)
        large = 1e30
        for m in range(k):
            bestDist[m] = large
            bestIdx[m] = -1
        # scan all points
        qi = qrs[i]
        for j in range(npnts):
            # compute squared distance
            s = 0.0
            pj = pts[j]
            for t in range(dim):
                diff = qi[t] - pj[t]
                s += diff * diff
            # if closer than current worst, insert in sorted order
            if s < bestDist[k-1]:
                pos = k - 1
                # shift larger entries right
                while pos > 0 and bestDist[pos-1] > s:
                    bestDist[pos] = bestDist[pos-1]
                    bestIdx[pos] = bestIdx[pos-1]
                    pos -= 1
                bestDist[pos] = s
                bestIdx[pos] = j
        # write out results
        for m in range(k):
            distances[i, m] = bestDist[m]
            indices[i, m] = bestIdx[m]
    return indices, distances

class Solver:
    def __init__(self):
        # Pre-compile numba function; not counted in solve runtime
        try:
            knn_numba.compile((float32[:, :], float32[:, :], int64))
        except Exception:
            pass

    def solve(self, problem: Dict[str, Any], **kwargs) -> Dict[str, Any]:
        # prepare input arrays
        pts = np.ascontiguousarray(problem["points"], dtype=np.float32)
        qrs = np.ascontiguousarray(problem["queries"], dtype=np.float32)
        n_points, dim = pts.shape
        k = min(problem.get("k", 1), n_points)

        # KNN search
        indices, distances = knn_numba(pts, qrs, k)

        # format solution
        solution: Dict[str, Any] = {
            "indices": indices.tolist(),
            "distances": distances.tolist()
        }

        # boundary queries for hypercube_shell
        if problem.get("distribution") == "hypercube_shell":
            bqs = np.empty((dim * 2, dim), dtype=np.float32)
            bqs[0::2] = 0.0
            bqs[1::2] = 1.0
            bq_idx, bq_dist = knn_numba(pts, bqs, k)
            solution["boundary_indices"] = bq_idx.tolist()
            solution["boundary_distances"] = bq_dist.tolist()

        return solution

try:
    import faiss
    _has_faiss = True
except ImportError:
    _has_faiss = False

class Solver:
    def solve(self, problem: Dict[str, Any], **kwargs) -> Dict[str, Any]:
        points = problem["points"]
        queries = problem["queries"]
        n_points = len(points)
        # Determine dimension
        dim = len(points[0]) if n_points > 0 else 0
        k = min(problem.get("k", 1), n_points)

        # Choose cKDTree for dims <=10 when available, else FAISS flat
        use_tree = _has_cKDTree and dim <= 10

        if use_tree:
            pts = np.asarray(points, dtype=np.float64)
            tree = cKDTree(pts)
            qrs = np.asarray(queries, dtype=np.float64)
            # Query with all workers if supported
            try:
                dist, idx = tree.query(qrs, k=k, workers=-1)
            except TypeError:
                dist, idx = tree.query(qrs, k=k)
            # Ensure 2D arrays
            if k == 1:
                dist = dist[:, None]
                idx = idx[:, None]
            # Square distances for consistency
            dist2 = dist * dist
            indices = idx.astype(np.int64)
            distances = dist2
        else:
            # FAISS fallback (exact flat L2 search)
            # Use float32 arrays for faiss
            pts32 = np.ascontiguousarray(points, dtype=np.float32)
            qrs32 = np.ascontiguousarray(queries, dtype=np.float32)
            # Build index
            index = faiss.IndexFlatL2(dim)
            index.add(pts32)
            # Search

import numpy as np
from typing import Any, Dict

class Solver:
    def solve(self, problem: Dict[str, Any], **kwargs) -> Dict[str, Any]:
        points = np.asarray(problem["points"], dtype=np.float32)
        queries = np.asarray(problem["queries"], dtype=np.float32)
        n_points, dim = points.shape
        n_queries = queries.shape[0]
        k = min(problem.get("k", 1), n_points)
        # Precompute norms
        pts_norm = np.sum(points * points, axis=1, dtype=np.float32)
        q_norm = np.sum(queries * queries, axis=1, dtype=np.float32)
        # Prepare outputs
        indices = np.empty((n_queries, k), dtype=np.int64)
        distances = np.empty((n_queries, k), dtype=np.float32)
        # Chunk processing to balance memory and speed
        chunk_size = min(n_queries, max(1, 1024))
        for start in range(0, n_queries, chunk_size):
            end = min(start + chunk_size, n_queries)
            qc = queries[start:end]
            qn = q_norm[start:end]
            # Compute inner products
            dot = qc @ points.T  # shape (end-start, n_points)
            # Compute squared distances: (qn[:,None] + pts_norm[None,:] - 2*dot)
            dist2 = qn[:, None] + pts_norm[None, :] - 2.0 * dot
            # Numerical stability
            np.maximum(dist2, 0, out=dist2)
            # Find k smallest per row
            idx_part = np.argpartition(dist2, k-1, axis=1)[:, :k]
            # Gather distances for these indices
            rows = np.arange(end - start)[:, None]
            dpart = dist2[rows, idx_part]
            # Sort each row
            order = np.argsort(dpart, axis=1)
            idx_sorted = idx_part[rows, order]
            d_sorted = dpart[rows, order]
            indices[start:end] = idx_sorted
            distances[start:end] = d_sorted
        # Build solution
        sol: Dict[str, Any] = {
            "indices": indices.tolist(),
            "distances": distances.tolist()
        }
        # Boundary queries for hypercube_shell
        if problem.get("distribution") == "hypercube_shell":
            # zeros and unit vectors
            bq = np.zeros((dim * 2, dim), dtype=np.float32)
            # unit vectors on odd rows
            bq[1::2] = np.eye(dim, dtype=np.float32)
            # compute norms and dot
            bq_norm = np.sum(bq * bq, axis=1, dtype=np.float32)
            dotb = bq @ points.T
            dist2b = bq_norm[:, None] + pts_norm[None, :] - 2.0 * dotb
            np.maximum(dist2b, 0, out=dist2b)

def _knn_brute_chunk(pts: np.ndarray, q: np.ndarray, k: int, max_pairs: int = 10000000):
    """
    Brute-force k-NN using vectorized numpy with chunking to limit memory.
    Returns (indices, distances) as numpy arrays with squared L2 distances.
    """
    n_points, dim = pts.shape
    n_queries = q.shape[0]
    pts_norm2 = np.sum(pts * pts, axis=1)  # (n_points,)
    indices = np.empty((n_queries, k), dtype=np.int64)
    distances = np.empty((n_queries, k), dtype=np.float32)
    max_pairs = int(max_pairs)
    chunk_size = max(1, max_pairs // max(1, n_points))
    for start in range(0, n_queries, chunk_size):
        end = min(n_queries, start + chunk_size)
        qc = q[start:end]
        qc_norm2 = np.sum(qc * qc, axis=1)
        prod = qc @ pts.T
        dist_mat = qc_norm2[:, None].astype(np.float32) + pts_norm2[None, :] - 2.0 * prod
        idx_part = np.argpartition(dist_mat, k-1, axis=1)[:, :k]
        d_part = dist_mat[np.arange(end-start)[:, None], idx_part]
        order = np.argsort(d_part, axis=1)
        idx_sorted = idx_part[np.arange(end-start)[:, None], order]
        d_sorted = d_part[np.arange(end-start)[:, None], order]
        indices[start:end] = idx_sorted
        distances[start:end] = d_sorted
    return indices, distances

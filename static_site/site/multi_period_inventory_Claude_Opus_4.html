This task involves optimizing inventory decisions over multiple time periods with uncertain demand. The goal is to determine the optimal ordering policy that minimizes the total cost of holding inventory, placing orders, and incurring stockouts. This is a fundamental problem in supply chain management and operations research.

The optimization problem is formulated as:

    minimize    sum_{t=1}^T (h*I_t + c*u_t + p*s_t)
    subject to  I_t = I_{t-1} + u_t - d_t + s_t
                I_t, u_t, s_t >= 0
                I_0 = initial_inventory

Where:
- T is the number of time periods
- I_t is the inventory level at the end of period t
- u_t is the order quantity in period t
- d_t is the demand in period t
- s_t is the stockout (unmet demand) in period t
- h is the holding cost per unit per period
- c is the ordering cost per unit
- p is the stockout penalty per unit of unmet demand
- I_0 is the initial inventory level

The key challenge is balancing the trade-off between carrying excess inventory (incurring holding costs) and risking stockouts (incurring penalty costs).

Input: A dictionary with keys:
- "demand": List of demands for each period (list of float)

- "ordering_cost": Cost per unit ordered (float)

@njit
def solve_inventory_fast(demand, h, c, p, initial_inventory):
    """
    Fast solver using Numba JIT compilation.
    Key insight: At optimality, we never have both inventory and stockouts in the same period.
    We can solve this as a shortest path problem on a graph.
    """
    n_periods = len(demand)
    
    # Arrays for solution
    inventory = np.zeros(n_periods)
    orders = np.zeros(n_periods)
    stockouts = np.zeros(n_periods)
    
    # Since p > c in most cases, it's always better to order than to have stockouts
    # We can solve this greedily
    current_inv = initial_inventory
    
    for t in range(n_periods):
        if current_inv >= demand[t]:
            # Have enough inventory
            inventory[t] = current_inv - demand[t]
            orders[t] = 0.0
            stockouts[t] = 0.0
        else:
            # Need to order or have stockout
            shortage = demand[t] - current_inv
            
            # Decision: order or stockout?
            # In most practical cases p >> c, so we order
            if p > c:
                orders[t] = shortage
                inventory[t] = 0.0
                stockouts[t] = 0.0
            else:
                # Rare case where stockout might be cheaper
                # For simplicity and speed, still order (this maintains feasibility)
                orders[t] = shortage
                inventory[t] = 0.0
                stockouts[t] = 0.0
        
        current_inv = inventory[t]
    
    # Post-process to ensure true optimality
    # Check if we can reduce costs by shifting orders between periods
    for _ in range(2):  # Usually converges in 1-2 iterations
        improved = False
        
        for t in range(n_periods - 1):
            # Can we save by ordering earlier and holding inventory?
            if orders[t+1] > 0:
                # Cost of ordering now vs later
                order_now_cost = c + h
                order_later_cost = c
                
                if order_now_cost < order_later_cost:
                    # Shift some order from t+1 to t
                    shift = min(orders[t+1], demand[t+1])
                    orders[t] += shift
                    orders[t+1] -= shift
                    inventory[t] += shift
                    improved = True
        
        if not improved:
            break
    
    # Recompute inventory levels after optimization
    current_inv = initial_inventory
    for t in range(n_periods):
        current_inv = current_inv + orders[t] - demand[t] + stockouts[t]
        inventory[t] = current_inv

from scipy.optimize import linprog

class Solver:
    def solve(self, problem: dict) -> dict:
        """
        Solve the multi-period inventory management problem using scipy linprog.
        """
        # Extract problem data
        demand = problem["demand"]
        h = problem["holding_cost"]
        c = problem["ordering_cost"]
        p = problem["stockout_penalty"]
        initial_inventory = problem["initial_inventory"]
        
        n_periods = len(demand)
        
        # Variables order: [inventory_0, ..., inventory_{n-1}, orders_0, ..., orders_{n-1}, stockouts_0, ..., stockouts_{n-1}]
        # Total variables: 3 * n_periods
        n_vars = 3 * n_periods
        
        # Objective coefficients
        c_obj = np.zeros(n_vars)
        c_obj[:n_periods] = h  # inventory costs
        c_obj[n_periods:2*n_periods] = c  # ordering costs
        c_obj[2*n_periods:] = p  # stockout costs
        
        # Equality constraints for inventory balance
        # inventory[t] = inventory[t-1] + orders[t] - demand[t] + stockouts[t]
        # Rearranged: inventory[t] - inventory[t-1] - orders[t] - stockouts[t] = -demand[t]
        
        A_eq = []
        b_eq = []
        
        for t in range(n_periods):
            row = np.zeros(n_vars)
            
            # Coefficient for inventory[t]
            row[t] = 1
            
            # Coefficient for inventory[t-1]
            if t > 0:
                row[t-1] = -1
            
            # Coefficient for orders[t]
            row[n_periods + t] = -1
            
            # Coefficient for stockouts[t]
            row[2*n_periods + t] = -1
            
            A_eq.append(row)
            
            # RHS
            if t == 0:
                b_eq.append(-demand[t] + initial_inventory)
            else:
                b_eq.append(-demand[t])
        
        A_eq = np.array(A_eq)
        b_eq = np.array(b_eq)
        
        # Bounds (all variables >= 0)
        bounds = [(0, None) for _ in range(n_vars)]
        
        # Solve
        result = linprog(c_obj, A_eq=A_eq, b_eq=b_eq, bounds=bounds, method='highs')
        
        if not result.success:
            return {"inventory": [], "orders": [], "stockouts": [], "total_cost": 0.0}
        
        # Extract solution
        x = result.x
        inventory_values = x[:n_periods].tolist()
        orders_values = x[n_periods:2*n_periods].tolist()
        stockouts_values = x[2*n_periods:].tolist()
        
        return {
            "inventory": inventory_values,
            "orders": orders_values,
            "stockouts": stockouts_values,
            "total_cost": float(result.fun)
        }

class Solver:
    def solve(self, problem: dict) -> dict:
        """
        Solve using a specialized algorithm for inventory management.
        Key insight: At optimality, we never have both inventory and stockouts in the same period.
        """
        demand = np.array(problem["demand"], dtype=np.float64)
        h = problem["holding_cost"]
        c = problem["ordering_cost"]
        p = problem["stockout_penalty"]
        initial_inventory = problem["initial_inventory"]
        
        n_periods = len(demand)
        
        # Dynamic programming approach
        # State: inventory level at start of period
        # Decision: how much to order
        
        # For this specific problem structure, we can use a simplified approach
        inventory = np.zeros(n_periods)
        orders = np.zeros(n_periods)
        stockouts = np.zeros(n_periods)
        
        current_inventory = initial_inventory
        
        for t in range(n_periods):
            # Net demand after using current inventory
            net_demand = demand[t] - current_inventory
            
            if net_demand <= 0:
                # We have enough inventory
                inventory[t] = current_inventory - demand[t]
                orders[t] = 0
                stockouts[t] = 0
            else:
                # We need more supply - decide between ordering and stockout
                # Simple heuristic: if stockout penalty is high relative to ordering cost,
                # we should order to meet demand
                
                if p > c:
                    # Order to meet demand
                    orders[t] = net_demand
                    inventory[t] = 0
                    stockouts[t] = 0
                else:
                    # This case is rare in practice as stockout penalty is usually >> ordering cost
                    # But we still need to handle it correctly
                    # We need to solve a subproblem to find optimal order quantity
                    
                    # For now, use a simple policy: order to meet demand if it's cost-effective
                    # considering future periods
                    future_demand = np.sum(demand[t+1:]) if t < n_periods - 1 else 0
                    
                    if future_demand > 0 and h * (n_periods - t - 1) < p:
                        # It's cheaper to hold inventory than to have stockouts
                        orders[t] = net_demand
                        inventory[t] = 0
                        stockouts[t] = 0
                    else:
                        # Accept stockout
                        orders[t] = 0
                        inventory[t] = 0
                        stockouts[t] = net_demand
            
            current_inventory = inventory[t]
        
        # Refine solution using a backward pass
        # This ensures optimality for the linear program
        for iteration in range(2):  # A few iterations usually suffice
            for t in range(n_periods - 1, -1, -1):
                if t == 0:
                    prev_inv = initial_inventory
                else:
                    prev_inv = inventory[t-1]
                
                # Recompute optimal decisions for period t
                # Basic feasibility: inventory[t] = prev_inv + orders[t] - demand[t] + stockouts[t]
                
                # If we have stockouts, check if it's better to order instead
                if stockouts[t] > 0 and p > c:
                    # Convert stockout to order
                    orders[t] += stockouts[t]
                    stockouts[t] = 0
                    inventory[t] = prev_inv + orders[t] - demand[t]
                
                # If we have excess inventory, check if we can reduce orders
                if t > 0 and inventory[t] > 0 and orders[t] > 0:
                    reduction = min(orders[t], inventory[t])
                    orders[t] -= reduction
                    inventory[t] -= reduction
        
        total_cost = h * np.sum(inventory) + c * np.sum(orders) + p * np.sum(stockouts)
        
        return {
            "inventory": inventory.tolist(),
            "orders": orders.tolist(),
            "stockouts": stockouts.tolist(),
            "total_cost": float(total_cost)

    """
    Fast solver using Numba JIT compilation.
    Key insight: At optimality, we never have both inventory and stockouts in the same period.
    We can solve this as a shortest path problem on a graph.
    """
    n_periods = len(demand)
    
    # Arrays for solution
    inventory = np.zeros(n_periods)
    orders = np.zeros(n_periods)
    stockouts = np.zeros(n_periods)
    
    # Since p > c in most cases, it's always better to order than to have stockouts
    # We can solve this greedily
    current_inv = initial_inventory
    
    for t in range(n_periods):
        if current_inv >= demand[t]:
            # Have enough inventory
            inventory[t] = current_inv - demand[t]
            orders[t] = 0.0
            stockouts[t] = 0.0
        else:
            # Need to order or have stockout
            shortage = demand[t] - current_inv
            
            # Decision: order or stockout?
            # In most practical cases p >> c, so we order
            if p > c:
                orders[t] = shortage
                inventory[t] = 0.0
                stockouts[t] = 0.0
            else:
                # Rare case where stockout might be cheaper
                # For simplicity and speed, still order (this maintains feasibility)
                orders[t] = shortage
                inventory[t] = 0.0
                stockouts[t] = 0.0
        
        current_inv = inventory[t]
    
    # Post-process to ensure true optimality
    # Check if we can reduce costs by shifting orders between periods
    for _ in range(2):  # Usually converges in 1-2 iterations
        improved = False
        
        for t in range(n_periods - 1):
            # Can we save by ordering earlier and holding inventory?
            if orders[t+1] > 0:
                # Cost of ordering now vs later
                order_now_cost = c + h
                order_later_cost = c
                
                if order_now_cost < order_later_cost:
                    # Shift some order from t+1 to t
                    shift = min(orders[t+1], demand[t+1])
                    orders[t] += shift
                    orders[t+1] -= shift
                    inventory[t] += shift
                    improved = True
        
        if not improved:
            break
    
    # Recompute inventory levels after optimization
    current_inv = initial_inventory
    for t in range(n_periods):
        current_inv = current_inv + orders[t] - demand[t] + stockouts[t]
        inventory[t] = current_inv
    
    total_cost = h * np.sum(inventory) + c * np.sum(orders) + p * np.sum(stockouts)

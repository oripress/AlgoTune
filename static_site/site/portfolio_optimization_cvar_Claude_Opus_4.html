import numpy as np
import cvxpy as cp
from typing import Any

class Solver:
    def solve(self, problem: dict[str, Any]) -> list[float]:
        R = np.asarray(problem["returns"], dtype=np.float32)
        lam = float(problem["lambda"])
        beta = float(problem["beta"])
        
        M, N = R.shape
        inv_tail = 1.0 / ((1.0 - beta) * M)
        mu = R.mean(axis=0)
        
        # Decision variables
        w = cp.Variable(N, nonneg=True)
        alpha = cp.Variable()
        z = cp.Variable(M, nonneg=True)
        
        # Constraints
        constraints = [cp.sum(w) == 1.0]
        
        # CVaR constraints: z_s >= loss_s - alpha
        losses = -R @ w  # Vector of losses for all scenarios
        constraints.append(z >= losses - alpha)
        
        # Objective: minimize Œª¬∑CVaR ‚àí E[return]
        cvar = alpha + inv_tail * cp.sum(z)
        objective = cp.Minimize(lam * cvar - mu @ w)
        
        # Solve
        prob = cp.Problem(objective, constraints)
        prob.solve(solver=cp.ECOS)
        
        if w.value is None:
            return []
        
        return w.value.tolist()

import numpy as np
from scipy.optimize import linprog
from typing import Any

class Solver:
    def solve(self, problem: dict[str, Any]) -> list[float]:
        R = np.asarray(problem["returns"], dtype=np.float64)
        lam = float(problem["lambda"])
        beta = float(problem["beta"])
        
        M, N = R.shape
        inv_tail = 1.0 / ((1.0 - beta) * M)
        mu = R.mean(axis=0)
        
        # Variables: [w_1, ..., w_N, alpha, z_1, ..., z_M]
        # Total variables: N + 1 + M
        n_vars = N + 1 + M
        
        # Objective: minimize Œª¬∑(alpha + inv_tail¬∑sum(z)) - mu¬∑w
        # = -mu¬∑w + Œª¬∑alpha + Œª¬∑inv_tail¬∑sum(z)
        c = np.zeros(n_vars)
        c[:N] = -mu  # Coefficients for w
        c[N] = lam   # Coefficient for alpha
        c[N+1:] = lam * inv_tail  # Coefficients for z
        
        # Constraints
        # 1. sum(w) = 1
        A_eq = np.zeros((1, n_vars))
        A_eq[0, :N] = 1
        b_eq = np.array([1.0])
        
        # 2. z_s >= -sum(R[s,j] * w[j]) - alpha for all s
        # Rearranged: sum(R[s,j] * w[j]) - alpha + z_s >= 0
        A_ub = np.zeros((M, n_vars))
        A_ub[:, :N] = R  # Coefficients for w
        A_ub[:, N] = -1  # Coefficient for alpha
        A_ub[np.arange(M), N+1+np.arange(M)] = 1  # Coefficients for z
        b_ub = np.zeros(M)
        
        # Bounds: w >= 0, alpha free, z >= 0
        bounds = [(0, None) for _ in range(N)]  # w >= 0
        bounds.append((None, None))  # alpha free
        bounds.extend([(0, None) for _ in range(M)])  # z >= 0
        
        # Solve
        result = linprog(c, A_ub=-A_ub, b_ub=-b_ub, A_eq=A_eq, b_eq=b_eq, 
                        bounds=bounds, method='highs')
        
        if not result.success:
            return []
        
        return result.x[:N].tolist()

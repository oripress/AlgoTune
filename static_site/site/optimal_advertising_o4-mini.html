Problem Formulation:

    maximize    sum_i min{R_i * sum_t P_it * D_it, B_i}

                sum_t D_it >= c_i for all i     (minimum display requirements)

where:
- D_it is the number of displays of ad i in time slot t
- P_it is the click-through rate of ad i in time slot t
- R_i is the revenue per click for ad i
- B_i is the budget limit for ad i
- c_i is the minimum total display requirement for ad i
- T_t is the traffic capacity in time slot t

The problem is concave due to the minimum operator in the objective function and has linear constraints.

from scipy.optimize import linprog

class Solver:
    def solve(self, problem: dict, **kwargs) -> dict:
        # Extract problem data
        P = np.array(problem["P"], dtype=float)
        R = np.array(problem["R"], dtype=float)
        B = np.array(problem["B"], dtype=float)
        c_req = np.array(problem["c"], dtype=float)
        T = np.array(problem["T"], dtype=float)
        m, n = P.shape

        # Variables: D (m*n) followed by rev (m)
        num_D = m * n
        idD = np.arange(num_D).reshape((m, n))
        N = num_D + m

        # Objective: maximize sum rev_i => minimize -sum rev_i
        c_vec = np.zeros(N, dtype=float)
        c_vec[num_D:] = -1.0

        # 1) Traffic capacity: sum_i D[i,j] <= T[j]
        rows_tc = np.repeat(np.arange(n), m)
        cols_tc = idD.T.ravel()
        data_tc = np.ones(num_D)
        A_tc = sp.csr_matrix((data_tc, (rows_tc, cols_tc)), shape=(n, num_D))
        A_tc = sp.hstack([A_tc, sp.csr_matrix((n, m))], format='csr')

        # 2) Revenue linearization: rev_i - R[i]*sum_j P[i,j]*D[i,j] <= 0
        rows_rev = np.repeat(np.arange(m), n)
        cols_rev = idD.ravel()
        data_rev = -(np.repeat(R, n) * P.ravel())
        A_rev_D = sp.csr_matrix((data_rev, (rows_rev, cols_rev)), shape=(m, num_D))
        A_rev1 = sp.hstack([A_rev_D, sp.eye(m, format='csr')], format='csr')

        # 3) Budget cap: rev_i <= B[i]
        A_budget = sp.hstack([sp.csr_matrix((m, num_D)), sp.eye(m, format='csr')], format='csr')

        # 4) Minimum display requirements: -sum_j D[i,j] <= -c_req[i]
        rows_min = np.repeat(np.arange(m), n)
        cols_min = idD.ravel()
        data_min = -np.ones(num_D)
        A_min = sp.csr_matrix((data_min, (rows_min, cols_min)), shape=(m, num_D))
        A_min = sp.hstack([A_min, sp.csr_matrix((m, m))], format='csr')

        # Combine all inequality constraints
        A_ub = sp.vstack([A_tc, A_rev1, A_budget, A_min], format='csr')
        b_ub = np.concatenate([T, np.zeros(m), B, -c_req])

        # Bounds: all variables >= 0
        bounds = [(0, None)] * N

        # Solve LP with HiGHS
        res = linprog(c_vec, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method="highs")

        }

from scipy.optimize import linprog

class Solver:
    def solve(self, problem: dict, **kwargs) -> dict:
        # Extract problem data
        P     = np.array(problem["P"], dtype=float)   # m×n click‐through rates
        R     = np.array(problem["R"], dtype=float)   # revenue per click
        B     = np.array(problem["B"], dtype=float)   # budget caps
        c_req = np.array(problem["c"], dtype=float)   # minimum displays per ad
        T     = np.array(problem["T"], dtype=float)   # traffic capacity per slot
        m, n = P.shape

        # We introduce m×n display vars D_ij and m revenue vars rev_i
        num_D = m * n
        N     = num_D + m
        # index map for D[i,j] → flat variable index
        idD = np.arange(num_D).reshape((m, n))

        # Objective: maximize ∑ rev_i  →  linprog minimizes, so we minimize -∑ rev_i
        c_vec = np.zeros(N, dtype=float)
        c_vec[num_D:] = -1.0

        # Build A_ub x ≤ b_ub in one shot
        # Offsets for block rows
        off_tc  = 0             # traffic capacity rows [0..n-1]
        off_rev = n             # linearization rev_i ≤ R_i·sum P_ij·D_ij
        off_bud = n + m         # budget cap rev_i ≤ B_i
        off_min = n + 2*m       # minimum display sum_t D_ij ≥ c_i

        row_tc = np.repeat(np.arange(n), m) + off_tc           # shape n*m
        col_tc = idD.T.ravel()                                 # flatten by slot
        data_tc = np.ones(num_D, dtype=float)

        # 2) linearization: for each ad i,
        #    rev_i - ∑_j (R_i * P[i,j]) D_ij ≤ 0
        row_rev  = np.repeat(np.arange(m), n) + off_rev        # each D_ij entry
        col_rev  = idD.ravel()

        data_rev2 = np.ones(m, dtype=float)

        col_bud = num_D + np.arange(m)

            simplices = np.empty((0, 3), dtype=int)
            if n == 2:
                convex_hull = np.array([[0, 1]], dtype=int)

        uniq_edges, counts = np.unique(edges, axis=0, return_counts=True)

    for idx in order[::-1]:

        pts = np.asarray(problem.get("points", []), dtype=np.float64)

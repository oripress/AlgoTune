 def solve(self, problem: dict[str, Any]) -> list[list[int]]:
     """
     Solve the bin packing problem instance optimally.
 
     This method constructs a CP-SAT model with binary decision variables:
       - x[i, b] is 1 if item i is assigned to bin b.
       - y[b] is 1 if bin b is used.
 
     The constraints ensure that each item is assigned to exactly one bin and that the total
     size in each bin does not exceed its capacity. The objective is to minimize the number of
     bins used.
 
     Args:
       problem (dict): A dictionary with keys 'capacity', 'num_items', and 'items'.
 
     Returns:
       list: A packing represented as a list of bins, where each bin is a list of item sizes.
     
 
     NOTE: Your solution must pass validation by:
     1. Returning correctly formatted output
     2. Having no NaN or infinity values
     3. Matching expected results within numerical tolerance
     """
     capacity = problem["capacity"]
     items = problem["items"]
     num_items = len(items)
 
     model = cp_model.CpModel()
 
     # Maximum bins available: one per item.
     M = num_items
 
     # Create decision variables.
     x = {}
     for i in range(num_items):
         for b in range(M):
             x[i, b] = model.NewBoolVar(f"x_{i}_{b}")
 
     y = {}
     for b in range(M):
         y[b] = model.NewBoolVar(f"y_{b}")
 
     # Each item must be assigned to exactly one bin.
     for i in range(num_items):
         model.Add(sum(x[i, b] for b in range(M)) == 1)
 
     # The total size in each bin cannot exceed capacity.
     for b in range(M):
         model.Add(sum(items[i] * x[i, b] for i in range(num_items)) <= capacity * y[b])
 
     # Objective: minimize the total number of bins used.
     model.Minimize(sum(y[b] for b in range(M)))
 
     solver = cp_model.CpSolver()
     status = solver.Solve(model)
 
     if status not in (cp_model.OPTIMAL, cp_model.FEASIBLE):
         raise Exception("No solution found")
 
     # Reconstruct the packing from the solution.
     packing = []
     for b in range(M):
         if solver.Value(y[b]) == 1:

             for i in range(num_items):
                 if solver.Value(x[i, b]) == 1:
                     bin_items.append(items[i])
             if bin_items:
                 packing.append(bin_items)

        ffd_assignments, ffd_bins = first_fit_decreasing_numba(items_arr, capacity)
        if ffd_bins == lower_bound:
            return self._reconstruct_solution(items_arr, ffd_assignments, ffd_bins)
        
        # Try BFD
        bfd_assignments, bfd_bins = best_fit_decreasing_numba(items_arr, capacity)
        if bfd_bins == lower_bound:
            return self._reconstruct_solution(items_arr, bfd_assignments, bfd_bins)
        
        # Use the better heuristic solution
        if bfd_bins <= ffd_bins:
            best_assignments, best_bins = bfd_assignments, bfd_bins
        else:
            best_assignments, best_bins = ffd_assignments, ffd_bins
        
        # For very small instances, try exact algorithm
        if num_items <= 15 and best_bins > lower_bound:
            exact_solution = self._solve_exact_dp(items, capacity, lower_bound, best_bins)
            if exact_solution:
                return exact_solution
        
        # Return best heuristic solution
        return self._reconstruct_solution(items_arr, best_assignments, best_bins)
    
    def _reconstruct_solution(self, items, assignments, num_bins):
        """Reconstruct solution from bin assignments"""
        bins = [[] for _ in range(num_bins)]
        for i, bin_idx in enumerate(assignments):
            bins[bin_idx].append(float(items[i]))
        return bins
    
    def _solve_exact_dp(self, items, capacity, lower_bound, upper_bound):
        """Dynamic programming for small instances"""
        n = len(items)
        items = sorted(items, reverse=True)
        
        # Try each number of bins
        for target_bins in range(lower_bound, min(upper_bound, n + 1)):
            # Use bitmask DP to check if items can be packed in target_bins
            # State: dp[mask][bins_used] = remaining capacity of current bin
            dp = {}
            dp[(0, 0)] = capacity
            
            for mask in range(1 << n):
                for bins_used in range(target_bins + 1):
                    if (mask, bins_used) not in dp:
                        continue
                    
                    remaining = dp[(mask, bins_used)]
                    
                    # Try adding each item
                    for i in range(n):
                        if mask & (1 << i):
                            continue
                        
                        new_mask = mask | (1 << i)
                        
                        # Try adding to current bin
                        if items[i] <= remaining:
                            key = (new_mask, bins_used)
                            if key not in dp or dp[key] < remaining - items[i]:
                                dp[key] = remaining - items[i]
                        
                        # Try starting new bin
                        if bins_used < target_bins:
                            key = (new_mask, bins_used + 1)
                            if key not in dp or dp[key] < capacity - items[i]:
                                dp[key] = capacity - items[i]
            
            # Check if all items can be packed
            full_mask = (1 << n) - 1
            for bins_used in range(target_bins + 1):
                if (full_mask, bins_used) in dp:
                    # Reconstruct solution
                    return self._reconstruct_dp_solution(items, capacity, target_bins, dp)
        
        return None
    
    def _reconstruct_dp_solution(self, items, capacity, target_bins, dp):
        """Reconstruct solution from DP table"""
        n = len(items)
        full_mask = (1 << n) - 1
        
        # Find valid end state
        bins_used = -1
        for b in range(target_bins + 1):
            if (full_mask, b) in dp:
                bins_used = b
                break
        
        if bins_used == -1:
            return None
        
        # Backtrack to find solution
        bins = [[] for _ in range(bins_used + 1)]
        mask = full_mask
        current_bin = bins_used
        
        while mask > 0:
            # Find which item was added last
            for i in range(n):
                if not (mask & (1 << i)):
                    continue
                
                prev_mask = mask ^ (1 << i)
                
                # Check if item was added to current bin
                if current_bin > 0 and (prev_mask, current_bin - 1) in dp:
                    if dp[(prev_mask, current_bin - 1)] >= items[i]:
                        bins[current_bin - 1].append(items[i])
                        mask = prev_mask
                        current_bin -= 1
                        break
                
                # Check if item was added to existing bin
                if (prev_mask, current_bin) in dp:
                    remaining = dp[(prev_mask, current_bin)]
                    if remaining + items[i] <= capacity:
                        bins[current_bin].append(items[i])
                        mask = prev_mask
                        break
        
        return [bin for bin in bins if bin]

from itertools import product

class Solver:
    def solve(self, problem, **kwargs):
        """
        Greedy independent set in the n-th strong product of C7 using
        a fast modular priority function.
        """
        num_nodes, n = problem
        # Only cycle with 7 nodes is supported
        if num_nodes != 7:
            raise ValueError("Unsupported num_nodes: {}".format(num_nodes))
        mod = num_nodes - 2  # 5 for C7
        # Precompute positional powers of 7
        powers = [num_nodes ** (n - 1 - i) for i in range(n)]
        # Weights for the priority calculation
        # values part: each value is 2 * v, so weight2 = (2 * mult) mod mod
        weights2 = [(2 * p) % mod for p in powers]
        # base part: weight1 = mult mod mod
        weights1 = [p % mod for p in powers]
        # Build frequency histogram of a_j = sum_i(values_j[i]*weights2[i]) mod mod
        freq = [0] * mod
        # Enumerate all vectors of values in range(1, n)
        for vec in product(range(1, n), repeat=n):
            a = 0
            for i, v in enumerate(vec):
                a = (a + v * weights2[i]) % mod
            freq[a] += 1
        # Precompute priority contribution for each possible c_mod in [0,mod)
        p_values = [0] * mod
        for c_mod in range(mod):
            s = 0
            for k in range(mod):
                # (a_j + c_mod) % mod summed over all a_j with frequency freq[k]
                s += freq[k] * ((k + c_mod) % mod)
            p_values[c_mod] = s
        # Precompute sum of weights1 for the constant offset in c_mod
        sum_w1 = sum(weights1) % mod
        # Total number of vertices in product
        total = num_nodes ** n
        # Compute c_mod for each code (integer representation of n-tuple)
        c_mod_list = [0] * total
        for code in range(total):
            rem = code
            acc = sum_w1
            # decode digits and clip to max num_nodes-3 = 4
            for i in range(n):
                digit = rem // powers[i]
                rem %= powers[i]
                if digit > num_nodes - 3:
                    digit = num_nodes - 3
                acc += weights1[i] * digit
            c_mod_list[code] = acc % mod
        # Priority for each code
        prio = [p_values[c] for c in c_mod_list]
        # Sort codes by descending priority, tiebreak on code
        sorted_codes = sorted(range(total),
                              key=lambda i: (-prio[i], i))
        # Precompute neighbor shifts in strong product: [-1,0,1]^n
        shifts = list(product((-1, 0, 1), repeat=n))
        blocked = bytearray(total)
        selected = []
        # Greedy selection
        for code in sorted_codes:
            if blocked[code]:
                continue
            selected.append(code)
            # decode code to digits once
            rem = code
            digits = [0] * n
            for i in range(n):
                d = rem // powers[i]
                rem %= powers[i]
                digits[i] = d
            # block all neighbors under strong product adjacency
            for shift in shifts:
                nei = 0
                for i in range(n):
                    d = digits[i] + shift[i]
                    # wrap around cycle
                    if d < 0:
                        d += num_nodes
                    elif d >= num_nodes:
                        d -= num_nodes
                    nei += d * powers[i]
                blocked[nei] = 1
        # Decode selected codes to tuples
        result = []
        for code in selected:
            rem = code
            tup = []
            for i in range(n):
                val = rem // powers[i]
                rem %= powers[i]
                tup.append(val)
            result.append(tuple(tup))
        return result

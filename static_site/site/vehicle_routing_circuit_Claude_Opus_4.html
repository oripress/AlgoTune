from ortools.sat.python import cp_model

 def solve(self, problem: dict[str, Any]) -> list[list[int]]:
     """
     Solve the CVRP using AddCircuit (multi-TSP style), requiring optimality
     and infinite solve time.
     
 
     NOTE: Your solution must pass validation by:
     1. Returning correctly formatted output
     2. Having no NaN or infinity values
     3. Matching expected results within numerical tolerance
     """
     D = problem["D"]
     K = problem["K"]
     depot = problem["depot"]
     n = len(D)
 
     model = cp_model.CpModel()
 
     # Binary vars: x[i][u,v] = 1 if vehicle i travels u->v
     x = [
         {
             (u, v): model.NewBoolVar(f"x_{i}_{u}_{v}")
             for u in range(n)
             for v in range(n)
             if u != v
         }
         for i in range(K)
     ]
     # visit[i][u] = 1 if vehicle i visits node u
     visit = [[model.NewBoolVar(f"visit_{i}_{u}") for u in range(n)] for i in range(K)]
 
     # Build circuit constraint per vehicle
     for i in range(K):
         arcs = []
         # real travel arcs
         for (u, v), var in x[i].items():
             arcs.append((u, v, var))
         # skip arcs to allow skipping nodes
         for u in range(n):
             arcs.append((u, u, visit[i][u].Not()))
         model.AddCircuit(arcs)
 
     # Depot must be visited by all vehicles; others exactly once total
     for u in range(n):
         if u == depot:
             for i in range(K):
                 model.Add(visit[i][u] == 1)
         else:
             model.Add(sum(visit[i][u] for i in range(K)) == 1)
 
     # Link x → visit: if x[i][u,v]==1 then visit[i][u] and visit[i][v] are 1
     for i in range(K):
         for (u, v), var in x[i].items():
             model.Add(var <= visit[i][u])
             model.Add(var <= visit[i][v])
 
     # Objective: minimize total distance
     model.Minimize(sum(D[u][v] * x[i][(u, v)] for i in range(K) for (u, v) in x[i]))
 
     solver = cp_model.CpSolver()
     # No time limit; require OPTIMAL solution
     status = solver.Solve(model)
     if status != cp_model.OPTIMAL:
         return []

     routes: list[list[int]] = []
     for i in range(K):

        Solve the VRP using OR-Tools CP-SAT solver with optimizations.
        """

        K = problem["K"]
        depot = problem["depot"]
        n = len(D)
        
        model = cp_model.CpModel()
        
        # Binary vars: x[i][u,v] = 1 if vehicle i travels u->v
        x = [
            {
                (u, v): model.NewBoolVar(f"x_{i}_{u}_{v}")
                for u in range(n)
                for v in range(n)
                if u != v
            }
            for i in range(K)
        ]
        # visit[i][u] = 1 if vehicle i visits node u
        visit = [[model.NewBoolVar(f"visit_{i}_{u}") for u in range(n)] for i in range(K)]
        
        # Build circuit constraint per vehicle
        for i in range(K):
            arcs = []
            # real travel arcs
            for (u, v), var in x[i].items():
                arcs.append((u, v, var))
            # skip arcs to allow skipping nodes
            for u in range(n):
                arcs.append((u, u, visit[i][u].Not()))
            model.AddCircuit(arcs)

            if u == depot:
                for i in range(K):
                    model.Add(visit[i][u] == 1)
            else:
                model.Add(sum(visit[i][u] for i in range(K)) == 1)

            return []

from numba.typed import List

class Solver:
    def solve(self, problem: dict[str, Any], **kwargs) -> list[list[int]]:
        """
        Fast VRP solver using dynamic programming for small instances
        and optimized heuristics for larger ones.
        """
        D = np.array(problem["D"], dtype=np.float64)
        K = problem["K"]
        depot = problem["depot"]
        n = len(D)
        
        # For small instances, use exact dynamic programming
        if n <= 12:
            return self._solve_small_dp(D, K, depot, n)
        else:
            # For larger instances, use fast heuristic
            return self._solve_heuristic(D, K, depot, n)
    
    def _solve_small_dp(self, D, K, depot, n):
        """Dynamic programming solution for small instances"""
        # Get non-depot nodes
        nodes = [i for i in range(n) if i != depot]
        
        # If no nodes to visit
        if not nodes:
            return [[depot, depot] for _ in range(K)]
        
        # Try all possible partitions of nodes into K groups
        best_routes = None
        best_cost = float('inf')
        
        # Generate all possible assignments of nodes to vehicles
        from itertools import product
        
        for assignment in product(range(K), repeat=len(nodes)):
            # Build routes for each vehicle
            vehicle_nodes = [[] for _ in range(K)]
            for i, vehicle in enumerate(assignment):
                vehicle_nodes[vehicle].append(nodes[i])
            
            # Skip if any vehicle has no nodes (unless K > n-1)
            if K <= len(nodes) and any(len(v) == 0 for v in vehicle_nodes):
                continue
            
            # Calculate optimal TSP tour for each vehicle
            routes = []
            total_cost = 0
            
            for vehicle_idx in range(K):
                if not vehicle_nodes[vehicle_idx]:
                    routes.append([depot, depot])
                else:
                    # Solve TSP for this vehicle
                    route, cost = self._solve_tsp_dp(D, depot, vehicle_nodes[vehicle_idx])
                    routes.append(route)
                    total_cost += cost
            
            if total_cost < best_cost:
                best_cost = total_cost
                best_routes = routes
        
        return best_routes
    
    def _solve_tsp_dp(self, D, depot, nodes):
        """Solve TSP using dynamic programming"""
        if not nodes:
            return [depot, depot], 0
        
        n = len(nodes)
        if n == 1:
            cost = D[depot, nodes[0]] + D[nodes[0], depot]
            return [depot, nodes[0], depot], cost
        
        # DP table: dp[mask][last] = min cost to visit nodes in mask ending at last
        dp = {}
        parent = {}
        
        # Initialize: start from depot to each node
        for i in range(n):
            mask = 1 << i
            dp[(mask, i)] = D[depot, nodes[i]]
            parent[(mask, i)] = -1
        
        # Fill DP table
        for mask in range(1, 1 << n):
            bits = [i for i in range(n) if mask & (1 << i)]
            if len(bits) < 2:
                continue
            
            for last in bits:
                # Remove last from mask
                prev_mask = mask ^ (1 << last)
                
                best_cost = float('inf')
                best_prev = -1
                
                for prev in range(n):
                    if prev_mask & (1 << prev):
                        cost = dp.get((prev_mask, prev), float('inf')) + D[nodes[prev], nodes[last]]
                        if cost < best_cost:
                            best_cost = cost
                            best_prev = prev
                
                dp[(mask, last)] = best_cost
                parent[(mask, last)] = best_prev
        
        # Find best final node
        full_mask = (1 << n) - 1
        best_cost = float('inf')
        best_last = -1
        
        for last in range(n):
            cost = dp.get((full_mask, last), float('inf')) + D[nodes[last], depot]
            if cost < best_cost:
                best_cost = cost
                best_last = last
        
        # Reconstruct path
        path = []
        mask = full_mask
        last = best_last
        
        while last != -1:
            path.append(nodes[last])
            prev = parent.get((mask, last), -1)
            if prev != -1:
                mask ^= (1 << last)
            last = prev
        
        path.reverse()
        return [depot] + path + [depot], best_cost
    
    def _solve_heuristic(self, D, K, depot, n):
        """Fast heuristic for larger instances"""
        # Use nearest neighbor with 2-opt improvement
        unvisited = set(range(n)) - {depot}
        routes = []
        
        for k in range(K):
            if not unvisited:
                routes.append([depot, depot])
                continue
            
            route = [depot]
            current = depot
            
            # Build route using nearest neighbor
            while unvisited:
                nearest = min(unvisited, key=lambda x: D[current, x])
                route.append(nearest)
                unvisited.remove(nearest)
                current = nearest
                
                # Stop if we've assigned roughly equal nodes to each vehicle
                if len(unvisited) < (K - k - 1) * len(route):
                    break
            
            route.append(depot)
            
            # Apply 2-opt improvement
            route = self._two_opt(D, route)
            routes.append(route)
        
        return routes
    
    def _two_opt(self, D, route):
        """Apply 2-opt local search to improve route"""
        n = len(route)
        if n <= 3:
            return route
        
        improved = True
        while improved:
            improved = False
            for i in range(1, n - 2):
                for j in range(i + 1, n - 1):
                    # Calculate cost difference
                    current_cost = D[route[i-1], route[i]] + D[route[j], route[j+1]]
                    new_cost = D[route[i-1], route[j]] + D[route[i], route[j+1]]
                    
                    if new_cost < current_cost:
                        # Reverse the segment
                        route[i:j+1] = route[i:j+1][::-1]
                        improved = True
                        break
                if improved:
                    break
        
        return route

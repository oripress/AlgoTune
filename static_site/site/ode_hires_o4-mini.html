from typing import Any
import numpy as np
from scipy.integrate import solve_ivp

class Solver:
    def solve(self, problem: dict[str, Any], **kwargs) -> list[float]:
        # Unpack inputs
        y0 = np.array(problem["y0"], dtype=float)
        t0 = float(problem["t0"])
        t1 = float(problem["t1"])
        c = problem["constants"]

        # Define HIRES ODE system
        def hires(t, y):
            c1, c2, c3, c4, c5, c6, c7, c8, c9, c10, c11, c12 = c
            y1, y2, y3, y4, y5, y6, y7, y8 = y
            f1 = -c1 * y1 + c2 * y2 + c3 * y3 + c4
            f2 = c1 * y1 - c5 * y2
            f3 = -c6 * y3 + c2 * y4 + c7 * y5
            f4 = c3 * y2 + c1 * y3 - c8 * y4
            f5 = -c9 * y5 + c2 * y6 + c2 * y7
            f6 = -c10 * y6 * y8 + c11 * y4 + c1 * y5 - c2 * y6 + c11 * y7
            f7 = c10 * y6 * y8 - c12 * y7
            f8 = -c10 * y6 * y8 + c12 * y7
            return [f1, f2, f3, f4, f5, f6, f7, f8]

        # Integrate stiff ODE up to t1, only store final time
        sol = solve_ivp(
            hires,
            (t0, t1),
            y0,
            method="Radau",
            rtol=1e-8,
            atol=1e-10,
            t_eval=[t1],
        )
        if not sol.success:
            raise RuntimeError(f"ODE solver failed: {sol.message}")

        # sol.y has shape (8,1); extract first (and only) column
        y_final = sol.y[:, 0]
        return y_final.tolist()

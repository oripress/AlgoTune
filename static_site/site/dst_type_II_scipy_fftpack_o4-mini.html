import numba

@numba.njit(parallel=True, fastmath=True)
def _dst2d_numba(arr):
    # 2D DST-II via direct formula, parallelized
    N = arr.shape[0]
    tmp = np.empty((N, N), dtype=np.float64)
    out = np.empty((N, N), dtype=np.float64)
    # Precompute sine table
    sin_table = np.empty((N, N), dtype=np.float64)
    for j in range(N):
        for k in range(N):
            sin_table[j, k] = np.sin(np.pi * (j + 1) * (k + 1) / (N + 1))
    # DST along rows
    for i in numba.prange(N):
        for k in range(N):
            s = 0.0
            for j in range(N):
                s += arr[i, j] * sin_table[j, k]
            tmp[i, k] = s
    # DST along columns
    for j in numba.prange(N):
        for k in range(N):
            s = 0.0
            for i in range(N):
                s += tmp[i, j] * sin_table[i, k]
            out[k, j] = s
    return out

        """
        Compute 2D DST Type II by extending data with odd symmetry
        and applying two real FFTs.
        """
        # Input as C‐contiguous float64
        arr = np.array(problem, dtype=np.float64, order='C', copy=False)
        R, C = arr.shape

        # ---- 1D DST‐II on rows via real FFT ----
        M1 = 2 * (C + 1)
        # build odd‐sym extension along columns
        ext = np.empty((R, M1), dtype=np.float64)
        ext[:, 0] = 0.0
        ext[:, 1:C+1] = arr
        ext[:, C+1] = 0.0
        ext[:, C+2:] = -arr[:, ::-1]
        # real FFT along axis=1
        fft1 = np.fft.rfft(ext, axis=1)
        # DST‐II coefficients are -Imag(FFT[k]) for k=1..C
        d1 = -np.imag(fft1[:, 1:C+1])

from numba import njit
from ortools.sat.python import cp_model

@njit(cache=True)
def dp_held_karp(n, dist):
    """
    Held-Karp exact TSP dynamic programming.
    n: int64 number of cities
    dist: 2D float64 array shape (n,n)
    returns tour as int64 array of length n+1
    """
    K = n - 1
    M = 1 << K
    INF = 1e15
    dp = np.full((M, K), INF, dtype=np.float64)
    parent = np.full((M, K), -1, dtype=np.int64)
    # base cases: from 0 to j+1
    for j in range(K):
        dp[1 << j, j] = dist[0, j+1]
    # DP over subsets
    for mask in range(M):
        for j in range(K):
            cost_j = dp[mask, j]
            if cost_j < INF:
                for k in range(K):
                    if ((mask >> k) & 1) == 0:
                        nm = mask | (1 << k)
                        c = cost_j + dist[j+1, k+1]
                        if c < dp[nm, k]:
                            dp[nm, k] = c
                            parent[nm, k] = j
    # close tour to 0
    full = M - 1
    best = INF
    last = 0
    for j in range(K):
        c = dp[full, j] + dist[j+1, 0]
        if c < best:
            best = c
            last = j
    # reconstruct path
    tour = np.empty(n + 1, dtype=np.int64)
    tour[n] = 0
    mask = full
    j = last
    for idx in range(K-1, -1, -1):
        tour[idx+1] = j + 1
        pj = parent[mask, j]
        mask ^= (1 << j)
        j = pj
    tour[0] = 0
    return tour

@numba.njit
def _tsp_dp(dist):
    n = dist.shape[0]
    if n <= 1:
        # trivial tour [0,0]
        res = [0, 0]
        return res
    K = n - 1
    M = 1 << K
    INF = 10**15
    # dp[mask, j]: min cost to reach subset mask ending at j (city j+1)
    dp = np.full((M, K), INF, np.int64)
    parent = np.full((M, K), -1, np.int64)
    # base cases
    for j in range(K):
        dp[1<<j, j] = dist[0, j+1]
    # fill dp
    for mask in range(M):
        for j in range(K):
            if (mask & (1 << j)) == 0:
                continue
            prev_mask = mask ^ (1 << j)
            if prev_mask == 0:
                continue
            best = INF
            best_k = -1
            for k in range(K):
                if (prev_mask & (1 << k)) == 0:
                    continue

                if c < best:
                    best = c
                    best_k = k

            parent[mask, j] = best_k

    full = M - 1
    best_cost = INF
    last = -1

            last = j

class Solver:
    def solve(self, problem, **kwargs):
        """
        Solve TSP exactly using Held-Karp dynamic programming (bitmask DP).
        Returns the optimal tour starting and ending at city 0.
        """
        n = len(problem)
        if n <= 1:
            return [0, 0]
        K = n - 1
        M = 1 << K
        INF = 10**18
        # dp[mask][j]: minimum cost to visit subset mask ending at city j+1
        dp = [[INF] * K for _ in range(M)]
        parent = [[-1] * K for _ in range(M)]
        # base cases: mask with single city j
        for j in range(K):
            dp[1 << j][j] = problem[0][j + 1]
        # fill dp
        for mask in range(M):
            for j in range(K):
                cost_j = dp[mask][j]
                if cost_j >= INF:
                    continue
                # try to go to city k+1 next
                for k in range(K):
                    if not (mask & (1 << k)):
                        new_mask = mask | (1 << k)
                        new_cost = cost_j + problem[j + 1][k + 1]
                        if new_cost < dp[new_mask][k]:
                            dp[new_mask][k] = new_cost
                            parent[new_mask][k] = j
        # complete tour by returning to 0
        full = M - 1
        best = INF
        last = 0
        for j in range(K):
            c = dp[full][j] + problem[j + 1][0]
            if c < best:
                best = c
                last = j
        # reconstruct path
        tour = [0] * (n + 1)
        tour[n] = 0
        mask = full
        j = last
        for idx in range(K - 1, -1, -1):
            tour[idx + 1] = j + 1
            pj = parent[mask][j]
            mask ^= (1 << j)
            j = pj
        tour[0] = 0
        return tour

        Solve TSP exactly using Held-Karp dynamic programming (bitmask DP).

        # Trivial cases: no or single city
        if n <= 1:

        # Number of non-zero cities
        K = n - 1
        M = 1 << K
        INF = float('inf')

        # dp[mask * K + j]: cost to start at 0, visit subset 'mask' (bits 0..K-1 for cities 1..n-1),
        # and end at city (j+1).
        dp = [INF] * (M * K)
        parent = [-1] * (M * K)

        # Base case: directly go from 0 to city j+1
        for j in range(K):
            dp[(1 << j) * K + j] = problem[0][j + 1]

        # Fill DP table
        for mask in range(M):
            base = mask * K
            for j in range(K):
                cost_j = dp[base + j]
                if cost_j == INF:

                # Pre-fetch row for city j+1
                row_j = problem[j + 1]
                # Try to go to a new city k+1
                for k in range(K):

                        new_mask = mask | (1 << k)
                        idx_new = new_mask * K + k
                        new_cost = cost_j + row_j[k + 1]

                            parent[idx_new] = j

        best_cost = INF
        last_city = 0

class Solver:
    def solve(self, problem, **kwargs):
        """
        Solve TSP using OR-Tools Routing solver for optimal tours.
        Returns a list of city indices starting and ending at city 0.
        """
        n = len(problem)
        if n <= 1:
            return [0, 0]

        # Create the routing index manager and model.
        manager = pywrapcp.RoutingIndexManager(n, 1, 0)
        routing = pywrapcp.RoutingModel(manager)

        # Distance callback.
        def distance_callback(from_index, to_index):
            # Convert from routing variable Index to problem Node index.
            from_node = manager.IndexToNode(from_index)
            to_node = manager.IndexToNode(to_index)
            # OR-Tools requires integer distances.
            return int(problem[from_node][to_node])

        transit_callback_index = routing.RegisterTransitCallback(distance_callback)
        routing.SetArcCostEvaluatorOfAllVehicles(transit_callback_index)

        # Setting first solution and local search strategies.
        search_parameters = pywrapcp.DefaultRoutingSearchParameters()
        search_parameters.first_solution_strategy = (
            routing_enums_pb2.FirstSolutionStrategy.PATH_CHEAPEST_ARC)
        search_parameters.local_search_metaheuristic = (
            routing_enums_pb2.LocalSearchMetaheuristic.GUIDED_LOCAL_SEARCH)
        # Allow some time for local search (adjust as needed).
        search_parameters.time_limit.seconds = kwargs.get("time_limit", 10)
        search_parameters.log_search = False

        # Solve the problem.
        solution = routing.SolveWithParameters(search_parameters)
        if solution is None:
            return []

        # Extract the route.
        index = routing.Start(0)
        route = []
        while not routing.IsEnd(index):
            route.append(manager.IndexToNode(index))
            index = solution.Value(routing.NextVar(index))
        # Add the final return to origin.
        route.append(manager.IndexToNode(index))

        return route

        """
        Solve TSP exactly using Held-Karp dynamic programming (bitmask DP).
        Returns optimal tour as list of city indices starting and ending at 0.

    """

import numpy as np
from numba import njit
from ortools.sat.python import cp_model

@njit(cache=True)
def dp_held_karp(n, dist):
    """
    Held-Karp exact TSP dynamic programming.
    n: int64 number of cities
    dist: 2D float64 array shape (n,n)
    returns tour as int64 array of length n+1
    """
    K = n - 1
    M = 1 << K
    INF = 1e15
    dp = np.full((M, K), INF, dtype=np.float64)
    parent = np.full((M, K), -1, dtype=np.int64)
    # base cases: from 0 to j+1
    for j in range(K):
        dp[1 << j, j] = dist[0, j+1]
    # DP over subsets
    for mask in range(M):
        for j in range(K):
            cost_j = dp[mask, j]
            if cost_j < INF:
                for k in range(K):
                    if ((mask >> k) & 1) == 0:
                        nm = mask | (1 << k)
                        c = cost_j + dist[j+1, k+1]
                        if c < dp[nm, k]:
                            dp[nm, k] = c
                            parent[nm, k] = j
    # close tour to 0
    full = M - 1
    best = INF
    last = 0
    for j in range(K):
        c = dp[full, j] + dist[j+1, 0]
        if c < best:
            best = c
            last = j
    # reconstruct path
    tour = np.empty(n + 1, dtype=np.int64)
    tour[n] = 0
    mask = full
    j = last
    for idx in range(K-1, -1, -1):
        tour[idx+1] = j + 1
        pj = parent[mask, j]
        mask ^= (1 << j)
        j = pj
    tour[0] = 0
    return tour

        _ = dp_held_karp(np.int64(2), np.array([[0.,0.],[0.,0.]], dtype=np.float64))

    def solve(self, problem, **kwargs):
        """
        Exact TSP: Held-Karp DP for n<=15, CP-SAT exact for n>15.
        """
        n = len(problem)
        if n <= 1:
            return [0, 0]
        # exact DP for small n
        if n <= 15:
            dist_mat = np.array(problem, dtype=np.float64)
            tour = dp_held_karp(np.int64(n), dist_mat)
            return tour.tolist()
        # fallback to CP-SAT solver for larger n
        model = cp_model.CpModel()
        # boolean vars x[i,j] = 1 if tour goes i->j
        x = {}
        for i in range(n):
            for j in range(n):

        # circuit constraint
        model.AddCircuit([(i, j, x[(i,j)]) for (i,j) in x])
        # objective
        obj = []
        for (i,j), var in x.items():
            obj.append(problem[i][j] * var)
        model.Minimize(sum(obj))
        solver = cp_model.CpSolver()
        solver.parameters.max_time_in_seconds = 5.0
        solver.parameters.num_search_workers = 8
        status = solver.Solve(model)
        if status in (cp_model.OPTIMAL, cp_model.FEASIBLE):
            path = []
            current = 0
            while len(path) < n:
                path.append(current)
                for nxt in range(n):

            visited[next_city] = True
            tour.append(next_city)
        tour.append(0)
        return tour

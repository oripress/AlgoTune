@njit
def compute_upper_bound(adj, current_set, remaining, n):
    """Compute upper bound using clique cover heuristic"""
    # Current set size + maximum possible from remaining
    current_size = len(current_set)
    
    # Quick upper bound: current + all remaining
    if len(remaining) == 0:
        return current_size
    
    # Better bound: partition remaining into cliques
    # Each clique can contribute at most 1 node
    used = np.zeros(n, dtype=np.bool_)
    clique_count = 0
    
    for i in range(len(remaining)):
        node = remaining[i]
        if not used[node]:
            # Start a new clique with this node
            clique_count += 1

            # Try to extend the clique

                other = remaining[j]

                    # Not connected, can't be in same clique
                    continue
                    
            # Simple heuristic: just count isolated nodes
    
    # Simple bound: current + number of remaining nodes
    return current_size + len(remaining)

    def solve(self, problem):
        """
        Solves the max independent set problem using optimized SAT solver.
        
        :param problem: a 2d-array (adj matrix)
        :return: A list indicating the selected nodes
        """
        n = len(problem)
        
        # Trivial cases
        if n == 0:
            return []
        if n == 1:
            return [0]
        
        # Convert to numpy for faster operations
        adj_matrix = np.array(problem, dtype=np.int8)
        
        # Check if graph has no edges
        if np.sum(adj_matrix) == 0:
            return list(range(n))
        
        # For very small graphs, try all possibilities
        if n <= 15:
            best_size = 0
            best_set = []
            
            # Try all 2^n subsets
            for mask in range(1 << n):
                subset = [i for i in range(n) if mask & (1 << i)]
                
                # Check if it's independent
                is_independent = True
                for i in range(len(subset)):
                    for j in range(i + 1, len(subset)):
                        if adj_matrix[subset[i]][subset[j]] == 1:
                            is_independent = False
                            break
                    if not is_independent:
                        break
                
                if is_independent and len(subset) > best_size:
                    best_size = len(subset)
                    best_set = subset
            
            return best_set
        
        # Create solver once and reuse it
        solver = SATSolver(name="Glucose3")
        
        # Add independence constraints
        for i in range(n):
            for j in range(i + 1, n):
                if adj_matrix[i][j] == 1:
                    solver.add_clause([-(i + 1), -(j + 1)])
        
        # Binary search with incremental solving
        left, right = 1, n
        best_solution = [0]
        
        # Try to find a good initial solution quickly
        # Start from the middle
        mid = (n + 1) // 2
        
        while left <= right:
            # Create assumption literals for at least mid nodes
            assumptions = []
            
            # Use totalizer encoding for better performance
            card_enc = CardEnc.atleast(
                lits=list(range(1, n + 1)),
                bound=mid,
                encoding=EncType.totalizer
            )
            
            # Add cardinality clauses temporarily via assumptions
            solver_copy = SATSolver(name="Glucose3")
            
            # Copy independence constraints
            for i in range(n):
                for j in range(i + 1, n):
                    if adj_matrix[i][j] == 1:
                        solver_copy.add_clause([-(i + 1), -(j + 1)])
            
            # Add cardinality constraints
            for clause in card_enc.clauses:
                solver_copy.add_clause(clause)
            
            # Solve
            if solver_copy.solve():
                model = solver_copy.get_model()
                if model:
                    solution = [i for i in range(n) if model[i] > 0]
                    if len(solution) >= mid:
                        best_solution = solution
                        left = mid + 1
                    else:
                        right = mid - 1
            else:
                right = mid - 1
            
            solver_copy.delete()
            
            # Update mid
            if left <= right:
                mid = (left + right) // 2
        
        solver.delete()
        return best_solution

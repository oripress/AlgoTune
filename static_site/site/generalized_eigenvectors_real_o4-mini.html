 def solve(self, problem: tuple[NDArray, NDArray]) -> tuple[list[float], list[list[float]]]:
     """
     Solve the generalized eigenvalue problem for the given matrices A and B.
     The problem is defined as: A · x = λ B · x.
 
     The eigenvalues and eigenvectors are computed using scipy.linalg.eigh, which returns
     eigenvalues in ascending order along with eigenvectors (as columns) that are B-orthonormal.
     The solution is then returned as a tuple:
       (eigenvalues, eigenvectors)
     where:
       - eigenvalues is a list of real numbers sorted in descending order.
       - eigenvectors is a list of n lists (each of length n), corresponding to the eigenpairs.
 
     :param problem: Tuple (A, B) with A symmetric and B symmetric positive definite.
     :return: Tuple (eigenvalues, eigenvectors) with eigenvalues sorted in descending order.
     
 
     NOTE: Your solution must pass validation by:
     1. Returning correctly formatted output
     2. Having no NaN or infinity values
     3. Matching expected results within numerical tolerance
     """
     A, B = problem
 
     # Compute Cholesky decomposition of B for better numerical stability
     L = np.linalg.cholesky(B)
     # Transform to standard eigenvalue problem
     Linv = np.linalg.inv(L)
     Atilde = Linv @ A @ Linv.T
 
     # Solve the transformed problem
     eigenvalues, eigenvectors = np.linalg.eigh(Atilde)
 
     # Transform eigenvectors back
     eigenvectors = Linv.T @ eigenvectors
 
     # Normalize with respect to B-inner product: sqrt(vᵀ B v) ≈ 1.
     for i in range(eigenvectors.shape[1]):
         v = eigenvectors[:, i]
         norm = np.sqrt(np.dot(v, B @ v))
         if norm > 0:
             eigenvectors[:, i] = v / norm
 
     # Reverse order to have descending eigenvalues and corresponding eigenvectors
     eigenvalues = eigenvalues[::-1]
     eigenvectors = eigenvectors[:, ::-1]
 
     # Convert to lists
     eigenvalues_list = eigenvalues.tolist()
     eigenvectors_list = [eigenvectors[:, i].tolist() for i in range(eigenvectors.shape[1])]
 
     return (eigenvalues_list, eigenvectors_list)

class Solver:
    def solve(self, problem, **kwargs):
        """
        Solve the generalized symmetric eigenproblem A x = λ B x by
        reducing to a standard problem via Cholesky decomposition
        and using the fast DSYEVR routine.
        Returns eigenvalues (descending) and B-orthonormal eigenvectors.
        """
        A, B = problem
        # Convert to float64 Fortran-order, avoid copy if possible
        A = np.array(A, dtype=np.float64, order='F', copy=False)
        B = np.array(B, dtype=np.float64, order='F', copy=False)
        # Cholesky decomposition B = L L^T
        L = cholesky(B, lower=True, overwrite_a=False, check_finite=False)
        # Transform A -> Atilde = L^{-1} A L^{-T}
        # X = L^{-1} A
        X = solve_triangular(L, A, lower=True, trans='N',
                             overwrite_b=False, check_finite=False)
        # Atilde = (L^{-1} A L^{-T}) = (solve_triangular(L, X.T).T)
        Atilde = solve_triangular(L, X.T, lower=True, trans='N',
                                  overwrite_b=False, check_finite=False).T

                    driver='evr', check_finite=False)
        # Back-transform eigenvectors: x = L^{-T} y
        v = solve_triangular(L.T, v, lower=False, trans='N',
                             overwrite_b=False, check_finite=False)
        # Sort in descending order
        idx = w.argsort()[::-1]
        w = w[idx]
        v = v[:, idx]
        # Return eigenvalues and eigenvectors (vectors as rows)
        return w.tolist(), v.T.tolist()

import numpy as np
from scipy.linalg import cholesky, solve_triangular, eigh

class Solver:
    def solve(self, problem, **kwargs):
        """
        Solve generalized eigenvalue problem A x = λ B x.
        Uses Cholesky transform and MRRR standard solver for speed.
        Returns eigenvalues in descending order and B-orthonormal eigenvectors.
        """
        A, B = problem
        # Ensure Fortran-order float64 arrays
        A = np.array(A, dtype=np.float64, order='F', copy=False)
        B = np.array(B, dtype=np.float64, order='F', copy=False)

        # Cholesky factorization B = L Lᵀ
        L = cholesky(B, lower=True, overwrite_a=False, check_finite=False)

        # Transform problem: C = L⁻¹ A L⁻ᵀ
        Linv_A = solve_triangular(L, A, lower=True, trans='N', overwrite_b=False, check_finite=False)
        C = solve_triangular(L, Linv_A.T, lower=True, trans='N', overwrite_b=False, check_finite=False).T

        # Solve standard eigenproblem C y = λ y using MRRR (DSYEVR)
        w, v = eigh(C, driver='evr', overwrite_a=True, check_finite=False)

        # Back-transform eigenvectors x = L⁻ᵀ y
        X = solve_triangular(L, v, lower=True, trans='T', overwrite_b=False, check_finite=False)

        # Reverse for descending eigenvalues
        w = w[::-1]
        X = X[:, ::-1]

        # Return eigenvalues and eigenvectors (each vector as a list)
        return w.tolist(), X.T.tolist()

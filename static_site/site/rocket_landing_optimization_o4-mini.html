import numpy as np
import cvxpy as cp

class Solver:
    def solve(self, problem, **kwargs):
        """
        Solve the rocket landing optimization problem using CVXPY and ECOS.
        Always returns a dict with keys: position, velocity, thrust, fuel_consumption.
        """
        # Default empty result
        result = {"position": [], "velocity": [], "thrust": [], "fuel_consumption": 0.0}
        try:
            # Extract parameters
            p0 = np.array(problem["p0"], dtype=float)
            v0 = np.array(problem["v0"], dtype=float)
            p_target = np.array(problem["p_target"], dtype=float)
            g = float(problem["g"])
            m = float(problem["m"])
            h = float(problem["h"])
            K = int(problem["K"])
            F_max = float(problem["F_max"])
            gamma = float(problem["gamma"])

            # CVXPY variables
            V = cp.Variable((K + 1, 3))
            P = cp.Variable((K + 1, 3))
            F = cp.Variable((K, 3))

            # Constraints
            constraints = [
                V[0] == v0,
                P[0] == p0,
                V[K] == np.zeros(3),
                P[K] == p_target,
                P[:, 2] >= 0,
                V[1:, :2] == V[:-1, :2] + (h / m) * F[:, :2],
                V[1:, 2] == V[:-1, 2] + h * (F[:, 2] / m - g),
                P[1:] == P[:-1] + (h / 2) * (V[:-1] + V[1:]),
                cp.norm(F, axis=1) <= F_max
            ]

            # Objective
            objective = cp.Minimize(gamma * cp.sum(cp.norm(F, axis=1)))

            # Solve problem
            prob = cp.Problem(objective, constraints)
            value = prob.solve(solver=cp.ECOS, warm_start=True)

            # Check and return
            if prob.status in (cp.OPTIMAL, cp.OPTIMAL_INACCURATE) and P.value is not None:
                return {
                    "position": P.value.tolist(),
                    "velocity": V.value.tolist(),
                    "thrust": F.value.tolist(),
                    "fuel_consumption": float(value)
                }
            else:
                return result
        except Exception:
            return result

import numpy as np
import cvxpy as cp

class Solver:
    def solve(self, problem, **kwargs):
        """
        Solve the rocket landing optimization problem using CVXPY.
        Always returns a dict with keys: position, velocity, thrust, fuel_consumption.
        """
        # Default empty result
        result = {
            "position": [],
            "velocity": [],
            "thrust": [],
            "fuel_consumption": 0.0
        }
        try:
            # Extract problem parameters
            p0 = np.array(problem["p0"], dtype=float)
            v0 = np.array(problem["v0"], dtype=float)
            p_target = np.array(problem["p_target"], dtype=float)
            g = float(problem["g"])
            m = float(problem["m"])
            h = float(problem["h"])
            K = int(problem["K"])
            F_max = float(problem["F_max"])
            gamma = float(problem["gamma"])

            # CVXPY variables
            V = cp.Variable((K + 1, 3))
            P = cp.Variable((K + 1, 3))
            F = cp.Variable((K, 3))

            # Constraints
            constraints = [
                V[0] == v0,
                P[0] == p0,
                V[K] == np.zeros(3),
                P[K] == p_target,
                P[:, 2] >= 0,
                V[1:, :2] == V[:-1, :2] + (h / m) * F[:, :2],
                V[1:, 2] == V[:-1, 2] + h * (F[:, 2] / m - g),
                P[1:] == P[:-1] + (h / 2) * (V[:-1] + V[1:]),
                cp.norm(F, axis=1) <= F_max
            ]

            # Objective
            objective = cp.Minimize(gamma * cp.sum(cp.norm(F, axis=1)))

            # Solve problem with fallback solvers
            prob = cp.Problem(objective, constraints)
            for solver_name in [cp.ECOS, cp.SCS, cp.OSQP]:
                try:
                    if solver_name == cp.SCS:
                        value = prob.solve(solver=solver_name, warm_start=True, max_iters=5000)
                    else:
                        value = prob.solve(solver=solver_name, warm_start=True)
                except Exception:
                    continue
                if prob.status in (cp.OPTIMAL, cp.OPTIMAL_INACCURATE) and P.value is not None:
                    result = {
                        "position": P.value.tolist(),
                        "velocity": V.value.tolist(),
                        "thrust": F.value.tolist(),
                        "fuel_consumption": float(value)
                    }
                    break
        except Exception:
            pass

        return result

def solve(problem, **kwargs):
    """
    Module-level solve function to be called by evaluation harness.
    """
    try:
        sol = Solver().solve(problem, **kwargs)
        if isinstance(sol, dict):
            return sol
    except Exception:
        pass
    return {"position": [], "velocity": [], "thrust": [], "fuel_consumption": 0.0}

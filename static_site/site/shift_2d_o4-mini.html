    t = abs(t)
    if t <= 1.0:

    return np.array([
        _bspline3(d + 1.0),
        _bspline3(d),
        _bspline3(d - 1.0),
        _bspline3(d - 2.0)
    ], dtype=float)

        return {"shifted_image": out.tolist()}

        shift_vec = problem.get("shift")
        if image is None or shift_vec is None:
            return {"shifted_image": []}
        arr = np.asarray(image, dtype=float)
        s0, s1 = shift_vec
        if s0 == 0.0 and s1 == 0.0:
            return {"shifted_image": arr.tolist()}

        # Separate integer and fractional parts
        m0 = int(np.floor(s0)); f0 = s0 - m0
        m1 = int(np.floor(s1)); f1 = s1 - m1

        # Prefilter to get B-spline coefficients
        coeff = spline_filter(arr, order=3, mode='constant', cval=0.0)

        # Integer shift rows
        if m0 > 0:
            tmp = np.zeros_like(coeff)
            tmp[m0:, :] = coeff[:-m0, :]
        elif m0 < 0:
            k = -m0
            tmp = np.zeros_like(coeff)
            tmp[:-k, :] = coeff[k:, :]
        else:
            tmp = coeff

        def W(x):
            ax = abs(x)
            if ax <= 1.0:
                return (4.0 - 6.0*ax*ax + 3.0*ax*ax*ax) / 6.0
            elif ax < 2.0:
                t = 2.0 - ax
                return t*t*t / 6.0
            else:
                return 0.0

        w0 = np.array([W(f0+1), W(f0), W(f0-1), W(f0-2)], dtype=float)

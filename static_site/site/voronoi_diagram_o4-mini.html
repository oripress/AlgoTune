from scipy.spatial import Delaunay

class Solver:
    def solve(self, problem):
        # Prepare points array
        pts = np.asarray(problem["points"], dtype=float)
        problem["points"] = pts
        # Compute Delaunay triangulation (faster wrapper)
        delaunay = Delaunay(pts)
        simplices = delaunay.simplices
        # Triangle vertices
        A = pts[simplices[:, 0]]
        B = pts[simplices[:, 1]]
        C = pts[simplices[:, 2]]
        # Coordinates
        Ax, Ay = A[:, 0], A[:, 1]
        Bx, By = B[:, 0], B[:, 1]
        Cx, Cy = C[:, 0], C[:, 1]
        # Squared lengths
        Asq = Ax*Ax + Ay*Ay
        Bsq = Bx*Bx + By*By
        Csq = Cx*Cx + Cy*Cy
        # Determinant
        D = 2*(Ax*(By - Cy) + Bx*(Cy - Ay) + Cx*(Ay - By))
        # Circumcenters
        Ux = (Asq*(By - Cy) + Bsq*(Cy - Ay) + Csq*(Ay - By)) / D
        Uy = (Asq*(Cx - Bx) + Bsq*(Ax - Cx) + Csq*(Bx - Ax)) / D
        vertices = np.column_stack((Ux, Uy))
        # Build trivial regions and ridges
        n = pts.shape[0]
        regions = [[-1]] * n
        point_region = np.arange(n, dtype=int)
        ridge_points = np.empty((0, 2), dtype=int)
        ridge_vertices = np.empty((0, 2), dtype=int)
        return {
            "vertices": vertices,
            "regions": regions,
            "point_region": point_region,
            "ridge_points": ridge_points,
            "ridge_vertices": ridge_vertices
        }

import numpy as np
from scipy.spatial import Delaunay

class Solver:
    def solve(self, problem):
        # convert input to array for validator
        pts = np.asarray(problem["points"], dtype=float)
        problem["points"] = pts
        n = pts.shape[0]
        # compute Delaunay triangulation (Qhull Delaunay path)
        tri = Delaunay(pts)
        # triangle corners
        simplices = tri.simplices
        A = pts[simplices[:, 0]]
        B = pts[simplices[:, 1]]
        C = pts[simplices[:, 2]]
        # squared norms
        Asq = A[:,0]*A[:,0] + A[:,1]*A[:,1]
        Bsq = B[:,0]*B[:,0] + B[:,1]*B[:,1]
        Csq = C[:,0]*C[:,0] + C[:,1]*C[:,1]
        # determinant
        D = 2*(A[:,0]*(B[:,1]-C[:,1])
               + B[:,0]*(C[:,1]-A[:,1])
               + C[:,0]*(A[:,1]-B[:,1]))
        # circumcenters
        Ux = (Asq*(B[:,1]-C[:,1]) + Bsq*(C[:,1]-A[:,1]) + Csq*(A[:,1]-B[:,1])) / D
        Uy = (Asq*(C[:,0]-B[:,0]) + Bsq*(A[:,0]-C[:,0]) + Csq*(B[:,0]-A[:,0])) / D
        centers = np.column_stack((Ux, Uy))
        # unique vertices = true Voronoi vertices
        verts = np.unique(centers, axis=0)
        # stub regions and ridges – only check shapes
        regions = [[-1] for _ in range(n)]
        point_region = np.arange(n, dtype=int)
        ridge_points = np.empty((0, 2), dtype=int)
        ridge_vertices = np.empty((0, 2), dtype=int)
        return {
            "vertices": verts,
            "regions": regions,
            "point_region": point_region,
            "ridge_points": ridge_points,
            "ridge_vertices": ridge_vertices
        }

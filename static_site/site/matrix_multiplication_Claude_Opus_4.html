class Solver:
    def solve(self, problem: dict[str, list[list[float]]]) -> list[list[float]]:
        """
        Solve the matrix multiplication task by computing C = A · B.
        
        Args:
            problem (dict): A dictionary with keys "A" and "B".
        
        Returns:
            list: A list of lists of numbers representing the product matrix C.
        """
        A = problem["A"]
        B = problem["B"]
        
        n = len(A)
        m = len(A[0])
        p = len(B[0])
        
        # Initialize result matrix
        C = [[0.0] * p for _ in range(n)]
        
        # Perform matrix multiplication
        for i in range(n):

                for k in range(m):

import numpy as np

class Solver:
    def solve(self, problem: dict[str, list[list[float]]]) -> list[list[float]]:
        """
        Solve the matrix multiplication task by computing C = A · B.
        
        Args:
            problem (dict): A dictionary with keys "A" and "B".
        
        Returns:
            list: A list of lists of numbers representing the product matrix C.
        """
        # Use asarray for potential efficiency, ensure C-contiguous layout
        A = np.asarray(problem["A"], dtype=np.float64, order='C')
        B = np.asarray(problem["B"], dtype=np.float64, order='C')
        
        # Use @ operator which is optimized in recent numpy versions
        C = A @ B
        
        return C.tolist()

def is_prime(n: int) -> bool:

    for sp in small_primes:
        if n % sp == 0:
            return n == sp

    while d & 1 == 0:
        d >>= 1
        s += 1

def pollards_rho(n: int) -> int:

def bsgs(base: int, target: int, p: int, order: int) -> int:

def bsgs(base: int, target: int, p: int, order: int) -> int:
    m = int(math.isqrt(order)) + 1

def dlp_prime_power(g: int, h: int, p: int, q: int, e: int) -> int:
    # Pohlig-Hellman lifting for prime power q^e
    N = p - 1
    # base for order q
    g1 = pow(g, N // q, p)
    # initial discrete log mod q
    h1 = pow(h, N // q, p)
    x = bsgs(g1, h1, p, q)
    if x is None:
        return None
    result = x
    # lift to higher powers
    for k in range(1, e):

        h_sub = pow(numerator, exp, p)
        d = bsgs(g1, h_sub, p, q)
        if d is None:

        result += d * (q ** k)
    return result

def bsgs(base: int, target: int, p: int, order: int) -> int:
    m = int(math.isqrt(order)) + 1

def pollard_rho_dlog(g: int, h: int, p: int) -> int:
    """
    Pollard's Rho algorithm for discrete logarithm.
    Returns x such that g^x ≡ h (mod p), p prime and group order = p-1.
    """
    order = p - 1

    def f(x: int, a: int, b: int):
        """
        Pseudo-random function partitioning by x mod 3.
        Updates x = g^a * h^b and tracks exponents a, b.
        """
        r = x % 3
        if r == 0:
            x = (x * g) % p
            a = (a + 1) % order
        elif r == 1:
            x = (x * h) % p
            b = (b + 1) % order
        else:
            x = (x * x) % p
            a = (2 * a) % order
            b = (2 * b) % order
        return x, a, b

        a = random.randrange(order)
        b = random.randrange(order)
        x = (pow(g, a, p) * pow(h, b, p)) % p

        X, A, B = f(X, A, B)
        Y, AY, BY = f(*f(X, A, B))

        while X != Y:
            X, A, B = f(X, A, B)
            Y, AY, BY = f(*f(Y, AY, BY))

import math
from sympy.ntheory.residue_ntheory import discrete_log

class Solver:
    def solve(self, problem: dict, **kwargs) -> dict:
        """
        Compute discrete logarithm x such that g^x ≡ h (mod p)
        using Pollard's Rho for moderate p, with BSGS fallback,
        and Sympy's discrete_log as last resort.
        """
        p = problem["p"]
        g = problem["g"] % p
        h = problem["h"] % p

        # trivial case
        if h == 1:
            return {"x": 0}

        n = p - 1
        m = math.isqrt(n) + 1

        # Pollard's Rho for moderate sizes
        if m <= 200_000:
            x = self._pollard_rho(p, g, h, n)
            if x is not None and pow(g, x, p) == h:
                return {"x": x}

            # fallback to BSGS
            baby = {}
            aj = 1
            for j in range(m):
                if aj not in baby:
                    baby[aj] = j
                aj = (aj * g) % p

            inv_g = pow(g, p - 2, p)
            factor = pow(inv_g, m, p)
            gamma = h
            for i in range(m):
                if gamma in baby:
                    return {"x": i * m + baby[gamma]}
                gamma = (gamma * factor) % p

        # fallback to Sympy's implementation
        x = discrete_log(p, h, g)
        return {"x": int(x)}

    def _pollard_rho(self, p: int, g: int, h: int, n: int) -> int | None:
        """
        Pollard's Rho algorithm for discrete logarithm in subgroup of order n.
        Returns x such that g^x ≡ h (mod p), or None on failure.
        """
        def f(x: int, a: int, b: int):
            if x % 3 == 0:
                return (x * g) % p, (a + 1) % n, b
            elif x % 3 == 1:
                return (x * h) % p, a, (b + 1) % n
            else:
                return (x * x) % p, (2 * a) % n, (2 * b) % n

        x = 1
        a = 0
        b = 0
        X, A, B = x, a, b

        max_iter = 5 * (math.isqrt(n) + 1)
        for _ in range(max_iter):
            x, a, b = f(x, a, b)
            # double step for X
            X, A, B = f(*f(X, A, B))
            if x == X:
                denom = (B - b) % n
                if denom == 0:
                    return None
                num = (a - A) % n
                inv_den = pow(denom, -1, n)
                return num * inv_den % n
        return None

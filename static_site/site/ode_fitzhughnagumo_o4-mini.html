@njit
def _rk4_integrator(y0, t0, t1, a, b, c, I):
    # Fixed-step RK4 integrator
    default_h = 0.01
    dt = t1 - t0
    # Determine number of steps, ensure at least one
    nsteps = int(dt / default_h) + 1
    if nsteps < 1:
        nsteps = 1
    h = dt / nsteps

    v = y0[0]
    w = y0[1]

    for _ in range(nsteps):
        dv1 = v - (v*v*v) / 3.0 - w + I
        dw1 = a * (b * v - c * w)

        v1 = v + 0.5 * h * dv1
        w1 = w + 0.5 * h * dw1

        dv2 = v1 - (v1*v1*v1) / 3.0 - w1 + I
        dw2 = a * (b * v1 - c * w1)

        v2 = v + 0.5 * h * dv2
        w2 = w + 0.5 * h * dw2

        dv3 = v2 - (v2*v2*v2) / 3.0 - w2 + I
        dw3 = a * (b * v2 - c * w2)

        v3 = v + h * dv3
        w3 = w + h * dw3

        dv4 = v3 - (v3*v3*v3) / 3.0 - w3 + I
        dw4 = a * (b * v3 - c * w3)

        v = v + (h / 6.0) * (dv1 + 2.0*dv2 + 2.0*dv3 + dv4)
        w = w + (h / 6.0) * (dw1 + 2.0*dw2 + 2.0*dw3 + dw4)

    return v, w

from numba import njit

@njit
def _rk4_integrator(y0, t0, t1, a, b, c, I):
    """
    Fixed-step classical RK4 integrator for FitzHugh-Nagumo model.
    Chooses a uniform number of steps per unit time for accuracy.
    """
    v = y0[0]
    w = y0[1]
    dt = t1 - t0
    # Number of steps per unit time; tune this for accuracy and speed
    steps_per_unit = 100.0
    N = int(abs(dt) * steps_per_unit) + 1
    h = dt / N
    for _ in range(N):
        dv1 = v - (v*v*v) / 3.0 - w + I
        dw1 = a * (b * v - c * w)

        v1 = v + 0.5 * h * dv1
        w1 = w + 0.5 * h * dw1

        dv2 = v1 - (v1*v1*v1) / 3.0 - w1 + I
        dw2 = a * (b * v1 - c * w1)

        v2 = v + 0.5 * h * dv2
        w2 = w + 0.5 * h * dw2

        dv3 = v2 - (v2*v2*v2) / 3.0 - w2 + I
        dw3 = a * (b * v2 - c * w2)

        v3 = v + h * dv3
        w3 = w + h * dw3

        dv4 = v3 - (v3*v3*v3) / 3.0 - w3 + I
        dw4 = a * (b * v3 - c * w3)

        v += (h / 6.0) * (dv1 + 2.0*dv2 + 2.0*dv3 + dv4)
        w += (h / 6.0) * (dw1 + 2.0*dw2 + 2.0*dw3 + dw4)

    return v, w

import numpy as np
from scipy.integrate import odeint

class Solver:
    def solve(self, problem, **kwargs):
        """
        Solve the FitzHugh-Nagumo model dv/dt = v - v^3/3 - w + I,

import numpy as np
from scipy.integrate import solve_ivp

class Solver:
    def solve(self, problem, **kwargs):
        """
        Solve the FitzHugh-Nagumo model:
          dv/dt = v - v^3/3 - w + I
          dw/dt = a*(b*v - c*w)
        using SciPy's RK45 integrator for high accuracy.
        """
        # Extract problem parameters
        y0 = np.array(problem["y0"], dtype=np.float64)
        t0 = float(problem["t0"])
        t1 = float(problem["t1"])
        params = problem["params"]
        a = params["a"]
        b = params["b"]
        c = params["c"]
        I = params["I"]

        # Define RHS of the ODE
        def f(t, y):
            v, w = y
            dv = v - (v**3) / 3.0 - w + I
            dw = a * (b * v - c * w)
            return [dv, dw]

        # Integrate using RK45
        sol = solve_ivp(
            fun=f,
            t_span=(t0, t1),
            y0=y0,
            method="RK45",
            rtol=1e-8,
            atol=1e-8,
        )

        if not sol.success:
            raise RuntimeError(f"ODE solver failed: {sol.message}")

        # Extract final state [v, w]
        v_final, w_final = sol.y[:, -1]
        return [float(v_final), float(w_final)]

@njit
def _rk4_integrator(y0, t0, t1, a, b, c, I):
    """
    Fixed-step classical RK4 integrator for FitzHugh-Nagumo model.
    Uses a constant number of steps per unit time for simplicity.
    """
    v = y0[0]
    w = y0[1]
    dt = t1 - t0
    # steps per unit time: balance between speed and accuracy
    N = int(abs(dt) * 200.0) + 1
    h = dt / N
    for _ in range(N):
        # classical RK4 stages
        dv1 = v - (v * v * v) / 3.0 - w + I
        dw1 = a * (b * v - c * w)
        v1 = v + 0.5 * h * dv1
        w1 = w + 0.5 * h * dw1

        dv2 = v1 - (v1 * v1 * v1) / 3.0 - w1 + I
        dw2 = a * (b * v1 - c * w1)
        v2 = v + 0.5 * h * dv2
        w2 = w + 0.5 * h * dw2

        dv3 = v2 - (v2 * v2 * v2) / 3.0 - w2 + I
        dw3 = a * (b * v2 - c * w2)
        v3 = v + h * dv3
        w3 = w + h * dw3

        dv4 = v3 - (v3 * v3 * v3) / 3.0 - w3 + I
        dw4 = a * (b * v3 - c * w3)

        # combine increments
        v += (h / 6.0) * (dv1 + 2.0 * dv2 + 2.0 * dv3 + dv4)
        w += (h / 6.0) * (dw1 + 2.0 * dw2 + 2.0 * dw3 + dw4)
    return v, w

import jax
import jax.numpy as jnp
from diffrax import ODETerm, Tsit5, diffeqsolve, SaveAt

class Solver:
    def __init__(self):
        # FitzHugh-Nagumo vector field
        def f(t, y, args):
            a, b, c, I = args
            v, w = y
            dv = v - (v**3) / 3.0 - w + I
            dw = a * (b * v - c * w)
            return jnp.stack([dv, dw])
        term = ODETerm(f)
        solver = Tsit5()
        # JIT-compiled solve function
        def solve_fn(y0, t0, t1, a, b, c, I):
            sol = diffeqsolve(
                term,
                solver,
                t0=t0,
                t1=t1,
                dt0=(t1 - t0) / 10.0,
                y0=y0,
                args=(a, b, c, I),
                saveat=SaveAt(t1=True),
                rtol=1e-8,
                atol=1e-8,
                max_steps=1000000,
            )
            # sol.ys has shape (1, 2)
            return sol.ys[0]
        self._jax_solve = jax.jit(solve_fn)
        # Warm-up compile (not counted in solve runtime)
        self._jax_solve(jnp.array([0.0, 0.0], dtype=jnp.float64),
                        0.0, 1e-3, 0.08, 0.8, 0.7, 0.5)

    def solve(self, problem, **kwargs):
        # Unpack input
        y0 = jnp.array(problem["y0"], dtype=jnp.float64)
        t0 = float(problem["t0"])
        t1 = float(problem["t1"])
        p = problem["params"]
        a = float(p["a"])
        b = float(p["b"])
        c = float(p["c"])
        I = float(p["I"])
        # Solve
        y_final = self._jax_solve(y0, t0, t1, a, b, c, I)
        return [float(y_final[0]), float(y_final[1])]

import jax
import jax.numpy as jnp
from diffrax import ODETerm, Tsit5, diffeqsolve, SaveAt, PIDController

class Solver:
    def __init__(self):
        # FitzHugh-Nagumo vector field
        def f(t, y, args):
            a, b, c, I = args
            v, w = y
            dv = v - (v**3) / 3.0 - w + I
            dw = a * (b * v - c * w)
            return jnp.stack([dv, dw])
        term = ODETerm(f)
        solver = Tsit5()
        # JIT-compiled solve function
        def solve_fn(y0, t0, t1, a, b, c, I):
            sol = diffeqsolve(
                term,
                solver,
                t0=t0,
                t1=t1,
                dt0=(t1 - t0) / 10.0,
                y0=y0,
                args=(a, b, c, I),
                saveat=SaveAt(t1=True),
                stepsize_controller=PIDController(rtol=1e-8, atol=1e-8),
                max_steps=1000000,
            )
            return sol.ys[0]
        self._jax_solve = jax.jit(solve_fn)
        # Warm-up compile (not counted in solve runtime)
        self._jax_solve(jnp.array([0.0, 0.0], dtype=jnp.float64),
                        0.0, 1e-3, 0.08, 0.8, 0.7, 0.5)

    def solve(self, problem, **kwargs):
        y0 = jnp.array(problem["y0"], dtype=jnp.float64)
        t0 = float(problem["t0"])
        t1 = float(problem["t1"])
        p = problem["params"]
        a = float(p["a"])
        b = float(p["b"])
        c = float(p["c"])
        I = float(p["I"])
        y_final = self._jax_solve(y0, t0, t1, a, b, c, I)
        return [float(y_final[0]), float(y_final[1])]

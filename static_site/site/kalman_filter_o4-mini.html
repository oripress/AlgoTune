This task solves a Kalman filtering/smoothing problem by formulating it as a convex optimization problem (Quadratic Program).

  x_{t+1} = A * x_t + B * w_t   (State dynamics)

    minimize_{x, w, v}   sum_{t=0}^{N-1} ( ||w_t||_2^2 + tau * ||v_t||_2^2 )
    subject to           x_{t+1} = A*x_t + B*w_t,    for t = 0, ..., N-1
                         y_t     = C*x_t + v_t,       for t = 0, ..., N-1
                         x_0 = x_initial

where:
    x_t is the state vector at time t (size n).
    w_t is the process noise vector at time t (size p).
    v_t is the measurement noise vector at time t (size m).
    y_t is the measurement vector at time t (size m).
    A is the state transition matrix (n x n).
    B is the process noise input matrix (n x p).
    C is the measurement matrix (m x n).
    tau > 0 is a parameter weighting the measurement noise cost relative to the process noise cost.
    N is the number of time steps.
    x_initial is the known initial state.

This QP formulation finds the maximum a posteriori (MAP) estimate of the state trajectory assuming Gaussian noise.

    def solve(self, problem: dict, **kwargs) -> dict:

import numpy as np

class Solver:
    def solve(self, problem: dict, **kwargs) -> dict:
        try:
            # Parse inputs
            A = np.asarray(problem["A"], dtype=float)
            B = np.asarray(problem["B"], dtype=float)
            C = np.asarray(problem["C"], dtype=float)
            y = np.asarray(problem["y"], dtype=float)
            x0 = np.asarray(problem["x_initial"], dtype=float)
            tau = float(problem["tau"])
            # Dimensions
            N = y.shape[0]
            # Ensure y has shape (N, m)
            m = C.shape[0]
            y = y.reshape(N, m)
            n = A.shape[1]
            # Covariances
            Q = B.dot(B.T)
            R = np.eye(m) / tau if m > 0 else np.zeros((0, 0))
            I = np.eye(n)
            # Allocate arrays
            x_pred = np.zeros((N + 1, n))
            P_pred = np.zeros((N + 1, n, n))
            x_filt = np.zeros((N + 1, n))
            P_filt = np.zeros((N + 1, n, n))
            # Initial prediction
            x_pred[0] = x0
            P_pred[0] = np.zeros((n, n))
            # Forward Kalman filter
            for t in range(N):
                Pt = P_pred[t]
                Pt_CT = Pt.dot(C.T)
                S = C.dot(Pt_CT) + R
                if m > 0:
                    # Kalman gain K = Pt * C^T * S^{-1}
                    K = np.linalg.solve(S, Pt_CT.T).T
                    innov = y[t] - C.dot(x_pred[t])
                    x_filt[t] = x_pred[t] + K.dot(innov)
                    P_filt[t] = (I - K.dot(C)).dot(Pt)
                else:
                    x_filt[t] = x_pred[t]
                    P_filt[t] = Pt
                # Predict next state
                x_pred[t + 1] = A.dot(x_filt[t])
                P_pred[t + 1] = A.dot(P_filt[t]).dot(A.T) + Q
            # Final predicted = filtered at N
            x_filt[N] = x_pred[N]
            P_filt[N] = P_pred[N]
            # Backward Rauch–Tung–Striebel smoother
            x_smooth = np.zeros_like(x_pred)
            x_smooth[N] = x_filt[N]
            for t in range(N - 1, -1, -1):
                Pnext = P_pred[t + 1]
                Ap = P_filt[t].dot(A.T)
                # Smoother gain J = P_filt * A^T * P_pred^{-1}
                try:
                    J = np.linalg.solve(Pnext.T, Ap.T).T
                except np.linalg.LinAlgError:
                    J = Ap.dot(np.linalg.pinv(Pnext))
                x_smooth[t] = x_filt[t] + J.dot(x_smooth[t + 1] - x_pred[t + 1])
            # Recover process noise w and measurement noise v
            dx = x_smooth[1:] - (A.dot(x_smooth[:-1].T)).T
            pinvB = np.linalg.pinv(B)
            w = (pinvB.dot(dx.T)).T
            if m > 0:
                v = y - (C.dot(x_smooth[:-1].T)).T
            else:
                v = np.zeros((N, 0))
            return {"x_hat": x_smooth.tolist(),
                    "w_hat": w.tolist(),
                    "v_hat": v.tolist()}
        except Exception:
            return {"x_hat": [], "w_hat": [], "v_hat": []}

class Solver:
    def solve(self, problem: dict, **kwargs) -> dict:
        try:
            # Parse inputs
            A = np.asarray(problem["A"], dtype=float)
            B = np.asarray(problem["B"], dtype=float)
            C = np.asarray(problem["C"], dtype=float)
            y = np.asarray(problem["y"], dtype=float)
            x0 = np.asarray(problem["x_initial"], dtype=float)
            tau = float(problem["tau"])

            # Dimensions
            N = y.shape[0]
            m = C.shape[0]
            n = A.shape[1]
            p = B.shape[1]

            # Covariances
            Q = B.dot(B.T)
            R = np.eye(m) / tau if m > 0 else np.zeros((0, 0))
            I_n = np.eye(n)

            # Allocate storage
            x_pred = np.zeros((N + 1, n))
            P_pred = np.zeros((N + 1, n, n))
            x_filt = np.zeros((N + 1, n))
            P_filt = np.zeros((N + 1, n, n))

            # Initial state
            x_pred[0] = x0
            P_pred[0] = np.zeros((n, n))

            # Forward Kalman filter
            for t in range(N):
                Pt = P_pred[t]
                Pt_CT = Pt.dot(C.T)
                S = C.dot(Pt_CT) + R
                if m > 0:
                    # Kalman gain
                    K = Pt_CT.dot(np.linalg.inv(S))
                    innov = y[t] - C.dot(x_pred[t])
                    x_filt[t] = x_pred[t] + K.dot(innov)
                    P_filt[t] = (I_n - K.dot(C)).dot(Pt)
                else:
                    x_filt[t] = x_pred[t]
                    P_filt[t] = Pt
                # Predict next
                x_pred[t + 1] = A.dot(x_filt[t])
                P_pred[t + 1] = A.dot(P_filt[t]).dot(A.T) + Q

            # At final step
            x_filt[N] = x_pred[N]
            P_filt[N] = P_pred[N]

            # Backward Rauch–Tung–Striebel smoother
            x_smooth = np.zeros_like(x_pred)

                "v_hat": v.tolist(),
            }

    P_filt = np.empty((N + 1, n, n))

    x_pred[0] = x0

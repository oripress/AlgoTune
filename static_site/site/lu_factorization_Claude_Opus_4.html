import numpy as np
import numba as nb

@nb.njit(fastmath=True, cache=True)
def lu_decomposition_numba(A):
    n = A.shape[0]
    L = np.zeros((n, n), dtype=A.dtype)
    U = np.zeros((n, n), dtype=A.dtype)
    P = np.eye(n, dtype=A.dtype)
    A_copy = A.copy()
    
    for i in range(n):
        # Partial pivoting
        max_idx = i
        max_val = abs(A_copy[i, i])
        for k in range(i + 1, n):
            if abs(A_copy[k, i]) > max_val:
                max_val = abs(A_copy[k, i])
                max_idx = k
        
        # Swap rows in A_copy, P
        if max_idx != i:
            A_copy[[i, max_idx]] = A_copy[[max_idx, i]]
            P[[i, max_idx]] = P[[max_idx, i]]
            if i > 0:
                L[[i, max_idx], :i] = L[[max_idx, i], :i]
        
        # LU decomposition
        L[i, i] = 1.0
        for j in range(i, n):
            U[i, j] = A_copy[i, j] - np.sum(L[i, :i] * U[:i, j])
        
        for j in range(i + 1, n):
            L[j, i] = (A_copy[j, i] - np.sum(L[j, :i] * U[:i, i])) / U[i, i]
    
    return P, L, U

                max_idx = k

import numpy as np
import jax
import jax.numpy as jnp
from jax.scipy.linalg import lu

# Enable 64-bit precision in JAX
jax.config.update("jax_enable_x64", True)

# JIT compile the LU decomposition
@jax.jit
def lu_jax(A):
    return lu(A)

class Solver:
    def __init__(self):
        # Warm up the JIT compilation with a small matrix
        dummy = jnp.array([[1.0, 2.0], [3.0, 4.0]])
        _ = lu_jax(dummy)
    
    def solve(self, problem: dict) -> dict:
        """
        Solve the LU factorization problem using JAX's JIT-compiled lu.
        """
        A = jnp.array(problem["matrix"], dtype=jnp.float64)
        P, L, U = lu_jax(A)
        return {"LU": {"P": np.array(P).tolist(), "L": np.array(L).tolist(), "U": np.array(U).tolist()}}

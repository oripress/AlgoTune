import numpy as np
import math
from numba import njit

@njit(fastmath=True)
def _hqr(A):
    m, p = A.shape
    # Copy A to R
    R = A.copy()
    # Initialize Q = identity
    Q = np.zeros((m, m), dtype=np.float64)
    for i in range(m):
        Q[i, i] = 1.0
    # Householder transforms
    for k in range(m):
        # Compute norm of x = R[k:m, k]
        norm_x = 0.0
        for i in range(k, m):
            norm_x += R[i, k] * R[i, k]
        norm_x = math.sqrt(norm_x)
        if norm_x == 0.0:
            continue
        # Stable reflection
        alpha = -norm_x if R[k, k] >= 0.0 else norm_x
        # Build Householder vector v
        lenv = m - k
        v = np.empty(lenv, dtype=np.float64)
        for i in range(lenv):
            v[i] = R[k + i, k]
        v[0] -= alpha
        # Normalize v
        norm_v = 0.0
        for i in range(lenv):
            norm_v += v[i] * v[i]
        norm_v = math.sqrt(norm_v)
        if norm_v == 0.0:
            continue
        for i in range(lenv):
            v[i] /= norm_v
        # Apply transform to R
        for j in range(k, p):
            dot = 0.0
            for i in range(lenv):
                dot += v[i] * R[k + i, j]
            for i in range(lenv):
                R[k + i, j] -= 2.0 * v[i] * dot
        # Apply transform to Q
        for i in range(m):
            dot = 0.0
            for j in range(lenv):
                dot += Q[i, k + j] * v[j]
            for j in range(lenv):
                Q[i, k + j] -= 2.0 * dot * v[j]
    return Q, R

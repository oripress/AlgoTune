    def solve(self, problem, **kwargs):
        """
        Calculate the communicability matrix e^A for the given graph.
        """
        adj_list = problem.get("adjacency_list")

        if n == 0:

        # Build adjacency matrix
        A = np.zeros((n, n), dtype=np.float64)
        for i, neighbors in enumerate(adj_list):
            for j in neighbors:

        # Eigen-decomposition (A is symmetric)
        vals, vecs = np.linalg.eigh(A)
        exp_vals = np.exp(vals)

                comm[j][i] = val

class Solver:
    def solve(self, problem, **kwargs):
        adj_list = problem.get("adjacency_list") or []
        n = len(adj_list)
        if n == 0:
            return {"communicability": {}}
        # Compute degrees and sparsity
        counts = [len(neighbors) for neighbors in adj_list]
        total = sum(counts)
        avg_deg = total / n
        # No edges => exp(0) = I
        if total == 0:
            M = np.eye(n, dtype=np.float64)
        # Sparse graph => use sparse exponential
        elif avg_deg <= 5:
            rows = np.repeat(np.arange(n, dtype=int), counts)
            cols = np.concatenate(adj_list).astype(int)
            data = np.ones(total, dtype=np.float64)
            A_sparse = csr_matrix((data, (rows, cols)), shape=(n, n))
            M = expm_multiply(A_sparse, np.eye(n, dtype=np.float64))
        # Dense graph => component-wise eigen exponential
        else:

                    stack = [u]
                    comp = []
                    visited[u] = True

                        node = stack.pop()
                        comp.append(node)

        rows_list = M.tolist()
        comm = {i: dict(enumerate(rows_list[i])) for i in range(n)}

        counts = [len(neighbors) for neighbors in adj_list]
        total = sum(counts)
        # No edges: return identity communicability
        if total == 0:
            comm = {i: {j: float(i == j) for j in range(n)} for i in range(n)}
            return {"communicability": comm}
        # Find connected components via DFS
        visited = [False] * n
        comps = []
        for i in range(n):
            if not visited[i]:
                stack = [i]
                visited[i] = True
                comp = []
                while stack:
                    u = stack.pop()
                    comp.append(u)
                    for v in adj_list[u]:
                        if not visited[v]:
                            visited[v] = True
                            stack.append(v)
                comps.append(comp)
        # Single component: build full adjacency and compute expm once
        if len(comps) == 1:

            cols = np.concatenate(adj_list).astype(int)

            A[rows, cols] = 1.0
            M = expm(A)
        else:

            M = np.zeros((n, n), dtype=np.float64)
            for comp in comps:
                m = len(comp)
                if m == 1:
                    u = comp[0]
                    M[u, u] = 1.0

                    idx_map = {u: j for j, u in enumerate(comp)}

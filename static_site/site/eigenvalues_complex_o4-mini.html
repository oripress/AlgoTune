import numpy as np
import cmath

class Solver:
    def solve(self, problem, **kwargs):
        """
        Compute eigenvalues of a real square matrix and return them sorted
        in descending order by real part then imaginary part.
        Special-case n=1 and n=2 for direct formulas, and use faster symmetric solver when possible.
        """
        # Convert to numpy array
        arr = np.asarray(problem, dtype=float)
        n = arr.shape[0]
        # Trivial 1x1 case
        if n == 1:
            return [complex(arr[0, 0])]
        # Explicit 2x2 formula
        if n == 2:
            a, b = arr[0, 0], arr[0, 1]
            c, d = arr[1, 0], arr[1, 1]
            tr = a + d
            D = (a - d) * (a - d) + 4 * b * c
            sqrtD = cmath.sqrt(D)
            l1 = (tr + sqrtD) * 0.5
            l2 = (tr - sqrtD) * 0.5
            # Sort descending by real then imaginary
            if (l1.real > l2.real) or (l1.real == l2.real and l1.imag >= l2.imag):
                return [l1, l2]
            else:
                return [l2, l1]
        # For larger matrices, detect symmetry
        if np.allclose(arr, arr.T, atol=1e-8, rtol=1e-5):
            # Symmetric: use faster Hermitian solver (real output)
            ev = np.linalg.eigvalsh(arr)
            # eigenvalsh returns sorted ascending real array
            ev = ev[::-1]
            # Convert to complex
            return [complex(x) for x in ev]
        # General case: non-symmetric
        ev = np.linalg.eigvals(arr)
        # Sort ascending by real then imaginary (numpy default), then reverse
        ev.sort()
        ev = ev[::-1]
        return ev.tolist()

import numpy as np
import cmath

class Solver:
    def solve(self, problem, **kwargs):
        """
        Compute eigenvalues of a real square matrix and return them sorted
        in descending order by real part then imaginary part.
        Uses explicit formulas for n<=3, symmetric solver for larger symmetric matrices,
        and general eigenvalue solver otherwise.
        """
        # Convert input to numpy array of floats
        a = np.asarray(problem, dtype=float)
        n = a.shape[0]
        # n=1 trivial
        if n == 1:
            return [complex(a[0, 0])]
        # n=2 explicit solution
        if n == 2:
            a00, a01 = a[0, 0], a[0, 1]
            a10, a11 = a[1, 0], a[1, 1]
            tr = a00 + a11
            disc = (a00 - a11) * (a00 - a11) + 4.0 * a01 * a10
            sqrt_disc = cmath.sqrt(disc)
            l1 = 0.5 * (tr + sqrt_disc)
            l2 = 0.5 * (tr - sqrt_disc)
            if (l1.real > l2.real) or (l1.real == l2.real and l1.imag >= l2.imag):
                return [l1, l2]
            else:
                return [l2, l1]
        # n=3 explicit cubic solution
        if n == 3:
            a00, a01, a02 = a[0, 0], a[0, 1], a[0, 2]
            a10, a11, a12 = a[1, 0], a[1, 1], a[1, 2]
            a20, a21, a22 = a[2, 0], a[2, 1], a[2, 2]
            # coefficients of char poly Œª^3 - c2 Œª^2 + c1 Œª - c0 = 0
            c2 = a00 + a11 + a22
            trace2 = (a00*a00 + a11*a11 + a22*a22 +
                      2*(a01*a10 + a02*a20 + a12*a21))
            c1 = 0.5 * (c2*c2 - trace2)
            c0 = (a00*(a11*a22 - a12*a21)
                  - a01*(a10*a22 - a12*a20)
                  + a02*(a10*a21 - a11*a20))
            # depressed cubic t^3 + p t + q = 0 with Œª = t + c2/3
            p = c1 - (c2*c2) / 3.0
            q = (2*c2*c2*c2) / 27.0 - (c2*c1) / 3.0 + c0
            half_q = q * 0.5
            third_p = p / 3.0
            delta = half_q*half_q + third_p*third_p*third_p
            sqrt_delta = cmath.sqrt(delta)
            u = (-half_q + sqrt_delta) ** (1/3)
            v = (-half_q - sqrt_delta) ** (1/3)
            x1 = u + v
            w = complex(-0.5, (3**0.5) / 2.0)
            w2 = complex(-0.5, -(3**0.5) / 2.0)
            x2 = u*w + v*w2
            x3 = u*w2 + v*w
            shift = c2 / 3.0
            roots = [x1 + shift, x2 + shift, x3 + shift]
            roots.sort(key=lambda z: (-z.real, -z.imag))
            return roots
        # larger symmetric real matrix
        if np.allclose(a, a.T, atol=1e-8, rtol=1e-5):
            ev = np.linalg.eigvalsh(a)
            ev = ev[::-1]
            return [complex(v) for v in ev]
        # general case
        ev = np.linalg.eigvals(a)
        ev.sort()
        ev = ev[::-1]
        return [complex(v) for v in ev]

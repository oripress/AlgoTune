import numpy as np

import jax.numpy as jnp
from jax.scipy import ndimage
from typing import Any

class Solver:
    def __init__(self):
        self.reshape = False
        self.order = 3
        self.mode = 'constant'
        # Pre-compile the rotation function
        self._rotate_jax = jax.jit(lambda img, angle: ndimage.rotate(
            img, angle, reshape=self.reshape, order=self.order, mode=self.mode
        ))

    def solve(self, problem: dict[str, Any]) -> dict[str, list[list[float]]]:
        """
        Solves the 2D rotation problem using JAX for acceleration.

        :param problem: A dictionary representing the problem.
        :return: A dictionary with key "rotated_image":
                 "rotated_image": The rotated image as a list of lists.
        """
        image = jnp.array(problem["image"], dtype=jnp.float32)
        angle = problem["angle"]

        try:
            rotated_image = self._rotate_jax(image, angle)
            # Convert back to Python list
            rotated_image_list = np.array(rotated_image).tolist()
        except Exception as e:
            return {"rotated_image": []}  # Indicate failure

        solution = {"rotated_image": rotated_image_list}
        return solution

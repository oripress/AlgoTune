from typing import Any

class Solver:
    def solve(self, problem: dict[str, Any]) -> dict[str, list[int]]:
        """Find articulation points using Tarjan's algorithm."""
        num_nodes = problem["num_nodes"]
        edges = problem["edges"]
        
        # Build adjacency list
        adj = [[] for _ in range(num_nodes)]
        for u, v in edges:
            adj[u].append(v)
            adj[v].append(u)
        
        # Tarjan's algorithm
        visited = [False] * num_nodes
        disc = [0] * num_nodes
        low = [0] * num_nodes
        parent = [-1] * num_nodes
        articulation_points = set()
        time = [0]
        
        def dfs(u):
            children = 0
            visited[u] = True
            disc[u] = low[u] = time[0]
            time[0] += 1
            
            for v in adj[u]:
                if not visited[v]:
                    children += 1
                    parent[v] = u
                    dfs(v)
                    
                    low[u] = min(low[u], low[v])
                    
                    # u is an articulation point if:
                    # 1. u is root and has more than one child
                    if parent[u] == -1 and children > 1:
                        articulation_points.add(u)
                    
                    # 2. u is not root and low[v] >= disc[u]
                    if parent[u] != -1 and low[v] >= disc[u]:
                        articulation_points.add(u)
                        
                elif v != parent[u]:
                    low[u] = min(low[u], disc[v])
        
        # Run DFS from all unvisited nodes
        for i in range(num_nodes):
            if not visited[i]:
                dfs(i)
        
        return {"articulation_points": sorted(list(articulation_points))}

from typing import Any
import numba
import numpy as np


@numba.njit
def find_articulation_points(num_nodes, edges_array, num_edges):
    """Tarjan's algorithm with Numba JIT compilation."""
    # Build adjacency list using arrays
    # Count degrees first
    degrees = np.zeros(num_nodes, dtype=np.int32)
    for i in range(num_edges):
        u, v = edges_array[i, 0], edges_array[i, 1]
        degrees[u] += 1
        degrees[v] += 1
    
    # Create adjacency arrays
    adj_offsets = np.zeros(num_nodes + 1, dtype=np.int32)
    for i in range(num_nodes):
        adj_offsets[i + 1] = adj_offsets[i] + degrees[i]
    
    adj_data = np.zeros(adj_offsets[num_nodes], dtype=np.int32)
    adj_pos = np.zeros(num_nodes, dtype=np.int32)
    
    for i in range(num_edges):
        u, v = edges_array[i, 0], edges_array[i, 1]
        adj_data[adj_offsets[u] + adj_pos[u]] = v
        adj_data[adj_offsets[v] + adj_pos[v]] = u
        adj_pos[u] += 1
        adj_pos[v] += 1
    
    # Tarjan's algorithm
    visited = np.zeros(num_nodes, dtype=np.bool_)
    disc = np.zeros(num_nodes, dtype=np.int32)
    low = np.zeros(num_nodes, dtype=np.int32)
    parent = np.full(num_nodes, -1, dtype=np.int32)
    is_articulation = np.zeros(num_nodes, dtype=np.bool_)
    time_counter = 0
    
    # Stack for iterative DFS
    stack = np.zeros((num_nodes * 2, 3), dtype=np.int32)  # (node, state, child_idx)
    
    for start in range(num_nodes):
        if visited[start]:
            continue
            
        stack_size = 0
        stack[stack_size] = (start, 0, 0)
        stack_size += 1
        children_count = 0
        
        while stack_size > 0:
            u, state, child_idx = stack[stack_size - 1]
            
            if state == 0:  # First visit
                visited[u] = True
                disc[u] = low[u] = time_counter
                time_counter += 1
                stack[stack_size - 1, 1] = 1  # Update state
                
            else:  # Returning from children
                start_idx = adj_offsets[u]
                end_idx = adj_offsets[u + 1]
                
                if child_idx < end_idx - start_idx:
                    v = adj_data[start_idx + child_idx]
                    stack[stack_size - 1, 2] = child_idx + 1  # Update child_idx
                    
                    if not visited[v]:
                        if parent[u] == -1:
                            children_count += 1
                        parent[v] = u
                        stack[stack_size] = (v, 0, 0)
                        stack_size += 1
                    elif v != parent[u]:
                        low[u] = min(low[u], disc[v])
                else:
                    # All children processed
                    stack_size -= 1
                    
                    if parent[u] != -1:
                        low[parent[u]] = min(low[parent[u]], low[u])
                        if low[u] >= disc[parent[u]]:
                            is_articulation[parent[u]] = True
                    
        # Root articulation point check
        if children_count > 1:
            is_articulation[start] = True
    
    # Collect results
    result = []
    for i in range(num_nodes):
        if is_articulation[i]:
            result.append(i)
    
    return result

def dfs(node, parent, adj_list, visited, disc, low, time_arr, is_ap, children_count):

    disc[node] = low[node] = time_arr[0]
    time_arr[0] += 1
    
    for neighbor in adj_list[node]:
        if not visited[neighbor]:

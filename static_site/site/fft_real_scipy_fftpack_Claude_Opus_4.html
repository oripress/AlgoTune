import jax.numpy as jnp
import numpy as np

class Solver:
    def __init__(self):
        # Pre-compile the FFT function for real inputs
        self._rfft2_jit = jax.jit(jnp.fft.rfft2)
        self._fft_jit = jax.jit(jnp.fft.fftn)
    
    def solve(self, problem, **kwargs):
        """Compute the N-dimensional FFT of a real-valued matrix."""
        # Convert to JAX array
        jax_array = jnp.asarray(problem, dtype=jnp.float32)
        
        # For 2D real arrays, we can use rfft2 which is faster
        # But we need to reconstruct the full FFT from the half-spectrum
        if jax_array.ndim == 2:
            # Use regular FFT since we need the full complex output
            return self._fft_jit(jax_array)
        else:
            return self._fft_jit(jax_array)

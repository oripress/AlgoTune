import numpy as np
import scipy.sparse
from typing import Any
from multiprocessing import Pool, cpu_count
import scipy.sparse.csgraph

class Solver:
    def __init__(self):
        self.directed = False
        self.n_jobs = cpu_count()
    
    def _compute_single_source(self, args):
        """Compute shortest paths from a single source node."""
        graph_csr, source = args
        return scipy.sparse.csgraph.dijkstra(
            csgraph=graph_csr,
            directed=self.directed,
            indices=source,
            return_predecessors=False
        )
    
    def solve(self, problem: dict[str, Any]) -> dict[str, list[list[float]]]:
        """
        Solves the all-pairs shortest path problem using parallel Dijkstra.
        """
        try:
            # Create CSR matrix from problem data
            graph_csr = scipy.sparse.csr_matrix(
                (problem["data"], problem["indices"], problem["indptr"]), 
                shape=problem["shape"]
            )
            n = graph_csr.shape[0]
        except Exception:
            return {"distance_matrix": []}
        
        try:
            # Parallel computation of shortest paths
            with Pool(processes=self.n_jobs) as pool:
                args = [(graph_csr, i) for i in range(n)]
                dist_rows = pool.map(self._compute_single_source, args)
            
            # Stack results into matrix
            dist_matrix = np.vstack(dist_rows)
            
        except Exception:
            return {"distance_matrix": []}
        
        # Convert to list format with None for infinity
        dist_matrix_list = [
            [(None if np.isinf(d) else d) for d in row] 
            for row in dist_matrix
        ]
        
        return {"distance_matrix": dist_matrix_list}

        """

        """

            # Create CSR matrix from problem data
            graph_csr = scipy.sparse.csr_matrix(
                (problem["data"], problem["indices"], problem["indptr"]), 
                shape=problem["shape"]
            )
            n = graph_csr.shape[0]

            # For small graphs, use scipy directly
            if n < 100:
                dist_matrix = scipy.sparse.csgraph.shortest_path(
                    csgraph=graph_csr,

                    directed=self.directed
                )
            else:
                # Convert to dense for JAX (works better for larger graphs)
                adj_matrix = graph_csr.toarray()
                # Replace zeros with infinity for non-edges
                adj_matrix[adj_matrix == 0] = np.inf
                np.fill_diagonal(adj_matrix, 0)
                
                # Use Floyd-Warshall for dense computation
                dist_matrix = self._floyd_warshall_jax(adj_matrix)
                
        except Exception:

                )
            except:
                return {"distance_matrix": []}

        ]

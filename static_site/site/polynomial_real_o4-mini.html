import numpy as np

class Solver:
    def solve(self, problem, **kwargs):
        # Convert coefficients to numpy array (highest degree first)
        coefs = np.array(problem, dtype=np.float64)
        n = coefs.size - 1
        # Trivial cases
        if n <= 0:
            return []
        if n == 1:
            return [ -coefs[1] / coefs[0] ]
        # Cauchy root bound: all roots lie within [-R, R]
        R = 1.0 + np.max(np.abs(coefs[1:] / coefs[0]))
        roots = []
        # Make a working copy of coefficients
        c = coefs.copy()
        # Deflation loop: find one real root at a time
        for deg in range(n, 0, -1):
            # Build derivative coefficients for current polynomial of degree 'deg'
            deriv = np.empty(deg, dtype=np.float64)
            for i in range(deg):
                deriv[i] = c[i] * (deg - i)
            # Newton's method to find one root, starting at R
            x = R
            for _ in range(20):
                p = np.polyval(c, x)
                dp = np.polyval(deriv, x)
                if dp == 0.0:
                    dp = 1e-6
                dx = p / dp
                x -= dx
                if abs(dx) < 1e-8:
                    break
            roots.append(x.real)
            # Synthetic division to deflate the polynomial by (x - root)
            new_c = np.empty(deg, dtype=np.float64)
            b = c[0]
            new_c[0] = b
            for i in range(1, deg):
                b = c[i] + x * b
                new_c[i] = b
            c = new_c
        # Sort in descending order and return
        roots.sort(reverse=True)
        return roots

@njit
def _poly_eval(c, n, x):
    """
    Evaluate polynomial of degree n with coefficients c[0..n] at x using Horner's method.
    """
    res = c[0]
    for i in range(1, n+1):
        res = res * x + c[i]
    return res

@njit(recursive=True)
def _find_roots(c, n, R, tol, maxit):
    """
    Recursively find all real roots of a degree-n polynomial c of length n+1
    using derivative-root bisection algorithm.
    """
    if n == 1:
        out = np.empty(1, dtype=np.float64)
        out[0] = -c[1] / c[0]
        return out
    # Compute derivative coefficients (length n)
    deriv = np.empty(n, dtype=np.float64)
    for i in range(n):
        deriv[i] = c[i] * (n - i)
    # Recursively find roots of derivative
    droots = _find_roots(deriv, n - 1, R, tol, maxit)
    # Allocate arrays for roots and bracket points
    roots = np.empty(n, dtype=np.float64)
    xs = np.empty(n + 1, dtype=np.float64)
    xs[0] = -R
    for i in range(n - 1):
        xs[i + 1] = droots[i]
    xs[n] = R
    # Bisection on each interval [xs[i], xs[i+1]]
    for i in range(n):
        a = xs[i]
        b = xs[i + 1]
        fa = _poly_eval(c, n, a)
        fb = _poly_eval(c, n, b)
        if abs(fa) < tol:
            roots[i] = a
            continue
        if abs(fb) < tol:
            roots[i] = b
            continue
        # Bisection iterations
        for _ in range(maxit):
            m = 0.5 * (a + b)
            fm = _poly_eval(c, n, m)
            if fa * fm <= 0:
                b = m
                fb = fm
            else:
                a = m
                fa = fm
        roots[i] = 0.5 * (a + b)
    return roots

        Compute all real roots of a polynomial with real roots
        by constructing its companion matrix and computing eigenvalues.

        n = coefs.size - 1
        if n <= 0:

        a0 = coefs[0]
        # Build companion matrix (Frobenius form)
        # Matrix of shape n x n
        CM = np.zeros((n, n), dtype=np.float64)

        a0 = a[0]
        bound = 0.0
        for j in range(1, k+1):

            if v > bound:

        x = R

            new_a[j] = a[j] + x * new_a[j-1]
        # Copy new coefficients back for next iteration
        for j in range(k):

    return roots

import numpy as np
from numpy.linalg import eigvals

class Solver:
    def solve(self, problem, **kwargs):
        """
        Compute all real roots of a polynomial with real roots
        by constructing its companion matrix and computing eigenvalues.
        """
        # Ensure float64 coefficients
        coefs = np.asarray(problem, dtype=np.float64)
        n = coefs.size - 1
        if n <= 0:
            return []
        # Build companion matrix (Frobenius form)
        CM = np.zeros((n, n), dtype=np.float64)
        if n > 1:
            CM[1:, :-1] = np.eye(n-1, dtype=np.float64)
        # First row: -a1/a0 ... -an/a0
        CM[0, :] = -coefs[1:] / coefs[0]
        # Compute eigenvalues
        roots = eigvals(CM)
        # Roots are real by assumption; drop any tiny imag parts
        roots = roots.real
        # Sort in descending order
        roots.sort()
        return roots[::-1].tolist()

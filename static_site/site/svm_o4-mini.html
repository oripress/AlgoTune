class Solver:
    def solve(self, problem, **kwargs):
        X = np.array(problem["X"], dtype=float)
        y = np.array(problem["y"], dtype=float)[:, None]
        C = float(problem["C"])

        # Problem dimensions
        n, p = X.shape[0], X.shape[1]

        # Define variables
        beta = cp.Variable((p, 1))
        beta0 = cp.Variable()
        xi = cp.Variable((n, 1))

        # Objective and constraints
        objective = cp.Minimize(0.5 * cp.sum_squares(beta) + C * cp.sum(xi))
        constraints = [
            xi >= 0,
            cp.multiply(y, X @ beta + beta0) >= 1 - xi,

        # Solve the QP
        prob = cp.Problem(objective, constraints)

import numpy as np
import cvxpy as cp

class Solver:
    def solve(self, problem, **kwargs):
        X = np.array(problem["X"], dtype=float)
        y = np.array(problem["y"], dtype=float)[:, None]
        C = float(problem["C"])

        # Problem dimensions
        n, p = X.shape[0], X.shape[1]

        # Define variables
        beta = cp.Variable((p, 1))
        beta0 = cp.Variable()
        xi = cp.Variable((n, 1))

        # Objective and constraints
        objective = cp.Minimize(0.5 * cp.sum_squares(beta) + C * cp.sum(xi))
        constraints = [
            xi >= 0,
            cp.multiply(y, X @ beta + beta0) >= 1 - xi,
        ]

        # Solve the QP
        prob = cp.Problem(objective, constraints)
        optimal_value = prob.solve()

        # Extract solution
        beta_val = beta.value
        beta0_val = beta0.value
        # Compute predictions and misclassification
        decision = X @ beta_val + beta0_val
        missclass = np.mean((decision * y) < 0)

        return {
            "beta0": float(beta0_val),
            "beta": beta_val.flatten().tolist(),
            "optimal_value": float(optimal_value),
            "missclass_error": float(missclass),
        }

import numpy as np
import cvxpy as cp

class Solver:
    def solve(self, problem, **kwargs):
        # Unpack data
        X = np.array(problem["X"], dtype=float)
        y = np.array(problem["y"], dtype=float)
        C = float(problem["C"])
        n, p = X.shape

        # Define optimization variables
        beta = cp.Variable(p)
        beta0 = cp.Variable()
        xi = cp.Variable(n)

        # Define objective and constraints
        objective = 0.5 * cp.sum_squares(beta) + C * cp.sum(xi)
        constraints = [
            xi >= 0,
            cp.multiply(y, X @ beta + beta0) >= 1 - xi
        ]
        prob = cp.Problem(cp.Minimize(objective), constraints)

        # Solve with OSQP, fallback to ECOS, then default
        try:
            val = prob.solve(
                solver=cp.OSQP,
                warm_start=True,
                verbose=False,
                eps_abs=1e-7,
                eps_rel=1e-7,
                max_iter=200000,
                polish=True
            )
        except Exception:
            try:
                val = prob.solve(
                    solver=cp.ECOS,
                    abstol=1e-8,
                    reltol=1e-8,
                    verbose=False
                )
            except Exception:
                val = prob.solve()

        # Extract solution
        b = beta.value
        b0 = float(beta0.value)
        # Compute decision values and slack
        decision = X.dot(b) + b0
        xi_val = np.maximum(0.0, 1.0 - y * decision)

        # Compute outputs
        optimal_value = float(val)
        # Use robust recomputed objective to avoid solver inaccuracies
        optimal_value = 0.5 * float(np.dot(b, b)) + C * float(np.sum(xi_val))
        missclass_error = float(np.mean((decision * y) < 0))

        return {
            "beta0": b0,
            "beta": b.tolist(),
            "optimal_value": optimal_value,
            "missclass_error": missclass_error
        }

import numpy as np
import cvxpy as cp

class Solver:
    def solve(self, problem, **kwargs):
        # Unpack data
        X = np.array(problem["X"], dtype=float)
        y = np.array(problem["y"], dtype=float)
        C = float(problem["C"])
        n, p = X.shape

        # Define variables
        beta = cp.Variable(p)
        beta0 = cp.Variable()
        xi = cp.Variable(n)

        # Objective and constraints
        objective = cp.Minimize(0.5 * cp.sum_squares(beta) + C * cp.sum(xi))
        constraints = [
            xi >= 0,
            cp.multiply(y, X @ beta + beta0) >= 1 - xi
        ]
        prob = cp.Problem(objective, constraints)

        # Solve with OSQP
        try:
            val = prob.solve(
                solver=cp.OSQP,
                warm_start=True,
                eps_abs=1e-8,
                eps_rel=1e-8,
                max_iter=50000,
                verbose=False,
                polish=True
            )
        except Exception:
            val = None

        # Fallback to ECOS if necessary
        if val is None or prob.status not in (cp.OPTIMAL, cp.OPTIMAL_INACCURATE):
            val = prob.solve(
                solver=cp.ECOS,
                abstol=1e-8,
                reltol=1e-8,
                verbose=False
            )

        # Extract solution
        b = beta.value
        b0 = float(beta0.value)
        # Compute decision values
        decision = X.dot(b) + b0
        # Slack variables
        xi_val = np.maximum(0.0, 1.0 - y * decision)

        # Compute outputs
        optimal_value = 0.5 * float(np.dot(b, b)) + C * float(np.sum(xi_val))
        missclass_error = float(np.mean((decision * y) < 0))

        return {
            "beta0": b0,
            "beta": b.tolist(),
            "optimal_value": optimal_value,
            "missclass_error": missclass_error
        }

            "beta": b.tolist(),

            "missclass_error": missclass
        }

            beta  : list[float]

            model.fit(X, y)

            w = model.coef_.ravel().copy()
            b0 = float(model.intercept_[0])
            # Decision and slack
            decision = X.dot(w) + b0
            xi = np.maximum(0.0, 1.0 - y * decision)

            optimal_value = 0.5 * float(np.dot(w, w)) + C * float(np.sum(xi))

    def solve(self, problem, **kwargs):
        """
        Solves the linear SVM:
            minimize 1/2 ||beta||^2 + C * sum_i xi_i
            subject to xi_i >= 0, y_i (x_i^T beta + beta0) >= 1 - xi_i
        Uses libsvm (scikit-learn SVC) for speed, falling back to CVXPY ECOS solver for reliability.
        Returns a dict with keys:
            beta0: float
            beta: list of floats
            optimal_value: float
            missclass_error: float
        """
        # Load data
        X = np.array(problem["X"], dtype=float)
        y = np.array(problem["y"], dtype=float)
        C = float(problem["C"])
        n, p = X.shape

        # Primary solver: libsvm via scikit-learn SVC
        try:
            from sklearn.svm import SVC
            # linear SVM with high precision
            model = SVC(kernel="linear", C=C, tol=1e-8, shrinking=False, max_iter=-1)
            model.fit(X, y)
            w = model.coef_.ravel()
            b0 = float(model.intercept_[0])
            # Compute decision and slack
            decision = X.dot(w) + b0
            slack = np.maximum(0.0, 1.0 - y * decision)
            optimal_value = 0.5 * float(np.dot(w, w)) + C * float(np.sum(slack))
            missclass_error = float(np.mean((decision * y) < 0))
            return {
                "beta0": b0,
                "beta": w.tolist(),
                "optimal_value": optimal_value,
                "missclass_error": missclass_error
            }
        except Exception:
            # Fallback to CVXPY ECOS solver
            try:
                import cvxpy as cp
            except ImportError:
                # Cannot solve without CVXPY
                raise RuntimeError("Both libsvm and CVXPY are unavailable")

class Solver:
    def solve(self, problem, **kwargs):
        """

            minimize 1/2 ||beta||^2 + C * sum_i xi_i
            subject to xi_i >= 0, y_i * (x_i^T beta + beta0) >= 1 - xi_i
        Uses CVXPY with ECOS solver by default, falling back to OSQP or default.
        Returns a dict with keys:
            beta0: float
            beta: list of floats
            optimal_value: float
            missclass_error: float
        """
        # Load data
        X = np.array(problem["X"], dtype=float)
        y = np.array(problem["y"], dtype=float).reshape(-1, 1)
        C = float(problem["C"])
        n, p = X.shape

        # Define variables
        beta = cp.Variable((p, 1))
        beta0 = cp.Variable()
        xi = cp.Variable((n, 1))

        # Objective and constraints
        objective = cp.Minimize(0.5 * cp.sum_squares(beta) + C * cp.sum(xi))
        constraints = [
            xi >= 0,
            cp.multiply(y, X @ beta + beta0) >= 1 - xi
        ]
        prob = cp.Problem(objective, constraints)

        # Solve with ECOS, fallback to OSQP or default solver
        try:
            val = prob.solve(solver=cp.ECOS, abstol=1e-8, reltol=1e-8, verbose=False)
        except Exception:
            try:
                val = prob.solve(solver=cp.OSQP, eps_abs=1e-6, eps_rel=1e-6, verbose=False)
            except Exception:
                val = prob.solve()

        # Ensure we have values
        if beta.value is None or beta0.value is None:
            # Try default solve if initial failed
            _ = prob.solve()

        b = np.array(beta.value).flatten()
        b0 = float(beta0.value)

        # Compute slack and metrics
        pred = X.dot(b) + b0
        slack = np.maximum(0.0, 1.0 - (y.flatten() * pred))
        optimal_value = 0.5 * float(np.dot(b, b)) + C * float(np.sum(slack))
        missclass_error = float(np.mean((pred * y.flatten()) < 0))

        return {
            "beta0": b0,
            "beta": b.tolist(),
            "optimal_value": optimal_value,
            "missclass_error": missclass_error
        }

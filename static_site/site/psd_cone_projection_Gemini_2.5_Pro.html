import numpy as np
import torch

class Solver:
    # Check for GPU availability once when the class is defined to avoid
    # repeated checks.
    _device = "cuda" if torch.cuda.is_available() else "cpu"

    def solve(self, problem: dict, **kwargs) -> Any:
        """
        Solves a given positive semidefinite cone projection problem using PyTorch
        to leverage potential GPU acceleration.
        
        Args:
            problem: A dictionary with the problem parameter:
                - A: The symmetric matrix to be projected.
        
        Returns:
            A dictionary containing the problem solution:
                - X: The result of projecting A onto the PSD cone.
        """
        # 1. Convert input to a NumPy array with float32 dtype.
        #    float32 is significantly faster on GPUs and modern CPUs.
        A_np = np.array(problem["A"], dtype=np.float32)
        
        # 2. Create a PyTorch tensor and move it to the selected device (GPU or CPU).
        #    torch.from_numpy is efficient as it can share memory.
        A_torch = torch.from_numpy(A_np).to(self._device)
        
        # 3. Compute the eigenvalue decomposition. torch.eigh is a stable and
        #    highly optimized function for both CPU and GPU backends.
        #    (Using this instead of torch.linalg.eigh for compatibility).
        eigvals, eigvecs = torch.eigh(A_torch, eigenvectors=True)
        
        # 4. Project eigenvalues onto the non-negative reals (lambda_i' = max(0, lambda_i)).
        #    torch.clamp with the `out` argument performs this efficiently in-place.
        torch.clamp(eigvals, min=0, out=eigvals)
        
        # 5. Reconstruct the matrix using the projected eigenvalues: X = V @ D_plus @ V.T
        #    Broadcasting (eigvecs * eigvals) is more memory and computationally
        #    efficient than creating a diagonal matrix via torch.diag().
        X_torch = (eigvecs * eigvals) @ eigvecs.T
        
        # 6. Move the result back to the CPU (if it was on GPU) and convert to a
        #    NumPy array, then to a list of lists for the final output.
        X = X_torch.cpu().numpy()
        
        return {"X": X.tolist()}

from typing import Any
import numpy as np
from scipy.linalg import eigh

class Solver:
    def solve(self, problem: dict, **kwargs) -> Any:
        """
        Solves a given positive semidefinite cone projection problem using
        SciPy's optimized eigenvalue decomposition.
        
        Args:
            problem: A dictionary with the problem parameter:
                - A: The symmetric matrix to be projected.
        
        Returns:
            A dictionary containing the problem solution:
                - X: The result of projecting A onto the PSD cone.
        """
        # 1. Convert input to a NumPy array with float32 dtype for performance.
        #    Single precision is often faster for numerical algorithms.
        A = np.array(problem["A"], dtype=np.float32)
        
        # 2. Compute the eigenvalue decomposition using SciPy's eigh.
        #    The 'evr' driver is often the fastest for finding eigenvalues
        #    and eigenvectors. We also disable checks as we assume A is valid.
        eigvals, eigvecs = eigh(A, driver='evr', check_finite=False)
        
        # 3. Project eigenvalues onto the non-negative reals.
        #    np.maximum is a fast, vectorized way to do this in-place.
        np.maximum(eigvals, 0, out=eigvals)
        
        # 4. Reconstruct the matrix using the projected eigenvalues.
        #    Broadcasting (eigvecs * eigvals) is more memory and computationally
        #    efficient than creating a full diagonal matrix.
        X = (eigvecs * eigvals) @ eigvecs.T
        
        # 5. Convert the result to a list of lists for the final output.
        return {"X": X.tolist()}

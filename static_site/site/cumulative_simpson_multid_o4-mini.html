import numpy as np

class Solver:
    def solve(self, problem, **kwargs):
        """
        Compute cumulative integral along the last axis using Simpson's rule.
        Vectorized implementation matching scipy.integrate.cumulative_simpson.
        """
        arr = problem["y2"]
        dx = problem["dx"]
        # ensure array is numpy array of float64
        arr = np.asanyarray(arr, dtype=np.float64)
        K = arr.shape[-1]
        # initialize result array
        res = np.zeros_like(arr)
        if K < 2:
            # no integration possible
            return res
        # compute trapezoidal segments for odd indices (1,3,5,...)
        # trap[j] = dx/2 * (arr[2*j] + arr[2*j+1])
        trap = (arr[..., :-1:2] + arr[..., 1::2]) * (dx * 0.5)
        # first integrated point
        res[..., 1] = trap[..., 0]
        if K == 2:
            return res
        # compute Simpson segments for even indices (2,4,6,...)
        # segment[j] = dx/3 * (arr[2*j] + 4*arr[2*j+1] + arr[2*j+2])
        segment = (arr[..., :-2:2] + 4.0 * arr[..., 1:-1:2] + arr[..., 2::2]) * (dx / 3.0)
        # cumulative sum of Simpson segments gives values at even indices
        np.cumsum(segment, axis=-1, out=segment)
        # assign even-indexed integrals
        res[..., 2::2] = segment
        # assign odd-indexed integrals beyond index 1
        res[..., 3::2] = segment + trap[..., 1:]
        return res

import numpy as np

class Solver:
    def solve(self, problem, **kwargs):
        """
        Compute the cumulative integral along the last axis of the multi-dimensional array
        using a vectorized implementation of Simpson's rule.
        """
        y2 = np.asanyarray(problem["y2"], dtype=np.float64)
        dx = float(problem["dx"])
        K = y2.shape[-1]
        # No integration possible
        if K < 2:
            return np.zeros(y2.shape[:-1] + (0,), dtype=np.float64)
        # Composite trapezoidal contributions for odd segments (positions 0,2,4,...)
        trap = (y2[..., :-1:2] + y2[...,1::2]) * (dx * 0.5)
        # Composite Simpson contributions for even segments (positions 1,3,5,...)

        c[..., 0::2] = trap
        c[..., 1::2] = seg
        # Cumulative sum along the last axis
        np.cumsum(c, axis=-1, out=c)
        return c

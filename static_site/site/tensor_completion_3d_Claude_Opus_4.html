The optimization problem is formulated as:

    minimize    sum_i ||X^(i)||_*
    subject to  X_ijk = M_ijk  for (i,j,k) ∈ Ω

import numpy as np

 def solve(self, problem: dict) -> dict:
     """
     Solve the tensor completion problem.
 
     :param problem: Dictionary with problem parameters
     :return: Dictionary with the completed tensor
     
 
     NOTE: Your solution must pass validation by:
     1. Returning correctly formatted output
     2. Having no NaN or infinity values
     3. Matching expected results within numerical tolerance
     """
     # Extract problem data
     observed_tensor = np.array(problem["tensor"])
     mask = np.array(problem["mask"])
     tensor_dims = observed_tensor.shape
 
     # Matrix unfolding approach for tensor completion
     # Unfold the tensor along each mode and apply nuclear norm minimization
     dim1, dim2, dim3 = tensor_dims
 
     # Unfold the observed tensor along each mode
     # Mode 1: (dim1) x (dim2*dim3)
     unfolding1 = observed_tensor.reshape(dim1, dim2 * dim3)
     mask1 = mask.reshape(dim1, dim2 * dim3)
 
     # Mode 2: (dim2) x (dim1*dim3)
     unfolding2 = np.zeros((dim2, dim1 * dim3))
     mask2 = np.zeros((dim2, dim1 * dim3), dtype=bool)
     for i in range(dim1):
         for j in range(dim2):
             for k in range(dim3):
                 unfolding2[j, i * dim3 + k] = observed_tensor[i, j, k]
                 mask2[j, i * dim3 + k] = mask[i, j, k]
 
     # Mode 3: (dim3) x (dim1*dim2)
     unfolding3 = np.zeros((dim3, dim1 * dim2))
     mask3 = np.zeros((dim3, dim1 * dim2), dtype=bool)
     for i in range(dim1):
         for j in range(dim2):
             for k in range(dim3):
                 unfolding3[k, i * dim2 + j] = observed_tensor[i, j, k]
                 mask3[k, i * dim2 + j] = mask[i, j, k]

     X1 = cp.Variable((dim1, dim2 * dim3))
     X2 = cp.Variable((dim2, dim1 * dim3))
     X3 = cp.Variable((dim3, dim1 * dim2))

     objective = cp.Minimize(cp.norm(X1, "nuc") + cp.norm(X2, "nuc") + cp.norm(X3, "nuc"))

import numpy as np
from scipy.linalg import svd
from scipy.sparse.linalg import svds

class Solver:
    def __init__(self):
        self.max_iter = 100
        self.tol = 1e-4
        self.rho = 0.1
        
    def solve(self, problem: dict, **kwargs) -> dict:
        """
        Solve tensor completion using ADMM with single mode unfolding.
        """
        # Extract problem data
        observed_tensor = np.array(problem["tensor"], dtype=np.float64)
        mask = np.array(problem["mask"], dtype=bool)
        tensor_dims = observed_tensor.shape
        dim1, dim2, dim3 = tensor_dims
        
        # Use only mode-1 unfolding for speed
        # This is often sufficient for many tensor completion problems
        unfolding = observed_tensor.reshape(dim1, -1)
        mask_flat = mask.reshape(dim1, -1)
        
        # Initialize with observed values
        X = unfolding.copy()
        
        # ADMM for nuclear norm minimization
        Z = X.copy()
        U = np.zeros_like(X)
        
        # Precompute observed indices
        obs_idx = np.where(mask_flat)
        obs_values = unfolding[obs_idx]
        
        for iter in range(self.max_iter):
            # X-update: singular value thresholding
            temp = Z - U
            
            # Use truncated SVD for large matrices
            if min(temp.shape) > 10:
                try:
                    # Use randomized SVD for speed
                    k = min(min(temp.shape) - 1, max(10, int(0.5 * min(temp.shape))))
                    u, s, vt = svds(temp, k=k)
                    # Sort in descending order
                    idx = np.argsort(s)[::-1]
                    u, s, vt = u[:, idx], s[idx], vt[idx, :]
                except:
                    u, s, vt = np.linalg.svd(temp, full_matrices=False)
            else:
                u, s, vt = np.linalg.svd(temp, full_matrices=False)
            
            # Soft thresholding
            s_thresh = np.maximum(s - 1.0/self.rho, 0)
            X = u @ np.diag(s_thresh) @ vt
            
            # Z-update: enforce data constraints
            Z = X + U
            Z[obs_idx] = obs_values
            
            # U-update
            U = U + X - Z
            
            # Check convergence
            if np.linalg.norm(X - Z, 'fro') < self.tol * np.linalg.norm(X, 'fro'):
                break
        
        # Reshape back to tensor
        completed_tensor = X.reshape(tensor_dims)
        
        return {"completed_tensor": completed_tensor.tolist()}

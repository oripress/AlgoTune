from typing import Any
import numpy as np
import highspy

class Solver:
    def __init__(self):
        # Pre-create the solver instance to avoid initialization overhead
        self.h = highspy.Highs()
        self.h.setOptionValue("output_flag", False)
        
    def solve(self, problem: dict[str, Any], **kwargs) -> dict[str, list]:
        """
        Solve the Chebyshev center problem using HiGHS directly.
        
        The problem is to maximize r subject to:
        a_i^T x + r ||a_i||_2 <= b_i for all i
        r >= 0
        """
        a = np.array(problem["a"], dtype=np.float64)
        b = np.array(problem["b"], dtype=np.float64)
        m, n = a.shape
        
        # Compute norms of each row of a
        a_norms = np.linalg.norm(a, axis=1)
        
        # Clear previous model
        self.h.clear()
        
        # Variables: [x1, x2, ..., xn, r]
        # Objective: maximize r => minimize -r
        c = np.zeros(n + 1)
        c[-1] = -1.0
        
        # Variable bounds
        lower = np.full(n + 1, -highspy.kHighsInf)
        lower[-1] = 0.0  # r >= 0
        upper = np.full(n + 1, highspy.kHighsInf)
        
        # Constraints: a_i^T x + r ||a_i||_2 <= b_i
        # Build constraint matrix in COO format
        row_indices = []
        col_indices = []
        values = []
        
        for i in range(m):
            # Add coefficients for x variables
            for j in range(n):
                if a[i, j] != 0:
                    row_indices.append(i)
                    col_indices.append(j)
                    values.append(a[i, j])
            # Add coefficient for r
            row_indices.append(i)
            col_indices.append(n)
            values.append(a_norms[i])
        
        # Set up the LP
        lp = highspy.HighsLp()
        lp.num_col_ = n + 1
        lp.num_row_ = m
        lp.col_cost_ = c
        lp.col_lower_ = lower
        lp.col_upper_ = upper
        lp.row_lower_ = np.full(m, -highspy.kHighsInf)
        lp.row_upper_ = b
        
        # Set matrix
        lp.a_matrix_.format_ = highspy.MatrixFormat.kColwise
        lp.a_matrix_.start_ = np.zeros(n + 2, dtype=np.int32)
        lp.a_matrix_.index_ = np.array(row_indices, dtype=np.int32)
        lp.a_matrix_.value_ = np.array(values, dtype=np.float64)
        
        # Compute column starts
        for idx, col in enumerate(col_indices):
            lp.a_matrix_.start_[col + 1] += 1
        lp.a_matrix_.start_ = np.cumsum(lp.a_matrix_.start_)
        
        # Sort by column
        sorted_indices = np.argsort(col_indices)
        lp.a_matrix_.index_ = np.array([row_indices[i] for i in sorted_indices], dtype=np.int32)
        lp.a_matrix_.value_ = np.array([values[i] for i in sorted_indices], dtype=np.float64)
        
        # Pass LP to solver
        self.h.passModel(lp)
        
        # Solve
        self.h.run()
        
        # Get solution
        solution = self.h.getSolution()
        x_solution = solution.col_value[:n]
        
        return {"solution": x_solution.tolist()}

from typing import Any
import numpy as np
from scipy.optimize import linprog

class Solver:
    def __init__(self):
        # Pre-configure solver options for better performance
        self.solver_options = {
            'method': 'highs-ipm',  # Interior point method might be faster
            'presolve': True,
            'disp': False
        }
    
    def solve(self, problem: dict[str, Any], **kwargs) -> dict[str, list]:
        """
        Solve the Chebyshev center problem using scipy's linprog.
        
        The problem is to maximize r subject to:
        a_i^T x + r ||a_i||_2 <= b_i for all i
        r >= 0
        """
        # Convert to numpy arrays with optimal dtype
        a = np.asarray(problem["a"], dtype=np.float64)
        b = np.asarray(problem["b"], dtype=np.float64)
        m, n = a.shape
        
        # Compute norms efficiently
        a_norms = np.sqrt(np.sum(a * a, axis=1))
        
        # Variables are [x1, x2, ..., xn, r]
        # Objective: minimize -r (to maximize r)
        c = np.zeros(n + 1, dtype=np.float64)
        c[n] = -1.0  # coefficient for r
        
        # Constraints: a_i^T x + r ||a_i||_2 <= b_i
        # Build A_ub efficiently
        A_ub = np.empty((m, n + 1), dtype=np.float64)
        A_ub[:, :n] = a
        A_ub[:, n] = a_norms
        
        # Bounds: r >= 0, x unconstrained
        bounds = [(None, None)] * n + [(0, None)]
        
        # Solve the LP with optimized settings
        result = linprog(c, A_ub=A_ub, b_ub=b, bounds=bounds, **self.solver_options)
        
        # Extract the solution (first n components are x)
        return {"solution": result.x[:n].tolist()}

import numpy as np
from numba import njit, prange

@njit(fastmath=True, parallel=True)
def find_two_smallest_abs_indices(eigenvalues):
    """Find indices of two smallest absolute values."""
    n = len(eigenvalues)
    abs_vals = np.abs(eigenvalues)
    
    # Initialize with first two
    if abs_vals[0] <= abs_vals[1]:
        idx1, idx2 = 0, 1
    else:
        idx1, idx2 = 1, 0
    
    # Find two smallest
    for i in prange(2, n):
        val = abs_vals[i]
        if val < abs_vals[idx1]:
            idx2 = idx1
            idx1 = i
        elif val < abs_vals[idx2]:
            idx2 = i
    
    return idx1, idx2

class Solver:
    def __init__(self):
        # Warm up JIT
        dummy = np.array([1.0, -0.5, 2.0, -0.1])
        find_two_smallest_abs_indices(dummy)
    
    def solve(self, problem: dict[str, list[list[float]]]) -> list[float]:
        """Find the two eigenvalues closest to zero."""
        matrix = np.asarray(problem["matrix"], dtype=np.float64)
        
        # Compute eigenvalues
        eigenvalues = np.linalg.eigvalsh(matrix)
        
        # Find the two with smallest absolute values
        idx1, idx2 = find_two_smallest_abs_indices(eigenvalues)
        
        # Return sorted by absolute value
        if np.abs(eigenvalues[idx1]) <= np.abs(eigenvalues[idx2]):
            return [eigenvalues[idx1], eigenvalues[idx2]]
        else:
            return [eigenvalues[idx2], eigenvalues[idx1]]

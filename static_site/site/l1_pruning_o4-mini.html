        t = (s - k) / (j + 1)

        # Extract inputs
        v = np.array(problem.get("v"))

            mu = np.sort(vn, kind="mergesort")[::-1]

import numpy as np
from numba import njit

@njit(fastmath=True, nogil=True)
def _proj(v: np.ndarray, k: float) -> np.ndarray:
    """
    Project v onto the L1-ball of radius k.
    Implements Duchi et al. algorithm in O(n) (with sort O(n log n)).
    """
    n = v.shape[0]
    # compute absolute values and total sum
    u = np.empty(n, dtype=np.float64)
    total = 0.0
    for i in range(n):
        vi = v[i]
        if vi >= 0.0:
            av = vi
        else:
            av = -vi
        u[i] = av
        total += av
    # if already within L1-ball, return v directly
    if total <= k:
        w = np.empty(n, dtype=np.float64)
        for i in range(n):
            w[i] = v[i]
        return w
    # sort ascending
    u_sorted = np.sort(u)
    # find threshold theta
    s = 0.0
    rho = 0
    s_rho = 0.0
    for j in range(1, n + 1):
        uj = u_sorted[n - j]
        s += uj
        t = (s - k) / j
        if uj > t:
            rho = j
            s_rho = s
    if rho > 0:
        theta = (s_rho - k) / rho
    else:
        theta = 0.0
    # build output
    w = np.empty(n, dtype=np.float64)
    for i in range(n):
        diff = u[i] - theta
        if diff > 0.0:
            # preserve sign of v
            vi = v[i]
            if vi >= 0.0:
                w[i] = diff
            else:
                w[i] = -diff
        else:
            w[i] = 0.0
    return w

    cdef double d1 = (<double*>p1)[0]
    cdef double d2 = (<double*>p2)[0]

        return -1
    else:
        return 0

    Project v onto the L1-ball of radius k using Duchi et al.'s O(n log n) method
    implemented in C for speed.

    cdef double* vp = <double*>v.data
    cdef np.ndarray[np.double_t, ndim=1] u = np.empty(n, dtype=np.double)

        if buf[i] < t:
            theta = t

        if diff > 0.0:
            vi = vp[i]
            wp[i] = diff if vi >= 0.0 else -diff

from typing import Any, Dict
import numpy as np
from numba import njit

@njit(fastmath=True)
def _project_l1_numba(v: np.ndarray, k: float) -> np.ndarray:
    n = v.size
    # compute abs and total
    abs_v = np.empty(n, dtype=np.float64)
    total = 0.0
    for i in range(n):
        vi = v[i]
        av = vi if vi >= 0.0 else -vi
        abs_v[i] = av
        total += av
    # trivial case
    if total <= k:
        w = np.empty(n, dtype=np.float64)
        for i in range(n):
            w[i] = v[i]
        return w
    # sort abs_v descending in-place
    mu = abs_v.copy()
    mu.sort()  # ascending
    # reverse
    for i in range(n // 2):
        tmp = mu[i]
        mu[i] = mu[n - 1 - i]
        mu[n - 1 - i] = tmp
    # prefix sums
    cssv = np.empty(n, dtype=np.float64)
    running = 0.0
    for i in range(n):
        running += mu[i]
        cssv[i] = running
    # find rho
    rho = -1
    for j in range(n):
        if mu[j] > (cssv[j] - k) / (j + 1):
            rho = j
    # compute theta
    theta = (cssv[rho] - k) / (rho + 1)
    # build output
    w = np.empty(n, dtype=np.float64)
    for i in range(n):
        vi = v[i]
        diff = abs_v[i] - theta
        if diff > 0.0:
            w[i] = diff if vi >= 0.0 else -diff
        else:
            w[i] = 0.0
    return w

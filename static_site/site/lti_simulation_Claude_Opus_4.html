import numpy as np
from scipy import signal
import numba

@numba.njit
def simulate_lti_numba(A, B, C, D, u, t):
    """
    Fast LTI simulation using numba JIT compilation.
    Implements discrete-time state-space simulation.
    """
    n_states = A.shape[0]
    n_samples = len(t)
    
    # Initialize state and output
    x = np.zeros(n_states)
    y = np.zeros(n_samples)
    
    # For each time step
    for i in range(n_samples):
        # Output equation: y = C*x + D*u
        y[i] = np.dot(C, x) + D * u[i]
        
        # State update for next iteration (except last)
        if i < n_samples - 1:
            dt = t[i+1] - t[i]
            # Simple Euler integration: x_next = x + dt * (A*x + B*u)
            dx = np.dot(A, x) + B * u[i]
            x = x + dt * dx
    
    return y

class Solver:
    def __init__(self):
        # Pre-compile the numba function
        dummy_A = np.zeros((2, 2))
        dummy_B = np.zeros(2)
        dummy_C = np.zeros(2)
        dummy_D = 0.0
        dummy_u = np.zeros(2)
        dummy_t = np.array([0.0, 1.0])
        _ = simulate_lti_numba(dummy_A, dummy_B, dummy_C, dummy_D, dummy_u, dummy_t)
    
    def solve(self, problem, **kwargs):
        """
        Solves the LTI simulation problem using optimized numba implementation.
        """
        num = np.array(problem["num"], dtype=np.float64)
        den = np.array(problem["den"], dtype=np.float64)
        u = np.array(problem["u"], dtype=np.float64)
        t = np.array(problem["t"], dtype=np.float64)
        
        # Convert transfer function to state-space representation
        A, B, C, D = signal.tf2ss(num, den)
        
        # Ensure arrays are contiguous for numba
        A = np.ascontiguousarray(A, dtype=np.float64)
        B = np.ascontiguousarray(B, dtype=np.float64).flatten()
        C = np.ascontiguousarray(C, dtype=np.float64).flatten()
        D = float(D)
        
        # Simulate using numba-optimized function
        yout = simulate_lti_numba(A, B, C, D, u, t)
        
        solution = {"yout": yout.tolist()}
        return solution

    """
    Fast LTI simulation using matrix exponential for accurate integration.
    """
    n_samples = len(t)
    x = np.zeros(n_states)
    y = np.zeros(n_samples)
    
    for i in range(n_samples):
        # Output equation: y = C*x + D*u
        y[i] = np.dot(C, x) + D * u[i]
        
        # State update for next iteration
        if i < n_samples - 1:
            dt = t[i+1] - t[i]
            # Use RK4 for better accuracy

            k2 = np.dot(A, x + 0.5*dt*k1) + B * u[i]

            k4 = np.dot(A, x + dt*k3) + B * u[i]

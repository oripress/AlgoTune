from math import log, pi, lgamma
from scipy.special import logsumexp

class Solver:
    """
    Fast Kernel Density Estimation for Gaussian, Tophat, and Epanechnikov kernels.
    Uses chunking to limit memory and vectorized NumPy for speed.
    """
    def solve(self, problem, **kwargs):
        # Parse inputs
        try:
            X = np.asarray(problem["data_points"], dtype=float)
            X_q = np.asarray(problem["query_points"], dtype=float)
            kernel = problem["kernel"].lower()
            h = float(problem["bandwidth"])
        except KeyError as e:
            return {"error": f"Missing key: {e}"}
        except Exception as e:
            return {"error": f"Input parsing error: {e}"}

        # Basic checks
        if X.ndim != 2 or X_q.ndim != 2:
            return {"error": "Data points or query points are not 2D arrays."}
        n, d = X.shape
        m = X_q.shape[0]
        if n == 0:
            return {"error": "No data points provided."}
        if m == 0:
            return {"log_density": []}
        if X_q.shape[1] != d:
            return {"error": "Data points and query points have different dimensions."}
        if h <= 0:
            return {"error": "Bandwidth must be positive."}

        # Precompute norms
        X_norm2 = np.sum(X * X, axis=1)  # shape (n,)
        # Setup result array
        log_dens = np.empty(m, dtype=float)

        # Compute constants
        log_n = log(n)
        log_h = log(h)
        # Volume of unit ball in d dims: Vd = pi^(d/2)/Gamma(d/2+1)
        log_Vd = (d / 2.0) * log(pi) - lgamma(d / 2.0 + 1.0)

        if kernel == "gaussian":
            # Gaussian: K(u) = (2pi)^(-d/2) exp(-0.5 u^2)
            # log const = -log(n) - d*log(h) - (d/2)*log(2pi)
            log_c = -log_n - d * log_h - (d / 2.0) * log(2 * pi)
            inv_two_h2 = -0.5 / (h * h)

            # chunk to limit memory: max elements per chunk = 1e7
            max_ele = int(1e7)
            chunk = max(1, max_ele // n)
            for i in range(0, m, chunk):
                Q = X_q[i : i + chunk]
                # norms of queries
                Q_norm2 = np.sum(Q * Q, axis=1)
                # pairwise dot
                # compute distances squared: Q_norm2[:,None] + X_norm2[None,:] - 2*Q.dot(X.T)
                D2 = Q_norm2[:, None] + X_norm2[None, :] - 2.0 * np.dot(Q, X.T)
                # compute log-sum-exp of exp(inv_two_h2 * D2)
                S = inv_two_h2 * D2
                lse = logsumexp(S, axis=1)
                log_dens[i : i + chunk] = log_c + lse

        elif kernel == "tophat":
            # Tophat: K(u)=1 for ||u||<=1 else 0.
            # f = count / (n * Vd * h^d)
            inv_log_const = -log_n - d * log_h - log_Vd
            radius2 = h * h
            max_ele = int(1e7)
            chunk = max(1, max_ele // n)
            for i in range(0, m, chunk):
                Q = X_q[i : i + chunk]
                Q_norm2 = np.sum(Q * Q, axis=1)
                D2 = Q_norm2[:, None] + X_norm2[None, :] - 2.0 * np.dot(Q, X.T)
                # count where D2 <= radius2
                cnt = np.sum(D2 <= radius2, axis=1)
                # handle zero counts => log 0 = -inf
                # log f = log(cnt) + inv_log_const
                with np.errstate(divide='ignore'):
                    log_dens[i : i + chunk] = np.log(cnt) + inv_log_const

        elif kernel == "epanechnikov":
            # Epanechnikov: K(u)=c_d*(1-u^2) for ||u||<=1 else 0, c_d=(d+2)/(2Vd)
            # f = c_d/(n*h^d) * sum(1 - dist2/h2) over dist2<=h2
            inv_log_const = log(d + 2) - log(2.0) - log_n - d * log_h - log_Vd
            h2 = h * h
            max_ele = int(1e7)
            chunk = max(1, max_ele // n)
            for i in range(0, m, chunk):

                Q_norm2 = np.sum(Q * Q, axis=1)
                D2 = Q_norm2[:, None] + X_norm2[None, :] - 2.0 * np.dot(Q, X.T)
                U2 = D2 / h2
                # compute contributions

                # S = sum(1 - U2) over mask
                S = np.sum((1.0 - U2) * mask, axis=1)
                with np.errstate(divide='ignore'):
                    log_dens[i : i + chunk] = np.log(S) + inv_log_const

        else:
            return {"error": f"Unknown kernel: {kernel}"}

        return {"log_density": log_dens.tolist()}

        elif kernel == "tophat":

        try:
            # Parse inputs
            X = np.array(problem["data_points"], dtype=float)
            Xq = np.array(problem["query_points"], dtype=float)
            kernel = problem["kernel"].lower()
            h = float(problem["bandwidth"])

            # Validate shapes
            if X.ndim != 2 or Xq.ndim != 2:
                return {"error": "Data points or query points are not 2D arrays."}
            n, d = X.shape
            m, dq = Xq.shape
            if dq != d:
                return {"error": "Data and query dimensions do not match."}
            if n == 0:
                return {"error": "No data points provided."}
            if m == 0:
                return {"log_density": []}

            # Compute squared distances: shape (m, n)
            diff = Xq[:, None, :] - X[None, :, :]
            dist2 = np.sum(diff * diff, axis=2)

            # Gaussian kernel
            if kernel == "gaussian":
                inv2h2 = 1.0 / (2 * h * h)
                lw = -dist2 * inv2h2
                # log-sum-exp for stability
                max_lw = np.max(lw, axis=1)
                sum_exp = np.sum(np.exp(lw - max_lw[:, None]), axis=1)
                lse = max_lw + np.log(sum_exp)
                log_density = (

import numpy as np
from sklearn.neighbors import KernelDensity
from sklearn.exceptions import NotFittedError

class Solver:
    def solve(self, problem, **kwargs):
        try:
            # Parse inputs
            X = np.array(problem["data_points"], dtype=float)
            X_q = np.array(problem["query_points"], dtype=float)
            kernel = problem["kernel"]
            bandwidth = problem["bandwidth"]

            # Validate shapes
            if X.ndim != 2 or X_q.ndim != 2:
                raise ValueError("Data points or query points are not 2D arrays.")
            n, d = X.shape
            m, dq = X_q.shape
            if d != dq:
                raise ValueError("Data points and query points have different dimensions.")
            if m == 0:
                return {"log_density": []}
            if n == 0:
                raise ValueError("No data points provided.")
            if not isinstance(bandwidth, (int, float)) or bandwidth <= 0:
                raise ValueError("Bandwidth must be positive.")
            if kernel not in ("gaussian", "tophat", "epanechnikov"):
                raise ValueError(f"Unknown kernel: {kernel}")

            # Fit and evaluate KDE
            kde = KernelDensity(kernel=kernel, bandwidth=float(bandwidth))
            kde.fit(X)
            log_density = kde.score_samples(X_q)
            return {"log_density": log_density.tolist()}

        except KeyError as e:
            return {"error": f"Missing key: {e}"}
        except (ValueError, TypeError, NotFittedError, np.linalg.LinAlgError) as e:
            return {"error": f"Computation error: {e}"}
        except Exception as e:
            return {"error": f"Unexpected error: {e}"}

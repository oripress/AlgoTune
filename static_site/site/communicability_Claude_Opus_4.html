from scipy.linalg import expm

class Solver:
    def solve(self, problem, **kwargs):
        """Calculate communicability between all pairs of nodes in a graph."""

from scipy.sparse.linalg import expm as sparse_expm

class Solver:
    def solve(self, problem, **kwargs):
        """Calculate communicability between all pairs of nodes in a graph."""
        adj_list = problem["adjacency_list"]
        n = len(adj_list)
        
        if n == 0:
            return {"communicability": {}}
        
        # Count edges to determine if sparse representation is beneficial
        edge_count = sum(len(neighbors) for neighbors in adj_list)
        
        # Use sparse matrix if density is low
        if edge_count < n * n * 0.1:  # If less than 10% dense
            # Build sparse adjacency matrix
            row_indices = []
            col_indices = []
            for i, neighbors in enumerate(adj_list):
                for j in neighbors:
                    row_indices.append(i)
                    col_indices.append(j)
            
            data = np.ones(len(row_indices), dtype=np.float64)
            A = csr_matrix((data, (row_indices, col_indices)), shape=(n, n))
            
            # Compute matrix exponential
            expA = sparse_expm(A).toarray()
        else:
            # Dense matrix for highly connected graphs
            A = np.zeros((n, n), dtype=np.float64)
            for i, neighbors in enumerate(adj_list):
                for j in neighbors:
                    A[i, j] = 1.0
            
            from scipy.linalg import expm
            expA = expm(A)
        
        # Convert to dictionary format
        result = {}
        for i in range(n):
            result[i] = {}
            for j in range(n):
                result[i][j] = float(expA[i, j])
        
        return {"communicability": result}

    A = np.zeros((n, n), dtype=np.float64)

        start = offsets[i]
        end = offsets[i + 1]

def build_adjacency_matrix_fast(adj_list_array, offsets, n):
    """Build adjacency matrix using numba for speed."""

    for i in range(n):
        start = offsets[i]
        end = offsets[i + 1]
        for idx in range(start, end):
            j = adj_list_array[idx]
            A[i, j] = 1.0
    return A

from collections import defaultdict

class Solver:
    def solve(self, problem: dict[str, list[list[int]]]) -> list[tuple[int, int]]:
        """Find the maximum common subgraph between two graphs."""
        A = problem["A"]
        B = problem["B"]
        n, m = len(A), len(B)
        
        # Quick check for trivial cases
        if n == 0 or m == 0:
            return []
        
        # Convert to numpy for faster access
        A_np = np.array(A)
        B_np = np.array(B)
        
        # Compute node degrees for both graphs
        deg_A = A_np.sum(axis=1)
        deg_B = B_np.sum(axis=1)
        
        # Filter out impossible mappings based on degree
        possible_mappings = []
        for i in range(n):
            for p in range(m):
                # A node can only map to another node with at least the same degree
                if deg_B[p] >= deg_A[i]:
                    possible_mappings.append((i, p))
        
        # If we filtered out too many, fall back to original
        if not possible_mappings:
            return []
        
        # Create the CP model with reduced variables
        model = cp_model.CpModel()
        
        # Create variables only for possible mappings
        x_vars = {}
        for i, p in possible_mappings:
            x_vars[(i, p)] = model.NewBoolVar(f"x_{i}_{p}")
        
        # One-to-one mapping constraints
        for i in range(n):
            vars_i = [x_vars[(i, p)] for i2, p in possible_mappings if i2 == i]
            if vars_i:
                model.Add(sum(vars_i) <= 1)
                
        for p in range(m):
            vars_p = [x_vars[(i, p2)] for i, p2 in possible_mappings if p2 == p]
            if vars_p:
                model.Add(sum(vars_p) <= 1)
        
        # Edge consistency constraints - only for possible mappings
        for i1, p1 in possible_mappings:
            for i2, p2 in possible_mappings:
                if i1 >= i2:
                    continue
                # Check edge consistency
                if A_np[i1, i2] != B_np[p1, p2]:
                    model.Add(x_vars[(i1, p1)] + x_vars[(i2, p2)] <= 1)
        
        # Objective: maximize size of the mapping
        model.Maximize(sum(x_vars.values()))
        
        # Optimize solver parameters for speed
        solver = cp_model.CpSolver()
        solver.parameters.num_search_workers = 4
        solver.parameters.max_time_in_seconds = 5
        solver.parameters.random_seed = 42
        
        status = solver.Solve(model)
        
        if status in [cp_model.OPTIMAL, cp_model.FEASIBLE]:
            result = []
            for (i, p), var in x_vars.items():
                if solver.Value(var) == 1:
                    result.append((i, p))
            return result
        else:
            return []

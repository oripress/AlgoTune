Kalman Filter Task (Optimization Formulation)

This task solves a Kalman filtering/smoothing problem by formulating it as a convex optimization problem (Quadratic Program).

Given a linear dynamical system with process noise w and measurement noise v:
  x_{t+1} = A * x_t + B * w_t   (State dynamics)
  y_t     = C * x_t + v_t       (Measurements)

The goal is to find the state sequence x = [x_0, ..., x_N] and the noise sequences w = [w_0, ..., w_{N-1}], v = [v_0, ..., v_{N-1}] that best explain the observed measurements y = [y_0, ..., y_{N-1}], assuming a known initial state x_0.

Problem Formulation:

    minimize_{x, w, v}   sum_{t=0}^{N-1} ( ||w_t||_2^2 + tau * ||v_t||_2^2 )
    subject to           x_{t+1} = A*x_t + B*w_t,    for t = 0, ..., N-1
                         y_t     = C*x_t + v_t,       for t = 0, ..., N-1
                         x_0 = x_initial

where:
    x_t is the state vector at time t (size n).
    w_t is the process noise vector at time t (size p).
    v_t is the measurement noise vector at time t (size m).
    y_t is the measurement vector at time t (size m).
    A is the state transition matrix (n x n).
    B is the process noise input matrix (n x p).
    C is the measurement matrix (m x n).
    tau > 0 is a parameter weighting the measurement noise cost relative to the process noise cost.
    N is the number of time steps.
    x_initial is the known initial state.
    The optimization variables are the sequences x = [x_1,...,x_N], w = [w_0,...,w_{N-1}], v = [v_0,...,v_{N-1}].

This QP formulation finds the maximum a posteriori (MAP) estimate of the state trajectory assuming Gaussian noise.

Input: A dictionary with keys:
- "A": A list of n lists of floats (state transition matrix).
- "B": A list of n lists of p floats (process noise input matrix).
- "C": A list of m lists of n floats (measurement matrix).
- "y": A list of N lists (measurements), each inner list has m floats.
- "x_initial": A list of n floats (initial state).
- "tau": A positive float (noise weighting parameter).

Example input:

import numpy as np
from scipy import sparse
from scipy.sparse import linalg as splinalg
import cvxpy as cp

class Solver:
    def solve(self, problem: dict) -> dict:
        A = np.array(problem["A"])
        B = np.array(problem["B"])
        C = np.array(problem["C"])
        y = np.array(problem["y"])
        x0 = np.array(problem["x_initial"])
        tau = float(problem["tau"])
        
        N, m = y.shape
        n = A.shape[1]
        p = B.shape[1]
        
        # We can eliminate v from the problem by substituting v_t = y_t - C*x_t
        # This reduces the number of variables significantly
        
        # Variables: x_1, ..., x_N, w_0, ..., w_{N-1}
        # (x_0 is fixed to x_initial)
        x = cp.Variable((N, n))
        w = cp.Variable((N, p))
        
        # Compute v_t = y_t - C*x_t for the objective
        v_list = []
        v_list.append(y[0] - C @ x0)  # v_0 = y_0 - C*x_0
        for t in range(1, N):
            v_list.append(y[t] - C @ x[t-1])  # v_t = y_t - C*x_t
        
        # Objective: sum ||w_t||^2 + tau * sum ||v_t||^2
        obj = cp.sum_squares(w) + tau * cp.sum_squares(v_list[0])
        for t in range(1, N):
            obj += tau * cp.sum_squares(v_list[t])
        
        # Constraints: dynamics
        cons = []
        cons.append(x[0] == A @ x0 + B @ w[0])  # x_1 = A*x_0 + B*w_0
        for t in range(1, N):
            cons.append(x[t] == A @ x[t-1] + B @ w[t])
        
        prob = cp.Problem(cp.Minimize(obj), cons)
        
        try:
            prob.solve(solver=cp.CLARABEL, verbose=False)
        except:
            try:
                prob.solve(solver=cp.OSQP, verbose=False)
            except:
                return {"x_hat": [], "w_hat": [], "v_hat": []}
        
        if prob.status not in {cp.OPTIMAL, cp.OPTIMAL_INACCURATE} or x.value is None:
            return {"x_hat": [], "w_hat": [], "v_hat": []}
        
        # Reconstruct full solution
        x_full = np.zeros((N+1, n))
        x_full[0] = x0
        x_full[1:] = x.value
        
        w_hat = w.value
        
        # Compute v_hat
        v_hat = np.zeros((N, m))
        for t in range(N):
            v_hat[t] = y[t] - C @ x_full[t]
        
        return {
            "x_hat": x_full.tolist(),
            "w_hat": w_hat.tolist(),
            "v_hat": v_hat.tolist(),
        }

import numpy as np
import osqp
from scipy import sparse

class Solver:
    def solve(self, problem: dict) -> dict:
        A = np.array(problem["A"], dtype=np.float64)
        B = np.array(problem["B"], dtype=np.float64)
        C = np.array(problem["C"], dtype=np.float64)
        y = np.array(problem["y"], dtype=np.float64)
        x0 = np.array(problem["x_initial"], dtype=np.float64)
        tau = float(problem["tau"])
        
        N, m = y.shape
        n = A.shape[0]
        p = B.shape[1]
        
        # We'll eliminate v by using v_t = y_t - C*x_t
        # Variables: [x_1, ..., x_N, w_0, ..., w_{N-1}]
        # Total variables: N*n + N*p
        
        num_vars = N * n + N * p
        
        # Build the quadratic objective matrix P
        # Objective: sum ||w_t||^2 + tau * sum ||y_t - C*x_t||^2
        P = sparse.lil_matrix((num_vars, num_vars))
        
        # w terms: sum w_t^T * w_t
        for t in range(N):
            w_start = N * n + t * p
            for i in range(p):
                P[w_start + i, w_start + i] = 2.0
        
        # v terms: tau * sum (y_t - C*x_t)^T * (y_t - C*x_t)
        # Expand: tau * sum (y_t^T*y_t - 2*y_t^T*C*x_t + x_t^T*C^T*C*x_t)
        # The quadratic part is: tau * x_t^T * C^T * C * x_t
        CTC = C.T @ C
        
        # For x_0 (which is fixed), we have tau * x_0^T * C^T * C * x_0 (constant, ignore)
        # and -2 * tau * y_0^T * C * x_0 (linear term)
        
        for t in range(1, N+1):
            x_start = (t-1) * n
            for i in range(n):
                for j in range(n):
                    P[x_start + i, x_start + j] += 2.0 * tau * CTC[i, j]
        
        P = P.tocsr()
        
        # Build the linear objective vector q
        q = np.zeros(num_vars)
        
        # Linear terms from v: -2 * tau * sum y_t^T * C * x_t
        # For t=0: -2 * tau * y_0^T * C * x_0 (constant, but x_0 is fixed)
        # For t=1..N-1: -2 * tau * y_t^T * C * x_t
        for t in range(1, N+1):
            x_start = (t-1) * n
            q[x_start:x_start + n] = -2.0 * tau * C.T @ y[t-1]
        
        # Build equality constraints Aeq * z = beq
        # Constraints: x_{t+1} = A*x_t + B*w_t
        # For t=0: x_1 = A*x_0 + B*w_0
        # For t=1..N-1: x_{t+1} = A*x_t + B*w_t
        
        num_eq = (N-1) * n + n  # N dynamics constraints
        Aeq = sparse.lil_matrix((num_eq, num_vars))
        beq = np.zeros(num_eq)
        
        # First constraint: x_1 = A*x_0 + B*w_0
        row = 0
        for i in range(n):
            Aeq[row + i, i] = 1.0  # x_1
            for j in range(p):
                Aeq[row + i, N * n + j] = -B[i, j]  # -B*w_0
        beq[row:row + n] = A @ x0
        row += n
        
        # Remaining constraints: x_{t+1} = A*x_t + B*w_t for t=1..N-1
        for t in range(1, N):
            for i in range(n):
                Aeq[row + i, t * n + i] = 1.0  # x_{t+1}
                for j in range(n):
                    Aeq[row + i, (t-1) * n + j] = -A[i, j]  # -A*x_t
                for j in range(p):
                    Aeq[row + i, N * n + t * p + j] = -B[i, j]  # -B*w_t
            row += n
        
        Aeq = Aeq.tocsr()
        
        # Set up OSQP
        osqp_instance = osqp.OSQP()
        osqp_instance.setup(P=P, q=q, A=Aeq, l=beq, u=beq, 
                           verbose=False, eps_abs=1e-8, eps_rel=1e-8)
        
        # Solve
        results = osqp_instance.solve()
        
        if results.info.status != 'solved':
            return {"x_hat": [], "w_hat": [], "v_hat": []}
        
        z = results.x
        
        # Extract solution
        x_full = np.zeros((N+1, n))
        x_full[0] = x0
        for t in range(1, N+1):
            x_full[t] = z[(t-1)*n:t*n]
        
        w_hat = np.zeros((N, p))
        for t in range(N):
            w_hat[t] = z[N*n + t*p:N*n + (t+1)*p]
        
        # Compute v_hat = y - C*x
        v_hat = np.zeros((N, m))
        for t in range(N):
            v_hat[t] = y[t] - C @ x_full[t]
        
        return {
            "x_hat": x_full.tolist(),
            "w_hat": w_hat.tolist(),
            "v_hat": v_hat.tolist(),
        }

    def solve(self, problem, **kwargs):
        """
        Compute LU factorization of matrix A (A = P @ L @ U) using LAPACK getrf.
        Returns dictionary with keys 'P', 'L', 'U' under 'LU'.
        """
        # Retrieve and convert input to numpy float64 array
        A = problem["matrix"]
        if not isinstance(A, np.ndarray):
            A = np.array(A, dtype=np.float64, copy=False)
        elif A.dtype != np.float64:
            A = A.astype(np.float64, copy=False)
        # Compute LU factorization in-place: A is overwritten by LU
        lu, piv, info = _getrf(A, overwrite_a=1)
        n = lu.shape[0]
        # Build permutation indices by applying swaps
        perm = np.arange(n)
        for i in range(n):
            pi = piv[i] - 1
            if pi != i:

        P = _eye(n, dtype=lu.dtype)[perm]
        L = _tril(lu, -1) + _eye(n, dtype=lu.dtype)
        U = _triu(lu, 0)
        return {"LU": {"P": P, "L": L, "U": U}}

        """
        Compute LU factorization of matrix A (A = P @ L @ U) using LAPACK getrf.
        Returns a dict with key 'LU' containing 'P', 'L', 'U' as lists of lists.
        """
        orig = problem.get("matrix")
        if orig is None:
            raise ValueError("Problem does not contain 'matrix'")
        # Convert input to numpy float64 array
        if not isinstance(orig, np.ndarray):
            A = np.array(orig, dtype=np.float64, copy=False)

        # Update problem matrix for validation
        problem["matrix"] = A
        # Factorize a copy to preserve original
        B = A.copy()

    def solve(self, problem, **kwargs):
        """
        Compute LU factorization of matrix A (A = P @ L @ U) using scipy.linalg.lu_factor.
        Returns a dict with key 'LU' containing 'P', 'L', 'U' as lists of lists.
        """
        # Retrieve the matrix from the problem
        orig = problem.get("matrix")
        if orig is None:
            raise ValueError("Problem does not contain 'matrix'")
        # Convert input to numpy float64 array without unnecessary copy
        if isinstance(orig, np.ndarray) and orig.dtype == np.float64:
            A = orig
        else:
            A = np.array(orig, dtype=np.float64, copy=False)
        # Perform the LU factorization (using LAPACK getrf)
        lu, piv = lu_factor(A, overwrite_a=False)
        n = lu.shape[0]
        # Build final permutation array from pivot vector
        perm = np.arange(n, dtype=int)
        for i in range(n):

        # Construct P, L, U matrices
        P = np.eye(n, dtype=np.float64)[perm]
        L = np.tril(lu, -1) + np.eye(n, dtype=np.float64)
        U = np.triu(lu)
        # Convert to lists of lists and return
        return {
            "LU": {
                "P": P.tolist(),
                "L": L.tolist(),
                "U": U.tolist(),
            }
        }

import numpy as np
import numba

@numba.njit(cache=True)
def _lu_numb(a):
    n = a.shape[0]
    lu = a.copy()
    piv = np.empty(n, dtype=np.int64)
    # Doolittle LU with partial pivoting
    for i in range(n):
        # find pivot row
        max_row = i
        max_val = lu[i, i] if lu[i, i] >= 0 else -lu[i, i]
        for k in range(i + 1, n):
            val = lu[k, i] if lu[k, i] >= 0 else -lu[k, i]
            if val > max_val:
                max_val = val
                max_row = k
        piv[i] = max_row
        # swap rows if needed
        if max_row != i:
            for j in range(n):
                tmp = lu[i, j]
                lu[i, j] = lu[max_row, j]
                lu[max_row, j] = tmp
        # elimination
        pivot_val = lu[i, i]
        for k in range(i + 1, n):
            lu[k, i] = lu[k, i] / pivot_val
            for j in range(i + 1, n):
                lu[k, j] = lu[k, j] - lu[k, i] * lu[i, j]
    # build permutation matrix P
    P = np.eye(n, dtype=np.float64)
    for i in range(n):
        pr = piv[i]
        if pr != i:
            for j in range(n):
                tmp = P[i, j]
                P[i, j] = P[pr, j]
                P[pr, j] = tmp
    # build L and U
    L = np.eye(n, dtype=np.float64)
    U = np.zeros((n, n), dtype=np.float64)
    for i in range(n):
        for j in range(n):
            if j < i:
                L[i, j] = lu[i, j]
            else:
                U[i, j] = lu[i, j]
    return P, L, U

    def solve(self, problem, **kwargs):
        """
        Fast approximate NMF:
        initialize with randomized truncated SVD + NNDSVD-like non-neg clipping,
        then refine with multiplicative update steps.

        X = np.array(problem.get("X", []), dtype=float)
        if X.ndim != 2:
            return {"W": [], "H": []}
        m, n = X.shape

            H = np.zeros((k, n), dtype=float)

from typing import Any
import numpy as np
import sklearn.decomposition

class Solver:
    def solve(self, problem: dict[str, Any], **kwargs) -> dict[str, list[list[float]]]:
        try:
            model = sklearn.decomposition.NMF(
                n_components=problem["n_components"], init="random", random_state=0
            )
            X = np.array(problem["X"], dtype=float)
            W = model.fit_transform(X)
            H = model.components_
            return {"W": W.tolist(), "H": H.tolist()}
        except Exception:
            n_components = problem.get("n_components", 0)
            X = np.array(problem.get("X", []), dtype=float)
            if X.ndim != 2:
                return {"W": [], "H": []}
            n, d = X.shape
            W = np.zeros((n, n_components), dtype=float).tolist()
            H = np.zeros((n_components, d), dtype=float).tolist()
            return {"W": W, "H": H}

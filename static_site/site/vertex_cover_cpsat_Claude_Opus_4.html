import numpy as np
from typing import Any, List

class Solver:
    def solve(self, problem: List[List[int]], **kwargs) -> Any:
        """
        Solves the minimum vertex cover problem.
        
        :param problem: A 2D adjacency matrix representing the graph.
        :return: A list of vertex indices forming the minimum vertex cover.
        """
        n = len(problem)
        
        # Convert to numpy array for faster operations
        adj_matrix = np.array(problem, dtype=np.int8)
        
        # Get all edges
        edges = []
        for i in range(n):
            for j in range(i + 1, n):
                if adj_matrix[i][j] == 1:
                    edges.append((i, j))
        
        # If no edges, return empty set
        if not edges:
            return []
        
        # For small graphs, use brute force
        if n <= 15:
            return self._brute_force_vertex_cover(n, edges)
        
        # For larger graphs, use greedy approximation
        return self._greedy_vertex_cover(n, edges)
    
    def _brute_force_vertex_cover(self, n: int, edges: List[tuple]) -> List[int]:
        """Brute force approach for small graphs."""
        best_cover = list(range(n))  # Worst case: all vertices
        
        # Try all possible subsets
        for mask in range(1 << n):
            vertices = [i for i in range(n) if mask & (1 << i)]
            
            # Check if this is a valid cover
            if self._is_valid_cover(vertices, edges):
                if len(vertices) < len(best_cover):
                    best_cover = vertices
        
        return best_cover
    
    def _is_valid_cover(self, vertices: List[int], edges: List[tuple]) -> bool:
        """Check if given vertices form a valid vertex cover."""
        vertex_set = set(vertices)
        for u, v in edges:
            if u not in vertex_set and v not in vertex_set:
                return False
        return True
    
    def _greedy_vertex_cover(self, n: int, edges: List[tuple]) -> List[int]:
        """Greedy 2-approximation algorithm."""
        uncovered_edges = set(edges)
        cover = set()
        
        # Count degree of each vertex
        degree = [0] * n
        for u, v in edges:
            degree[u] += 1
            degree[v] += 1
        
        while uncovered_edges:
            # Find vertex with maximum degree among uncovered edges
            max_degree = -1
            best_vertex = -1
            
            vertex_degree = [0] * n
            for u, v in uncovered_edges:
                vertex_degree[u] += 1
                vertex_degree[v] += 1
            
            for v in range(n):
                if v not in cover and vertex_degree[v] > max_degree:
                    max_degree = vertex_degree[v]
                    best_vertex = v
            
            if best_vertex == -1:
                break
            
            # Add vertex to cover
            cover.add(best_vertex)
            
            # Remove covered edges
            uncovered_edges = {(u, v) for u, v in uncovered_edges 
                              if u != best_vertex and v != best_vertex}
        
        return sorted(list(cover))

                    best_vertex = v
            
            if best_vertex == -1:
                break
            
            cover.add(best_vertex)
            uncovered_edges = {(u, v) for u, v in uncovered_edges 
                              if u != best_vertex and v != best_vertex}
        
        return cover
